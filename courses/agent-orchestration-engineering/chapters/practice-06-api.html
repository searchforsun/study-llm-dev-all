FastAPI

python
@app.post("/v1/agent/chat")
async def chat(req: ChatRequest, x_thread_id: str = Header(...)):
    config = {"configurable": {"thread_id": x_thread_id}}
    result = await graph.ainvoke({"messages": req.messages}, config)
    return ChatResponse(messages=result["messages"])
