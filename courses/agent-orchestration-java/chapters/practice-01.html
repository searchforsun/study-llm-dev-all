本章要点速记

Sequential/Parallel/Routing 模式

三种基础编排模式覆盖多 Agent 协作的大部分场景。Sequential 按顺序执行，Parallel 同时执行，Routing 根据条件分配到不同 Agent。

// Sequential 模式
@Component
public class SequentialOrchestrator {

    public String execute(String userId) {
        // 先查用户
        AgentResult userResult = userAgent.execute("查用户信息: " + userId);
        String dept = userResult.getData("department");

        // 再查工单
        AgentResult ticketResult = ticketAgent.execute(
            "查 " + userId + " 的待处理工单");

        // 最后汇总
        return summaryAgent.execute(
            "汇总: 用户部门=" + dept + ", 工单=" + ticketResult.getOutput());
    }
}

// Parallel 模式
@Component
public class ParallelOrchestrator {

    public Map executeParallel(String userId) {
        CompletableFuture hrFuture =
            CompletableFuture.supplyAsync(() ->
                hrAgent.execute("查询 HR 信息: " + userId));

        CompletableFuture itFuture =
            CompletableFuture.supplyAsync(() ->
                itAgent.execute("查询 IT 信息: " + userId));

        CompletableFuture ticketFuture =
            CompletableFuture.supplyAsync(() ->
                ticketAgent.execute("查询工单: " + userId));

        // 等待所有并行任务完成
        Map results = Map.of(
            "hr", hrFuture.join(),
            "it", itFuture.join(),
            "ticket", ticketFuture.join()
        );
        return results;
    }
}

// Routing 模式（基于 StateGraph）
public class RoutingWorkflow {
    public static StateGraph build() {
        StateGraph graph = new StateGraph<>(String.class);

        graph.addNode("router", state -> {
            if (state.contains("HR") || state.contains("请假")) {
                return "hr_agent";
            } else if (state.contains("IT") || state.contains("VPN")) {
                return "it_agent";
            }
            return "general_agent";
        });

        graph.addConditionalEdge("router",
            Function.identity(),
            Map.of("hr_agent", "hr_agent",
                   "it_agent", "it_agent",
                   "general_agent", "general_agent")
        );
        return graph;
    }
}

主管-工人模式

主管-工人模式中，Supervisor Agent 负责任务规划、分配和结果整合，Worker Agent 专注于执行特定子任务。这是最常用的多 Agent 模式。

java
// 主管 Agent：拆解任务并分配给工人
@Component
public class SupervisorAgent {

    private final ChatClient chatClient;
    private final List workers;

    public AgentResult execute(String task) {
        // 1. 规划：用 LLM 拆解任务
        String plan = chatClient.prompt()
            .user("""
                将以下任务拆解为子任务列表（JSON 数组格式）：
                任务：%s
                可用的工人：查用户、查日程、发邮件、查工单

                返回格式：[{"worker":"查用户","task":"查询张三的信息"}, ...]
                """.formatted(task))
            .call()
            .content();

        // 2. 分配：解析 JSON 并分发给对应工人
        List subTasks = parsePlan(plan);
        List> futures = new ArrayList<>();

        for (SubTask subTask : subTasks) {
            WorkerAgent worker = findWorker(subTask.worker());
            futures.add(CompletableFuture.supplyAsync(() ->
                worker.execute(subTask.task())
            ));
        }

        // 3. 收集结果
        List results = futures.stream()
            .map(CompletableFuture::join)
            .toList();

        // 4. 汇总：LLM 整合所有结果
        String summary = chatClient.prompt()
            .user("请整合以下结果：\n" + formatResults(results))
            .call()
            .content();

        return new AgentResult(summary, results);
    }
}

CorpAssist 场景

员工说"帮我查一下张三的工单状态，然后通知他处理"。Supervisor 拆解为：1) QueryUserWorker(查张三的信息) → 2) QueryTicketWorker(查张三工单) → 3) SendEmailWorker(给张某发邮件)。三个 Worker 中 1 和 2 可以并行（没有数据依赖），3 需要 1 和 2 的结果。Supervisor 自动优化执行顺序。

Handoff 交接

Handoff 是一个 Agent 将任务和上下文交给另一个 Agent 的机制。交接时传递 State（当前进度）、Context（已收集的信息）和 Artifacts（中间产物）。

java
// Handoff 上下文
public class HandoffContext {
    private final String fromAgent;
    private final String toAgent;
    private final Map state;     // 当前状态
    private final List history;         // 对话历史
    private final String handoffReason;          // 交接原因

    // 构建交接上下文
    public static HandoffContext create(
            String from, String to, String reason,
            Map state, List history) {
        return new HandoffContext(from, to, reason, state, history);
    }
}

// Agent 间的 Handoff 工具
@Component
public class HandoffTools {

    @Tool(name = "handoffToHR",
          description = "将对话转交给 HR 专员 Agent，处理请假、薪酬等人力资源问题")
    public HandoffResult handoffToHR(
            @ToolParam(description = "交接原因，说明为什么需要 HR 处理") String reason,
            @ToolParam(description = "当前已收集的信息，JSON 格式") String context) {

        return new HandoffResult("hr_agent", reason, context);
    }

    @Tool(name = "handoffToIT",
          description = "将对话转交给 IT 支持 Agent，处理技术问题")
    public HandoffResult handoffToIT(
            @ToolParam(description = "交接原因") String reason,
            @ToolParam(description = "当前已收集的信息") String context) {

        return new HandoffResult("it_agent", reason, context);
    }
}

// Handoff 执行器
@Component
public class HandoffExecutor {

    private final Map agentRegistry;

    public AgentResult executeHandoff(HandoffContext context) {
        Agent targetAgent = agentRegistry.get(context.getToAgent());
        if (targetAgent == null) {
            return AgentResult.error("未知 Agent: " + context.getToAgent());
        }

        // 将上下文注入目标 Agent 的记忆
        targetAgent.getMemory().add(context.getHistory());

        // 执行
        return targetAgent.execute(
            "【经由 " + context.getFromAgent() + " 转交】" +
            context.getHandoffReason() +
            "\n已收集信息: " + context.getState()
        );
    }
}

状态共享与冲突解决

多 Agent 并行执行时可能同时修改同一资源（如同时查和修改同一工单）。需要状态隔离和冲突检测机制。

问题表现解决方案
写冲突两个 Agent 同时修改同一工单的状态悲观锁：分布式锁（Redis RedLock）
读过期Agent A 读到的状态被 Agent B 改了乐观锁：版本号校验（CAS）
状态覆盖并行结果合并时后到的覆盖了先到的MergedState：按字段合并而非整体覆盖
循环依赖Agent A 等 B，B 等 A 的死锁依赖图分析 + 超时中断
java
// 分布式状态存储
@Component
public class DistributedStateStore {

    private final RedisTemplate redis;

    // 乐观锁：版本号 CAS
    public boolean updateWithVersion(String key,
                                      Function updater) {
        for (int retry = 0; retry < 3; retry++) {
            String current = redis.opsForValue().get(key);
            String version = extractVersion(current);
            String newValue = updater.apply(current);
            // CAS: 仅当版本号一致时更新
            boolean success = redis.opsForValue()
                .setIfEquals(key, current, newValue);
            if (success) return true;
        }
        return false;
    }
