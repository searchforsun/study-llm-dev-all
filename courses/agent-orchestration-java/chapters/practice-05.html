@Configuration
public class CorpAgentConfig {

    @Bean
    public ChatClient corpChatClient(ChatClient.Builder builder) {
        return builder
            .defaultSystem("""
                你是 CorpAssist 办公助手，帮助员工处理日常工作。

                可用工具：
                1. 日历：创建日程、查询日程、取消日程
                2. 邮件：发送邮件、查询收件箱、搜索邮件
                3. 工单：查询工单状态、创建工单、更新工单
                4. 用户：查询员工信息、查询部门信息

                工作流程：
                1. 先理解用户意图，拆解为具体任务
                2. 按顺序调用需要的工具
                3. 如果某个工具返回错误，尝试修正参数后重试
                4. 最终用中文给出完整、友好的回答
                5. 涉及外部操作（发邮件、创建工单）必须先让用户确认

                Token 预算：单次对话不超过 4096 tokens
                最大步骤：不超过 10 步
                """)
            .build();
    }

    @Bean
    public ReactAgent corpAssistant(
            ChatClient chatClient,
            List allTools) {

        return ReactAgent.builder()
            .chatClient(chatClient)
            .tools(allTools)
            .maxIterations(15)
            .maxTokenUsage(8192)
            .toolRetryAttempts(2)
            .build();
    }
