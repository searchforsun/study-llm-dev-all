Micrometer 指标埋点

Micrometer 是 Spring Boot 内置的指标采集库。Agent 的关键指标包括步数、Token 消耗、延迟、工具调用频率，全部通过 Micrometer 暴露到 Prometheus。

java
// Agent 指标采集器
@Component
public class AgentTelemetry {

    private final MeterRegistry registry;
    private final Map timerCache = new ConcurrentHashMap<>();
    private final Map counterCache = new ConcurrentHashMap<>();

    public AgentTelemetry(MeterRegistry registry) {
        this.registry = registry;
    }

    // 1. 步数指标
    public void recordStep(String agentId, int stepNumber) {
        registry.counter("agent.steps",
            "agent", agentId,
            "step", String.valueOf(stepNumber)
        ).increment();
    }

    // 2. Token 指标（按模型拆分）
    public void recordTokens(String agentId, String model,
                              int promptTokens, int completionTokens) {
        registry.counter("agent.tokens",
            "agent", agentId,
            "model", model,
            "type", "prompt"
        ).increment(promptTokens);

        registry.counter("agent.tokens",
            "agent", agentId,
            "model", model,
            "type", "completion"
        ).increment(completionTokens);
    }

    // 3. 延迟指标（按工具拆分）
    public void recordToolLatency(String toolName, long ms) {
        registry.timer("agent.tool.latency",
            "tool", toolName
        ).record(Duration.ofMillis(ms));
    }

    // 4. 工具调用结果
    public void recordToolResult(String toolName, boolean success) {
        registry.counter("agent.tool.calls",
            "tool", toolName,
            "result", success ? "success" : "failure"
        ).increment();
    }
}

// 拦截器自动采集指标
@Component
public class MetricInterceptor implements ToolCallbackInterceptor {

    @Override
    public void beforeCall(ToolCallback callback, Map args) {
        // 开始计时
        RequestContextHolder.set("start_time", System.nanoTime());
    }

    @Override
    public void afterCall(ToolCallback callback, Object result) {
        long start = RequestContextHolder.get("start_time");
        long ms = (System.nanoTime() - start) / 1_000_000;

        telemetry.recordToolLatency(callback.getName(), ms);
        telemetry.recordToolResult(callback.getName(), result != null);
    }
}

// OTel 配置
@Configuration
public class TelemetryConfig {

    @Bean
    public Tracer tracer() {
        return OpenTelemetrySdk.builder()
            .setTracerProvider(
                SdkTracerProvider.builder()
                    .addSpanProcessor(
                        BatchSpanProcessor.builder(
                            OtlpGrpcSpanExporter.builder()
                                .setEndpoint("http://otel-collector:4317")
                                .build()
                        ).build())
                    .build())
            .build()
            .getTracer("corp-agent");
    }
}

// Agent 执行追踪
@Service
public class TracedAgentService {

    private final Tracer tracer;

    @WithSpan("agent.execute")  // OTel 自动 span
    public AgentResponse execute(String userId, String query) {
        // 创建 Trace
        Span span = tracer.spanBuilder("agent.execute")
            .setAttribute("user.id", userId)
            .setAttribute("query.length", query.length())
            .startSpan();

        try (Scope scope = span.makeCurrent()) {
            // Agent 执行
            AgentResponse response = agent.process(userId, query);
            span.setAttribute("response.length",
                response.getContent().length());
            span.setStatus(StatusCode.OK);
            return response;
        } catch (Exception e) {
            span.setStatus(StatusCode.ERROR, e.getMessage());
            span.recordException(e);
            throw e;
        } finally {
            span.end();
        }
    }
}

// 工具调用自动埋点（通过 AOP）
@Aspect
@Component
public class ToolTracingAspect {

    @Around("@annotation(com.alibaba.spring.ai.tool.annotation.Tool)")
    public Object traceTool(ProceedingJoinPoint pjp) throws Throwable {
        Span span = tracer.spanBuilder("tool." + getToolName(pjp))
            .setSpanKind(SpanKind.CLIENT)
            .startSpan();

        try (Scope scope = span.makeCurrent()) {
            Object result = pjp.proceed();
            span.setStatus(StatusCode.OK);
            return result;
        } catch (Exception e) {
            span.setStatus(StatusCode.ERROR);
            span.recordException(e);
            throw e;
        } finally {
            span.end();
        }
    }
}

Prompt 回放调试

Prompt 回放保存每次 LLM 调用的完整输入输出，按 session 组织，用于回放调试。当 Agent 输出错误时，回放当时的 Prompt 定位问题。

java
// Prompt 录制器
@Component
public class PromptRecorder {

    private final JdbcTemplate jdbc;

    // 录制每次 LLM 调用
    public void record(PromptRecord record) {
        jdbc.update("""
            INSERT INTO prompt_records
            (session_id, step, span_id, model, system_prompt,
             user_prompt, tool_definitions, completion,
             prompt_tokens, completion_tokens, created_at)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, NOW())
            """, record.sessionId(), record.step(),
            record.spanId(), record.model(),
            record.systemPrompt(), record.userPrompt(),
            toJson(record.toolDefinitions()),
            record.completion(),
            record.promptTokens(), record.completionTokens()
        );
    }

    // 按 session 回放
    public List replaySession(String sessionId) {
        return jdbc.query("""
            SELECT * FROM prompt_records
            WHERE session_id = ?
            ORDER BY step ASC
            """, new PromptRecordMapper(), sessionId);
    }

    // 回放界面（Web UI）
    @GetMapping("/api/debug/replay/{sessionId}")
    public String replayPage(@PathVariable String sessionId) {
        List records = replaySession(sessionId);
        StringBuilder html = new StringBuilder();
        html.append("
Prompt 回放: ").append(sessionId).append("
");
        for (PromptRecord rec : records) {
            html.append("
")
                .append("
Step ").append(rec.step()).append("
")
                .append("
System: ").append(esc(rec.systemPrompt()))
                .append("
")
                .append("
User: ").append(esc(rec.userPrompt()))
                .append("
")
                .append("
Tools: ").append(esc(rec.toolDefinitions()))
                .append("
")
                .append("
Response: ").append(esc(rec.completion()))
                .append("
")
                .append("
");
        }
        return html.toString();
    }
}

// Agent 调用时注入录制
@Component
public class RecordingInterceptor implements ChatClient.RequestInterceptor {

    @Override
    public void intercept(Request request) {
        // 自动录制所有 LLM 调用
        recorder.record(new PromptRecord(
            request.getSessionId(),
            request.getStep(),
            Span.current().getSpanContext().getSpanId(),
            request.getModel(),
            request.getSystemPrompt(),
            request.getUserPrompt(),
            request.getTools(),
            null, 0, 0
        ));
    }
}

面试可能问

全链路 Trace 中，Agent 的 span 如何与 LLM API 的 span 关联？

通过 Trace Context 传播。Agent 发起 LLM API 调用时，将当前 Trace 的 traceId 和 spanId 以 HTTP Header 的形式（W3C Trace Context: traceparent）传递给 LLM API 网关。LLM API 如果也接了 OTel，会在自己的系统中创建子 span，并关联到 Agent 的 span。这样在 Jaeger 中可以看到完整的调用链：Agent→LLM API→模型推理。CorpAssist 通过 Spring Cloud Gateway 自动在 HTTP 调用中注入 traceparent header。

Prompt 回放存储量很大（每天 10 万次 LLM 调用），怎么管理存储成本？

分层存储策略：1) 在线保留（7 天）——完整 Prompt 记录，用于实时调试，存储在 PostgreSQL；2) 近线保留（30 天）——压缩记录（去掉冗余的 system prompt，只保留差异部分），存储在 Parquet；3) 离线保留（1 年）——只保留 session 级别的摘要（步数、Token 消耗、成功/失败），不保留具体 Prompt 文本，存储在 HDFS。压缩后每天存储量从 10GB 降到 500MB。

Agent 回答质量下降怎么通过观测工具定位根因？

三步定位：1) 检视告警——看 QPS/延迟/Token 消耗是否有突增，如果有则是系统级问题（如模型降级）；2) 回放 Prompt——找到质量差的 session，回放 LLM 输入输出，看是 System Prompt 没生效还是上下文被截断了；3) 查看 Trace——如果 Agent 调用了工具但结果不对，看 Trace 中工具调用的参数和返回结果。常见根因：Context Window 溢出导致早期指令被遗忘、工具返回格式不符合预期、System Prompt 被用户 Prompt 注入覆盖。

判断练习

场景：Agent 在某个特定查询上返回错误，但回放 Prompt 时发现 LLM 的回复完全正确，问题出在哪？

问题可能出在 Prompt 回放和实际执行不同步。检查：1) Prompt 回放记录是否完整包括了 tool_results？如果只录了 LLM 的请求没录工具的返回结果，看不到 LLM 观察到的数据；2) 工具调用的结果是否被截断或序列化错误？如果工具返回了长列表但只保留了摘要，LLM 看到的上下文和回放记录不一致。确保 Prompt 回放同时记录 LLM 请求和工具返回结果（系统消息）。

场景：延迟 P99 从 3 秒飙升到 20 秒，从 Trace 看每个节点耗时都正常，怎么排查？

每个节点耗时正常但整体延迟异常——可能是"等待"时间（非阻塞时间）。排查方向：1) 限流排队——检查 Sentinel 的 blocked_requests 指标，可能请求在限流队列中等待；2) 线程池排队——Agent 的线程池是否已满，请求在等待线程；3) LLM API 的 queue time——看 Trace 中 LLM API 的 span 是否包含了"queue_time"属性（表示在 API 端排队）；4) 数据库连接池——JDBC 连接池是否耗尽。CorpAssist 经验：60% 的延迟异常是线程池或连接池等待，不是 LLM 本身变慢。
