熔断策略

熔断器防止级联故障。当 LLM API 或 MCP Server 连续失败达到阈值时，熔断开启，请求快速失败（不调用后端），给后端恢复时间。

java
import io.github.resilience4j.circuitbreaker.CircuitBreakerConfig;
import io.github.resilience4j.circuitbreaker.CircuitBreakerRegistry;
import java.time.Duration;

@Configuration
public class CircuitBreakerConfig {

    @Bean
    public CircuitBreakerRegistry circuitBreakerRegistry() {
        CircuitBreakerConfig config = CircuitBreakerConfig.custom()
            // 滑动窗口统计最近 10 次调用
            .slidingWindowSize(10)
            // 失败率超过 50% 开启熔断
            .failureRateThreshold(50)
            // 熔断开启后等待 30 秒尝试半开
            .waitDurationInOpenState(Duration.ofSeconds(30))
            // 半开状态下允许 3 次尝试
            .permittedNumberOfCallsInHalfOpenState(3)
            // 记录哪些异常为失败
            .recordExceptions(TimeoutException.class,
                HttpServerErrorException.class)
            // 忽略哪些异常不算失败
            .ignoreExceptions(IllegalArgumentException.class)
            .build();

        return CircuitBreakerRegistry.of(config);
    }

    @Bean
    public CircuitBreaker llmCircuitBreaker(CircuitBreakerRegistry registry) {
        return registry.circuitBreaker("llm-api");
    }

    // 使用熔断器包裹 LLM 调用
    @Component
    public class ResilientLlmClient {

        private final CircuitBreaker circuitBreaker;
        private final ChatClient primaryClient;
        private final ChatClient fallbackClient;

        public String callLlm(String prompt) {
            // 熔断器会自动统计失败率，熔断时走 fallback
            Supplier decorated = Decorators.ofSupplier(
                () -> primaryClient.prompt().user(prompt).call().content()
            ).withCircuitBreaker(circuitBreaker)
             .withFallback(throwable ->
                 fallbackClient.prompt().user(prompt).call().content()
             ).decorate();
            return decorated.get();
        }
    }
}

// 熔断事件监听
@Component
public class CircuitBreakerEventListener {

    @EventListener
    public void onStateTransition(CircuitBreakerOnStateTransitionEvent event) {
        String newState = event.getStateTransition().getToState().name();
        if ("OPEN".equals(newState)) {
            alertService.sendAlert(
                "LLM 熔断已开启，降级到备用模型");
        } else if ("HALF_OPEN".equals(newState)) {
            log.info("LLM 熔断尝试恢复");
        } else if ("CLOSED".equals(newState)) {
            notificationService.notify("LLM 熔断已恢复");
        }
    }
}

模型 Fallback 链路

模型 Fallback 链路是容灾的核心——当首选模型不可用时，自动降级到备用模型。

flowchart TB
    Q[用户查询] --> L1{GPT-4.1 可用?}
    L1 -->|是| GPT[GPT-4.1 回答]
    L1 -->|否: 超时/错误| L2{DeepSeek-V3 可用?}
    L2 -->|是| DS[DeepSeek-V3 回答
带降级标记]
    L2 -->|否| L3{本地 BGE 模型  可用?}
    L3 -->|是| BGE[BGE 基础模型回答
仅支持简单查询]
    L3 -->|否| FALLBACK[返回"AI服务暂时不可用"]
java
@Component
public class ModelFallbackChain {

    private final List providers;
    private final CircuitBreaker circuitBreaker;

    public ModelFallbackChain(CircuitBreaker circuitBreaker) {
        this.circuitBreaker = circuitBreaker;
        this.providers = List.of(
            new OpenAiProvider("gpt-4.1", 15.0),         // 首选
            new DeepSeekProvider("deepseek-chat", 0.5),   // 备用 1
            new LocalOllamaProvider("qwen2.5:7b", 0.0)    // 备用 2
        );
    }

    public FallbackResponse callWithFallback(String prompt) {
        List attemptedProviders = new ArrayList<>();

        for (LlmProvider provider : providers) {
            attemptedProviders.add(provider.getName());
            try {
                // 熔断器内调用
                String result = Decorators.ofSupplier(
                    () -> provider.call(prompt)
                ).withCircuitBreaker(circuitBreaker)
                 .withFallback(t -> null)
                 .get();

                if (result != null) {
                    boolean isDegraded = attemptedProviders.size() > 1;
                    return new FallbackResponse(
                        result,
                        provider.getName(),
                        isDegraded,
                        attemptedProviders
                    );
                }
            } catch (Exception e) {
                log.warn("Provider {} failed: {}",
                    provider.getName(), e.getMessage());
            }
        }

        // 所有模型都不可用
        return new FallbackResponse(
            "AI 服务暂时不可用，请稍后重试",
            "none", true, attemptedProviders
        );
    }
}

// 降级标记：告知用户当前使用的是降级模型
public record FallbackResponse(
    String content,
    String provider,
    boolean degraded,
    List attemptedProviders
) {
    public String toUserMessage() {
        if (degraded) {
            return content + "\n\n（当前使用" + provider +
                   "提供服务，响应质量可能有所降低）";
        }
        return content;
    }
}

限流与削峰

限流防止突发流量压垮后端服务。Sentinel 是阿里开源的限流组件，支持 QPS 限流、并发线程数限流、热点参数限流。

java
// Sentinel 限流配置
@Configuration
public class SentinelConfig {

    // 按租户限流：每个租户 QPS 不超过 50
    @PostConstruct
    public void initFlowRules() {
        List rules = new ArrayList<>();

        FlowRule agentRule = new FlowRule();
        agentRule.setResource("corp_agent_query");
        agentRule.setGrade(RuleConstant.FLOW_GRADE_QPS);
        agentRule.setCount(200);      // 全局 QPS 200
        agentRule.setLimitApp("default");
        rules.add(agentRule);

        // 热点参数限流（按 tenantId）
        ParamFlowRule tenantRule = new ParamFlowRule("corp_agent_query");
        tenantRule.setParamIdx(0);    // 第一个参数是 tenantId
        tenantRule.setCount(50);      // 每个租户 QPS 50
        tenantRule.setDurationInSec(1);
        rules.add(tenantRule);

        FlowRuleManager.loadRules(rules);
    }
}

// 限流后的处理
@Service
public class AgentQueryService {

    @SentinelResource(value = "corp_agent_query",
                      blockHandler = "handleBlocked",
                      fallback = "handleFallback")
    public AgentResponse query(String tenantId, String question) {
        return agentService.process(tenantId, question);
    }

    // 被限流时调用（blockHandler）
    public AgentResponse handleBlocked(
            String tenantId, String question, BlockException ex) {
        // 返回排队提示
        return AgentResponse.builder()
            .content("系统繁忙，请稍后重试（当前排队人数: " +
                     getQueueSize() + "）")
            .build();
    }

    // 服务异常时调用（fallback）
    public AgentResponse handleFallback(
            String tenantId, String question, Throwable ex) {
        return AgentResponse.builder()
            .content("AI 服务暂时异常，请稍后重试")
            .build();
    }

    // 削峰：请求队列，异步处理
    private final Queue requestQueue =
        new ConcurrentLinkedQueue<>();

    @Scheduled(fixedDelay = 100) // 每 100ms 处理一批
    public void processQueue() {
        int batchSize = Math.min(10, requestQueue.size());
        for (int i = 0; i < batchSize; i++) {
            AgentRequest req = requestQueue.poll();
            if (req != null) {
                CompletableFuture.runAsync(() ->
                    agentService.process(req.tenantId(), req.question())
                );
            }
        }
    }
}

@Component
public class ZoneAwareRouter {
    private final Map zones;

    public ServiceInstance chooseInstance(String serviceId) {
        List allInstances = discoveryClient
            .getInstances(serviceId);

        // 按机房分组
        Map> byZone = allInstances.stream()
            .collect(Collectors.groupingBy(
                inst -> inst.getMetadata()
                    .getOrDefault("zone", "default")));

        // 优先本机房
        String myZone = System.getProperty("zone");
        List local = byZone.get(myZone);
        if (local != null && !local.isEmpty()
            && isZoneHealthy(myZone)) {
            return loadBalancer.choose(local);
        }
        // 本机房不健康 → 切换到健康的其他机房
        for (var entry : byZone.entrySet()) {
            if (isZoneHealthy(entry.getKey())) {
                return loadBalancer.choose(entry.getValue());
            }
        }
        throw new NoAvailableInstanceException("所有机房都不可用");
    }
}

面试可能问

熔断器的三种状态（CLOSED/OPEN/HALF_OPEN）转换逻辑是什么？

CLOSED（关闭）：正常状态，所有请求通过。每次失败计数器 +1，当滑动窗口内的失败率超过阈值（如 50%），状态转换到 OPEN。OPEN（开启）：所有请求快速失败，不调用后端。等待配置的时间（30 秒）后，转换到 HALF_OPEN。HALF_OPEN（半开）：允许少量请求（3 次）试探后端是否恢复。如果试探成功（所有请求都成功），转换到 CLOSED；如果有任何一个失败，回到 OPEN 重新等待 30 秒。

Fallback 链路中，从 GPT-4.1 降级到 DeepSeek 时，质量和延迟的变化？

质量：DeepSeek-V3 在中文和企业场景下质量接近 GPT-4.1（准确率约 92% vs 96%），但复杂推理和代码生成差距明显。延迟：GPT-4.1 平均 3s，DeepSeek-V3 平均 1.5s。降级策略：1) 简单查询（分类/提取/事实问答）降级到 DeepSeek，用户几乎无感；2) 复杂查询（推理/比较/分析）不自动降级，提示用户等待或返回部分结果。CorpAssist 的 RouterQueryEngine 自动识别查询复杂度，只有简单查询才启用 DeepSeek fallback。

多活部署下，两个机房的 Agent 同时处理同一用户的请求，怎么保证一致性？

通过会话亲和性负载均衡：同一用户的请求始终路由到同一机房的同一实例（基于 sessionId hash）。但如果机房故障，用户会被路由到另一个机房，此时 ChatMemory 在 Redis 中是共享的（跨机房 Redis 集群），所以新的机房可以看到完整的对话历史。关键依赖：Redis 必须是跨机房部署的集群（CRDT 或主从），确保数据一致性。

判断练习

场景：熔断器频繁开启和关闭（抖动），影响用户体验，怎么优化参数？

原因是滑动窗口太小（10 次），偶尔的失败就触发熔断。优化：1) 增大滑动窗口到 50 次（更平滑）；2) 提高失败率阈值从 50% 到 70%（容忍更多失败）；3) 增加半开状态的等待时间从 30s 到 60s；4) 确保只熔断"不可恢复"的异常（如 HTTP 503），忽略"可恢复"的异常（如 429 限流，应该重试而不是熔断）。CorpAssist 参数：窗口 30 次，阈值 60%，半开等待 45 秒。

场景：限流后用户排队等待，但队列积压到 1000 个请求时内存不足，怎么设计削峰？

三层削峰：1) 请求进入队列前先进行 Token Bucket 限流（桶大小=200，每秒补充=50），超出的拒绝；2) 通过的请求放入有界队列（ArrayBlockingQueue 容量=200），队列满时拒绝；3) 用户收到限流响应后，客户端自动退避重试（Retry-After header 指定等待时间）。这样不会无限积压，即使流量峰值 10 倍于正常，也只拒绝超出容量的部分。

Micrometer 指标埋点

Micrometer 是 Spring Boot 内置的指标采集库。Agent 的关键指标包括步数、Token 消耗、延迟、工具调用频率，全部通过 Micrometer 暴露到 Prometheus。

java
// Agent 指标采集器
@Component
public class AgentTelemetry {

    private final MeterRegistry registry;
    private final Map timerCache = new ConcurrentHashMap<>();
    private final Map counterCache = new ConcurrentHashMap<>();

    public AgentTelemetry(MeterRegistry registry) {
        this.registry = registry;
    }

    // 1. 步数指标
    public void recordStep(String agentId, int stepNumber) {
        registry.counter("agent.steps",
            "agent", agentId,
            "step", String.valueOf(stepNumber)
        ).increment();
    }

    // 2. Token 指标（按模型拆分）
    public void recordTokens(String agentId, String model,
                              int promptTokens, int completionTokens) {
        registry.counter("agent.tokens",
            "agent", agentId,
            "model", model,
            "type", "prompt"
        ).increment(promptTokens);

        registry.counter("agent.tokens",
            "agent", agentId,
            "model", model,
            "type", "completion"
        ).increment(completionTokens);
    }

    // 3. 延迟指标（按工具拆分）
    public void recordToolLatency(String toolName, long ms) {
        registry.timer("agent.tool.latency",
            "tool", toolName
        ).record(Duration.ofMillis(ms));
    }

    // 4. 工具调用结果
    public void recordToolResult(String toolName, boolean success) {
        registry.counter("agent.tool.calls",
            "tool", toolName,
            "result", success ? "success" : "failure"
        ).increment();
    }
}

@Service
public class TracedAgentService {

    private final Tracer tracer;

    @WithSpan("agent.execute")  // OTel 自动 span
    public AgentResponse execute(String userId, String query) {
        // 创建 Trace
        Span span = tracer.spanBuilder("agent.execute")
            .setAttribute("user.id", userId)
            .setAttribute("query.length", query.length())
            .startSpan();

        try (Scope scope = span.makeCurrent()) {
            // Agent 执行
            AgentResponse response = agent.process(userId, query);
            span.setAttribute("response.length",
                response.getContent().length());
            span.setStatus(StatusCode.OK);
            return response;
        } catch (Exception e) {
            span.setStatus(StatusCode.ERROR, e.getMessage());
            span.recordException(e);
            throw e;
        } finally {
            span.end();
        }
    }
}

面试可能问

全链路 Trace 中，Agent 的 span 如何与 LLM API 的 span 关联？

通过 Trace Context 传播。Agent 发起 LLM API 调用时，将当前 Trace 的 traceId 和 spanId 以 HTTP Header 的形式（W3C Trace Context: traceparent）传递给 LLM API 网关。LLM API 如果也接了 OTel，会在自己的系统中创建子 span，并关联到 Agent 的 span。这样在 Jaeger 中可以看到完整的调用链：Agent→LLM API→模型推理。CorpAssist 通过 Spring Cloud Gateway 自动在 HTTP 调用中注入 traceparent header。

Prompt 回放存储量很大（每天 10 万次 LLM 调用），怎么管理存储成本？

分层存储策略：1) 在线保留（7 天）——完整 Prompt 记录，用于实时调试，存储在 PostgreSQL；2) 近线保留（30 天）——压缩记录（去掉冗余的 system prompt，只保留差异部分），存储在 Parquet；3) 离线保留（1 年）——只保留 session 级别的摘要（步数、Token 消耗、成功/失败），不保留具体 Prompt 文本，存储在 HDFS。压缩后每天存储量从 10GB 降到 500MB。

Agent 回答质量下降怎么通过观测工具定位根因？

三步定位：1) 检视告警——看 QPS/延迟/Token 消耗是否有突增，如果有则是系统级问题（如模型降级）；2) 回放 Prompt——找到质量差的 session，回放 LLM 输入输出，看是 System Prompt 没生效还是上下文被截断了；3) 查看 Trace——如果 Agent 调用了工具但结果不对，看 Trace 中工具调用的参数和返回结果。常见根因：Context Window 溢出导致早期指令被遗忘、工具返回格式不符合预期、System Prompt 被用户 Prompt 注入覆盖。

判断练习

场景：Agent 在某个特定查询上返回错误，但回放 Prompt 时发现 LLM 的回复完全正确，问题出在哪？

问题可能出在 Prompt 回放和实际执行不同步。检查：1) Prompt 回放记录是否完整包括了 tool_results？如果只录了 LLM 的请求没录工具的返回结果，看不到 LLM 观察到的数据；2) 工具调用的结果是否被截断或序列化错误？如果工具返回了长列表但只保留了摘要，LLM 看到的上下文和回放记录不一致。确保 Prompt 回放同时记录 LLM 请求和工具返回结果（系统消息）。

场景：延迟 P99 从 3 秒飙升到 20 秒，从 Trace 看每个节点耗时都正常，怎么排查？

每个节点耗时正常但整体延迟异常——可能是"等待"时间（非阻塞时间）。排查方向：1) 限流排队——检查 Sentinel 的 blocked_requests 指标，可能请求在限流队列中等待；2) 线程池排队——Agent 的线程池是否已满，请求在等待线程；3) LLM API 的 queue time——看 Trace 中 LLM API 的 span 是否包含了"queue_time"属性（表示在 API 端排队）；4) 数据库连接池——JDBC 连接池是否耗尽。CorpAssist 经验：60% 的延迟异常是线程池或连接池等待，不是 LLM 本身变慢。

面试可能问

CrewAI 的 Crew 和 SAA Graph 在概念上有什么对应关系？

CrewAI 的 Crew 是一个"Agent 团队"，包含多个 Agent 和 Task 列表，Process 定义执行模式（顺序/层级）。在 SAA Graph 中，Crew 对应 CompiledGraph，Agent 映射到 Node，Task 映射到 Node 中的处理逻辑，Process 映射到 Edge/ConditionalEdge。Task 间的依赖关系在 CrewAI 中用上下文传递，在 SAA Graph 中用 State 传递。CrewAI 更抽象（只需定义 Agent 和 Task 即可运行），SAA Graph 更灵活（可以精细控制每一步的流转）。

AutoGen 的 GroupChat 和 SAA Graph 的多 Agent 编排有什么本质区别？

AutoGen 的 GroupChat 是基于"对话"的：多个 Agent 轮流发言，Manager LLM 决定谁发言。适合辩论、头脑风暴等需要多轮对话的场景。SAA Graph 是基于"工作流"的：预定义的节点和有向边决定执行路径。适合有明确步骤的业务流程。区别本质：AutoGen 是 Agent 间自由对话（不确定性强），SAA Graph 是预定义路径（确定性强）。CorpAssist 的审批流程更适合 Graph 的确定性，而创新讨论场景可能适合 GroupChat。

Dify 的低代码平台能否用于 CorpAssist 的 Agent？有什么局限性？

Dify 适合：1) 非技术人员创建简单的对话应用；2) 快速原型验证（几天内做出 MVP）；3) 不需要深度定制的工作流。局限性：1) 无法与 Java 微服务集成（Dify 是 Python 独立部署）；2) 自定义代码只能通过"代码节点"以 Python 脚本形式运行，无法调用 Java 类库；3) 高并发场景性能有限（单节点约 200 QPS）；4) 无法使用 Nacos/Sentinel/Spring Cloud 等中间件。CorpAssist 的 Java 后端用 Dify 不合适，但如果业务团队想自己搭建简单的 HR 问答机器人，Dify 可以单独部署使用。

多步规划与反思

第二问："Agent 做多步规划时，怎么保证不走偏？如何加入反思（Reflection）机制？"

java
// 带反思机制的 Agent
@Component
public class ReflectiveAgent {

    private final ChatClient planner;    // 规划 LLM
    private final ChatClient executor;   // 执行 LLM
    private final ChatClient reflector;  // 反思 LLM

    public AgentResponse execute(String task) {
        // Step 1: 规划（生成执行计划）
        String plan = planner.prompt()
            .user("为以下任务制定执行计划：" + task)
            .call()
            .content();

        AgentState state = new AgentState(task, plan);
        int maxReflections = 3;

        for (int step = 0; step < 10; step++) {
            // Step 2: 执行（执行单步）
            String result = executor.prompt()
                .user("当前步骤: " + plan.split("\n")[step] +
                      "\n上下文: " + state.getContext())
                .call()
                .content();

            state.addStepResult(step, result);

            // Step 3: 反思（检查结果是否偏离计划）
            if (step > 0 && step % 2 == 0) {
                String reflection = reflector.prompt()
                    .user("检查当前进度是否符合原计划:\n" +
                          "计划: " + plan + "\n" +
                          "已完成: " + state.getSummary() + "\n" +
                          "偏离了吗？如果偏离请给出修正方案")
                    .call()
                    .content();

                if (reflection.contains("偏离")) {
                    // 修正计划
                    plan = planner.prompt()
                        .user("原计划偏离了，当前进度: " +
                              state.getSummary() +
                              "。请重新规划剩余步骤")
                        .call()
                        .content();
                    state.addReflection(reflection);
                }
            }
        }
        return new AgentResponse(state);
    }
}

// 三层降级策略
@Component
public class ToolDegradationStrategy {

    @Tool(name = "sendEmail", description = "发送邮件")
    @Retry(name = "email", maxAttempts = 3,
           backoff = @Backoff(delay = 1000))
    public EmailResult sendEmail(String to, String subject, String body) {
        try {
            return emailClient.send(to, subject, body);
        } catch (TimeoutException e) {
            // 第 1 层：重试（由 @Retry 处理）
            throw e;
        } catch (PermissionDeniedException e) {
            // 第 2 层：替代方案
            return saveToDraft(to, subject, body);
        } catch (ServiceUnavailableException e) {
            // 第 3 层：人工介入
            createManualTicket("send_email", to, subject, body);
            return new EmailResult("manual",
                "邮件系统暂时不可用，已创建人工处理工单");
        }
    }

    // LLM 级别的降级提示
    private String getDegradationPrompt() {
        return """
            工具调用规则（降级相关）：
            1. 如果 sendEmail 返回超时，等 2 秒后重试一次
            2. 如果重试仍然失败，告诉用户"邮件系统繁忙，已保存为草稿"
            3. 绝对不要连续重试同一个工具超过 2 次
            4. 如果用户提供了多种联系方式（邮件+企业微信），
               一种失败时尝试另一种
            """;
    }

// 三层记忆系统
@Component
public class MemorySystem {

    // 工作记忆：当前会话上下文
    public class WorkingMemory {
        private final Map context = new HashMap<>();
        private final List recentActions = new ArrayList<>();

        public void recordAction(String action, String result) {
            recentActions.add(action + "→" + result);
            if (recentActions.size() > 20) recentActions.remove(0);
        }
    }

    // 情景记忆：跨会话的对话历史
    public class EpisodicMemory {
        private final ChatMemory chatMemory;

        public List recall(String sessionId) {
            return chatMemory.get(sessionId, 20);
        }

        public void store(String sessionId, Message msg) {
            chatMemory.add(sessionId, msg);
        }
    }

    // 语义记忆：用户偏好和知识
    public class SemanticMemory {
        private final VectorStore vectorStore;

        public void rememberPreference(String userId, String preference) {
            // 将偏好转为向量存入 Milvus
            Document doc = new Document(preference,
                Map.of("user_id", userId, "type", "preference"));
            index.insert(doc);
        }

        public List recallPreferences(String userId) {
            retriever.setFilters(
                new MetadataFilters(List.of(
                    new ExactMatchFilter("user_id", userId),
                    new ExactMatchFilter("type", "preference")
                )));
            return retriever.retrieve("用户偏好").stream()
                .map(n -> n.getText())
                .toList();
        }

面试中被追问"如果 LLM 一直说需要调用工具但参数都不对，怎么处理？"

答：这是 Agent 常见的"幻觉循环"问题。三层防护：1) 工具层——工具参数做服务端校验，不符合 Schema 的工具调用拒绝并返回具体错误（"name 字段不能为空"），LLM 看到错误后会修正参数；2) Agent 层——toolRetryAttempts=2，重试 2 次后如果还是错的，停止执行并返回"无法完成此操作"；3) 监控层——检测连续参数错误的模式（N 次调用同一工具失败），告警并可能自动切换到备用模型。

面试官问"CorpAssist 的 Agent 如果出现幻觉（自己编造工具返回结果），怎么检测和防止？"

检测：1) 工具结果验证——所有工具返回结果通过 JSON Schema 验证，不符合 Schema 的视为无效；2) 对比检测——同时对同一问题用不同模型推理，对比结果一致性（如 GPT-4.1 vs DeepSeek）；3) 可信度评分——Agent 在返回答案时附带"自信度"评分，低分答案需要二次确认。防止：1) 在 System Prompt 中强调"不编造数据，如果工具没有返回就说明不知道"；2) 对所有写操作（发邮件/创建工单）增加审批节点；3) 审计日志自动对比 Agent 输出的数据和工具实际返回的数据。

为什么学这章

本章要点速记

1) ReAct 循环 = 思考(Thought)→行动(Action)→观察(Observation) 循环直到最终回答；2) 工具定义需包含 name/description/parameters 三要素；3) maxIterations 防止 Agent 无限循环（默认 10）；4) System Prompt 中固定 Agent 角色和行为边界；5) Spring AI Alibaba 的 ReactAgent 基于 Spring AI 的 AbstractToolCallSupport。

ReAct 思考-行动-观察

ReAct 循环是 Agent 的核心模式。每次循环：LLM 生成思考（Thought），决定调用哪个工具（Action），执行工具并获取结果（Observation），然后继续思考。循环持续到 LLM 认为任务完成并生成最终答案（Final Answer）。Spring AI Alibaba 的 ReactAgent 在 Spring AI 的 ChatClient 基础上增加了循环控制。

思考10秒：什么场景应该选 ReAct Agent 而不是 LangGraph？

ReAct Agent 适合：1) 单 Agent 顺序执行工具调用（CorpAssist 的办公助手场景）；2) 不需要复杂并行/条件分支；3) 团队对 Spring AI 熟悉。LangGraph 适合：1) 需要多个 Agent 协作（主管-工人模式）；2) 需要人在回路中断和恢复；3) 状态需要持久化和回放。简单原则：5 步以内的顺序工具调用用 ReAct，需要人工介入或多 Agent 用 LangGraph。

面试可能问

ReAct Agent 陷入死循环时如何检测和恢复？

检测：1) 设置 maxIterations=10，到达后强制中断；2) 检测相同工具+参数的调用模式——连续 3 次调用同一个工具且参数相同，判定为死循环，中断并触发降级。恢复：1) 中断时返回当前已收集的信息（部分结果）；2) 在 System Prompt 中加入"如果在 5 步内没有进展，请直接总结已有的信息并提交"；3) 添加工具调用 dedup 逻辑——对相同输入返回缓存结果。

Spring AI Alibaba 的 ReactAgent 与 Spring AI 的 ChatClient 是什么关系？

ReactAgent 构建在 ChatClient 之上。ChatClient 提供与 LLM 的通信（组装 Prompt、处理 Response、管理 ChatHistory）。ReactAgent 在 ChatClient 的基础上增加了 ReAct 循环控制：每次 LLM 返回结果后，检查是否有 tool_calls，如果有则执行工具并将结果作为新的 Message 发送给 LLM，重复此过程直到 LLM 返回 Final Answer 或达到 maxIterations。相当于 ChatClient 是"一回合对话"，ReactAgent 是"多回合自动对话"。

工具返回的 Observation 太长导致 Token 超限怎么办？

三种策略：1) 截断——工具返回结果前截断到 1000 字符，超出部分用"{{截断}}..."标记；2) 摘要——工具返回结构数据时提取关键字段丢弃细节（如工单列表只返回数量和摘要不返回全文）；3) 压缩——Agent 在发送给 LLM 前自动压缩历史消息（Spring AI 的 ChatMemory 支持 summarize 模式）。CorpAssist 三种都用：工具层截断 + Agent 层摘要 + ChatMemory 压缩。

public record CorpState(
    String ticketId,
    String status,
    String assignee,
    String report
) implements AgentState {}

// 2. 定义节点
public class QueryTicketNode implements Node {
    @Override
    public CorpState apply(CorpState state) {
        // 模拟查工单
        String status = "待处理";  // 来自工单系统
        String assignee = "张三";
        return new CorpState(
            state.ticketId(), status, assignee, null
        );
    }
}

public class NotifyAssigneeNode implements Node {
    @Override
    public CorpState apply(CorpState state) {
        // 发送邮件通知
        String report = "已通知处理人: " + state.assignee();
        return new CorpState(
            state.ticketId(), state.status(),
            state.assignee(), report
        );
    }
}

// 3. 构建 Graph
public class TicketWorkflow {
    public static StateGraph buildGraph() {
        StateGraph graph = new StateGraph<>(CorpState::new);

        // 添加节点
        graph.addNode("query_ticket", new QueryTicketNode());
        graph.addNode("notify", new NotifyAssigneeNode());
        graph.addNode("report_completed", state ->
            new CorpState(state.ticketId(), state.status(),
                state.assignee(), "工单已完成")
        );
        graph.addNode("report_closed", state ->
            new CorpState(state.ticketId(), state.status(),
                state.assignee(), "工单已关闭")
        );

        // 添加入口和边
        graph.setEntryPoint("query_ticket");

        // 条件边：根据工单状态路由
        graph.addConditionalEdge(
            "query_ticket",
            (Function) state -> {
                return switch (state.status()) {
                    case "待处理" -> "notify";
                    case "已完成" -> "report_completed";
                    case "已关闭" -> "report_closed";
                    default -> "report_completed";
                };
            },
            Map.of(
                "notify", "notify",
                "report_completed", "report_completed",
                "report_closed", "report_closed"
            )
        );

        // 普通边：顺序执行
        graph.addEdge("notify", "report_completed");
        graph.setFinishPoint("report_completed");

        return graph;
    }
}

思考10秒：Agent Graph 和 Saga 可以结合使用吗？

可以。架构设计：Agent Graph 负责"决策"——判断工作流的路径、调用哪些工具、何时需要人工介入。Saga 负责"执行"——Agent Graph 中的工具节点调用微服务，微服务通过 Saga 保证数据一致性。例如 CorpAssist 的报销审批流程：Agent Graph 判断是否需要审批、推送到谁；Saga 保证审批通过后，财务系统和报销系统的数据一致性。

面试可能问

StateGraph 和 LangGraph 的主要区别是什么？

StateGraph（Spring AI Alibaba）是 LangGraph 的 Java 移植版本，概念上兼容：节点/边/条件边/State/EntryPoint/FinishPoint。主要区别：1) Java 版强类型——State 用 record 或 POJO 定义，编译期检查；2) 序列化——Java 版 State 需要可序列化（JSON/Java Serialization），支持跨节点传递；3) 集成——Java 版与 Spring AI 的 ChatClient/ToolCallback 无缝集成；4) 生态——LangGraph 的 checkpoint/回放功能更丰富。功能覆盖度 Java 版约 80%，核心流控能力一致。

Graph 工作流中节点执行失败怎么恢复？

三层恢复策略：1) 节点内部重试——通过 NodeInterceptor 的 onError 捕获异常，判断可重试的异常（网络超时）进行 3 次退避重试；2) 回退节点——在 Graph 中定义 error_handler 节点，ConditionalEdge 从失败节点路由到 error_handler，清理或回滚状态；3) 人工介入——复杂错误路由到 human_approval 节点，等待人工处理后继续。State 的不可变性保证了重试时可以从任意节点恢复。

多个条件边组合时出现死路（没有匹配的路径），Graph 会怎么处理？

StateGraph 的 compile 阶段会做静态分析，检查所有节点是否可达。如果有节点没有边连接，compile 会抛出 IllegalGraphException。运行时如果 ConditionalEdge 的返回值没有对应的边，框架会使用默认路由（如果设置了 defaultEdge）或抛出 NoEdgeMatchException。最佳实践：每个 ConditionalEdge 都设一个 defaultEdge 指向 error_handler 节点，处理未预期的状态值。

判断练习

场景：Graph 中有一个"审批"节点需要等待外部系统响应（异步回调），怎么设计？

方案：在审批节点中，将审批请求发送到外部系统后，State 标记为 pending_approval，ConditionalEdge 路由到 wait_node。wait_node 设置轮询或回调机制（CompletableFuture + 超时），每 5 秒检查审批状态。超时 60 秒后路由到 timeout_handler。审批完成后 ConditionalEdge 路由到 continue_processing。这个模式叫做"悬挂节点"——等待外部异步结果时 Graph 不阻塞，通过轮询/回调恢复执行。

场景：条件边的路由逻辑需要查询外部数据才能确定，怎么设计？

在 ConditionalEdge 的 Function 中调用外部服务是反模式——Lambda 不应有副作用。正确做法：增加一个"路由决策"节点，在节点中查询外部数据，将决策结果写入 State（如 routeTo = "path_a"）。ConditionalEdge 只从 State 中读取 routeTo 字段做简单判断。这样职责清晰：节点做"数据获取"，边做"路由决定"。

错误重试策略

工具调用可能因网络超时、参数不合法、业务逻辑异常而失败。Spring AI Alibaba 提供多层级错误处理。

java
import com.alibaba.spring.ai.tool.annotation.Tool;
import io.github.resilience4j.retry.annotation.Retry;
import io.github.resilience4j.circuitbreaker.annotation.CircuitBreaker;

@Component
public class ResilientTools {

    @Tool(name = "sendEmail", description = "发送邮件")
    @Retry(name = "emailService", maxAttempts = 3,
           backoff = @Backoff(delay = 1000, multiplier = 2))
    @CircuitBreaker(name = "emailService",
                    fallbackMethod = "sendEmailFallback")
    public EmailResult sendEmail(
            @ToolParam(description = "收件人邮箱") String to,
            @ToolParam(description = "邮件主题") String subject,
            @ToolParam(description = "邮件正文") String body) {

        // 调用邮件服务
        return emailClient.send(to, subject, body);
    }

    // 降级方法
    public EmailResult sendEmailFallback(
            String to, String subject, String body, Throwable t) {
        // 转存到待发送队列
        pendingEmailQueue.add(new PendingEmail(to, subject, body));
        return new EmailResult("queued", "已转存待发送队列");
    }

    // Agent 级别错误处理
    @Bean
    public ReactAgent resilientAgent(ChatClient chatClient,
                                      List tools) {
        return ReactAgent.builder()
            .chatClient(chatClient)
            .tools(tools)
            .toolRetryAttempts(2)         // 工具级重试
            .toolTimeout(Duration.ofSeconds(10))
            .systemPrompt("""
                工具调用规则：
                1. 如果工具返回错误，检查参数格式是否正确后重试一次
                2. 重试仍失败，告诉用户错误信息并建议替代方案
                3. 不要连续重试同一个工具超过2次
                """)
            .build();
    }
}

面试可能问

@Tool 方法返回复杂对象（嵌套 List/Map），Schema 怎么生成？

Spring AI Alibaba 通过 Jackson 序列化自动处理。返回类型如果是 List、Map、Record 或 POJO，Schema 生成器会用 Jackson ObjectMapper 推断类型结构。例如返回 List<EventResult> 会在 Schema 的 "response" 字段中生成嵌套数组结构。建议返回类型用 Record 或 POJO（纯数据容器），避免用原始类型。框架也会检查返回类型是否可序列化（实现 Serializable 或有无参构造）。

工具调用返回的结果太长（如查询返回 100 条工单），Agent 怎么处理？

三种策略：1) 工具层截断——如果返回列表超过 5 条，只返回 5 条 + 总数提示"共 100 条，显示前 5 条"；2) 分页——工具设计时加 page/size 参数，让 LLM 按需请求；3) 摘要——工具在返回前用 LLM 对长列表做摘要，只返回摘要和数量。CorpAssist 推荐策略 1+3：显示前 5 条 + 统计信息"共 100 条工单，其中待处理 20 条、已完成 80 条"。

多个 @Tool 方法来自不同的 @Component，重复的工具名会怎么样？

Spring 启动时会检测到 ToolCallback 的 name 冲突，抛出 BeanDefinitionParsingException。解决方案：1) 确保不同组件的工具名全局唯一（推荐前缀命名，如 email_send、ticket_query）；2) 如果需要同名的不同实现，使用 @Primary 指定默认实现，或通过 @Qualifier 注入时区分。CorpAssist 规范：工具名采用 _ 格式，如 email_send、calendar_create。

场景：查询工具返回了空结果，Agent 应该继续下一步还是终止流程？

场景：工具 Schema 导出后发现参数类型错误（int 被识别为 string），怎么排查？

原因：@ToolParam 未指定 type 属性，框架使用了方法参数类型的默认映射（Java int → JSON integer，String → JSON string）。如果参数类型是自定义类（如 LocalDateTime），框架可能不知道如何映射。解决：1) 参数类型用基础类型（String/Long/Integer）传输，方法内部转换；2) 在 @ToolParam 中明确 type = "string", format = "date-time"；3) 注册自定义 Jackson 序列化器处理特殊类型。推荐实践：所有工具参数用 String 类型传，方法内解析。

1) ChatMemory 类型：InMemory（单实例）/ Redis（分布式）/ JDBC（持久化）；2) Redis 存储 message 过期时间默认 24h；3) 多租户用 sessionId 前缀区分 tenant；4) 记忆压缩降低 Token 消耗 30-50%；5) summarization 模式用 LLM 生成对话摘要替代完整历史。

Redis 持久化

RedisChatMemory 将对话消息存储在 Redis Hash 中，每个 session 对应一个 key，每条消息是 Hash 的一个 field。通过 TTL 实现自动过期。

java
// RedisChatMemory 详细配置
@Configuration
public class RedisChatMemoryConfig {

    @Bean
    public ChatMemory chatMemory(RedisConnectionFactory factory) {
        return RedisChatMemory.builder()
            .redisConnectionFactory(factory)

            .build();

    @Bean
    public SessionIdProvider sessionIdProvider() {
        return request -> {
            String tenantId = request.getHeader("X-Tenant-Id");
            String userId = request.getHeader("X-User-Id");
            // 格式: tenant:user:uuid → 支持多租户隔离
            return String.format("%s:%s:%s",
                tenantId, userId, UUID.randomUUID());
        };
    }

多租户 Session 隔离

Session 隔离确保不同用户、不同租户的对话历史互不可见。通过 session ID 的命名规范实现。

@Component
public class TenantAwareAdvisor implements RequestResponseAdvisor {

    @Override
    public AdvisedRequest before(AdvisedRequest request) {
        // 从请求上下文获取租户和用户信息
        String tenantId = TenantContext.getCurrentTenant();
        String userId = SecurityContextHolder.get()
            .getAuthentication().getName();

            .withSystemText(String.format(
                "当前租户: %s, 用户: %s\n%s",
                tenantId, userId, request.systemText()
            ))
            .withSessionId(sessionId)
            .build();
    }

    @Override
    public void after(AdvisedRequest request, AdvisedResponse response) {
        // 记录审计日志
        auditLogger.log(request.sessionId(),
            request.userText(), response.getResponse());

记忆压缩策略

对话历史越长，Token 消耗越大。LLM 的 Context Window 有限（128K），且长历史中早期信息可能被遗忘。记忆压缩通过摘要或裁剪降低 Token 消耗。

// 策略 2：摘要记忆（用 LLM 生成对话摘要）
@Component
public class SummarizingChatMemory implements ChatMemory {

    private final ChatMemory delegate;
    private final ChatClient summarizingClient;

    public SummarizingChatMemory(ChatMemory delegate,
                                  ChatClient.Builder builder) {
        this.delegate = delegate;
        this.summarizingClient = builder
            .defaultSystem("将对话历史压缩为100字以内的摘要，保留关键信息：工单号、处理人、决策")
            .build();
    }

    @Override
    public void add(String sessionId, Message message) {
        delegate.add(sessionId, message);
        // 每 10 条消息触发一次压缩
        if (delegate.get(sessionId, Integer.MAX_VALUE).size() >= 10) {
            compress(sessionId);
        }
    }

    private void compress(String sessionId) {
        List history = delegate.get(sessionId, 10);
        String fullHistory = history.stream()
            .map(Message::getContent)
            .collect(Collectors.joining("\n"));

        // LLM 生成摘要
        String summary = summarizingClient.prompt()
            .user(fullHistory)
            .call()
            .content();

        // 替换前 8 条为摘要，保留最后 2 条完整消息
        delegate.clear(sessionId);
        delegate.add(sessionId, new Message("system",
            "【历史摘要】" + summary));
        for (Message recent : history.subList(8, 10)) {
            delegate.add(sessionId, recent);
        }
    }

    @Override
    public List get(String sessionId, int lastN) {
        return delegate.get(sessionId, lastN);
    }

    @Override
    public void clear(String sessionId) {
        delegate.clear(sessionId);
    }
}

面试可能问

ChatMemory 的 add() 和 get() 实现有哪些注意事项？

add()：需要线程安全（ConcurrentHashMap 或 Redis 原子操作），消息需要有序（带递增索引），设置合理的 TTL 避免内存泄漏。get()：注意返回顺序（按时间升序），lastN 参数控制返回条数避免超长。Spring AI 的 MessageChatMemoryAdvisor 在每次 LLM 调用前自动执行 get()，调用后自动执行 add()，开发者不需要手动操作。在 Redis 实现中，add() 用 HSET，get() 用 HVALS 排序，O(N) 复杂度在 1000 条以内表现良好。

多租户场景下，sessionId 怎么设计才能防止数据泄露？

sessionId 设计为 {tenantId}:{userId}:{sessionUuid}。关键安全点：1) 从 JWT Token 中提取 tenantId 和 userId，不信任客户端传入的 sessionId；2) 在 TenantAwareAdvisor 的 before 方法中强制重写 sessionId，确保租户前缀正确；3) Redis key 按租户前缀分区，数据泄露不影响其他租户；4) 管理 API 需要校验当前用户是否有权限访问该 sessionId 所属的租户。

窗口记忆和摘要记忆能同时使用吗？怎么配合？

能。分层策略：1) 短期窗口——保留最近 20 轮完整对话（用于当前上下文）；2) 中期摘要——每隔 10 轮将前 8 轮压缩为摘要，保留最后 2 轮完整；3) 长期摘要——每天结束时将所有历史压缩为一份"日摘要"，用于第二天的上下文。CorpAssist 实现：RedisChatMemory 保留 48 小时完整历史（滑动窗口 20 轮），超过 48 小时的消息用定时任务压缩为日摘要存入单独字段。

判断练习

场景：Redis 宕机后 Agent 服务还能工作吗？对话历史会怎样？

Redis 宕机后，ChatMemory 的 get() 会抛出异常（Connection refused）。MessageChatMemoryAdvisor 默认不会阻止请求继续——它会忽略异常，Agent 以"无历史"的状态处理当前请求（不报错但失去上下文）。用户不会收到错误，但 Agent 不记得之前的对话内容。恢复策略：1) 给 Advisor 添加 Redis 连接失败的回退，切换到 InMemoryChatMemory（临时）；2) 添加重试逻辑（如果 Redis 只是暂时不可用）；3) 监控告警 Redis 连接状态。

场景：用户对话了 50 轮后 Token 消耗太大，但窗口记忆设为 20 轮会丢失早期的重要上下文（如工单编号），怎么解决？

方案：对早期信息做结构化提取而不是简单丢弃。在 System Prompt 中要求 Agent 每轮结束后自动汇总关键信息（工单编号、审批人、决策），以 JSON 格式存入记忆的 system message 中。这样即使滑动窗口丢弃了早期对话，关键的结构化信息仍然保留。CorpAssist 的 memory 包含两个部分：1) 结构化摘要（持久化 key-value 存储，保留所有关键信息）；2) 最近 20 轮对话（短时上下文，滑动窗口）。

主管-工人模式

主管-工人模式中，Supervisor Agent 负责任务规划、分配和结果整合，Worker Agent 专注于执行特定子任务。这是最常用的多 Agent 模式。

java
// 主管 Agent：拆解任务并分配给工人
@Component
public class SupervisorAgent {

    private final ChatClient chatClient;
    private final List workers;

    public AgentResult execute(String task) {
        // 1. 规划：用 LLM 拆解任务
        String plan = chatClient.prompt()
            .user("""
                将以下任务拆解为子任务列表（JSON 数组格式）：
                任务：%s
                可用的工人：查用户、查日程、发邮件、查工单

                返回格式：[{"worker":"查用户","task":"查询张三的信息"}, ...]
                """.formatted(task))
            .call()
            .content();

        // 2. 分配：解析 JSON 并分发给对应工人
        List subTasks = parsePlan(plan);
        List> futures = new ArrayList<>();

        for (SubTask subTask : subTasks) {
            WorkerAgent worker = findWorker(subTask.worker());
            futures.add(CompletableFuture.supplyAsync(() ->
                worker.execute(subTask.task())
            ));
        }

        // 3. 收集结果
        List results = futures.stream()
            .map(CompletableFuture::join)
            .toList();

        // 4. 汇总：LLM 整合所有结果
        String summary = chatClient.prompt()
            .user("请整合以下结果：\n" + formatResults(results))
            .call()
            .content();

        return new AgentResult(summary, results);
    }
}
