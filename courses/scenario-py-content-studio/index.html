# pipeline_state.py — 审核状态机
from enum import Enum
from dataclasses import dataclass, field
from typing import Optional

class ContentStatus(Enum):
    PENDING = "pending"
    SAFETY_CHECK = "safety_check"
    HUMAN_REVIEW = "human_review"
    APPROVED = "approved"
    REJECTED = "rejected"
    REVISE = "revise"
    PUBLISHED = "published"

@dataclass
class ContentItem:
    id: str
    text: str
    status: ContentStatus = ContentStatus.PENDING
    safety_result: Optional[dict] = None
    review_comment: str = ""
    revision_count: int = 0

class PipelineStateMachine:
    VALID_TRANSITIONS = {
        ContentStatus.PENDING: [ContentStatus.SAFETY_CHECK, ContentStatus.REJECTED],
        ContentStatus.SAFETY_CHECK: [ContentStatus.HUMAN_REVIEW, ContentStatus.REJECTED],
        ContentStatus.HUMAN_REVIEW: [ContentStatus.APPROVED, ContentStatus.REVISE],
        ContentStatus.REVISE: [ContentStatus.PENDING],
        ContentStatus.APPROVED: [ContentStatus.PUBLISHED],
    }

    def transition(self, item: ContentItem, to: ContentStatus) -> ContentItem:
        allowed = self.VALID_TRANSITIONS.get(item.status, [])
        if to not in allowed:
            raise ValueError(f"Cannot transition from {item.status.value} to {to.value}")
        item.status = to
        return item

# safety_engine.py — 内容安全检测
import re
from typing import List, Dict

class SafetyEngine:
    def __init__(self):
        self.sensitive_words = ["赌博", "色情", "暴力", "毒品", "枪支", "诈骗"]
        self.pii_patterns = {
            "phone": r"1[3-9]\d{9}",
            "id_card": r"\d{18}",
            "email": r"\w+@\w+\.\w+",
        }
        self.ad_law_patterns = [
            r"最好|第一|最[^好]|绝对|100%|永不|根治|彻底消除",
            r"全网最低|销量第[一二三]|首选|唯一",
        ]

    def check_sensitive(self, text: str) -> List[str]:
        return [w for w in self.sensitive_words if w in text]

    def check_pii(self, text: str) -> Dict[str, List[str]]:
        found = {}
        for name, pat in self.pii_patterns.items():
            matches = re.findall(pat, text)
            if matches:
                found[name] = matches
        return found

    def check_ad_law(self, text: str) -> List[Dict]:
        findings = []
        for pat in self.ad_law_patterns:
            matches = re.findall(pat, text)
            for m in matches:
                findings.append({"pattern": pat, "matched": m, "type": "ad_law"})
        return findings

    def full_check(self, text: str) -> Dict:
        result = {
            "passed": True,
            "sensitive": self.check_sensitive(text),
            "pii": self.check_pii(text),
            "ad_law": self.check_ad_law(text),
        }
        if result["sensitive"] or result["ad_law"]:
            result["passed"] = False
        if result["pii"]:
            result["passed"] = False  # PII默认拦截
        return result

engine = SafetyEngine()
result = engine.full_check("我们的产品是最好的，100%有效！联系邮箱: test@test.com")
print(f"安全检测: {'通过' if result['passed'] else '未通过'}")

人工复核与驳回重写

安全检测通过后，内容进入人工复核队列。审核员可以批准或驳回。驳回时需附反馈，LLM根据反馈重写。

Python：驳回重写回路

# review_loop.py — 驳回重写
import json
from openai import OpenAI

client = OpenAI(base_url="https://dashscope.aliyuncs.com/compatible-mode/v1")

class ReviewController:
    def __init__(self):
        self.pending_review = []  # 待复核队列

    def submit_for_review(self, item: ContentItem):
        self.pending_review.append(item)

    def approve(self, item: ContentItem) -> ContentItem:
        item.status = ContentStatus.APPROVED
        self.pending_review = [i for i in self.pending_review if i.id != item.id]
        return item

    def reject(self, item: ContentItem, feedback: str) -> ContentItem:
        item.review_comment = feedback
        item.status = ContentStatus.REVISE
        item = self._revise(item)
        return item

    def _revise(self, item: ContentItem) -> ContentItem:
        prompt = f"""请根据以下反馈重写内容。
原文：{item.text}
反馈要求：{item.review_comment}
请保持原文的主要信息和结构，仅根据反馈修改："""
        response = client.chat.completions.create(
            model="qwen-plus",
            messages=[{"role": "user", "content": prompt}],
        )
        item.text = response.choices[0].message.content
        item.revision_count += 1
        item.status = ContentStatus.PENDING
        return item

常见问题（FAQ）

安全检测能完全替代人工审核吗？
不能。自动化安全检测可拦截90%以上的违规内容，但品牌语调、营销合理性等维度仍需人工判断。推荐规则：自动化拦截硬违规（敏感词/PII/广告法），软违规（品牌一致性/风格）走人工复核。
驳回重写会无限循环吗？
必须设置重试上限（推荐3次）。超过上限自动标记为「人工处理中」，由资深审核员直接修改或重写。同时记录每次驳回原因，积累数据优化模板。
怎么衡量审核流水线的效率？
核心指标：1) 安全检测通过率；2) 人工复核批准率；3) 平均审核时长；4) 驳回重写次数分布；5) 发布后违规率。目标：批准率>80%，审核时长<4小时，发布后违规率<0.1%。

# user_profile.py — 用户画像系统
from pydantic import BaseModel
from typing import List, Optional
from datetime import datetime

class UserProfile(BaseModel):
    user_id: str
    age_group: str            # 18-25 / 26-35 / 36-50 / 50+
    gender: str               # male / female / unknown
    city: str
    interests: List[str]      # 兴趣标签
    purchase_history: List[str]  # 最近购买
    preferred_style: str      # concise / detailed / humorous
    preferred_channel: str    # email / sms / in_app
    lifetime_value: float     # 用户生命周期价值

class ProfileBuilder:
    def __init__(self):
        self.profiles: dict = {}

    def build_from_events(self, user_id: str, events: List[dict]) -> UserProfile:
        tags = self._extract_tags(events)
        purchases = [e["item"] for e in events if e["type"] == "purchase"]
        return UserProfile(
            user_id=user_id,
            age_group=self._infer_age(events),
            gender=self._infer_gender(events),
            city=self._infer_city(events),
            interests=tags,
            purchase_history=purchases[-5:],
            preferred_style=self._infer_style(events),
            preferred_channel="email",
            lifetime_value=sum(e.get("amount", 0) for e in events),
        )

    def _extract_tags(self, events: List[dict]) -> List[str]:
        tag_scores = {}
        category_map = {
            "电子产品": ["科技", "数码", "效率"],
            "服装": ["时尚", "穿搭"],
            "书籍": ["阅读", "学习"],
        }
        for e in events:
            for cat, tags in category_map.items():
                if cat in str(e.get("category", "")):
                    for t in tags:
                        tag_scores[t] = tag_scores.get(t, 0) + 1
        return sorted(tag_scores, key=tag_scores.get, reverse=True)[:5]

    def _infer_style(self, events: List[dict]) -> str:
        click_rates = {"concise": 0, "detailed": 0, "humorous": 0}
        for e in events:
            style = e.get("preferred_style", "")
            if style in click_rates:
                click_rates[style] += 1
        return max(click_rates, key=click_rates.get) if any(click_rates.values()) else "concise"

# ab_test.py — A/B测试框架
import random
from dataclasses import dataclass, field
from typing import List, Dict

@dataclass
class ABTestResult:
    variant: str
    impressions: int = 0
    clicks: int = 0
    conversions: int = 0

    @property
    def ctr(self) -> float:
        return self.clicks / self.impressions if self.impressions > 0 else 0

    @property
    def cvr(self) -> float:
        return self.conversions / self.impressions if self.impressions > 0 else 0

class ABTestService:
    def __init__(self, experiment_name: str, traffic_split: float = 0.5):
        self.name = experiment_name
        self.split = traffic_split  # 个性化流量占比
        self.results = {
            "control": ABTestResult("control"),
            "personalized": ABTestResult("personalized"),
        }

    def assign(self, user_id: str) -> str:
        """将用户分配到对照组或实验组"""
        random.seed(hash(user_id) % (2**31))
        return "personalized" if random.random() < self.split else "control"

    def record(self, variant: str, clicked: bool, converted: bool):

        r.impressions += 1
        if clicked:
            r.clicks += 1
        if converted:
            r.conversions += 1

        p = self.results["personalized"]
        lift = (p.ctr / c.ctr - 1) * 100 if c.ctr > 0 else 0

            "personalized_ctr": f"{p.ctr:.2%}",

            "control_cvr": f"{c.cvr:.2%}",

为什么不在Spring里直接调LLM？

SSE（Server-Sent Events）是单向服务器推送，实现简单、原生HTTP支持、自动重连。WebSocket支持双向通信。内容生成场景只需要服务器推送到前端，SSE足够。SSE天然支持text/event-stream格式。
双栈部署需要哪些基础设施？
Python侧需要独立部署（Docker + FastAPI + uvicorn），Spring侧保持原有部署方式。中间加一层API网关统一入口。Python和Spring通过内部网络通信，建议走Kubernetes Service名而非IP。

import re
from typing import Dict, Optional
import httpx

class APIGuard:
    def __init__(self, inventory_api: str, policy_api: str):
        self.inventory_api = inventory_api
        self.policy_api = policy_api

    def scan_promises(self, text: str) -> list:
        """扫描内容中的承诺性表述"""
        patterns = [
            (r"(?P.{2,20})库存充足|有货|现货", "inventory"),
            (r"价格.*?(?P\d+(\.\d+)?)元", "price"),
            (r"(?P\d+)天.*?退货|退款|保修", "policy"),
            (r"(?P\d+)折|打.*?折|优惠.*?\d+%", "discount"),
        ]
        promises = []
        for pat, ptype in patterns:
            for match in re.finditer(pat, text):
                promises.append({"type": ptype, "text": match.group(), "span": match.span()})
        return promises

    async def verify(self, text: str, sku: str) -> str:
        """真查并替换承诺"""
        # 查询真实库存
        async with httpx.AsyncClient() as client:
            inv_resp = await client.get(f"{self.inventory_api}/{sku}")
            inv_data = inv_resp.json()

        # 查询真实政策
        policy_resp = await client.get(f"{self.policy_api}/{sku}")
        policy_data = policy_resp.json()

        # 替换承诺
        text = re.sub(
            r"库存充足|有货|现货",
            f"当前库存{inv_data['stock']}件" if inv_data['stock'] > 0 else "需确认库存",

# adversarial_defense.py — 对抗样本防御
import re, unicodedata, base64

class AdversarialDefense:
    def normalize(self, text: str) -> str:
        """Unicode归一化"""
        text = unicodedata.normalize("NFKC", text)
        # 去零宽字符
        text = re.sub(r"[​‌‍⁠﻿]", "", text)
        # 去多余空格
        text = re.sub(r"\s+", "", text)
        return text

    def check_base64(self, text: str) -> list:
        """检测Base64编码的敏感内容"""
        found = []
        b64_pattern = r"[A-Za-z0-9+/]{20,}={0,2}"
        for match in re.finditer(b64_pattern, text):
            try:
                decoded = base64.b64decode(match.group()).decode("utf-8", errors="ignore")
                if self._is_sensitive(decoded):
                    found.append({"type": "base64", "decoded": decoded[:50]})
            except Exception:
                pass
        return found

    def check_reversed(self, text: str) -> list:
        words = text.split()
        for w in words:
            if self._is_sensitive(w[::-1]):
                return [{"type": "reversed", "original": w[::-1]}]
        return []

    def _is_sensitive(self, text: str) -> bool:
        sensitive = ["暴力", "赌博", "色情", "毒品", "枪支"]
        return any(s in text for s in sensitive)

    def full_check(self, text: str) -> dict:
        normalized = self.normalize(text)
        return {
            "base64": self.check_base64(text),
            "reversed": self.check_reversed(text),
            "normalized_sensitive": self._is_sensitive(normalized),
        }

from typing import List, Dict

class MemoryTier:
    LAYER_0 = "system"     # 固定，不可覆盖
    LAYER_1 = "profile"    # 用户画像，持久化
    LAYER_2 = "session"    # 短期记忆，滑动窗口

class TieredMemory:
    def __init__(self):
        self.system_prompt = ""
        self.profile = {}
        self.session: List[Dict] = []

    def set_system(self, prompt: str):
        self.system_prompt = prompt

    def update_profile(self, key: str, value: str):
        self.profile[key] = value

    def add_session(self, turn: Dict):
        self.session.append(turn)
        if len(self.session) > 10:
            self.session.pop(0)

    def build_context(self) -> str:
        return f"{self.system_prompt}\n{self.profile}\n{self.session}"

class OOCDetector:
    def __init__(self, role_boundary: dict):
        self.role = role_boundary

    def check(self, response: str) -> dict:
        violations = []
        if self.role.get("forbidden_actions"):
            for action in self.role["forbidden_actions"]:
                if action in response:
                    violations.append({"type": "forbidden_action", "match": action})
        if self.role.get("allowed_topics"):
            if not any(t in response for t in self.role["allowed_topics"]):
                violations.append({"type": "off_topic", "detail": "超出允许话题范围"})
        return {"is_ooc": len(violations) > 0, "violations": violations}

常见问题（FAQ）

API真查会不会增加响应延迟？
会，约增加50-200ms。优化：1) 对高频查询商品用Redis缓存库存数据（TTL 30s）；2) 将API查询与LLM生成并行执行；3) 非承诺性内容跳过真查。
对抗样本检测的漏报率多高？
单纯依赖规则（正则/归一化）漏报率约15-20%。建议叠加LLM判別层：将疑似内容送给qwen-turbo做二次判別。规则层拦截80%，LLM层再拦截15%，综合漏报率降至5%以下。
记忆分层后用户画像更新太慢怎么办？
用户画像更新应该走异步事件（用户行为→MQ→画像更新），不阻塞推理路径。画像的TTL设为24小时，确保次日看到最新数据。紧急更新（如用户主动修改偏好）走同步路径。

本章结论

  • API真查：所有承诺性声明必须通过真实API验证，不能依赖LLM输出。
  • 对抗样本防御：Unicode归一化+Base64检测+反转检测+LLM二次判別。
  • 记忆分层：System > Profile > Session三层，上层保护下层。
  • OOC检测：角色边界定义+禁止行为+允许话题，越界时拦截或修正。
  • 生产坑是常态：上线前做红队测试，覆盖承诺/对抗/角色三类场景。

# rbac_middleware.py — 行级权限
from enum import Enum
from typing import List, Optional
from fastapi import Request, HTTPException

class Role(Enum):
    ADMIN = "admin"       # 租户管理员：全部权限
    EDITOR = "editor"     # 编辑：创建和修改内容
    REVIEWER = "reviewer" # 审核员：审核权限
    VIEWER = "viewer"     # 查看者：只读

class Permission(str, Enum):
    CONTENT_READ = "content:read"
    CONTENT_WRITE = "content:write"
    CONTENT_APPROVE = "content:approve"
    TEMPLATE_READ = "template:read"
    TEMPLATE_WRITE = "template:write"
    USER_MANAGE = "user:manage"

ROLE_PERMISSIONS = {
    Role.ADMIN: list(Permission),
    Role.EDITOR: [Permission.CONTENT_READ, Permission.CONTENT_WRITE,
                  Permission.TEMPLATE_READ, Permission.TEMPLATE_WRITE],
    Role.REVIEWER: [Permission.CONTENT_READ, Permission.CONTENT_APPROVE],
    Role.VIEWER: [Permission.CONTENT_READ, Permission.TEMPLATE_READ],
}

class RowLevelAuth:
    def __init__(self):
        self.tenant_map = {}  # user_id -> tenant_id

    def check_permission(self, user_id: str, permission: Permission) -> bool:
        role = self._get_role(user_id)
        return permission in ROLE_PERMISSIONS.get(role, [])

    def check_row_access(self, user_id: str, content_owner_id: str,
                         content_tenant_id: str) -> bool:
        """检查行级数据访问权限"""
        user_tenant = self.tenant_map.get(user_id)
        role = self._get_role(user_id)
        # 管理员可看租户内所有数据
        if role == Role.ADMIN and user_tenant == content_tenant_id:
            return True
        # 普通用户只能看自己的数据
        return user_id == content_owner_id

    def _get_role(self, user_id: str) -> Role:
        # 实际从DB查询
        return Role.EDITOR

async def auth_middleware(request: Request, call_next):
    user_id = request.headers.get("x-user-id")
    tenant_id = request.headers.get("x-tenant-id")
    if not user_id or not tenant_id:
        raise HTTPException(401, "缺少认证信息")
    request.state.user_id = user_id
    request.state.tenant_id = tenant_id
    return await call_next(request)

# audit_logger.py — 审计日志
from datetime import datetime
from typing import Optional
import json, uuid

class AuditLogger:
    def __init__(self):
        self.logs = []  # 生产环境写入数据库

    def log(self, event_type: str, user_id: str, tenant_id: str,
            resource_id: str, detail: Optional[dict] = None,
            result: str = "success"):
        entry = {
            "id": uuid.uuid4().hex[:12],
            "event_type": event_type,
            "user_id": user_id,
            "tenant_id": tenant_id,
            "resource_id": resource_id,
            "detail": json.dumps(detail, ensure_ascii=False),
            "result": result,
            "timestamp": datetime.utcnow().isoformat(),
        }
        self.logs.append(entry)
        # 生产写入审计DB或消息队列
        return entry

    def query(self, tenant_id: str, event_type: Optional[str] = None,
              limit: int = 100) -> list:
        results = [l for l in self.logs if l["tenant_id"] == tenant_id]
        if event_type:
            results = [l for l in results if l["event_type"] == event_type]
        return results[-limit:]

from sqlalchemy import create_engine, Column, String, JSON, DateTime
from sqlalchemy.orm import declarative_base, Session
from sqlalchemy.sql import text

Base = declarative_base()

class Content(Base):
    __tablename__ = "contents"
    id = Column(String, primary_key=True)
    tenant_id = Column(String, nullable=False, index=True)
    template_id = Column(String)
    text = Column(JSON)
    status = Column(String)
    created_at = Column(DateTime)

class TenantRepository:
    def __init__(self, engine_url: str):
        self.engine = create_engine(engine_url)

    def get_contents(self, tenant_id: str, user_id: str,
                     role: str, limit: int = 50) -> List[Content]:
        with Session(self.engine) as session:
            query = session.query(Content).filter(
                Content.tenant_id == tenant_id
            )
            # 行级：非管理员只能看自己的内容
            if role != "admin":
                query = query.filter(Content.created_by == user_id)
            return query.order_by(Content.created_at.desc()).limit(limit).all()

    def create_content(self, tenant_id: str, user_id: str,
                       template_id: str, text: dict) -> Content:
        import uuid
        content = Content(
            id=uuid.uuid4().hex[:12],
            tenant_id=tenant_id, template_id=template_id,
            text=text, status="pending",
        )
        with Session(self.engine) as session:
            session.add(content)
            session.commit()
        return content

常见问题（FAQ）

行级隔离和库级隔离怎么选？
行级隔离（共享表+tenant_id）适合SaaS初期，运维简单。库级隔离适合对安全性要求高的大型客户。推荐混合：中小客户行级隔离，大客户独立库。上线初期先行级，后续可迁移。
审计日志会不会影响写入性能？
审计日志应该异步写入（MQ或日志文件），不要阻塞主流程。可以用单独的审计数据库，或写入日志系统（如ELK）。检索审计日志时按tenant_id+时间范围查询。
行级权限的SQL过滤怎么做？
用AOP或中间件模式：在每个查询前自动追加tenant_id和owner_id条件。Python端可用SQLAlchemy的filter自动拼接，Java端可用Spring Data的@TenantFilter注解或Hibernate Filter。

from dataclasses import dataclass

@dataclass
class CostModel:
    monthly_volume: int      # 月生成量
    tokens_per_item: int     # 每条平均tokens
    llm_price_per_1k: float  # LLM单价
    embed_price_per_1k: float = 0.001
    review_rate: float = 0.3
    review_price: float = 5.0
    infra_monthly: float = 500

    def monthly_llm_cost(self) -> float:
        total_tokens = self.monthly_volume * self.tokens_per_item
        return total_tokens / 1000 * self.llm_price_per_1k

    def monthly_review_cost(self) -> float:
        return self.monthly_volume * self.review_rate * self.review_price

    def monthly_total(self) -> dict:
        llm = self.monthly_llm_cost()
        review = self.monthly_review_cost()
        return {
            "llm": round(llm, 2),
            "review": round(review, 2),
            "infra": self.infra_monthly,
            "total": round(llm + review + self.infra_monthly, 2),
        }

model = CostModel(monthly_volume=10000, tokens_per_item=500,
                  llm_price_per_1k=0.002)

from typing import Dict

class ContentROI:
    def __init__(self, monthly_volume: int, human_cost_per_item: float,
                 ai_cost_per_item: float, review_rate: float):
        self.volume = monthly_volume
        self.human_cost = human_cost_per_item
        self.ai_cost = ai_cost_per_item
        self.review_rate = review_rate

    def calculate(self, safety_check_rate: float = 1.0,
                  error_rate_human: float = 0.05, error_rate_ai: float = 0.02) -> Dict:
        # 人工方案总成本
        human_total = self.volume * self.human_cost
        human_error_cost = self.volume * error_rate_human * 100  # 每错误100元

        # AI方案总成本
        ai_gen_cost = self.volume * self.ai_cost
        review_cost = self.volume * self.review_rate * self.human_cost * 0.2
        safety_cost = self.volume * safety_check_rate * 0.5  # 品牌安全API每条0.5元
        ai_error_cost = self.volume * error_rate_ai * 100

        total_ai = ai_gen_cost + review_cost + safety_cost + ai_error_cost
        total_human = human_total + human_error_cost
        saving = total_human - total_ai

        return {
            "human_monthly": round(total_human, 2),
            "ai_monthly": round(total_ai, 2),
            "monthly_saving": round(saving, 2),
            "roi_percent": round(saving / total_ai * 100, 1),
            "safety_cost": round(safety_cost, 2),
        }

calc = ContentROI(monthly_volume=5000, human_cost_per_item=20,
                  ai_cost_per_item=2, review_rate=0.3)
result = calc.calculate()
print(f"AI方案月总成本: {result['ai_monthly']}元")
print(f"其中安全审核: {result['safety_cost']}元")
print(f"ROI: {result['roi_percent']}%")

# compliance_checker.py — 合规检测
import re
from typing import List, Dict

class ComplianceChecker:
    def __init__(self, sensitive_words: List[str], brand_rules: Dict[str, str]):
        self.sensitive = sensitive_words
        self.brand_rules = brand_rules

    def check_sensitive(self, text: str) -> List[str]:
        return [w for w in self.sensitive if w in text]

    def check_brand_tone(self, text: str, brand: str) -> float:
        keywords = self.brand_rules.get(brand, "").split(",")
        matched = sum(1 for k in keywords if k in text)
        return matched / len(keywords) if keywords else 1.0

    def check_pii(self, text: str) -> List[str]:
        patterns = {
            "phone": r"1[3-9]\d{9}",
            "id_card": r"\d{18}",
            "email": r"\w+@\w+\.\w+",
        }
        found = []
        for name, pat in patterns.items():
            if re.search(pat, text):
                found.append(name)
        return found

    def audit(self, text: str, brand: str) -> Dict:
        return {
            "has_sensitive": len(self.check_sensitive(text)) > 0,
            "tone_score": self.check_brand_tone(text, brand),
            "has_pii": len(self.check_pii(text)) > 0,
            "passed": len(self.check_sensitive(text)) == 0
                     and len(self.check_pii(text)) == 0
                     and self.check_brand_tone(text, brand) > 0.8,
        }

常见问题（FAQ）

AI内容生成会不会导致品牌调性不一致？
会的，所以需要风格控制机制。通过品牌语调Profile和模板约束，可以将一致性提升到90%以上。但敏感场景（如危机公关）仍需人工把关。
小公司没有专门的合规团队，能上AI内容生成吗？
可以，但必须配置自动化合规检测。建议从低风险内容开始（如内部周报、产品说明书），等安全审核管道跑通后再扩展到营销文案等高风险领域。
ROI测算中AI的错误率怎么预估？
建议先做POC：用100条样本测试生成质量。通常AI初始错误率3-5%，经过模板调优后可降到1-2%。保守估计按5%算，后续根据线上数据修正。

# pipeline_orchestrator.py — 数据流编排
import asyncio
from dataclasses import dataclass
from typing import Optional

@dataclass
class ContentTask:
    task_id: str
    template_name: str
    variables: dict
    status: str = "pending"
    result: Optional[str] = None

class PipelineOrchestrator:
    def __init__(self, template_engine, llm_client, review_client):
        self.template_engine = template_engine
        self.llm = llm_client
        self.review = review_client

    async def process_single(self, task: ContentTask) -> ContentTask:
        # 1. 模板渲染
        prompt = self.template_engine.render(task.template_name, task.variables)
        # 2. LLM生成
        response = await self.llm.generate(prompt)
        # 3. 安全检测（快速通过则直接返回）
        review_result = await self.review.check(response)
        if review_result["passed"]:
            task.result = response
            task.status = "approved"
        else:
            task.status = "rejected"
        return task

    async def process_batch(self, tasks: list[ContentTask],
                            concurrency: int = 5) -> list[ContentTask]:
        sem = asyncio.Semaphore(concurrency)
        async def process(t: ContentTask):
            async with sem:
                return await self.process_single(t)
        return await asyncio.gather(*[process(t) for t in tasks])

常见问题（FAQ）

为什么选FastAPI而不是Flask？
FastAPI原生异步，调用LLM SDK（httpx异步）时不阻塞Worker线程。Flask的同步模型在LLM高延迟场景下会消耗大量线程资源。另外FastAPI内建Pydantic校验，对模板变量的入参校验天然适合。
RabbitMQ和Redis Stream怎么选？
内容工坊推荐RabbitMQ：1) 消息持久化和ACK机制更成熟；2) 死信队列方便处理失败任务；3) 管理台支持人工干预。Redis Stream适合对吞吐要求极高、可接受少量丢消息的场景。
模板引擎选Jinja2会不会性能不够？
Jinja2渲染性能在毫秒级，对于内容生成场景不是瓶颈。真正的瓶颈在LLM调用（秒级）和审核API。如果真有极端吞吐需求，可以预编译模板或用缓存渲染结果。

# chunk_strategies.py — 分块策略实现
from typing import List
import re

class FixedChunker:
    """固定大小分块"""
    def __init__(self, chunk_size: int = 512, overlap: int = 50):
        self.chunk_size = chunk_size
        self.overlap = overlap

    def chunk(self, text: str) -> List[str]:
        chunks, start = [], 0
        while start < len(text):
            end = start + self.chunk_size
            chunks.append(text[start:end])
            start += self.chunk_size - self.overlap
        return chunks

class SemanticChunker:
    """语义分块 — 按段落边界切分"""
    def __init__(self, max_size: int = 512, separator: str = r"\n\n"):
        self.max_size = max_size
        self.separator = separator

    def chunk(self, text: str) -> List[str]:
        paragraphs = re.split(self.separator, text)
        chunks, current = [], ""
        for para in paragraphs:
            if len(current) + len(para) < self.max_size:
                current += para + "\n\n"
            else:
                if current:
                    chunks.append(current.strip())
                current = para + "\n\n"
        if current:
            chunks.append(current.strip())
        return chunks

# 参数实验
def run_experiment(text: str, strategies: dict) -> dict:
    results = {}
    for name, chunker in strategies.items():
        chunks = chunker.chunk(text)
        results[name] = {
            "chunk_count": len(chunks),
            "avg_size": sum(len(c) for c in chunks) / len(chunks),
            "min_size": min(len(c) for c in chunks),
            "max_size": max(len(c) for c in chunks),
        }
    return results

# special_chunker.py — 特殊内容分块
from typing import List
import re

class CodeBlockChunker:
    """代码块按函数边界分块"""
    def __init__(self, max_lines: int = 200):
        self.max_lines = max_lines

    def chunk_python(self, code: str) -> List[str]:
        func_pattern = r"(def |class |@)"
        chunks, current = [], []
        for line in code.split("\n"):
            if re.match(func_pattern, line) and current:
                chunks.append("\n".join(current))
                current = []
            current.append(line)
            if len(current) >= self.max_lines:
                chunks.append("\n".join(current))
                current = []
        if current:
            chunks.append("\n".join(current))
        return chunks

class TableChunker:
    """表格保持完整"""
    def __init__(self, max_rows: int = 30):
        self.max_rows = max_rows

    def chunk(self, table_text: str) -> List[str]:
        rows = table_text.split("\n")
        if len(rows) <= self.max_rows:
            return [table_text]
        chunks = []
        for i in range(0, len(rows), self.max_rows):
            chunks.append(rows[0] + "\n" + "\n".join(rows[i:i + self.max_rows]))
        return chunks

def grid_search(text: str, questions: List[str]) -> List[Dict]:
    results = []
    for size in [256, 512, 768]:

                             if chunks else [0.0])
            results.append({"size": size, "overlap": overlap,
                           "recall": recall, "chunks": len(chunks)})
    return sorted(results, key=lambda r: r["recall"], reverse=True)

# hybrid_retriever.py — 混合检索
from rank_bm25 import BM25Okapi
from openai import OpenAI
import numpy as np
from typing import List, Dict

client = OpenAI(base_url="https://dashscope.aliyuncs.com/compatible-mode/v1")

class HybridRetriever:
    def __init__(self, bm25_weight: float = 0.3, vector_weight: float = 0.7):
        self.bm25_weight = bm25_weight
        self.vector_weight = vector_weight
        self.documents: List[str] = []
        self.bm25 = None

    def index(self, docs: List[str]):
        self.documents = docs
        tokenized = [doc.split() for doc in docs]
        self.bm25 = BM25Okapi(tokenized)

    def retrieve(self, query: str, top_k: int = 5) -> List[Dict]:
        # BM25分数
        tokenized_query = query.split()
        bm25_scores = self.bm25.get_scores(tokenized_query)
        bm25_norm = bm25_scores / (np.max(bm25_scores) + 1e-8)

        # 向量分数
        q_emb = client.embeddings.create(
            model="text-embedding-v3", input=query
        ).data[0].embedding
        vec_scores = np.array([
            np.dot(q_emb, self._get_emb(d))
            for d in self.documents
        ])
        vec_norm = vec_scores / (np.max(vec_scores) + 1e-8)

        # 融合
        combined = bm25_norm * self.bm25_weight + vec_norm * self.vector_weight
        top_indices = np.argsort(combined)[-top_k:][::-1]

        return [
            {"doc": self.documents[i], "score": float(combined[i]),
             "bm25": float(bm25_norm[i]), "vector": float(vec_norm[i])}
            for i in top_indices
        ]

    def _get_emb(self, text: str) -> np.ndarray:
        resp = client.embeddings.create(model="text-embedding-v3", input=text[:512])
        return np.array(resp.data[0].embedding)

# reranker.py — BGE-Reranker精排
from sentence_transformers import CrossEncoder
from typing import List, Dict

class BGEReranker:
    def __init__(self, model_name: str = "BAAI/bge-reranker-v2-m3"):
        self.model = CrossEncoder(model_name)

    def rerank(self, query: str, candidates: List[Dict], top_k: int = 3) -> List[Dict]:
        pairs = [(query, c["doc"][:512]) for c in candidates]
        scores = self.model.predict(pairs)
        for i, c in enumerate(candidates):
            c["rerank_score"] = float(scores[i])

        return sorted(candidates, key=lambda c: c["rerank_score"], reverse=True)[:top_k]

    name: str
    template_text: str
    variables: Dict[str, str]        # 变量名→类型

class TemplateEngine:
    def __init__(self, template_dir: str = "./templates"):
        self.env = Environment(loader=FileSystemLoader(template_dir))
        self.templates: Dict[str, ContentTemplate] = {}

    def register(self, template: ContentTemplate):
        path = Path(f"./templates/{template.name}.j2")

        path.write_text(template.template_text)
        self.templates[template.name] = template

        tpl = self.env.get_template(f"{name}.j2")
        return tpl.render(**variables)

    def list_templates(self) -> List[Dict]:
        return [{"name": t.name, "variables": t.variables,
                 "llm_fill": t.llm_fill, "desc": t.description}
                for t in self.templates.values()]

# batch_generator.py — MQ批量生成
import asyncio, json, uuid
from dataclasses import dataclass, field
from typing import Optional

@dataclass
class BatchTask:
    task_id: str = field(default_factory=lambda: uuid.uuid4().hex[:12])
    template_name: str = ""
    variables: dict = field(default_factory=dict)
    status: str = "pending"   # pending/processing/done/failed
    result: Optional[str] = None
    error: Optional[str] = None

class BatchGenerator:
    def __init__(self, engine: TemplateEngine, llm_client, max_concurrency: int = 5):
        self.engine = engine
        self.llm = llm_client
        self.sem = asyncio.Semaphore(max_concurrency)

    async def process_task(self, task: BatchTask) -> BatchTask:
        async with self.sem:
            task.status = "processing"
            try:
                # 1. 渲染固定变量
                prompt = self.engine.render(task.template_name, task.variables)
                # 2. LLM填充动态内容
                response = await self.llm.generate(prompt)
                task.result = response
                task.status = "done"
            except Exception as e:
                task.status = "failed"
                task.error = str(e)
            return task

    async def process_batch(self, tasks: list[BatchTask]) -> list[BatchTask]:
        return await asyncio.gather(*[self.process_task(t) for t in tasks])

    def create_batch(self, template: str, variable_list: list[dict]) -> list[BatchTask]:
        return [BatchTask(template_name=template, variables=v) for v in variable_list]

# 使用示例

    {"product_name": f"商品{i}", "price": 99.0 + i,
     "stock": 100, "features": ["功能A"], "call_to_action": "立即购买"}
    for i in range(100)
]

# progress_tracker.py — 进度追踪
import redis.asyncio as redis
import json

class ProgressTracker:
    def __init__(self, redis_url: str = "redis://localhost:6379/0"):
        self.redis = None
        self.redis_url = redis_url

    async def connect(self):
        self.redis = await redis.from_url(self.redis_url)

    async def init_batch(self, batch_id: str, total: int):
        await self.redis.hset(f"batch:{batch_id}", mapping={
            "total": total, "done": 0, "failed": 0, "status": "running"
        })

    async def mark_done(self, batch_id: str):
        await self.redis.hincrby(f"batch:{batch_id}", "done", 1)

    async def mark_failed(self, batch_id: str):
        await self.redis.hincrby(f"batch:{batch_id}", "failed", 1)

    async def get_progress(self, batch_id: str) -> dict:
        data = await self.redis.hgetall(f"batch:{batch_id}")
        if not data:
            return {"total": 0, "done": 0, "failed": 0, "percent": 0}
        total = int(data[b"total"])
        done = int(data.get(b"done", 0))
        failed = int(data.get(b"failed", 0))
        return {
            "total": total, "done": done, "failed": failed,
            "percent": round((done + failed) / total * 100, 1),
            "status": data.get(b"status", b"running").decode(),
        }

批量任务的并发数设多少合适？
取决于LLM API的QPS上限。qwen-plus建议并发5-10，qwen-turbo可到20-30。设太高会被限流429，设太低吞吐不够。建议用令牌桶做自适应限流。

1) 给LLM填充变量加few-shot示例；2) 对LLM输出用正则约束格式；3) 对关键变量（如价格）用固定变量而非LLM填充。

# auto_evaluator.py — 自动打分
import json, re
from openai import OpenAI
from typing import Dict, Optional

client = OpenAI(base_url="https://dashscope.aliyuncs.com/compatible-mode/v1")

class QualityScore:
    def __init__(self, grammar: float, coherence: float,
                 relevance: float, brand_consistency: float):
        self.grammar = grammar
        self.coherence = coherence
        self.relevance = relevance
        self.brand_consistency = brand_consistency
        self.overall = (grammar + coherence + relevance + brand_consistency) / 4

    def to_dict(self) -> Dict:
        return {
            "grammar": self.grammar, "coherence": self.coherence,
            "relevance": self.relevance,
            "brand_consistency": self.brand_consistency,
            "overall": round(self.overall, 2),
        }

class AutoEvaluator:
    def __init__(self, brand_guidelines: Optional[Dict] = None):
        self.brand = brand_guidelines or {}

    def evaluate(self, text: str, topic: str) -> QualityScore:
        prompt = f"""你是一个内容质量评估专家。请评估以下内容。

评估维度（每个维度0-1分）：
1. grammar: 语法是否正确，有无错别字或语病
2. coherence: 内容是否连贯，逻辑是否通顺
3. relevance: 是否切题，覆盖核心信息
4. brand_consistency: 是否符合品牌语调

品牌指南：{json.dumps(self.brand, ensure_ascii=False)}
主题：{topic}
内容：{text[:1500]}

返回JSON：{{"grammar": float, "coherence": float, "relevance": float, "brand_consistency": float}}"""

        resp = client.chat.completions.create(
            model="qwen-plus", temperature=0,
            messages=[{"role": "user", "content": prompt}],
        )
        match = re.search(r'\{.*\}', resp.choices[0].message.content, re.DOTALL)
        if match:
            data = json.loads(match.group())
            return QualityScore(**data)
        return QualityScore(0, 0, 0, 0)

# rubric_manager.py — Rubric评分管理
from dataclasses import dataclass, field
from typing import List, Dict

@dataclass
class RubricDimension:
    name: str
    score: int  # 1-5
    comment: str

@dataclass
class RubricScore:
    content_id: str
    reviewer: str
    dimensions: List[RubricDimension] = field(default_factory=list)
    overall: int = 0
    passed: bool = False

    @property
    def avg_score(self) -> float:
        return sum(d.score for d in self.dimensions) / len(self.dimensions)

class RubricManager:
    DIMENSIONS = ["准确性", "完整性", "可读性", "品牌一致性", "行动号召力"]

    def create_rubric(self, content_id: str, reviewer: str) -> RubricScore:
        return RubricScore(content_id=content_id, reviewer=reviewer)

    def score(self, rubric: RubricScore, dim_name: str,
              score: int, comment: str = "") -> RubricScore:
        if dim_name not in self.DIMENSIONS:
            raise ValueError(f"未知维度: {dim_name}")
        if score < 1 or score > 5:
            raise ValueError("分数必须在1-5之间")
        rubric.dimensions.append(RubricDimension(dim_name, score, comment))
        return rubric

    def finalize(self, rubric: RubricScore, pass_threshold: float = 3.5) -> RubricScore:
        rubric.overall = round(rubric.avg_score, 1)
        rubric.passed = rubric.avg_score >= pass_threshold
        return rubric

# bad_case.py — 坏例管理
from dataclasses import dataclass, field
from datetime import datetime
from typing import List, Optional

@dataclass
class BadCase:
    id: str
    content_id: str
    text: str
    template_name: str
    model: str
    temperature: float
    scores: Dict[str, float]
    failure_type: str  # grammar/coherence/relevance/brand
    failure_reason: str
    created_at: str = field(default_factory=lambda: datetime.now().isoformat())
    fix_action: str = ""

class BadCaseManager:
    def __init__(self):
        self.cases: List[BadCase] = []

    def add_case(self, case: BadCase):
        self.cases.append(case)

    def analyze_by_template(self, template_name: str) -> Dict:
        related = [c for c in self.cases if c.template_name == template_name]
        if not related:
            return {"total": 0, "by_type": {}}
        by_type = {}
        for c in related:
            by_type[c.failure_type] = by_type.get(c.failure_type, 0) + 1
        return {
            "total": len(related),
            "by_type": by_type,
            "top_failure": max(by_type, key=by_type.get),
        }

    def suggest_fix(self, template_name: str) -> List[str]:
        analysis = self.analyze_by_template(template_name)
        suggestions = []
        failure = analysis.get("top_failure")
        if failure == "grammar":
            suggestions.append("降低temperature到0.3以下")
            suggestions.append("在模板中增加语法约束说明")
        elif failure == "coherence":
            suggestions.append("增加模板中的段落结构指引")
            suggestions.append("给LLM填充字段加few-shot示例")
        elif failure == "brand":
            suggestions.append("在Prompt中强化品牌语调描述")
            suggestions.append("增加禁止词列表")
        return suggestions

不推荐
  • 只依赖自动打分，无人为判断
  • 评分结果不关联生成参数
  • 坏例不归档，每次都从零排查
  • 评估了但不驱动优化
推荐
  • 自动打分全覆盖+人工Rubric抽样
  • 每条评分记录关联模板/模型/温度
  • 坏例按类型归因+自动修复建议
  • 周维度评估报告驱动持续优化

常见问题（FAQ）

自动打分的准确率有多高？
用50条人工标注样本测试，qwen-plus打分的四个维度与人工评分相关系数约0.75。语法和品牌一致性相关性高（0.85），相关性和完整性相关性略低（0.65）。自动打分适合批量筛选，高价值内容仍需人工Rubric。
坏例采集多少条开始有意义？
建议至少100条。按失败类型分布可以看到主要问题。比如grammar类占40%说明温度过高或模板缺少约束，brand类占30%说明品牌语调Prompt不够。
评估系统如何与模板优化联动？
每周输出评估报告：每个模板的平均分+主要坏例类型。根据数据：1) 调整temperature；2) 在模板中增加/修改约束；3) 给LLM变量加few-shot。优化后下一周对比分数变化。

本章结论

  • 自动+人工互补：自动打分全覆盖批量场景，人工Rubric负责高价值内容。
  • 四维度评估：语法/连贯性/相关性/品牌一致性，overall为四维均值。
  • 坏例归因：按类型分类+关联生成参数，找到根因而非表面问题。
  • 驱动优化：周维度评估报告指导temperature调整、模板约束和few-shot完善。
  • 持续迭代：评估不是一次性工作，是持续优化的闭环。
