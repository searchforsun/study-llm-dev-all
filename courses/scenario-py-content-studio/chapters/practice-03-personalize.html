用户画像系统

用户画像是个性化内容生成的基础。画像系统包括静态属性（年龄、地域）、动态行为（浏览、购买）和兴趣标签。

# user_profile.py — 用户画像系统
from pydantic import BaseModel
from typing import List, Optional
from datetime import datetime

class UserProfile(BaseModel):
    user_id: str
    age_group: str            # 18-25 / 26-35 / 36-50 / 50+
    gender: str               # male / female / unknown
    city: str
    interests: List[str]      # 兴趣标签
    purchase_history: List[str]  # 最近购买
    preferred_style: str      # concise / detailed / humorous
    preferred_channel: str    # email / sms / in_app
    lifetime_value: float     # 用户生命周期价值

class ProfileBuilder:
    def __init__(self):
        self.profiles: dict = {}

    def build_from_events(self, user_id: str, events: List[dict]) -> UserProfile:
        tags = self._extract_tags(events)
        purchases = [e["item"] for e in events if e["type"] == "purchase"]
        return UserProfile(
            user_id=user_id,
            age_group=self._infer_age(events),
            gender=self._infer_gender(events),
            city=self._infer_city(events),
            interests=tags,
            purchase_history=purchases[-5:],
            preferred_style=self._infer_style(events),
            preferred_channel="email",
            lifetime_value=sum(e.get("amount", 0) for e in events),
        )

    def _extract_tags(self, events: List[dict]) -> List[str]:
        tag_scores = {}
        category_map = {
            "电子产品": ["科技", "数码", "效率"],
            "服装": ["时尚", "穿搭"],
            "书籍": ["阅读", "学习"],
        }
        for e in events:
            for cat, tags in category_map.items():
                if cat in str(e.get("category", "")):
                    for t in tags:
                        tag_scores[t] = tag_scores.get(t, 0) + 1
        return sorted(tag_scores, key=tag_scores.get, reverse=True)[:5]

    def _infer_style(self, events: List[dict]) -> str:
        click_rates = {"concise": 0, "detailed": 0, "humorous": 0}
        for e in events:
            style = e.get("preferred_style", "")
            if style in click_rates:
                click_rates[style] += 1
        return max(click_rates, key=click_rates.get) if any(click_rates.values()) else "concise"

动态Prompt组装

动态Prompt将用户画像注入模板引擎，使相同模板产出个性化内容。

Python：个性化Prompt生成

python
# personalizer.py — 个性化内容生成
from openai import OpenAI

client = OpenAI(base_url="https://dashscope.aliyuncs.com/compatible-mode/v1")

class Personalizer:
    def __init__(self, profile_builder: ProfileBuilder):
        self.profiles = profile_builder

    def build_prompt(self, profile: UserProfile, content_type: str, topic: str) -> str:
        style_guide = {
            "concise": "语言简洁精炼，核心信息前置",
            "detailed": "详细说明，分点列举，提供完整信息",
            "humorous": "轻松幽默，适当使用修辞和调侃",
        }
        return f"""你是一个个性化内容助手。请为以下用户生成{content_type}。

### 用户画像
- 年龄段: {profile.age_group} | 城市: {profile.city}
- 兴趣: {', '.join(profile.interests)}
- 最近购买: {', '.join(profile.purchase_history[-3:])}
- 偏好风格: {style_guide.get(profile.preferred_style, '通用')}
- 用户价值: ¥{profile.lifetime_value:.0f}

### 内容要求
主题: {topic}
风格: {profile.preferred_style}
渠道: {profile.preferred_channel}

请针对该用户的兴趣点和购买历史，生成个性化内容。
如果用户有相关购买历史，做交叉销售推荐。"""
        return prompt

    def generate(self, profile: UserProfile, content_type: str, topic: str) -> str:
        prompt = self.build_prompt(profile, content_type, topic)
        resp = client.chat.completions.create(
            model="qwen-plus",
            messages=[{"role": "user", "content": prompt}],
            temperature=0.4,
        )
        return resp.choices[0].message.content

# 使用示例
builder = ProfileBuilder()
profile = builder.build_from_events("u001", [
    {"type": "purchase", "item": "智能办公助手基础版", "amount": 299, "category": "电子产品"},
    {"type": "view", "item": "AI Agent进阶课", "category": "书籍"},
])
personalizer = Personalizer(builder)
content = personalizer.generate(profile, "营销邮件", "Pro版升级优惠")
print(content[:200])

Java：Spring AI个性化配置

java
// PersonalizationService.java
@Service
public class PersonalizationService {
    private final UserProfileRepository profileRepo;
    private final ChatClient chatClient;

    public String generatePersonalizedContent(String userId, String topic) {
        UserProfile profile = profileRepo.findById(userId).orElseThrow();
        String prompt = buildPersonalizedPrompt(profile, topic);

        return chatClient.prompt()
            .user(prompt)
            .call()
            .content();
    }

    private String buildPersonalizedPrompt(UserProfile p, String topic) {
        return String.format("""
            为用户生成个性化内容：
            用户: %s, 年龄: %s, 兴趣: %s
            主题: %s
            风格: %s
            """, p.getUserId(), p.getAgeGroup(),
            String.join(",", p.getInterests()), topic, p.getPreferredStyle());
    }
}

内容A/B测试

个性化效果需要通过A/B测试验证。将用户随机分组，分别展示个性化内容和通用内容，对比点击率/转化率。

Python：A/B测试框架

python
# ab_test.py — A/B测试框架
import random
from dataclasses import dataclass, field
from typing import List, Dict

@dataclass
class ABTestResult:
    variant: str
    impressions: int = 0
    clicks: int = 0
    conversions: int = 0

    @property
    def ctr(self) -> float:
        return self.clicks / self.impressions if self.impressions > 0 else 0

    @property
    def cvr(self) -> float:
        return self.conversions / self.impressions if self.impressions > 0 else 0

class ABTestService:
    def __init__(self, experiment_name: str, traffic_split: float = 0.5):
        self.name = experiment_name
        self.split = traffic_split  # 个性化流量占比
        self.results = {
            "control": ABTestResult("control"),
            "personalized": ABTestResult("personalized"),
        }

    def assign(self, user_id: str) -> str:
        """将用户分配到对照组或实验组"""
        random.seed(hash(user_id) % (2**31))
        return "personalized" if random.random() < self.split else "control"

    def record(self, variant: str, clicked: bool, converted: bool):
        r = self.results[variant]
        r.impressions += 1
        if clicked:
            r.clicks += 1
        if converted:
            r.conversions += 1

    def report(self) -> Dict:
        c = self.results["control"]
        p = self.results["personalized"]
        lift = (p.ctr / c.ctr - 1) * 100 if c.ctr > 0 else 0
        return {
            "control_ctr": f"{c.ctr:.2%}",
            "personalized_ctr": f"{p.ctr:.2%}",
            "ctr_lift": f"{lift:.1f}%",
            "control_cvr": f"{c.cvr:.2%}",
            "personalized_cvr": f"{p.cvr:.2%}",
        }

不推荐
  • 所有用户看到相同内容
  • 画像仅用年龄性别两个维度
  • 无A/B测试，上线靠感觉
  • 个性化不考虑隐私合规
推荐
  • 基于多维度画像的个性化内容
  • 画像含静态+行为+兴趣三层
  • A/B测试验证个性化效果
  • 用户数据脱敏，符合隐私法规

常见问题（FAQ）

用户画像的冷启动怎么做？新用户没有行为数据。
新用户用默认画像（通用风格）+ 基于注册信息的粗粒度画像（如城市、年龄段）。随着用户交互累积，逐步细化。可在首次交互中加入兴趣选择页面加速画像构建。
个性化内容会不会让用户觉得被监控？
可能。需要在用户协议中明确说明数据用途，并提供个性化开关。内容中避免出现「根据您的浏览记录」这类表述，用「为您推荐」替代。
A/B测试需要多少样本量才有统计意义？
取决于预期提升幅度。通常每个分组至少1000个样本。建议持续运行1-2周，覆盖工作日和周末的行为差异。使用p值检验置信度。

本章结论

  • 画像三层：静态属性+行为数据+兴趣标签，逐层丰富用户画像。
  • 动态Prompt：同一模板+不同画像=不同内容，核心是将画像结构化注入Prompt。
  • A/B验证：流量拆分+指标追踪，用数据证明个性化效果。
  • 冷启动：新用户用默认画像，通过交互逐步积累个性化数据。
  • 隐私合规：数据脱敏+用户选择权，避免过度个性化引起的不适。
