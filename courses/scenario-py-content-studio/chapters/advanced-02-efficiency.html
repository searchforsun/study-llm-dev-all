# metrics_dashboard.py — 指标看板
from dataclasses import dataclass, field
from datetime import datetime, timedelta
from typing import List, Dict
import statistics

@dataclass
class DailyMetrics:
    date: str
    total_generations: int = 0
    auto_approved: int = 0
    human_reviewed: int = 0
    rejected: int = 0
    avg_latency_ms: float = 0
    total_cost: float = 0
    recall_rate: float = 0

class MetricsDashboard:
    def __init__(self):
        self.daily: Dict[str, DailyMetrics] = {}

    def record(self, metrics: DailyMetrics):
        self.daily[metrics.date] = metrics

    def optimization_narrative(self, start_date: str, end_date: str) -> str:
        start_m = self.daily.get(start_date)
        end_m = self.daily.get(end_date)
        if not start_m or not end_m:
            return "数据不足"

        recall_lift = (end_m.recall_rate - start_m.recall_rate) / start_m.recall_rate * 100
        latency_reduction = (start_m.avg_latency_ms - end_m.avg_latency_ms) / start_m.avg_latency_ms * 100

        return (
            f"【优化复盘】{start_date} 至 {end_date}\n"
            f"召回率: {start_m.recall_rate:.0%} → {end_m.recall_rate:.0%} (提升{recall_lift:.0f}%)\n"
            f"延迟: {start_m.avg_latency_ms:.0f}ms → {end_m.avg_latency_ms:.0f}ms (降低{latency_reduction:.0f}%)\n"
            f"日均处理: {start_m.total_generations} → {end_m.total_generations} 条\n"
            f"关键改进: 引入BM25混合检索 + BGE-Reranker精排"
        )

# cost_model.py — 成本模型
from dataclasses import dataclass

@dataclass
class CostModel:
    monthly_volume: int      # 月生成量
    tokens_per_item: int     # 每条平均tokens
    llm_price_per_1k: float  # LLM单价
    embed_price_per_1k: float = 0.001
    review_rate: float = 0.3
    review_price: float = 5.0
    infra_monthly: float = 500

    def monthly_llm_cost(self) -> float:
        total_tokens = self.monthly_volume * self.tokens_per_item
        return total_tokens / 1000 * self.llm_price_per_1k

    def monthly_review_cost(self) -> float:
        return self.monthly_volume * self.review_rate * self.review_price

    def monthly_total(self) -> dict:
        llm = self.monthly_llm_cost()
        review = self.monthly_review_cost()
        return {
            "llm": round(llm, 2),
            "review": round(review, 2),
            "infra": self.infra_monthly,
            "total": round(llm + review + self.infra_monthly, 2),
        }

model = CostModel(monthly_volume=10000, tokens_per_item=500,
                  llm_price_per_1k=0.002)
costs = model.monthly_total()
print(f"月总成本: ¥{costs['total']} (LLM: ¥{costs['llm']}, 审核: ¥{costs['review']}, 基础设施: ¥{costs['infra']})")

Java：SLA监控

java
// SLAMonitor.java — SLA监控
@Component
public class SLAMonitor {
    private final MeterRegistry meterRegistry;

    public SLAMonitor(MeterRegistry meterRegistry) {
        this.meterRegistry = meterRegistry;
    }

    @EventListener
    public void onGenerationComplete(GenerationEvent event) {
        // 成功率
        meterRegistry.counter("content.generate.success").increment();
        // P95延迟
        meterRegistry.timer("content.generate.latency")
            .record(event.getDuration(), TimeUnit.MILLISECONDS);
    }

    public SLAReport getSLAReport() {
        return new SLAReport(
            getSuccessRate(),     // > 99.5%
            getP95Latency(),      // < 3000ms
            getDailyThroughput()  // > 5000条
        );
    }

    public record SLAReport(double successRate, double p95Latency, long dailyThroughput) {}
}

常见问题（FAQ）

召回率到多少才算合格？
取决于业务场景。客服问答场景要求>90%，内容生成素材检索>80%即可。关键是基线一致：同一测试集上的对比才有效。建议每月用固定测试集跑一次回归。
成本模型里的人工审核费怎么估算？
审核费 = 审核单价 x 审核率 x 月产量。审核单价通常为AI生成单价的5-10倍。目标是通过自动化将审核率从50%降到20%以下。
SLA的P95延迟怎么统计？
采集每分钟的请求延迟数据，按从小到大排序，取95%位置的数值。建议用Prometheus+Grafana采集展示。告警阈值设为P95大于目标值持续5分钟。

本章结论

  • 数据叙事三段式：基线→改进→结果，每个数字可验证。
  • 成本三维：LLM调用+人工审核+基础设施，缺一不可。
  • SLA三指标：成功率>99.5%、P95延迟<3s、日均处理量。
  • 交付清单：12项核验清单，确保内容工坊完整上线。
  • 持续复盘：月度复盘+季度目标迭代，用数据驱动优化。
