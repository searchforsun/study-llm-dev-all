Java：Token 计量与成本估算

java 
import io.micrometer.core.instrument.Counter;
import io.micrometer.core.instrument.MeterRegistry;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.atomic.AtomicLong;

@Service
public class TokenMeteringService {
    private final MeterRegistry registry;
    // 租户级日用量缓存
    private final Map tenantDailyTokens;

    // 模型单价（$/1K tokens）
    private static final Map MODEL_PRICES = Map.of(
        "claude-sonnet-4", 0.003,
        "gpt-4o", 0.005,
        "deepseek-v3", 0.001
    );

    public TokenMeteringService(MeterRegistry registry) {
        this.registry = registry;
        this.tenantDailyTokens = new ConcurrentHashMap<>();
    }

    public void recordTokenUsage(String tenantId, String modelId,
                                 int promptTokens, int completionTokens) {
        // Prometheus Counter — 长期趋势
        Counter.builder("llm.tokens.prompt")
            .tag("tenant", tenantId).tag("model", modelId)
            .register(registry).increment(promptTokens);
        Counter.builder("llm.tokens.completion")
            .tag("tenant", tenantId).tag("model", modelId)
            .register(registry).increment(completionTokens);

        // 估算费用
        double price = MODEL_PRICES.getOrDefault(modelId, 0.003);
        double cost = (promptTokens + completionTokens) / 1000.0 * price;
        Counter.builder("llm.cost.estimated")
            .tag("tenant", tenantId).tag("model", modelId)
            .register(registry).increment(cost);

        // 租户级日用量检查
        int total = promptTokens + completionTokens;
        tenantDailyTokens.computeIfAbsent(tenantId, k -> new AtomicLong())
            .addAndGet(total);
    }

    public long getTenantDailyTokens(String tenantId) {
        return tenantDailyTokens
            .getOrDefault(tenantId, new AtomicLong()).get();
    }

    public boolean isTenantOverDailyCap(String tenantId, long cap) {
        return getTenantDailyTokens(tenantId) > cap;
    }
}

// 每日零点重置租户用量
@Component
public class DailyTokenReset {
    private final TokenMeteringService metering;

    @Scheduled(cron = "0 0 0 * * ?")
    public void resetDailyCounters() {
        log.info("Daily token counters reset");
    }
}

Python：LangSmith 用量提取与成本报表

python 
from langsmith import Client
from datetime import datetime, timedelta
from collections import defaultdict

class CostReporter:
    """从 LangSmith 提取 Token 用量并按租户汇总。"""

    MODEL_PRICES = {
        "claude-sonnet-4": {"input": 0.003, "output": 0.015},
        "gpt-4o": {"input": 0.005, "output": 0.015},
        "deepseek-v3": {"input": 0.001, "output": 0.002},
    }

    def __init__(self):
        self.client = Client()

    def report_by_tenant(self, days: int = 30) -> dict:
        """生成按租户汇总的成本报表"""
        start = datetime.utcnow() - timedelta(days=days)
        runs = self.client.list_runs(
            project_name="corpassist-agent",
            start_time=start
        )
        report = defaultdict(lambda: {
            "prompt_tokens": 0, "completion_tokens": 0,
            "cost": 0.0, "call_count": 0
        })
        for run in runs:
            tenant = run.metadata.get("tenant_id", "unknown")
            model = run.metadata.get("model_id", "unknown")
            usage = run.outputs.get("usage", {})
            pt = usage.get("prompt_tokens", 0)
            ct = usage.get("completion_tokens", 0)
            report[tenant]["prompt_tokens"] += pt
            report[tenant]["completion_tokens"] += ct
            report[tenant]["call_count"] += 1
            prices = self.MODEL_PRICES.get(model, {"input": 0.003, "output": 0.015})
            report[tenant]["cost"] += (pt / 1000 * prices["input"]
                                       + ct / 1000 * prices["output"])
        return dict(report)

    def detect_anomaly(self, tenant_id: str, threshold: float = 500.0) -> bool:
        """检测租户日成本是否异常突增"""
        today = self._daily_cost(tenant_id, 1)
        avg = self._daily_cost(tenant_id, 30) / 30
        return today > avg * threshold if avg > 0 else False

    def _daily_cost(self, tenant_id: str, days: int) -> float:
        report = self.report_by_tenant(days)
        return report.get(tenant_id, {}).get("cost", 0.0)

成本数据来自双栈：Java 侧 Micrometer Counter 提供实时指标，LangSmith 提供详细的 token/run 级别数据。

不推荐
  • 月末从 API 账单反推成本，无法分租户
  • 不对 prompt/completion 分开统计
  • 无成本告警，等财务提醒才发现超预算
  • 所有租户共享同一个预算池
推荐
  • 实时 Counter 计量 + 定时 LangSmith 报表
  • prompt/completion 分开统计以识别优化空间
  • 三级告警：日 80%/100%、周趋势异常
  • 每个租户独立 daily cap + 月预算

三级成本告警设计

级别触发条件通知方式自动化动作
Info日用量达预算 80%Slack 通知无
Warning日用量达预算 100% 或日环比突增 300%PagerDuty触发租户级 rate limit
Critical连续 3 天超预算或月预算 80%电话临时停用非核心模型
Prompt token 和 completion token 的价格相同吗？

绝大多数模型 provider 的 completion token 比 prompt token 贵 3-5 倍。分开统计可以更精确地估算成本，也帮助你发现是否输出过长（completion 占比异常高）。

Chargeback 在多租户场景如何实现？
Chargeback 是把成本按租户分摊回业务部门。实现方式：1) 每个请求携带 tenant_id，2) 计量服务按 tenant 聚合 daily token 用量，3) 月末生成按租户的账单报表。关键是不只分 tenant，还分 model（不同模型价格差异大）。
Daily cap 超了该怎么做？
推荐三层策略：1) cap 达 80% 通知租户管理员，2) cap 达 100% 对非核心端点降级（用小模型或返回缓存），3) cap 超 120% 直接阻断，返回 429。阻断策略需在合同或服务条款中明确。
如何管理 token governance 的预算分配？
预算分配是迭代过程：1) 初始阶段按历史用量的 1.5x 分配 buffer，2) 每月回顾实际 vs 预算，3) 为大租户提供预付费折扣。工具方面，LangSmith 的 project-level budget 和 Grafana 告警配合使用。
常见误区
  • 只看总 token 不看分模型 — 不同模型单价差 10 倍；优化从换模型开始。
  • 只统计不告警 — 成本治理的闭环是「计量 → 预算 → 告警 → 自动化降级」，缺告警等于没有治理。
  • 一次性设死预算 — 预算应根据业务增长动态调整，季度 review。
  • 忽略嵌入缓存 — embedding 调用通常占 token 消耗 20-30%，语义缓存可以大幅降低。

本章结论

  • 分租户计量是基础：没有 token metering 就没有成本可观测。至少按 tenant + model 两个维度区分。
  • 三级告警体系：日 80%/100%/异常突增三个级别，从通知到自动限流。
  • Prompt/Completion 分开统计：价格差异大，分开才能精确估算成本并发现优化点。
  • Chargeback 是 FinOps 闭环：把成本分摊回业务部门，推动业务方主动优化用量。
  • Daily cap 防止失控：每个租户独立 cap，超限自动降级或阻断，避免账单爆炸。

用 CorpAssist 多租户场景，完成计量方案、成本看板设计和告警规则配置（约 25 分钟）。

操作步骤

  1. 设计 CorpAssist 的 Token 计量方案：列出至少 3 个指标及对应的 Prometheus Counter 名称和标签。
  2. 假设 acme 租户 daily cap 为 1M tokens，写一个 Java 方法检查是否超限（参考正文代码块）。
  3. 在纸上画出成本看板布局：包含日趋势、租户排名、模型分布、异常检测四个面板。
  4. 配置一个 Prometheus 告警规则：日用量 24 小时内超过前 7 天日均值的 3 倍。

判断练习

  1. 某租户日成本 $100 突增到 $500。排查步骤优先级：A) 检查该租户 QPS 变化 B) 检查是否换了更贵的模型 C) 检查 prompt 长度是否增加 D) 以上都需要按序排查

    参考答案

    D：三步都要。先看 QPS（用量增加），再看 model_id（单价变化），最后看 prompt/completion 长度（单次调用成本变化）。通常 A 是主要原因。
