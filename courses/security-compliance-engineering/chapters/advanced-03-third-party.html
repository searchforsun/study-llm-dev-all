第三方评估与数据出境

CorpAssist · 第三方依赖（30 秒） CorpAssist 同时使用阿里云通义千问（国内）和 OpenAI（美国）做 LLM 供应商。当用户输入包含 PII 时，如果路由到 OpenAI，数据就出了境。合规要求：金融客户的数据必须走国内模型，不能出境。需要在应用层做数据出境检测和路由控制。

Java：数据出境控制

java 
import java.util.Map;
import java.util.Set;

@Service
public class DataCrossBorderController {
    private final PIIDetector piiDetector;

    // 境内模型白名单
    private static final Set DOMESTIC_MODELS = Set.of(
        "qwen-max", "deepseek-v3", "baidu-ernie", "glm-4"
    );

    // 境外模型
    private static final Set OVERSEAS_MODELS = Set.of(
        "claude-sonnet-4", "gpt-4o", "gemini-pro"
    );

    public DataCrossBorderController(PIIDetector pii) {
        this.piiDetector = pii;
    }

    // 判断是否需要阻断请求
    public CrossBorderDecision check(String question, String modelId,
                                      String tenantRegion) {
        boolean containsPII = !piiDetector.detect(question).isEmpty();
        boolean isOverseasModel = OVERSEAS_MODELS.contains(modelId);
        boolean isRestrictedRegion = tenantRegion.equals("CN");

        // 中国用户 + 境外模型 + 含 PII → 必须阻断
        if (isRestrictedRegion && isOverseasModel && containsPII) {
            return CrossBorderDecision.block("数据包含 PII，禁止出境");
        }

        // 中国用户 + 境外模型 + 无 PII → 警告但放行
        if (isRestrictedRegion && isOverseasModel) {
            return CrossBorderDecision.warn("数据将发送到境外模型");
        }

        return CrossBorderDecision.allow();
    }

    public record CrossBorderDecision(
        boolean allowed, boolean warn, String message
    ) {
        public static CrossBorderDecision allow() {
            return new CrossBorderDecision(true, false, null);
        }
        public static CrossBorderDecision warn(String msg) {
            return new CrossBorderDecision(true, true, msg);
        }
        public static CrossBorderDecision block(String msg) {
            return new CrossBorderDecision(false, false, msg);
        }
    }
}

// 模型路由（根据区域选择 LLM）
@Component
public class RegionalModelRouter {
    private static final Map REGION_MODEL = Map.of(
        "CN", "qwen-max",    // 国内：阿里云
        "US", "claude-sonnet-4",  // 美国：Claude
        "EU", "gpt-4o"       // 欧洲：GPT-4o
    );

    public String resolveModel(String tenantRegion) {
        return REGION_MODEL.getOrDefault(tenantRegion, "gpt-4o");
    }
}

Python：DPIA 数据保护影响评估

python 
from dataclasses import dataclass, field
from typing import List, Optional


@dataclass
class DPIAAssessment:
    """数据保护影响评估报告。"""
    system_name: str
    assessor: str
    date: str

    # 数据处理清单
    data_categories: List[str] = field(default_factory=list)
    processing_purposes: List[str] = field(default_factory=list)

    # 第三方
    third_parties: List[str] = field(default_factory=list)
    cross_border: bool = False
    cross_border_destinations: List[str] = field(default_factory=list)

    # 风险与措施
    risks: List[str] = field(default_factory=list)
    mitigations: List[str] = field(default_factory=list)

    # 结论
    residual_risk: str = "medium"
    approved: bool = False


class DPIATemplate:
    """DPIA 报告生成器。"""

    @staticmethod
    def for_corpassist() -> DPIAAssessment:
        return DPIAAssessment(
            system_name="CorpAssist 智能客服",
            assessor="安全团队",
            date="2026-05-29",
            data_categories=[
                "用户问题（可能含 PII）",
                "企业知识库文档",
                "系统日志（含用户 ID、租户 ID）",
                "向量 embedding",
            ],
            processing_purposes=[
                "LLM 回复生成",
                "知识库检索",
                "服务质量监控",
                "合规审计",
            ],
            third_parties=[
                "LLM API 提供商（OpenAI / Anthropic / 阿里云）",
                "向量数据库（Milvus Cloud）",
                "云基础设施（AWS / 阿里云）",
                "监控服务（LangSmith）",
            ],
            cross_border=True,
            cross_border_destinations=["美国", "新加坡"],
            risks=[
                "PII 通过 LLM API 出境",
                "知识库内容在第三方日志中可见",
                "向量 embedding 可能重建原始数据",
            ],
            mitigations=[
                "PII 检测和脱敏后发送给 LLM",
                "第三方合同禁止用数据训练模型",
                "embedding 存储加密",
                "数据出境评估和审批",
            ],
            residual_risk="medium",
            approved=False,
        )

    @staticmethod
    def generate_report(assessment: DPIAAssessment) -> str:
        lines = [
            f"# DPIA 数据保护影响评估报告",
            f"系统: {assessment.system_name}",
            f"评估人: {assessment.assessor}",
            f"日期: {assessment.date}",
            "",
            "## 1. 数据处理清单",
        ]
        for c in assessment.data_categories:
            lines.append(f"- {c}")
        lines.extend(["", "## 2. 第三方依赖"])
        for t in assessment.third_parties:
            lines.append(f"- {t}")
        lines.extend(["", "## 3. 数据出境"])
        lines.append(f"是否出境: {assessment.cross_border}")
        for d in assessment.cross_border_destinations:
            lines.append(f"- 目的地: {d}")
        lines.extend(["", "## 4. 风险与缓解措施"])
        for r, m in zip(assessment.risks, assessment.mitigations):
            lines.append(f"- 风险: {r}")
            lines.append(f"  缓解: {m}")
        lines.extend(["", f"## 5. 残留风险: {assessment.residual_risk}"])
        return "\n".join(lines)

第三方评估流程

第三方风险评估流程
flowchart TB
  A[发现新第三方] --> B[填写评估表]
  B --> C{数据是否出境?}
  C -->|是| D[DPIA 评估]
  C -->|否| E[标准评估]
  D --> F{风险可接受?}
  E --> F
  F -->|是| G[签署合同条款]
  F -->|否| H[寻找替代方案]
  G --> I[加入第三方清单]
  I --> J[定期复评]

每个第三方依赖必须经过评估流程：发现 → 评估 → 签约 → 监控 → 复评。

LLM API 提供商的数据保护承诺在哪里？
在服务条款和 DPA（数据处理协议）中。关键条款：1) 是否用 API 数据训练模型，2) 数据传输是否加密，3) 数据存储位置，4) 服务终止时数据删除方式。OpenAI API 承诺不训练，Anthropic 提供 DPA 签署。
Subprocessor 指什么？
Subprocessor 是第三方再委托的第四方。例如 LLM API 商使用 AWS 作为基础设施，AWS 就是 subprocessor。合同应要求第三方披露其 subprocessor 清单，并在增加新的 subprocessor 时通知客户。
向量数据库中的 embedding 算不算 PII？
在法律上存在灰色地带。技术上 embedding 可能通过逆向工程重建部分原始信息。GDPR 建议将 embedding 视为潜在 PII。CorpAssist 实践中：对含 PII 的文档不生成 embedding，或在生成前脱敏。

本章结论

  • 第三方清单：6 类第三方依赖，每类有风险等级和控制措施。
  • 数据出境控制：PII 检测 + 区域路由 + 出境阻断三层控制。
  • DPIA 评估：数据分类 →第三方→ 出境 → 风险 → 缓解的完整框架。
  • 合同条款：禁止用数据训练、数据加密、subprocessor 披露。
  • 区域路由：根据用户区域自动选择境内/境外模型。
