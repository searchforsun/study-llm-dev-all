OPA 与规则

CorpAssist · 策略变更（30 秒） 安全团队需要更新话题过滤规则：新增「加密货币」到禁止话题列表。传统方式：改代码 → 提 PR → 重新部署。策略即代码方式：改 rego rule → 提 PR → CI policy 自动测试 → 合并后配置中心热更新——无需重启服务。

rego 
# corpassist-security-policy.rego
package corpassist.security

import future.keywords.if
import future.keywords.in

# 输入定义
input = {
    "question": "用户输入",
    "role": "用户角色",
    "tenant": "租户 ID",
    "tool_calls": ["工具调用列表"]
}

# 默认允许
default allow = false

# 规则 1: 阻止敏感话题
deny[reason] {
    some topic in ["政治", "宗教", "性", "暴力", "毒品", "赌博", "加密货币"]
    contains(input.question, topic)
    reason = sprintf("敏感话题: %s", [topic])
}

# 规则 2: 工具角色鉴权
deny[reason] {
    some tool in input.tool_calls
    not allowed_tool_for_role(input.role, tool)
    reason = sprintf("角色 %s 无权使用工具 %s", [input.role, tool])
}

allowed_tool_for_role(role, tool) {
    role == "agent"
    tool in ["order_query", "ticket_query", "faq_search"]
}

allowed_tool_for_role(role, tool) {
    role == "senior_agent"
    tool in ["order_query", "ticket_query", "faq_search", "refund_create"]
}

# 规则 3: PII 检测
contains_pii(text) {
    regex.match("\\d{18}", text)
}

contains_pii(text) {
    regex.match("1[3-9]\\d{9}", text)
}

# 规则 4: 拒绝带有 PII 的输入
deny[reason] {
    contains_pii(input.question)
    reason = "输入包含个人身份信息"
}

# 输出：汇总所有拒绝原因
allow {
    count(deny) == 0
}

Java：OPA 策略执行客户端

java 
import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.node.ObjectNode;
import org.springframework.web.client.RestTemplate;

@Service
public class OpaPolicyClient {
    private final RestTemplate restTemplate;
    private final ObjectMapper mapper;
    private final String opaUrl;

    public OpaPolicyClient(@Value("${opa.url}") String opaUrl) {
        this.restTemplate = new RestTemplate();
        this.mapper = new ObjectMapper();
        this.opaUrl = opaUrl;
    }

    // 评估安全策略
    public PolicyResult evaluate(String question, String role,
                                  List toolCalls) {
        ObjectNode input = mapper.createObjectNode();
        input.put("question", question);
        input.put("role", role);
        input.put("tenant", "acme");
        var tools = input.putArray("tool_calls");
        toolCalls.forEach(tools::add);

        var response = restTemplate.postForEntity(
            opaUrl + "/v1/data/corpassist/security/allow",
            Map.of("input", input),
            JsonNode.class
        );

        JsonNode body = response.getBody();
        boolean allowed = body.path("result").asBoolean(false);
        List reasons = new ArrayList<>();
        if (!allowed) {
            var denyResp = restTemplate.postForEntity(
                opaUrl + "/v1/data/corpassist/security/deny",
                Map.of("input", input),
                JsonNode.class
            );
            JsonNode denyResult = denyResp.getBody().path("result");
            if (denyResult.isArray()) {
                denyResult.forEach(n -> reasons.add(n.asText()));
            }
        }
        return new PolicyResult(allowed, reasons);
    }

    // 热加载策略（OPA 自动检测文件变化，无需重启）
    public void reloadPolicy() {
        restTemplate.postForEntity(
            opaUrl + "/v1/policies/corpassist-security",
            null, Void.class);
    }

    public record PolicyResult(boolean allowed, List reasons) {}
}

// 在输入审核中使用
@Component
public class PolicyBasedModeration {
    private final OpaPolicyClient opa;

    public ModerationResult check(String question, String role) {
        var result = opa.evaluate(question, role, List.of());
        if (!result.allowed()) {
            return ModerationResult.blocked(
                String.join("; ", result.reasons()));
        }
        return ModerationResult.passed();
    }
}

Python：策略版本管理

python 
import hashlib
import git
from typing import Dict, Any


class PolicyVersionManager:
    """策略版本管理器 —— 策略作为代码进行版本管理。"""

    def __init__(self, repo_path: str = "./policies"):
        self.repo_path = repo_path
        self.repo = git.Repo(repo_path)

    def get_current_version(self) -> Dict[str, Any]:
        """返回当前策略版本信息"""
        head = self.repo.head
        return {
            "commit": head.commit.hexsha,
            "message": head.commit.message.strip(),
            "author": str(head.commit.author),
            "date": head.commit.committed_datetime.isoformat(),
            "files_changed": [item.a_path for item in head.commit.diff("HEAD~1")],
        }

    def get_policy_hash(self, policy_name: str) -> str:
        """返回策略文件的 SHA256"""
        path = f"{self.repo_path}/{policy_name}.rego"
        with open(path, "rb") as f:
            return hashlib.sha256(f.read()).hexdigest()

    def deploy_policy(self, version: str) -> bool:
        """部署特定版本的策略到 OPA"""
        log.info("Deploying policy version %s", version)
        # git checkout tag/version
        # POST to OPA API for hot reload
        return True

    def diff_policies(self, old_commit: str, new_commit: str) -> str:
        """对比两个版本间的策略差异"""
        old = self.repo.commit(old_commit)
        new = self.repo.commit(new_commit)
        diffs = []
        for diff in old.diff(new):
            if diff.a_path.endswith(".rego"):
                diffs.append(f"  {diff.change_type}: {diff.a_path}")
        return "\n".join(diffs)

flowchart LR

  B --> C[CI 测试]
  C -->|pass| D[Merge to main]

  H --> A

策略变更走完整 CI/CD pipeline：修改 → review → 测试 → 合并 → 自动部署。

OPA 和 Spring AI Advisor 怎么配合？
OPA 做决策引擎，Advisor 做执行点。Advisor 收到请求后调用 OPA 获取决策结果（allow/deny + reasons），根据结果决定放行、阻断或替换。策略变更只需改 Rego 文件，无需改代码。
Rego 学习曲线怎么样？
Rego 是声明式查询语言，和 SQL 类似但有差异。团队中需要 1-2 人掌握 Rego 编写策略，其他人通过 YAML/JSON 配置策略（可以通过工具将 YAML 转成 Rego）。对于复杂策略（跨多条件联合判断）推荐直接写 Rego。
GitOps policy 怎么实现？
Git 仓库作为策略的唯一真相源。OPA 监听 Git 仓库的变更，合并后自动加载新策略（通过 bundle API 或 sidecar 模式）。所有策略变更有完整的审批和回滚记录。

本章结论

  • OPA/Rego：统一策略语言，覆盖话题过滤、工具鉴权、PII 检测。
  • 策略 = 代码：git 版本管理、PR review、CI 测试、自动部署。
  • OPA 做决策：不执行只决策，应用根据 allow/deny 执行对应动作。
  • 热更新：策略变更后 OPA 自动加载，无需重启服务。
  • GitOps：Git 仓库作为策略唯一真相源，变更可追溯。
