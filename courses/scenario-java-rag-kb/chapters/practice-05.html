集成测试设计（Testcontainers）

Spring AI + Testcontainers 可以启动真实的 Milvus 和 ES 容器进行集成测试。这是最可信的检索质量验证方式。

java
@SpringBootTest
@Testcontainers
class RetrievalIntegrationTest {

    @Container
    static MilvusContainer milvus = new MilvusContainer("milvusdb/milvus:v2.4.0")
        .withStartupTimeout(Duration.ofMinutes(5));

    @Container
    static ElasticsearchContainer es = new ElasticsearchContainer(
            "docker.elastic.co/elasticsearch/elasticsearch:8.15.0")
        .withEnv("xpack.security.enabled", "false")
        .withEnv("ES_JAVA_OPTS", "-Xms512m -Xmx512m");

    @Autowired
    private HybridSearchService hybridSearch;

    @Autowired
    private RerankService rerankService;

    private static List goldenQueries;

    @BeforeAll
    static void setup() {
        // 加载黄金评测集
        goldenQueries = List.of(
            new TestQuery("产假天数", List.of("doc-hr-001", "doc-hr-002")),
            new TestQuery("年假计算方法", List.of("doc-hr-003")),
            new TestQuery("加班费标准", List.of("doc-hr-004", "doc-hr-005")),
            new TestQuery("HR-2024-001 条款", List.of("doc-hr-001")),
            new TestQuery("医疗保险报销比例", List.of("doc-hr-006"))
        );
    }

    @Test
    void testHybridRetrievalRecall() {
        int hits = 0;
        for (TestQuery tq : goldenQueries) {
            List results = hybridSearch.hybridSearch(
                tq.query(), 5);
            boolean hasRelevant = results.stream()
                .anyMatch(d -> tq.relevantDocIds().contains(d.getId()));
            if (hasRelevant) hits++;
        }

        double recall = (double) hits / goldenQueries.size();
        System.out.println("Recall@5: " + (recall * 100) + "%");
        assertThat(recall).isGreaterThanOrEqualTo(0.85);
    }

    @Test
    void testRerankImprovement() {
        for (TestQuery tq : goldenQueries) {
            // 不加 Rerank
            List withoutRerank = hybridSearch.hybridSearch(
                tq.query(), 5);
            // 加 Rerank
            List withRerank = rerankService.rerank(
                tq.query(), withoutRerank);

            // 验证 Rerank 后的第一个结果相关性分数更高
            double beforeScore = withoutRerank.isEmpty() ? 0 :
                (double) withoutRerank.get(0).getMetadata()
                    .getOrDefault("distance", 0.0);
            double afterScore = withRerank.isEmpty() ? 0 :
                (double) withRerank.get(0).getMetadata()
                    .getOrDefault("rerank_score", 0.0);

            System.out.printf("Query: %s | Before: %.4f | After: %.4f%n",
                tq.query(), beforeScore, afterScore);
        }
    }
}

Testcontainers 的优点：测试环境与生产环境完全一致（同样的 Milvus/ES 版本），测试结果可重复。CI/CD 流水线中自动运行，每次代码变更都验证 Recall@K 指标不退化。

Advisor 断言：验证引用存在

除了检索质量的集成测试，还需要验证回答生成阶段的 Advisor 行为是否正确。通过断言检查响应的 attributes 中是否包含正确的引用信息。

java
@SpringBootTest
class AdvisorAssertionTest {

    @Autowired
    private ChatClient chatClient;

    @MockBean
    private VectorStore vectorStore;

    @Test
    void testCitationGuardAdvisorProducesCitations() {
        // 准备模拟的检索结果
        Document mockDoc = Document.builder()
            .id("doc-hr-001")
            .text("产假天数根据国家规定为128天。")
            .metadata(Map.of(
                "source", "hr_policy_2024.pdf",
                "page", "3",
                "version", "2024"
            ))
            .build();

        when(vectorStore.search(any(SearchRequest.class)))
            .thenReturn(List.of(mockDoc));

        // 执行问答
        Answer answer = chatClient.prompt()
            .user("产假多少天？")
            .advisors(new QuestionAnswerAdvisor(vectorStore,
                SearchRequest.builder().topK(5).build()))
            .call()
            .entity(Answer.class);

        // 断言引用存在
        assertThat(answer.citations())
            .isNotEmpty()
            .anyMatch(c -> "hr_policy_2024.pdf".equals(c.source()));

        // 断言回答内容包含引用标注
        assertThat(answer.content())
            .contains("128")
            .contains("引用来源");
    }

    @Test
    void testConfidenceCheckAdvisorRejectsLowConfidence() {
        // 模拟低相关性检索结果
        Document lowScoreDoc = Document.builder()
            .id("doc-low-001")
            .text("不相干的文档内容")
            .metadata(Map.of(
                "distance", 0.15,
                "source", "unrelated.pdf"
            ))
            .build();

        when(vectorStore.search(any(SearchRequest.class)))
            .thenReturn(List.of(lowScoreDoc));

        // 执行问答
        Answer answer = chatClient.prompt()
            .user("产假多少天？")
            .advisors(
                new QuestionAnswerAdvisor(vectorStore,
                    SearchRequest.builder().topK(5).build()),
                new ConfidenceCheckAdvisor(...)
            )
            .call()
            .entity(Answer.class);

        // 断言被拒答
        assertThat(answer.content())
            .contains("没有找到足够相关的信息");
    }
}

Advisor 断言测试是「测试金字塔」中比集成测试更高一层的测试——不仅验证检索是否正确，还验证回答生成过程是否遵循了业务规则（引用/拒答/disclaimer）。

与 Python RAGAS 评测结果对照

Python 的 RAGAS 框架提供了全面的 RAG 评测指标（Faithfulness / Answer Relevance / Context Precision）。评测结果应该通过 HTTP 回传 Java 侧，触发回归测试。

java
@RestController
@RequestMapping("/api/v1/eval")
public class EvalReportController {

    private final RegressionTestRunner regressionTestRunner;

    @PostMapping("/ragas-report")
    public ResponseEntity receiveRagasReport(
            @RequestBody RagasReport report) {

        log.info("Received RAGAS report: " +
            "faithfulness={}, relevance={}, precision={}",
            report.faithfulness(),
            report.answerRelevance(),
            report.contextPrecision());

        // 规则：如果 RAGAS 分数低于阈值，自动运行回归测试
        if (report.faithfulness() < 0.80 || report.contextPrecision() < 0.85) {
            RegressionTestResult result = regressionTestRunner.runAll();
            log.warn("RAGAS report triggered regression tests: {}",
                result.summary());

            // 如果回归测试也失败，创建告警
            if (result.hasFailures()) {
                alertService.sendAlert("RAG quality regression detected. " +
                    "RAGAS faithfulness: " + report.faithfulness() +
                    ", Regression failures: " + result.failureCount());
            }
        }

        return ResponseEntity.ok("Report received");
    }
}

record RagasReport(
    double faithfulness,
    double answerRelevance,
    double contextPrecision,
    double contextRecall,
    List badCases
) {}

record BadCase(String question, String answer,
               List contexts, double score) {}

CorpAssist 实践中，RAGAS 触发回归测试的阈值为：faithfulness < 0.80 或 contextPrecision < 0.85。回归测试失败时会自动创建 Jira 工单并通知值班工程师。这种 Java-Python 交叉验证机制确保了 RAG 系统持续达到质量标准。

索引蓝绿部署（别名切换）

当文档更新或分块参数变化时，需要重建索引。蓝绿部署使用别名实现零停机索引切换。

java
@Service
public class BlueGreenIndexService {
    private static final String ALIAS = "corpassist-policies-active";
    private static final String BLUE = "corpassist-policies-blue";
    private static final String GREEN = "corpassist-policies-green";

    private final MilvusServiceClient milvusClient;
    private final ElasticsearchClient esClient;

    // 切换到下一个索引
    public void switchIndex() {
        String current = identifyActiveIndex();
        String next = current.equals(BLUE) ? GREEN : BLUE;

        log.info("Switching active index: {} -> {}", current, next);

        // 1. 将别名从当前索引移除
        removeAlias(current);

        // 2. 将别名赋给新索引
        assignAlias(next);

        // 3. 验证新索引可用
        validateIndex(next);

        log.info("Active index switched to: {}", next);
    }

    // 初始创建索引
    public void initializeIndexes() {
        // 创建 blue 和 green 两个索引集合
        createCollection(BLUE);
        createCollection(GREEN);

        // 初始将别名指向 green（空索引可正常服务）
        assignAlias(GREEN);
    }

    // 重建索引(在新索引上)
    public void rebuildIndex(Collection documents) {
        String next = identifyInactiveIndex();
        log.info("Rebuilding index: {}", next);

        // 清空旧数据
        clearCollection(next);

        // 批量写入
        vectorStore.add(documents);

        // 验证数据完整性
        int docCount = countDocuments(next);
        log.info("Index {} rebuilt with {} documents", next, docCount);

        // 自动切换
        switchIndex();

        // 清理旧索引中的过期数据
        clearCollection(identifyInactiveIndex());
    }
}

蓝绿部署的关键优势：重建索引耗时（大型文档库可能需要数小时）完全不影响在线问答服务。旧索引在重建期间继续服务，新索引就绪后通过别名切换秒级生效。如果新索引有问题，别名可以立即切换回旧索引实现回滚。

面试可能问

Q1: Testcontainers 集成测试和单元测试的比例怎么分配？

测试金字塔：单元测试 70% + 集成测试 20% + E2E 测试 10%。单元测试测试单个组件（Splitter/MetadataEnricher），集成测试测试检索链路完整正确性（用真实 Milvus/ES），E2E 测试从 HTTP 请求到回答的端到端链路。Testcontainers 适合集成测试层。

Q2: 蓝绿部署切换时如果新索引有问题，怎么快速回滚？

别名切换的特性：切换别名到新索引后，旧索引的数据并没有删除。如果发现问题，只需要将别名再切回旧索引即可（秒级）。旧索引在切换后保留 72 小时作为回滚窗口，72 小时后自动清理。清理前发通知确认是否还有回滚需求。

判断练习

场景：CI 流水线中集成测试经常因为 Testcontainers 拉取镜像慢而超时。怎么优化？

优化方向：① CI 环境预拉取镜像到本地缓存；② 使用镜像预热策略——在 CI 启动时先拉取镜像再并行运行其他步骤；③ 增加 Testcontainers 超时时间（Ryuk 默认 30s 改为 60s）；④ 考虑使用 withReuse(true) 在开发环境复用容器。
