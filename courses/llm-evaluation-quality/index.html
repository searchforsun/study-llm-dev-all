回归门禁流程
flowchart LR
  MR[Open PR] --> EVAL[Trigger eval CI]
  EVAL --> COMPARE{比 baseline}
  COMPARE -->|all delta ≥ -0.02| PASS[Required check pass]

每次 MR 自动触发的回归门禁：计算 delta vs baseline，任何指标退化超 -0.02 则阻塞合并。

版本对比：Baseline Pin 与 Delta 计算

version diff：新 report vs baseline pin。baseline 是上次通过的评测报告，每次 MR 计算指标 delta。

代码：Baseline 管理与 Delta 计算

python 
# regression_gate.py — 回归门禁引擎
import json
from typing import Dict, Optional, List

class BaselineManager:
    """Baseline 管理"""

    def __init__(self, baseline_path: str = "./data/baseline_report.json"):
        self.baseline_path = baseline_path

    def get_baseline(self) -> Optional[Dict]:
        try:
            with open(self.baseline_path, "r") as f:
                return json.load(f)
        except FileNotFoundError:
            return None

    def update_baseline(self, report: Dict):
        """通过 MR 后更新 baseline"""
        with open(self.baseline_path, "w") as f:
            json.dump(report, f, indent=2, ensure_ascii=False)

class RegressionGate:
    """回归门禁"""

    def __init__(self, max_delta: float = -0.02,
                 hard_case_max_delta: float = -0.01):
        self.max_delta = max_delta
        self.hard_case_max_delta = hard_case_max_delta
        self.baseline_mgr = BaselineManager()

    def check(self, current_report: Dict) -> Dict:
        """检查是否存在回归"""
        baseline = self.baseline_mgr.get_baseline()
        if not baseline:
            return {"gate_passed": True, "reason": "No baseline yet"}

        deltas = {}
        regressions = []

        for metric in ["faithfulness", "context_recall", "answer_relevancy"]:
            if metric in current_report.get("aggregate", {}) and \
               metric in baseline.get("aggregate", {}):
                delta = (current_report["aggregate"][metric] -
                         baseline["aggregate"][metric])
                deltas[metric] = round(delta, 4)
                if delta < self.max_delta:
                    regressions.append({
                        "metric": metric,
                        "delta": delta,
                        "threshold": self.max_delta,
                    })

        gate_passed = len(regressions) == 0
        return {
            "gate_passed": gate_passed,
            "deltas": deltas,
            "regressions": regressions,
            "reason": "All metrics passed" if gate_passed else
                      f"Regression in: {', '.join(r['metric'] for r in regressions)}",
        }

    def generate_comment(self, result: Dict) -> str:
        """生成 MR 评论"""
        lines = ["## Eval Gate Report", ""]
        for metric, delta in result.get("deltas", {}).items():
            emoji = "✅" if delta >= self.max_delta else "❌"
            lines.append(f"{emoji} {metric}: {delta:+.4f}")
        if not result["gate_passed"]:
            lines.append("")
            lines.append("Regression detected. Please fix before merge.")
        return "\n".join(lines)

阻塞策略：Deploy Hold 与通知

metric regression 超 -0.02 或 hard case 失败 → deploy hold。

不推荐
  • warn only，不阻塞发布流水线
  • 手工审批 bypass 门禁
  • 不看 baseline delta，只看绝对分
  • 无通知机制，owner 不知情
推荐
  • block CI + 通知 quality owner
  • bypass 需要 quality owner 审批并记录
  • 对比 baseline delta + 难例子集
  • Slack/钉钉通知 regression 详情

Merge Queue 与 Release Candidate

merge queue 确保多个 MR 按序排队评测，避免并发冲突。release candidate 是标记为可发布的版本，必须通过全量回归集（含难例库）。

代码：Merge Queue 状态检查脚本

python 
# merge_queue_check.py — Merge Queue 状态检查
import subprocess
import json

class MergeQueueCheck:
    """Merge Queue 状态检查"""

    @staticmethod
    def get_pending_checks() -> list:
        """获取待执行的状态检查"""
        result = subprocess.run(
            ["gh", "pr", "list", "--json", "headRefName,statusCheckRollup"],
            capture_output=True, text=True
        )
        if result.returncode != 0:
            return []
        prs = json.loads(result.stdout)
        pending = []
        for pr in prs:
            for check in pr.get("statusCheckRollup", []):
                if check.get("conclusion") == "PENDING" and \
                   check.get("name") in ["ragas-eval", "spring-eval-subset"]:
                    pending.append({
                        "pr": pr["headRefName"],
                        "check": check["name"],
                    })
        return pending

    @staticmethod
    def is_merge_queue_healthy() -> dict:
        """检查 merge queue 是否健康"""
        pending = MergeQueueCheck.get_pending_checks()
        return {
            "healthy": len(pending) == 0,
            "pending_checks": pending,
            "queue_depth": len(pending),
        }

本章结论

回归门禁是质量治理的自动化防线。baseline pin + delta 计算 + merge queue 确保每次变更不降低质量。

  • CI 卡点：ragas-eval + spring-eval-subset 作为 required check。
  • 版本对比：delta vs baseline，任何退化超 -0.02 触发回归。
  • 阻塞策略：block + notify quality owner，bypass 需审批留痕。
  • Merge Queue：排队评测，RC 触发全量回归。

# sla_monitor.py — SLA 监控
from datetime import datetime, timedelta
from typing import List, Dict

class SLAMonitor:
    """SLA 违约监控"""

    def __init__(self, p95_threshold_ms: float = 2500,
                 consecutive_days: int = 3):
        self.p95_threshold = p95_threshold_ms
        self.consecutive_days = consecutive_days

    def check_p95_breach(self, daily_p95: List[Dict]) -> Dict:
        """检查 P95 是否连续超标"""
        recent = daily_p95[-self.consecutive_days:]
        if len(recent) < self.consecutive_days:
            return {"breach": False, "reason": "Insufficient data"}

        breach_days = [d for d in recent if d["p95_ms"] > self.p95_threshold]
        if len(breach_days) >= self.consecutive_days:
            return {
                "breach": True,
                "severity": "CRITICAL",
                "details": breach_days,
                "action": "Create incident, page on-call",
            }
        return {"breach": False, "reason": "Within threshold"}

    def check_faithfulness_regression(self, weekly_scores: List[float],
                                      baseline: float = 0.90) -> Dict:
        """检查 faithfulness 连续下降"""
        if len(weekly_scores) < 3:
            return {"breach": False}
        recent = weekly_scores[-3:]
        if all(s < baseline for s in recent) and \
           recent[0] > recent[-1]:
            return {
                "breach": True,
                "severity": "WARNING",
                "action": "Notify quality owner, start root cause analysis",
            }
        return {"breach": False}

# incident_manager.py — Incident 管理
from datetime import datetime
from typing import Dict, Optional

class IncidentManager:
    """Incident 自动创建与跟踪"""

    def __init__(self, sla_config: Dict):
        self.sla_config = sla_config
        self.active_incidents = []

    def create_incident(self, metric: str, value: float,
                        threshold: float, details: str) -> Dict:
        incident = {
            "id": f"INC-{datetime.utcnow().strftime('%Y%m%d-%H%M')}",
            "metric": metric,
            "value": value,
            "threshold": threshold,
            "details": details,
            "status": "open",
            "created_at": datetime.utcnow().isoformat(),
            "resolved_at": None,
        }
        self.active_incidents.append(incident)
        return incident

    def resolve(self, incident_id: str):
        for inc in self.active_incidents:
            if inc["id"] == incident_id:
                inc["status"] = "resolved"
                inc["resolved_at"] = datetime.utcnow().isoformat()
                break

  • ToB 验收无评测集归档 — 无法复现验收时的质量。

本章结论

质量治理把技术指标转化为管理动作。owner + 周报 + SLA + 验收形成完整的组织保障。

  • Owner：每个场景唯一 quality owner，对指标和发布决策负责。
  • 周报：趋势比单点重要，含 faithfulness + badcase + A/B 状态。
  • SLA 违约：连续超标自动创建 incident，24h postmortem。
  • ToB 验收：Eval OKR + 人工报告 + Badcase 清单 + 评测集归档。

DeepEval 简介

deepeval（Confident AI）是面向 LLM 应用的 Python 评测框架：把「输入 / 实际输出 / 检索上下文」组织为 Test Case，再对每条用例调用一个或多个 auto metric 打分，最后像单元测试一样 assert 通过与否。

CorpAssist S1 · 制度引用合规（30 秒） 产品要求：凡给出天数/金额，答案必须带 [pol-YYYY-NN] 引用。RAGAS 的 faithfulness 能抓「是否贴上下文」，但抓不到引用格式。DeepEval 允许你叠加一条 deterministic 规则 metric，与 LLM-as-judge 并存。
维度RAGAS（本课主链）DeepEval
核心抽象 Dataset + eval pipeline + RAG 指标 TestCase + Metric + assert
CI 体验 pytest 聚合 + JSON 报告（Python 服务评测流水线） deepeval test run / pytest 插件，失败即 non-zero exit
内置 RAG 指标 faithfulness、context recall 等一等公民 Answer Relevancy、Faithfulness 等 + llm evaluator
Spring 侧 Python 出 JSON，Spring 对齐 schema 通常 Python 专用；Spring 仍走集成测试（Spring AI 服务评测）

封装为可复用 metric plugin

  • 单一职责：一个 metric 只答一个问题（引用 / 拒答 / 长度），便于坏例归因（坏例归因方法论）。

custom metric 与 RAGAS 指标冲突时听谁的？

核心 RAG 阈值（faithfulness、context recall）优先；extensions 是附加 AND 条件。两者都 pass 才允许 merge，避免「格式对但内容错」。

判断：custom metric 得 1.0 但 faithfulness 0.4，能否发布？

不能。引用格式只是合规层；内容幻觉仍由 faithfulness 拦截。门禁应 AND 核心 RAG 阈值，而非用规则分替代三角。

# dashboard_merger.py — 双栈报告合并
import json
from typing import Dict, List

class DashboardMerger:
    """双栈报告合并器"""

    METRICS = ["faithfulness", "answer_relevancy", "context_recall"]

    @staticmethod
    def merge(python_report: Dict, spring_report: Dict) -> Dict:
        """合并双栈报告"""
        dashboard = {
            "timestamp": python_report.get("timestamp"),
            "dataset_version": python_report.get("dataset_version"),
            "python": {},
            "spring": {},
            "parity": {},
            "recommendation": {},
        }

        for metric in DashboardMerger.METRICS:
            py_val = python_report.get("aggregate", {}).get(metric)
            sp_val = spring_report.get("aggregate", {}).get(metric)
            dashboard["python"][metric] = py_val
            dashboard["spring"][metric] = sp_val

            if py_val is not None and sp_val is not None:
                delta = round(py_val - sp_val, 4)
                dashboard["parity"][metric] = {
                    "delta": delta,
                    "ok": abs(delta) <= 0.02,
                }

        return dashboard

    @staticmethod
    def classify(value: float, thresholds: Dict[str, float]) -> str:
        """红/黄/绿分类"""
        if value >= thresholds.get("green", 0.90):
            return "green"
        elif value >= thresholds.get("yellow", 0.85):
            return "yellow"
        return "red"

    @staticmethod
    def generate_dashboard_html(dashboard: Dict) -> str:
        """生成看板 HTML 片段"""
        lines = [""]
        lines.append("")
        for metric in DashboardMerger.METRICS:
            py = dashboard["python"].get(metric, "N/A")
            sp = dashboard["spring"].get(metric, "N/A")
            parity = dashboard["parity"].get(metric, {})
            delta = parity.get("delta", "N/A")
            ok = parity.get("ok", False)
            emoji = "✅" if ok else "❌"
            lines.append(f"")
        lines.append("
指标PythonSpringParity
{metric}{py}{sp}{emoji} {delta}
")
        return "\n".join(lines)

# release_recommender.py — 发布推荐引擎
from typing import Dict, List

class ReleaseRecommender:
    """自动发布推荐"""

    THRESHOLDS = {
        "green": 0.90,
        "yellow": 0.85,
    }

    def __init__(self, dashboard: Dict):
        self.dashboard = dashboard

    def recommend(self) -> Dict:
        """生成发布推荐"""
        python = self.dashboard.get("python", {})
        spring = self.dashboard.get("spring", {})
        parity = self.dashboard.get("parity", {})

        all_metrics = list(python.keys())
        statuses = {}

        for metric in all_metrics:
            py_val = python.get(metric, 0)
            sp_val = spring.get(metric, 0)
            avg_val = (py_val + sp_val) / 2
            if avg_val >= self.THRESHOLDS["green"]:
                statuses[metric] = "green"
            elif avg_val >= self.THRESHOLDS["yellow"]:
                statuses[metric] = "yellow"
            else:
                statuses[metric] = "red"

        parity_ok = all(
            p.get("ok", False) for p in parity.values()
        )
        has_red = "red" in statuses.values()
        has_yellow = "yellow" in statuses.values()

        if not has_red and parity_ok and not has_yellow:
            return {"decision": "Go", "reason": "全部指标绿区，parity 正常"}
        elif not has_red and parity_ok:
            return {"decision": "Canary",
                    "reason": f"黄区指标: {[k for k,v in statuses.items() if v=='yellow']}"}
        else:
            red_metrics = [k for k, v in statuses.items() if v == "red"]
            return {"decision": "No-Go",
                    "reason": f"红区指标: {red_metrics}"}

代码：趋势图表数据生成

python 
# trend_chart.py — 趋势数据生成
import json
from typing import List, Dict
from datetime import datetime, timedelta

class TrendGenerator:
    """看板趋势数据"""

    @staticmethod
    def load_weekly_reports(report_dir: str, weeks: int = 8) -> List[Dict]:
        """加载近 N 周的评测报告"""
        reports = []
        for i in range(weeks):
            week_start = datetime.utcnow() - timedelta(weeks=i+1)
            fname = f"report_{week_start.strftime('%Y%m%d')}.json"
            try:
                with open(f"{report_dir}/{fname}", "r") as f:
                    reports.append(json.load(f))
            except FileNotFoundError:
                continue
        return reports

    @staticmethod
    def build_trend_data(reports: List[Dict]) -> Dict:
        """构建趋势数据"""
        trend = {"weeks": [], "faithfulness": [], "context_recall": [],
                 "answer_relevancy": [], "p95_ms": []}
        for r in reversed(reports):
            trend["weeks"].append(r.get("run_id", "")[-8:])
            agg = r.get("aggregate", {})
            trend["faithfulness"].append(agg.get("faithfulness"))
            trend["context_recall"].append(agg.get("context_recall"))
            trend["answer_relevancy"].append(agg.get("answer_relevancy"))
            latency = r.get("latency", {})
            trend["p95_ms"].append(latency.get("p95_ms"))
        return trend

FAQ：质量看板常见问题

看板应该每天更新还是每周更新？

建议每日自动更新（CI 触发），但周报评审时用每周聚合。每日更新能快速发现异常，每周评审能看清趋势。

Dashboard 展示在 Grafana 还是内部页面？

都可以。Grafana 适合实时指标（P95、thumbs_up），内部页面适合日/周聚合的离线指标。建议两者结合：Grafana 看实时，内部页面看趋势。

双栈指标不一致时，看板以谁的为准？

同时展示，但以 Python 侧为主（因为 Python 跑全量 RAGAS）。Spring 侧作为 parity 验证。两边的 delta 在 ±0.02 以内视为一致。

学完你能

本章先记住 4 件事
  • 先 baseline：没有基准线的「提升」不可信

  • A/B 证据：离线 + 在线双重验证

量化叙事结构

quant narrative 回答面试官的核心问题：「你怎么知道变好了？」叙事结构：问题 → baseline → 改动 → 离线回归 → A/B → 业务指标。eval storytelling 要有 ns1 proof。

面试官：你怎么证明 Prompt v7 比 v6 更好？ 候选人答：「我们用固定的评测集 corpassist-s1-v3（200 条）在 Python 和 Spring 双栈上跑了 RAGAS。faithfulness 从 0.90 到 0.92 (+0.02)，同时线上 A/B 10% 流量观察 3 天，thumbs_up 从 85% 提升到 87% (p=0.03, n=3200)。P95 延迟从 2050ms 变为 2100ms（+50ms，在 SLA 范围内）。」这就是一个完整的量化叙事。
量化叙事框架
flowchart LR
  BASELINE[建设 baseline] --> CHANGE[实施改动]
  CHANGE --> OFF[离线回归]
  OFF --> AB[A/B 在线验证]
  AB --> METRIC[业务指标]
  METRIC --> STORY[量化故事]

  BASELINE --> |fixed dataset| PIN[version pin]
  OFF --> |RAGAS| SCORES[faithfulness +0.02]
  AB --> |guardrails| THUMBS[thumbs_up +2%, p=0.03]
  METRIC --> |business| TRANSFER[转人工 -1%]

量化叙事需要三个层级的证据：离线指标 → 在线 A/B → 业务指标。缺一不可。

Before/After 图表设计

chart story 的核心原则：

  • 双轴图：左轴 faithfulness，右轴 P95 延迟

  • 置信区间：A/B 指标展示 error bar
  • 数据集版本：标注 dataset_version 变化
  • 样本量：标注每个数据点的样本量
不推荐
  • 只有一句「提升 20%」没有上下文
  • 用不同尺度的 Y 轴误导
  • 不标样本量，让人怀疑统计意义
  • 只展示最好的一周
推荐
  • 曲线 + 置信区间 + 样本量标注
  • 前后一致的 Y 轴尺度
  • 标注数据集版本和变更事件
  • 展示完整趋势而非 cherry-pick

A/B 证据：说服力的关键

ab evidence 是面试中最有说服力的材料。离线指标可以「调参调出来」，但 A/B 是用户真实反馈。

证据类型示例说服力

A/B + 离线一致两者都正向非常高
业务指标提升转人工率 -1%, 工单解决时长 -20%最高

metric honesty 是所有工程叙事的基础。不夸大、不 cherry-pick、不隐藏退化。面试官欣赏坦诚的候选人。

代码：Metric Honesty 检查器

python 
# metric_honesty.py — 指标诚实性检查
class MetricHonestyCheck:
    """防止指标被 cherry-pick"""

    @staticmethod
    def check_report(report: dict) -> list:
        """检查报告是否包含必要的诚实信息"""
        warnings = []
        if "sample_count" not in report:
            warnings.append("缺少样本量")
        if "dataset_version" not in report:
            warnings.append("缺少数据集版本")
        if "aggregate" in report:
            agg = report["aggregate"]
            if not any(v <= 0.95 for v in agg.values() if v is not None):
                warnings.append("所有指标 > 0.95，可能评测集过简单")
        if "per_sample" in report:
            scores = [s.get("faithfulness", 1) for s in report["per_sample"]]
            if max(scores) - min(scores) < 0.05:
                warnings.append("样本间差异过小，可能数据泄露")
        return warnings

# interview_narrative.py — 面试叙事模板
class InterviewNarrative:
    """生成量化叙事模板"""

    @staticmethod
    def build(problem: str, baseline: dict, change: str,
              offline_result: dict, ab_result: dict,
              limitation: str) -> str:
        """构建完整的量化叙事"""
        lines = [
            f"## 问题：{problem}",
            "",
            "### Baseline",
            f"- 评测集：{baseline.get('dataset', 'corpassist-s1-v3')}（{baseline.get('n', 200)} 条）",
            f"- Faithfulness: {baseline['faithfulness']}",
            f"- Context Recall: {baseline['context_recall']}",
            "",
            f"### 改动：{change}",
            "",
            "### 离线结果",
            f"- Faithfulness: {offline_result['faithfulness_old']} → {offline_result['faithfulness_new']} (Δ{offline_result['faithfulness_delta']})",
            f"- Context Recall: {offline_result['recall_old']} → {offline_result['recall_new']} (Δ{offline_result['recall_delta']})",
            "",
            "### A/B 结果",
            f"- 流量：{ab_result.get('traffic', '10%')}",
            f"- 时长：{ab_result.get('duration', '3 天')}",
            f"- Thumbs Up: {ab_result['thumbs_control']}% → {ab_result['thumbs_treatment']}% (p={ab_result['p_value']})",
            "",
            "### 局限",
            f"- {limitation}",
            "",
            "### 下一步",
            "- 持续监控 x 指标",
            "- 下一轮迭代计划",
        ]
        return "\n".join(lines)

RAG 指标定义

CorpAssist · S1 制度问答评审 法务上传《远程办公管理办法》后，产品问：「模型说审批 3 天，制度里写的是 5 个工作日——这算检索错还是生成错？」本章指标帮你定位链路环节，而不是笼统说「模型不行」。

RAG 评测通常从「检索是否找对」「生成是否忠实」「答案是否切题」三个维度展开。CorpAssist 采用 RAGAS 兼容命名，便于与后续 RAGAS 与自动评测 流水线对接。

核心 RAG 指标对照

指标衡量什么低分常见原因CorpAssist 阈值示意
context recall 检索片段是否覆盖 gold 所需信息 分块切坏、Embedding 模型不匹配、TopK 过小 ≥ 0.85（回归集）
faithfulness 回答是否仅依据检索上下文、无胡编 Prompt 未约束「仅依据上下文」、模型过度发挥 ≥ 0.90
answer relevance 回答是否解决用户问题 检索偏题、生成跑题、多轮上下文干扰 ≥ 0.88
引用覆盖率 制度问答是否带 chunk 引用 未强制 citation 格式、检索为空仍生成 S1 必达 100% 有引用或明确拒答

反例 vs 推荐：指标选型

不推荐
  • 只报 BLEU/ROUGE 与 gold 文本字面重合
  • 用「用户满意度」替代可回归指标
  • 检索与生成混在一个「准确率」里
  • 拒答样本从评测集删除
推荐
  • RAGAS 三维 + 引用率 + 拒答正确率
  • 难例库单独统计，不与简单题混平均
  • 检索失败与生成失败分开归因（见 坏例归因方法论）
  • 拒答作为一等公民写入 gold 标签

Agent 成功率与 SLA

除 RAG 外，CorpAssist S2 智能客服 与 S3 Agent 任务自动化 需要 agent success：任务是否在步数预算内完成、工具调用是否成功、最终状态是否符合业务规则。

  • 工具调用成功率：schema 合法且 HTTP 2xx；失败要区分「模型选错工具」与「下游 API 超时」。
  • 任务完成率：用户意图「改收货地址」是否在规则校验后真正写入订单系统。
  • 步数预算：超过 N 步强制转人工或拒答，避免 Agent 空转烧 Token。CorpAssist 设定 S3 Agent 步数上限为 8 步。

# agent_metrics.py — Agent 成功率计算
from dataclasses import dataclass, field
from typing import List, Optional

@dataclass
class AgentTrace:
    """Agent 单次执行追踪"""
    task_id: str
    steps: int
    max_steps: int = 8
    tool_calls: List[dict] = field(default_factory=list)
    final_status: str = ""  # success / failure / timeout
    error_type: Optional[str] = None  # tool_error / plan_error / timeout

def compute_agent_success(traces: List[AgentTrace]) -> dict:
    """计算 Agent 成功率指标组"""
    total = len(traces)
    if total == 0:
        return {}
    success = sum(1 for t in traces if t.final_status == "success")
    tool_errors = sum(1 for t in traces if t.error_type == "tool_error")
    timeout = sum(1 for t in traces if t.final_status == "timeout")
    avg_steps = sum(t.steps for t in traces) / total

    return {
        "agent_success_rate": success / total,
        "tool_error_rate": tool_errors / total,
        "timeout_rate": timeout / total,
        "avg_steps": avg_steps,
    }
Agent 场景还要跑 RAGAS 吗？

若 Agent 内部含 RAG 子链（查制度再答），应对该子链单独跑 context recall / faithfulness；整体任务用 agent success + 端到端 rubrics。不要用一个平均分掩盖子链路失败。

双栈同口径：Python 与 Spring 共用指标

Python 侧常负责索引与 RAGAS 批跑；Spring 侧常负责 BFF 与集成测试。若 Python 报 0.92、Spring 报 0.78，多半是索引版本、Prompt 版本或检索参数不一致，而非「Java 比较笨」。

  • 维护评测集 JSONL 与 dataset_version
  • 批跑 RAGAS，产出 json artifact 上传 CI
  • 索引 pipeline 版本写入报告 metadata
  • 提供 baseline 报告供 Spring 对照
  • 集成测试读取同一 JSONL 子集
  • Advisor 链路与生产一致，禁止 mock 检索结果做「假绿」
  • 报告字段名与 Python schema 逐 key 对齐
  • 产出 parity diff 报告

常见误区
  • Spring 测试用内存 H2 向量库、Python 用 Milvus — 分数不可比。
  • 只对比平均分，不看同一题的双栈 diff 列表。
  • 延迟只测 LLM，不含检索与 Rerank。

本章结论

指标是 NS1 飞轮的「刻度尺」。发版前必须同时看质量分与 P95，且 Python/Spring 共用 schema。

  • RAG 指标定义：context recall、faithfulness、answer relevance 分工明确；拒答与引用率单独统计。
  • Agent 成功率：S2/S3 看任务完成与工具成功，RAG 子链仍要拆出 RAGAS 指标。
  • SLA：P95/P99 与可用性写入 yaml 门禁，不与离线分混为一谈。
  • 双栈同口径：同一 dataset_version、同一索引与 Prompt pin，字段级 diff 排查 parity。

- Pass：每个事实可在 contexts 找到原文支持

- Pass：明确说明「制度库中未找到相关依据」

# hardcase_manager.py — 难例库管理
import json
from datetime import datetime
from typing import List, Dict

class HardCaseManager:
    """难例库管理器"""

    def __init__(self, path: str = "./data/hard_cases.jsonl"):
        self.path = path
        self.cases: List[Dict] = []

    def load(self):
        """加载难例库"""
        self.cases = []
        try:
            with open(self.path, "r", encoding="utf-8") as f:
                for line in f:
                    self.cases.append(json.loads(line))
        except FileNotFoundError:
            pass
        return self

    def add_case(self, question: str, contexts: List[str],
                 gold_answer: str, trace_id: str,
                 failure_type: str = "faithfulness"):
        """新增难例"""
        case = {
            "id": f"hc-{len(self.cases)+1:04d}",
            "question": question,
            "contexts": contexts,
            "gold_answer": gold_answer,
            "trace_id": trace_id,
            "failure_type": failure_type,
            "added_at": datetime.utcnow().isoformat(),
            "source": "prod_badcase",
        }
        self.cases.append(case)
        return case

    def save(self):
        """持久化"""
        with open(self.path, "w", encoding="utf-8") as f:
            for case in self.cases:
                f.write(json.dumps(case, ensure_ascii=False) + "\n")

    def run_gate(self, eval_results: List[Dict]) -> bool:
        """门禁：难例子集所有指标不得低于阈值"""
        failed = []
        for r in eval_results:
            if r["faithfulness"] < 0.85:
                failed.append(r)
        return len(failed) == 0, failed

    contexts: List[str]
    metadata: Optional[dict] = None

@dataclass
class EvalDataset:
    """版本化的评测数据集"""
    name: str
    version: str
    samples: List[EvalSample] = field(default_factory=list)

    def add(self, sample: EvalSample):
        self.samples.append(sample)

    def compute_hash(self) -> str:
        """计算数据集 hash 用于 version pin"""
        content = json.dumps(
            [s.__dict__ for s in self.samples],
            sort_keys=True, ensure_ascii=False
        )
        return hashlib.sha256(content.encode()).hexdigest()[:12]

    def save_jsonl(self, path: str):
        with open(path, "w", encoding="utf-8") as f:
            for s in self.samples:
                f.write(json.dumps(s.__dict__, ensure_ascii=False) + "\n")

    def build_manifest(self) -> dict:
        return {
            "name": self.name,
            "version": self.version,
            "sha256": self.compute_hash(),
            "total": len(self.samples),
        }

评测集是 NS1 飞轮的燃料。没有版本化的评测集，所有指标数字都不可信。

指标测量对象范围依赖 LLMCorpAssist 阈值

Python 脚本：RAGAS 评测管线实现

以下代码实现 CorpAssist S1 的自动评测管线，包含评测执行、报告生成与阈值检查。

python 
# ragas_eval.py — RAGAS 评测管线
from datasets import Dataset
from ragas import evaluate
from ragas.metrics import faithfulness, answer_relevancy, context_recall
from ragas.llms import LangchainLLMWrapper
from ragas.embeddings import LangchainEmbeddingsWrapper
from langchain_openai import ChatOpenAI, OpenAIEmbeddings
from typing import List, Dict, Optional
import json

class RAGEvaluator:
    """CorpAssist S1 RAGAS 评测器"""

    def __init__(self, llm_model: str = "gpt-4o-mini",
                 embedding_model: str = "text-embedding-3-small"):
        self.evaluator_llm = LangchainLLMWrapper(
            ChatOpenAI(model=llm_model, temperature=0)
        )
        self.evaluator_emb = LangchainEmbeddingsWrapper(
            OpenAIEmbeddings(model=embedding_model)
        )
        self.metrics = [faithfulness, answer_relevancy, context_recall]
        for metric in self.metrics:
            metric.llm = self.evaluator_llm
            if hasattr(metric, "embeddings"):
                metric.embeddings = self.evaluator_emb

    def evaluate(self, dataset: Dataset,
                 column_map: Optional[Dict] = None) -> Dict[str, float]:
        if column_map is None:
            column_map = {
                "question": "question",
                "answer": "answer",
                "contexts": "contexts",
                "ground_truth": "ground_truth",
            }
        result = evaluate(dataset=dataset, metrics=self.metrics,
                         column_map=column_map)
        return dict(result)

    def print_report(self, scores: Dict[str, float]):
        print("=" * 50)
        print("CorpAssist S1 RAG 评测报告")
        print("=" * 50)
        for metric, score in scores.items():
            status = "PASS" if score >= 0.85 else "FAIL"
            bar = "█" * int(score * 20) + "░" * (20 - int(score * 20))
            print(f"{metric:25s} {bar} {score:.4f} [{status}]")
        print(f"\n平均得分: {sum(scores.values())/len(scores):.4f}")

    def check_gate(self, scores: Dict[str, float],
                   thresholds: Dict[str, float]) -> bool:
        """阈值门禁检查"""
        all_pass = True
        for metric, threshold in thresholds.items():
            if metric in scores and scores[metric] < threshold:
                print(f"BLOCK: {metric} = {scores[metric]:.4f} < {threshold}")
                all_pass = False
        return all_pass

CI 集成脚本

python 
# ci_eval.py — CI 集成评测入口
from ragas_eval import RAGEvaluator
from datasets import Dataset
import json, sys

THRESHOLDS = {
    "faithfulness": 0.90,
    "answer_relevancy": 0.88,
    "context_recall": 0.85,
}

def run_ci_eval(dataset_path: str = "./data/eval_dataset.jsonl",
                rag_endpoint: str = "http://localhost:8000/ask") -> bool:
    """CI 评测入口，返回 True 表示通过"""
    import requests

    # 加载评测集
    eval_samples = []
    with open(dataset_path, "r", encoding="utf-8") as f:
        for line in f:
            eval_samples.append(json.loads(line))

    # 用 RAG 管线生成回答
    answers, contexts_list = [], []
    for sample in eval_samples:
        try:
            resp = requests.post(rag_endpoint,
                json={"question": sample["question"]}, timeout=30)
            data = resp.json()
            answers.append(data.get("answer", ""))
            contexts_list.append(data.get("contexts", []))
        except Exception as e:
            print(f"Error: {e}")
            answers.append("")
            contexts_list.append([])

    # 构建 Dataset
    eval_data = {
        "question": [s["question"] for s in eval_samples],
        "answer": answers,
        "contexts": contexts_list,
    }
    if "ground_truth" in eval_samples[0]:
        eval_data["ground_truth"] = [s.get("ground_truth", "") for s in eval_samples]

    dataset = Dataset.from_dict(eval_data)

    # 运行评测
    evaluator = RAGEvaluator()
    scores = evaluator.evaluate(dataset)
    evaluator.print_report(scores)

    # 阈值门禁
    all_pass = evaluator.check_gate(scores, THRESHOLDS)

    # 保存报告
    report = {"scores": scores, "passed": all_pass,
              "thresholds": THRESHOLDS,
              "dataset_version": "corpassist-s1-v3"}
    with open("./data/eval_report.json", "w") as f:
        json.dump(report, f, indent=2, ensure_ascii=False)

    return all_pass

if __name__ == "__main__":
    passed = run_ci_eval()
    sys.exit(0 if passed else 1)

结果解读与人工校验

RAGAS 自动分与人工 rubrics 之间常有偏差。常见的偏差模式：

  • Faithfulness 虚高：LLM judge 对轻微胡编不敏感，尤其是编造的数字与原文相近时
  • Relevance 偏低：LLM judge 可能过于严格，将合理 paraphrase 判为不相关
  • Context Recall 不稳定：依赖 ground_truth 的完整性，如果 gold 写得不够详细会偏低

建议：定期抽取 10% 样本做人工校验，计算自动分与人工分的 Spearman 相关性。若相关系数 < 0.7，需调整 judge prompt 或更换模型。

RAGAS 评测每次都要调 LLM，成本如何控制？

200 条样本 × 3 个指标 × 每条约 300 token ≈ 180K token。使用 gpt-4o-mini 每次约 $0.05-0.10。如果做 ablation 实验（20 种参数组合），成本会达到 $1-2。策略：1) CI 用完整集；2) 本地开发用 10 条快速验证；3) 缓存已评测的样本结果。

RAGAS 指标一定要 ground_truth 吗？

faithfulness 和 answer_relevance 不需要；context_recall 需要。如果评测集没有 gold，可以只跑 faithful + relevance。但对于 S1 制度问答，建议 always 包含 ground_truth 以获得完整的检索质量评估。

  • Python 脚本：RAGAS evaluate() 封装为 CI 可直接调用的入口。
  • CI 集成：GitHub Actions + threshold gate，低于阈值 exit 1 block merge。
  • 结果解读：区分 faithfulness 虚高与 relevance 偏严，定期人工校验自动分。

    """人工评测抽样器"""

    def __init__(self, eval_report_path: str):
        self.df = pd.read_json(eval_report_path, lines=True)

    def border_samples(self, metric: str = "faithfulness",
                       low: float = 0.7, high: float = 0.85) -> pd.DataFrame:
        """筛选边界样本——RAGAS 评分不确定的区域"""
        return self.df[(self.df[metric] >= low) & (self.df[metric] < high)]

    def stratified_sample(self, n: int = 100) -> pd.DataFrame:
        """分层抽样：按类别比例"""
        if "category" not in self.df.columns:
            return self.df.sample(n=n, random_state=42)
        return self.df.groupby("category", group_keys=False)\
            .apply(lambda g: g.sample(
                min(len(g), max(1, int(n * len(g) / len(self.df)))),
                random_state=42))

    def hard_case_full(self, hard_case_ids: List[str]) -> pd.DataFrame:
        """难例全量"""
        return self.df[self.df["id"].isin(hard_case_ids)]

    def estimate_cost(self, n_samples: int,
                      time_per_sample_min: float = 1.5,
                      hourly_rate: float = 30) -> dict:
        """估算人工评测成本"""
        hours = n_samples * time_per_sample_min / 60
        return {
            "n_samples": n_samples,
            "estimated_hours": round(hours, 1),
            "estimated_cost": round(hours * hourly_rate, 1),
            "cost_per_sample": round(time_per_sample_min / 60 * hourly_rate, 2),
        }

# inter_annotator.py — 一致性计算
from sklearn.metrics import cohen_kappa_score
from typing import List, Tuple

class AnnotatorAgreement:
    """标注一致性分析"""

    @staticmethod
    def cohen_kappa(annotator_a: List[int],
                    annotator_b: List[int]) -> float:
        """计算 Cohen's Kappa"""
        return cohen_kappa_score(annotator_a, annotator_b)

    @staticmethod
    def agreement_report(annotations: List[Tuple[str, List[int], List[int]]],
                         threshold: float = 0.6) -> dict:
        """生成一致性报告"""
        report = {}
        for dimension, a_scores, b_scores in annotations:
            kappa = AnnotatorAgreement.cohen_kappa(a_scores, b_scores)
            report[dimension] = {
                "kappa": round(kappa, 3),
                "pass": kappa >= threshold,
                "action": "ok" if kappa >= threshold else "retrain + re-annotate",
            }
        return report

    @staticmethod
    def disagreement_samples(a_scores: List[int], b_scores: List[int],
                             indices: List[str]) -> List[str]:
        """找出分歧样本"""
        return [idx for idx, (a, b) in zip(indices, zip(a_scores, b_scores))
                if a != b]

# 使用示例
annotations = [
    ("faithfulness", [1, 0, 1, 1, 0], [1, 0, 0, 1, 0]),
    ("refusal", [1, 1, 1, 0, 1], [1, 1, 1, 0, 0]),
]
report = AnnotatorAgreement.agreement_report(annotations)
print(report)
# {'faithfulness': {'kappa': 0.615, 'pass': True, 'action': 'ok'}, ...}

成本：人工评测的经济模型

eval cost ≈ 条数 × 单条分钟 × 人力单价。用 RAGAS 筛出边界样本再人工，可省 60%+ 成本。对于 CorpAssist S1，200 条人工评测的月度预算建议控制在 $200 以内。

成本优化策略总结

  • 自动筛选边界：RAGAS 评分在 0.7-0.85 之间的样本优先人工
  • 分层降采样：稳定 intent 降低抽检率，新功能/高风险加倍
  • 难例全量：已知 badcase 全部人工，但难例库通常只有 20-50 条
  • 迭代复用：评分稳定的样本逐步移出人工集，只在版本变更时重新抽检

def cost_estimation(total_samples=500,
                    auto_metric_cost=0.08,
                    human_cost_per_sample=0.75,
                    border_ratio=0.2):
    """对比自动 vs 人工评测成本"""
    auto_total = total_samples * auto_metric_cost
    # 策略1: 全人工
    full_human = total_samples * human_cost_per_sample
    # 策略2: 自动 + 人工边界
    border_samples = int(total_samples * border_ratio)
    hybrid = auto_total + border_samples * human_cost_per_sample

    return {
        "full_auto": auto_total,
        "full_human": full_human,
        "hybrid": hybrid,
        "savings": round((full_human - hybrid) / full_human * 100, 1),
    }

print(cost_estimation())

人工是否还要评每条 RAGAS？

不需要。RAGAS 全量跑自动化；人工仅抽检 + 边界集 + 发版前 sign-off 子集 + 难例全量。典型比例：自动 100% + 人工 10-20%。

仲裁流程与质量改进

adjudication 是人工评测的最后一道关。当两位标注员不一致时，由第三方（通常是法务或产品负责人）做最终裁定。

  • 分歧记录：每次仲裁保留分歧原因，作为 quality audit 的依据
  • 培训回流：高频分歧点更新 annotation rubric，标注员再培训
  • 盲审：blind review 确保标注员不知道对方评分
  • 度量：使用 likert scale（1-5 分）而非仅二元通过/不通过
常见误区
  • 单人标注即 gold 标准—无一致性保障
  • Rubric 太模糊导致标注员频繁分歧
  • 忽略拒答类的评测—拒答错误比答错更影响信任
  • 人工评测结果不与 RAGAS 自动分做相关性分析

本章结论

人工评测是 RAGAS 自动分的必要补充，但必须标准化、抽样化、审计化。

  • Rubrics：可操作、二元门、每条 ≤1 分钟可判。
  • 抽样：按风险分层，边界样本加倍抽检，难例全量覆盖。
  • 一致性：Cohen κ ≥ 0.6 才能合并标注；否则暂停培训。
  • 成本：自动 100% + 人工 10-20% 是性价比最优组合。
  • 仲裁：分歧由第三方裁决并记录，持续改进 rubric。

# report_generator.py — 结构化报告生成
import json
import subprocess
from datetime import datetime
from typing import Dict, List

class EvalReport:
    """评测报告生成器"""

    @staticmethod
    def get_git_sha() -> str:
        return subprocess.check_output(
            ["git", "rev-parse", "--short", "HEAD"]
        ).decode().strip()

    @classmethod
    def build(cls, scores: Dict[str, float],
              thresholds: Dict[str, float],
              dataset_version: str,
              passed: bool,
              per_sample: List[Dict] = None) -> Dict:
        return {
            "run_id": f"eval-{datetime.utcnow().strftime('%Y%m%d-%H%M%S')}",
            "git_sha": cls.get_git_sha(),
            "dataset_version": dataset_version,
            "timestamp": datetime.utcnow().isoformat(),
            "aggregate": scores,
            "thresholds": thresholds,
            "passed": passed,
            "sample_count": len(per_sample) if per_sample else 0,
            "per_sample": per_sample or [],
        }

    @staticmethod
    def save(report: Dict, path: str = "./data/eval_report.json"):
        with open(path, "w", encoding="utf-8") as f:
            json.dump(report, f, indent=2, ensure_ascii=False)
        print(f"Report saved: {path}")

    @staticmethod
    def compare(current: Dict, baseline: Dict) -> Dict:
        """对比当前报告与 baseline"""
        deltas = {}
        for metric in current["aggregate"]:
            if metric in baseline["aggregate"]:
                deltas[metric] = round(
                    current["aggregate"][metric] - baseline["aggregate"][metric], 4
                )
        return {
            "deltas": deltas,
            "regression": any(v < 0 for v in deltas.values()),
        }

Regression Set 与版本管理

regression set 是 gold 评测集中最稳定的子集。每次发版前跑回归集，确保核心能力不退化。

  • 稳定子集：从完整评测集选 50-100 条最稳定的 QA 对
  • 难例全量：hard_cases.jsonl 全量跑，不允许任何退化
  • 版本锁定：regression set 版本号与 dataset_version 独立，更少更新
  • 基线存储：每次通过的评测报告作为下一次 baseline
regression set 与完整评测集什么关系？

regression set ⊆ 完整集。regression set 用于快速门禁（5 分钟），完整集用于详细报告（15 分钟）。MR 时跑 regression，发版前跑完整集。

// EvalReportService.java — 评测报告生成
@Service
public class EvalReportService {

    public EvalReport generate(List<EvalResult> results, EvalThresholds thresholds) {
        var aggregate = new HashMap<String, Double>();
        aggregate.put("faithfulness", avg(results, EvalResult::getFaithfulness));
        aggregate.put("answer_relevancy", avg(results, EvalResult::getAnswerRelevancy));
        aggregate.put("context_recall", avg(results, EvalResult::getContextRecall));

        boolean passed = aggregate.entrySet().stream()
            .allMatch(e -> e.getValue() >= thresholds.getMin(e.getKey()));

        return EvalReport.builder()
            .runId("spring-eval-" + LocalDateTime.now())
            .gitSha(getGitSha())
            .datasetVersion(thresholds.getDatasetVersion())
            .aggregate(aggregate)
            .passed(passed)
            .sampleCount(results.size())
            .build();
    }

    private double avg(List<EvalResult> results,
                       ToDoubleFunction<EvalResult> extractor) {
        return results.stream().mapToDouble(extractor).average().orElse(0);
    }
}

# traffic_split.py — 分流实现
import hashlib

class TrafficSplitter:
    """一致性 hash 分流器"""

    def __init__(self, experiment_id: str, variants: dict):
        self.experiment_id = experiment_id
        self.variants = variants  # {"control": 0.9, "treatment": 0.1}

    def get_variant(self, user_id: str) -> str:
        """根据 user_id 返回实验变体"""
        hash_input = f"{self.experiment_id}:{user_id}"
        hash_hex = hashlib.md5(hash_input.encode()).hexdigest()
        hash_int = int(hash_hex[:8], 16)
        total_weight = sum(self.variants.values())
        normalized = (hash_int % 10000) / 10000 * total_weight

        cumulative = 0
        for variant, weight in self.variants.items():
            cumulative += weight
            if normalized < cumulative:
                return variant
        return list(self.variants.keys())[-1]

    def get_experiment_context(self, user_id: str) -> dict:
        """获取实验上下文用于 trace"""
        return {
            "experiment_id": self.experiment_id,
            "variant": self.get_variant(user_id),
        }

experiment id 贯穿 trace：从网关到 BFF 再到 LLM 调用，每个 span 都携带 experiment_id 和 variant。方便事后按实验维度聚合指标。

# ab_metrics.py — A/B 指标聚合
import numpy as np
from scipy import stats
from typing import List, Dict

class ABMetrics:
    """A/B 实验指标聚合"""

    @staticmethod
    def proportion_test(control_success: int, control_total: int,
                        treatment_success: int, treatment_total: int) -> dict:
        """比例检验（thumbs-up, 转人工率等）"""
        p_control = control_success / control_total
        p_treatment = treatment_success / treatment_total
        z_stat, p_value = stats.proportions_ztest(
            [control_success, treatment_success],
            [control_total, treatment_total],
        )
        return {
            "control_rate": round(p_control, 4),
            "treatment_rate": round(p_treatment, 4),
            "delta": round(p_treatment - p_control, 4),
            "z_stat": round(z_stat, 4),
            "p_value": round(p_value, 4),
            "significant": p_value < 0.05,
        }

    @staticmethod
    def latency_test(control_ms: List[float],
                     treatment_ms: List[float]) -> dict:
        """延迟分布检验（Mann-Whitney U）"""
        u_stat, p_value = stats.mannwhitneyu(control_ms, treatment_ms)
        return {
            "control_p50": round(np.median(control_ms), 1),
            "treatment_p50": round(np.median(treatment_ms), 1),
            "control_p95": round(np.percentile(control_ms, 95), 1),
            "treatment_p95": round(np.percentile(treatment_ms, 95), 1),
            "p_value": round(p_value, 4),
            "significant": p_value < 0.05,
        }

    @staticmethod
    def sample_size_needed(baseline_rate: float,
                           effect_size: float = 0.02,
                           alpha: float = 0.05,
                           power: float = 0.8) -> int:
        """计算所需样本量"""
        from statsmodels.stats.proportion import sample_size_proportions_diff
        return int(sample_size_proportions_diff(
            effect_size, baseline_rate, alpha=alpha, power=power
        ))

statistical power：在 80% power 和 95% confidence 下，检测 2% 的 thumbs-up 变化至少需要约 3000 样本。低于此样本量做出的结论不可靠。

先想 10 秒：treatment thumbs_up 提升 2% 但 p=0.08，你该全量吗？

不应急于全量。p=0.08 意味着 8% 概率是随机波动。应继续积累样本直到 p < 0.05 或跑满 7 天。但可以检查 guardrail 指标是否有退化，若无退化可以谨慎扩量到 25% 继续观察。

# auto_rollback.py — 自动回滚判断
class AutoRollback:
    """自动回滚决策"""

    def __init__(self, config: dict):
        self.guardrails = config.get("guardrails", {})

    def should_rollback(self, metrics: dict) -> dict:
        """检查所有 guardrail 指标"""
        reasons = []
        for metric_name, guardrail in self.guardrails.items():
            if metric_name not in metrics:
                continue
            current = metrics[metric_name]
            reason = self._check(metric_name, current, guardrail)
            if reason:
                reasons.append(reason)

        return {
            "should_rollback": len(reasons) > 0,
            "reasons": reasons,
        }

    def _check(self, name: str, current: dict, guardrail: dict) -> str:
        """单指标检查"""
        control_val = current.get("control", 0)
        treatment_val = current.get("treatment", 0)

        if "max_delta_vs_control" in guardrail:
            delta = treatment_val - control_val
            max_delta = guardrail["max_delta_vs_control"]
            if abs(delta) > max_delta and delta < 0:
                return f"{name}: treatment({treatment_val}) < control({control_val}), delta {delta:.3f} > {max_delta}"
        if "max_ratio_vs_control" in guardrail:
            ratio = treatment_val / control_val if control_val else 999
            if ratio > guardrail["max_ratio_vs_control"]:
                return f"{name}: ratio {ratio:.2f} > {guardrail['max_ratio_vs_control']}"
        return ""

    def execute_rollback(self, experiment_id: str):
        """执行回滚：切 100% 流量回 control"""
        print(f"ROLLBACK: {experiment_id} — 100% traffic to control")

Online Eval 与离线评测的配合

online eval 与离线 RAGAS 评测是互补关系：

  • 离线：每次 MR 跑 RAGAS，确保核心指标不退化。优点是可复现、低成本、快速反馈。
  • 在线：A/B 验证用户真实反馈。优点是反映真实体验，缺点是样本量大、周期长。
  • 配合：离线通过 → A/B 10% → 放量 50% → 全量。离线不通过则不发版。
常见误区
  • 离线 0.92 就觉得线上一定更好——离线集无法覆盖线上分布偏移。
  • A/B 只跑 1 小时就下结论——样本量不足，统计不显著。
  • 无 guardrail 只看正向指标——忽视延迟和拒答率的负面影响。
  • 回滚手动操作——凌晨 2 点发现问题时响应不及时。

本章结论

A/B 测试是离线评测的必要补充。10% canary → guardrail → 扩量 → 全量，配合自动回滚保障线上质量。

  • 分流：user_id 一致性 hash，sticky variant。
  • 指标看板：thumbs-up + P95 + 拒答率 + 转人工率多维度监控。
  • 回滚：guardrail 触发即自动切回 control，无需新发版。
  • 统计：计算所需样本量，p < 0.05 才下结论。

坏例分类体系

error taxonomy 把 badcase 分为三类：检索失败、生成失败、Prompt/指令失败。每类再细分，形成可操作的归因树。

CorpAssist S1 · 坏例归因 线上监控发现「异地办公审批天数」的 faithfulness 从 0.95 降到 0.72。trace 显示：检索到的 5 个 chunk 中只有 1 个包含天数，其余 4 个都是无关的考勤规则。归因到「检索失败 — 分块粒度过粗」。

flowchart TB
  B[Badcase] --> RET[检索失败]
  B --> GEN[生成失败]
  B --> PROMPT[Prompt 失效]

  RET --> RET1[TopK 过小]
  RET --> RET2[分块切坏]
  RET --> RET3[Embedding 不匹配]
  RET --> RET4[Query 改写错误]

  GEN --> GEN1[Faithfulness 胡编]

  GEN --> GEN3[引用缺失/错误]
  GEN --> GEN4[拒答不该拒]

# root_cause.py — 坏例归因引擎
from dataclasses import dataclass
from typing import List, Optional

@dataclass
class EvalTrace:
    """单条评测的完整 trace"""
    question_id: str
    question: str
    retrieved_chunks: List[str]
    chunk_scores: List[float]
    answer: str
    gold_answer: str
    faithfulness: float
    context_recall: float

class RootCauseAnalyzer:
    """坏例归因分析"""

    @staticmethod
    def analyze(trace: EvalTrace) -> dict:
        """逐层归因"""
        if trace.context_recall < 0.7:
            return RootCauseAnalyzer._retrieval_analysis(trace)
        elif trace.faithfulness < 0.8 and trace.context_recall >= 0.7:
            return {
                "category": "generation",
                "code": "GEN1",
                "description": "检索到相关信息但生成偏离",
                "action": "检查 LLM temperature 和 system prompt 约束",
            }
        elif trace.faithfulness >= 0.8 and trace.faithfulness < 0.9:
            return {
                "category": "prompt",
                "code": "PR1",
                "description": "边界准确率，可能是 Prompt 约束模糊",
                "action": "Review system prompt 对忠实度的强调",
            }
        else:
            return {
                "category": "unknown",
                "code": "UNK",
                "description": "需人工 review",
                "action": "人工标注归因",
            }

    @staticmethod
    def _retrieval_analysis(trace: EvalTrace) -> dict:
        if not trace.chunk_scores:
            return {"category": "retrieval", "code": "RET4",
                    "description": "检索结果为空", "action": "检查向量库连接"}
        max_score = max(trace.chunk_scores)
        if max_score < 0.5:
            return {"category": "retrieval", "code": "RET3",
                    "description": f"最高分仅 {max_score:.2f}，embedding 不匹配",
                    "action": "尝试不同 embedding 模型"}
        if len(trace.chunk_scores) < 3:
            return {"category": "retrieval", "code": "RET1",
                    "description": "TopK 过小，仅返回 1-2 个 chunk",
                    "action": "增大 top_k"}
        return {"category": "retrieval", "code": "RET2",
                "description": "chunk 粒度问题",
                "action": "调整 chunk_size 和 overlap"}

    @classmethod
    def batch_analyze(cls, traces: List[EvalTrace]) -> dict:
        """批量归因统计"""
        results = {"retrieval": 0, "generation": 0, "prompt": 0, "unknown": 0}
        details = []
        for trace in traces:
            result = cls.analyze(trace)
            results[result["category"]] = results.get(result["category"], 0) + 1
            details.append({"question_id": trace.question_id, **result})
        results["total"] = sum(results.values())
        results["details"] = details
        return results

  CHECK_RECALL -->|否| RET_FAIL[检索失败]
  CHECK_RECALL -->|是| CHECK_FAITH{faithfulness ≥ 0.8?}

# trace_store.py — Trace 存储
import json
from typing import List, Dict, Optional

class TraceStore:
    """RAG Trace 存储与查询"""

    def __init__(self, path: str = "./data/traces.jsonl"):
        self.path = path

    def save_trace(self, trace: dict):
        with open(self.path, "a", encoding="utf-8") as f:
            f.write(json.dumps(trace, ensure_ascii=False) + "\n")

    def query_by_metric(self, metric: str = "faithfulness",
                        max_value: float = 0.8, limit: int = 20) -> List[Dict]:
        """按指标筛选低分 trace"""
        results = []
        with open(self.path, "r", encoding="utf-8") as f:
            for line in f:
                record = json.loads(line)
                if record.get(metric, 1.0) <= max_value:
                    results.append(record)
                    if len(results) >= limit:
                        break
        return results

    def get_retrieval_detail(self, trace_id: str) -> Optional[Dict]:
        """获取指定 trace 的检索详情"""
        with open(self.path, "r", encoding="utf-8") as f:
            for line in f:
                record = json.loads(line)
                if record.get("trace_id") == trace_id:
                    return record.get("retrieval", {})
        return None

# case_library.py — 案例库
from dataclasses import dataclass, field
from typing import List
import json
from datetime import datetime

@dataclass
class FailureCase:
    """归因案例"""
    id: str
    question: str
    trace_id: str
    failure_category: str  # retrieval / generation / prompt
    failure_code: str      # RET1 / GEN1 / PR1
    symptom: str
    root_cause: str
    fix_action: str
    fix_verified: bool = False
    tags: List[str] = field(default_factory=list)
    created_at: str = ""

    def to_dict(self):
        return {
            "id": self.id, "question": self.question,
            "trace_id": self.trace_id,
            "failure_category": self.failure_category,
            "failure_code": self.failure_code,
            "symptom": self.symptom,
            "root_cause": self.root_cause,
            "fix_action": self.fix_action,
            "fix_verified": self.fix_verified,
            "tags": self.tags,
            "created_at": self.created_at or datetime.utcnow().isoformat(),
        }

class CaseLibrary:
    """案例库"""
    def __init__(self, path: str = "./data/case_library.jsonl"):
        self.path = path
        self.cases: List[FailureCase] = []

    def load(self):
        with open(self.path, "r", encoding="utf-8") as f:
            for line in f:
                data = json.loads(line)
                self.cases.append(FailureCase(**data))
        return self

    def search(self, question: str) -> List[FailureCase]:
        return [c for c in self.cases if question in c.question or
                any(tag in question for tag in c.tags)]

    def add(self, case: FailureCase):
        self.cases.append(case)
        with open(self.path, "a", encoding="utf-8") as f:
            f.write(json.dumps(case.to_dict(), ensure_ascii=False) + "\n")

    def stats(self) -> dict:
        counts = {}
        for case in self.cases:
            cat = case.failure_category
            counts[cat] = counts.get(cat, 0) + 1
        return counts

Fix Loop：闭环改进

fix loop 是归因后的闭环：归因 → 修复 → 验证 → 入库。每个 badcase 必须走完这个循环才能关闭。

归因闭环
flowchart LR
  BC[Badcase 发现] --> RCA[归因分析]
  RCA --> FIX[制定修复方案]
  FIX --> DEPLOY[修改并验证]
  DEPLOY --> VERIFY{验证通过?}
  VERIFY -->|是| CL[入库案例库]
  VERIFY -->|否| RCA
  CL --> EVAL[加入难例库]

每个 badcase 走完归因 → 修复 → 验证 → 入库闭环。验证通过后加入评测集防止回归。

先想 10 秒：80% 的 badcase 归因到 retrieval，你该优化什么？

优先优化检索链路：chunk 策略、embedding 模型、top_k 参数。如果检索解决了大部分问题，生成端的投入应推迟。数据驱动的归因能帮团队做正确的优先级排序。

本章结论

系统化的归因方法论让 badcase 排查从「靠猜」变成「靠 trace 数据」。分类树 + trace 分析 + 案例库形成知识沉淀。

  • 检索 vs 生成 vs Prompt：分类树叶子节点精确归因。
  • Trace 分析：OpenTelemetry / LangSmith 提供全链路数据。
  • 案例库：归因结果沉淀为可检索的 knowledge base。
  • Fix Loop：归因 → 修复 → 验证 → 入库，形成闭环。

from dataclasses import dataclass, field
from typing import List, Optional
import json
from datetime import datetime

@dataclass
class FeedbackEvent:
    """用户反馈事件"""
    event: str  # thumbs_up / thumbs_down / comment
    trace_id: str
    question: str
    labels: List[str] = field(default_factory=list)
    comment: Optional[str] = None
    user_segment: Optional[str] = None

class FeedbackCollector:
    """反馈收集器"""

    def __init__(self, path: str = "./data/feedback.jsonl"):
        self.path = path

    def record(self, event: FeedbackEvent):
        """记录反馈"""
        record = {
            "event": event.event,
            "trace_id": event.trace_id,
            "question": event.question,
            "labels": event.labels,
            "comment": event.comment,
            "user_segment": event.user_segment,
            "timestamp": event.timestamp or datetime.utcnow().isoformat(),
        }
        with open(self.path, "a", encoding="utf-8") as f:
            f.write(json.dumps(record, ensure_ascii=False) + "\n")

    def get_thumbs_down(self, since: Optional[str] = None) -> list:
        """获取 thumbs-down 列表"""
        results = []
        with open(self.path, "r", encoding="utf-8") as f:
            for line in f:
                record = json.loads(line)
                if record["event"] == "thumbs_down":
                    if since and record.get("timestamp", "") < since:
                        continue
                    results.append(record)
        return results

    def label_distribution(self) -> dict:
        """标签分布统计"""
        from collections import Counter
        labels = Counter()
        with open(self.path, "r", encoding="utf-8") as f:
            for line in f:
                record = json.loads(line)
                for label in record.get("labels", []):
                    labels[label] += 1
        return dict(labels)

# reflow_pipeline.py — 坏例回流流水线
from feedback_collector import FeedbackEvent
from case_library import FailureCase, CaseLibrary
from build_eval_dataset import EvalSample, EvalDataset
import json
from datetime import datetime, timedelta

class ReflowPipeline:
    """坏例回流管理"""

    def __init__(self, feedback_path: str, case_lib_path: str,
                 eval_path: str, hard_case_path: str):
        self.feedback_path = feedback_path
        self.case_lib = CaseLibrary(case_lib_path)
        self.eval_path = eval_path
        self.hard_case_path = hard_case_path

    def daily_triage(self) -> int:
        """每日 triage：将新反馈转为评测样本"""
        since = (datetime.utcnow() - timedelta(days=1)).isoformat()
        new_cases = 0

        with open(self.feedback_path, "r", encoding="utf-8") as f:
            for line in f:
                fb = json.loads(line)
                if fb["event"] != "thumbs_down":
                    continue
                if fb.get("timestamp", "") < since:
                    continue

                # 创建归因案例
                case = FailureCase(
                    id=f"fb-{fb['trace_id']}",
                    question=fb["question"],
                    trace_id=fb["trace_id"],
                    failure_category="unknown",
                    failure_code="TRIAGE",
                    symptom=fb.get("comment", ""),
                    root_cause="",
                    fix_action="",
                    tags=fb.get("labels", []),
                )
                self.case_lib.add(case)
                new_cases += 1

        return new_cases

    def promote_to_hardcase(self, case_id: str, gold_answer: str,
                            contexts: list):
        """将确认的 badcase 升级为 hard case"""
        case = next((c for c in self.case_lib.cases if c.id == case_id), None)
        if not case:
            return

        sample = EvalSample(
            question=case.question,
            gold_answer=gold_answer,
            contexts=contexts,
            metadata={"source": "prod_badcase", "trace_id": case.trace_id},
        )
        # 追加到 hard_cases.jsonl
        with open(self.hard_case_path, "a", encoding="utf-8") as f:
            f.write(json.dumps(sample.__dict__, ensure_ascii=False) + "\n")

优先级排序：P0/P1/P2

priority queue：影响用户数 × 严重度 × 合规风险。不是所有 badcase 都需要立即修复，优先级决定迭代节奏。

# priority_queue.py — 优先级排序
from dataclasses import dataclass
from typing import List

@dataclass
class PrioritizedCase:
    """带优先级的 badcase"""
    case_id: str
    question: str
    severity: int  # 1-5
    user_count: int
    compliance_risk: bool
    labels: List[str]

    def priority_score(self) -> int:
        """计算优先级分数"""
        score = self.severity * self.user_count
        if self.compliance_risk:
            score *= 3
        return score

    def priority_level(self) -> str:
        if self.compliance_risk and self.severity >= 4:
            return "P0"
        elif self.priority_score() >= 50:
            return "P1"
        else:
            return "P2"

class PriorityQueue:
    """优先级队列"""

    def __init__(self):
        self.cases: List[PrioritizedCase] = []

    def add(self, case: PrioritizedCase):
        self.cases.append(case)

    def sorted(self) -> List[PrioritizedCase]:
        return sorted(self.cases,
                      key=lambda c: c.priority_score(), reverse=True)

    def next_to_fix(self) -> PrioritizedCase:
        return self.sorted()[0] if self.cases else None

  MON[周一: 归因评审] --> TUE[周二: Python 索引 Canary]

每周 release train：周一评审 badcase 优先级，周二 Python 轨先发，周三 Spring 轨跟随，周四全量，周五周报回顾。

不推荐
  • 反馈只看总数，不看分布
  • badcase 修复后不入库
  • 无优先级排序，随机修 bug
  • Python 和 Spring 各自发各自的版
推荐
  • 反馈按标签、用户群、trace 深度分析
  • 修复后的 badcase 加入难例库 + 案例库
  • P0/P1/P2 明确定义和 SLA
  • 双栈同 release train，版本对齐

本章结论

用户反馈是 NS1 飞轮的起点。从埋点到回流到修复到验证，形成每周迭代的 closed loop。

  • 埋点与标注：feedback event 带 trace_id、labels、user_segment。
  • 坏例回流：T+1 triage，T+3 入库难例库或评测集。
  • 优先级排序：P0 合规风险 / P1 高频错误 / P2 体验问题。
  • 双栈发布节奏：weekly release train，Python 先发 Spring 跟随。
