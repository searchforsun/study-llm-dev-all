四大生产坑详解

CorpAssist · 生产事故复盘 上线第一周，用户说「改地址」→ Agent 收集了 order_id 和 new_address → 用户下一个问题说「其实钱怎么退」→ Agent 的 DST 没清理旧槽位，把「退款」意图和「改地址」的槽位混在一起，调用 refund 工具时传错了参数。这就是典型的幻觉改单。

Python：DST 双校验实现

python
# dst_validator.py — DST 双校验：LLM 提取 + 规则校验
import re
from typing import Dict, Any

class DSTValidator:
    """DST 双校验器：LLM 提取槽位后，用规则校验合法性"""

    INTENT_SCHEMA = {
        "refund": {"required": ["order_id", "amount"], "optional": ["reason"]},
        "change_address": {"required": ["order_id", "new_address"], "optional": ["contact_phone"]},
        "query_order": {"required": ["order_id"], "optional": []},
    }

    def __init__(self):
        self.current_intent = None
        self.slots = {}
        self.last_intent = None

    def update(self, new_intent: str, new_slots: Dict[str, Any]) -> Dict[str, Any]:
        """更新 DST 状态，含意图切换检测和槽位校验"""
        # 意图切换检测
        if self.current_intent and new_intent != self.current_intent:
            print(f"[DST] 意图切换: {self.current_intent} → {new_intent}")
            self._clear_slots_for_intent(new_intent)

        self.last_intent = self.current_intent
        self.current_intent = new_intent

        # 合并新槽位
        for k, v in new_slots.items():
            self.slots[k] = v

        # 校验合法性
        self._validate()

        return self.slots

    def _clear_slots_for_intent(self, new_intent: str):
        """切换意图时清理无关槽位"""
        required_for_new = self.INTENT_SCHEMA.get(new_intent, {}).get("required", [])
        optional_for_new = self.INTENT_SCHEMA.get(new_intent, {}).get("optional", [])
        allowed = set(required_for_new + optional_for_new)
        self.slots = {k: v for k, v in self.slots.items() if k in allowed}
        print(f"[DST] 清理槽位，保留: {allowed}")

    def _validate(self):
        """规则校验槽位值"""
        for k, v in self.slots.items():
            if k == "order_id" and not re.match(r"^O\d{8}$", str(v)):
                print(f"[DST] 警告: order_id 格式异常 {v}")
                del self.slots[k]
            if k == "amount":
                try:
                    amt = float(v)
                    if amt <= 0 or amt > 99999.99:
                        del self.slots[k]
                except (ValueError, TypeError):
                    del self.slots[k]
            if k == "contact_phone" and not re.match(r"^1\d{10}$", str(v)):
                del self.slots[k]

    def get_required_missing(self) -> list:
        """返回缺失的必填字段"""
        schema = self.INTENT_SCHEMA.get(self.current_intent, {})
        required = schema.get("required", [])
        return [f for f in required if f not in self.slots]

    def is_ready(self) -> bool:
        """所有必填字段是否已收集"""
        return len(self.get_required_missing()) == 0


validator = DSTValidator()
# 模拟场景：用户先改地址，后切换退款
validator.update("change_address", {"order_id": "O2025001", "new_address": "北京朝阳"})
print(f"当前槽位: {validator.slots}")  # 含 order_id, new_address
validator.update("refund", {"amount": 299})
print(f"意图切换后槽位: {validator.slots}")  # 不含 new_address
print(f"缺失字段: {validator.get_required_missing()}")

# API 结果强制写回
def tool_result_formatter(tool_name: str, result: dict) -> str:
    """工具结果模板化写回，LLM 不修改内容"""
    templates = {
        "refund": "退款成功：订单 {order_id}，金额 ¥{amount}，预计 {days} 个工作日到账。",
        "address": "地址修改成功：{order_id} → {new_address}",
    }
    tmpl = templates.get(tool_name, str(result))
    return tmpl.format(**result)

print(tool_result_formatter("refund", {"order_id": "O2025001", "amount": 299, "days": 3}))

Java Spring 侧：截图分级路由

// VisionRouter.java — 截图分级路由
@Component
public class VisionRouter {

    enum Complexity { SIMPLE, MEDIUM, COMPLEX }
    enum ModelTier { TURBO, PLUS, MAX }

    private final Map<ModelTier, String> modelMap = Map.of(
        ModelTier.TURBO, "qwen-vl-turbo",
        ModelTier.PLUS, "qwen-vl-plus",
        ModelTier.MAX, "qwen-vl-max"
    );

    public ModelTier route(String imageUrl) {
        Complexity complexity = classifyComplexity(imageUrl);
        return switch (complexity) {
            case SIMPLE -> ModelTier.TURBO;   // ¥0.003/次，60% 流量
            case MEDIUM -> ModelTier.PLUS;    // ¥0.008/次，30% 流量
            case COMPLEX -> ModelTier.MAX;    // ¥0.02/次，10% 流量
        };
    }

    private Complexity classifyComplexity(String imageUrl) {
        // 简化版：按图片尺寸分类（生产用更复杂的分类器）
        // SIMPLE: 文字截图 < 200KB
        // MEDIUM: 含简单表格 200KB-1MB
        // COMPLEX: 复杂场景图 > 1MB
        return Complexity.SIMPLE;  // 示例返回
    }

    public String analyze(String imageUrl) {
        ModelTier tier = route(imageUrl);
        String model = modelMap.get(tier);
        // 调用 VL 模型
        return String.format("使用 %s 分析，成本 ¥%.3f", model, tier.ordinal() == 0 ? 0.003 : tier.ordinal() == 1 ? 0.008 : 0.02);
    }
}

// CacheConfig.java — 语义缓存配置
@Configuration
public class CacheConfig {
    @Bean
    public CacheManager cacheManager(RedisConnectionFactory factory) {
        RedisCacheConfiguration config = RedisCacheConfiguration.defaultCacheConfig()
            .entryTtl(Duration.ofHours(1))
            .prefixCacheNameWith("faq:");
        return RedisCacheManager.builder(factory).cacheDefaults(config).build();
    }
}
不推荐
  • 纯 LLM 管理全部 DST 状态，无规则约束
  • LLM 无约束地转述工具调用结果
  • 所有图片用最强模型分析
  • FAQ 请求全部走 gpt-4o 推理
推荐
  • LLM 提取 + 规则校验 + intent 变更清旧槽
  • 工具结果模板化，LLM 仅做格式编排
  • 截图分级：简单用 turbo (60%)，复杂用 max (10%)
  • 缓存 + qwen-turbo 处理 80% 请求
先想 10 秒：生产环境中 intent 变更检测的准确率怎么保证？

不能只靠 LLM 自己说「intent 变了」。CorpAssist 的做法是：1) 规则层检测关键词变化（如新消息中出现「退款」但之前是「改地址」→ 标记变更）；2) 模型层对比新旧消息的 embedding 相似度，骤降说明 topic shift；3) 坐席确认——intent 变更时主动问用户「您是要切换到另一个问题吗？」。

常见问题（FAQ）

DST 双校验和纯规则 DST 有什么区别？
纯规则 DST 要求每条规则精确匹配，覆盖率低。双校验是「LLM 提取 + 规则验证」：LLM 负责从自然语言中提取槽位（高召回），规则负责校验槽位格式和业务逻辑（高精度），两者互补。
截图分级会不会漏掉重要的视觉信息？
分级的目标是「大多数用便宜模型，少数复杂场景用贵模型」。分级分类器本身会用 qwen-vl-plus 做校准：如果 turbo 模型的输出置信度低于 0.7，自动升级到 plus 或 max 重新分析。确保不会因为省钱而降低质量。
语义缓存的相似度阈值设多少合适？
CorpAssist 的实践是 0.92。太高（如 0.98）命中率太低，太低（如 0.85）容易命中语义相近但答案不同的内容。建议在生产用 0.90-0.95 之间调试，同时缓存 key 包含意图标签以减少误命中。
高 QPS 场景下如果缓存全失效怎么办？
缓存雪崩是真实风险。防护措施：1) 缓存 key 的 TTL 加随机偏移（±10% 避免同时过期）；2) 本地缓存兜底（Caffeine，1000 条容量）；3) 如果 QPS 超过模型配额，开启弹性降级——转人工而非排队等待模型。

  • DST 双校验：LLM 提取 + 规则校验双保险，intent 变更时清理旧槽位。

  • 截图分级：turbo 60% + plus 30% + max 10%，节省 70% 视觉成本。
  • 缓存 + 小模型：语义缓存 ≥60% 命中率，小模型处理 ≥80% 请求。

  1. 生产监控发现截图分析日均成本从 ¥200 涨到 ¥800，你怎么排查？

    参考答案

    三步排查：1) 检查截图分级各级流量占比——如果 max 从 10% 涨到 50%，说明分级分类器有问题；2) 检查是否有新的截图来源（如新功能上传了高清图片）；3) 检查是否有攻击——恶意用户上传超大图片耗尽配额。修复：升级分级分类器 + 限制单图大小 + 限流。
