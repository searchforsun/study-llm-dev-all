Python 自动评测实现

python
# auto_eval.py — CorpAssist 自动评测脚本
import json
import csv
from datetime import datetime
from typing import List, Dict
from dataclasses import dataclass

@dataclass
class EvalSample:
    id: str
    message: str
    expected_intent: str
    expected_tool: str
    expected_success: bool
    expected_csat: int  # 1-5

@dataclass
class EvalResult:
    sample_id: str
    predicted_intent: str
    intent_correct: bool
    tool_called: str
    tool_success: bool
    latency_ms: float

class AutoEvalPipeline:
    """自动评测管线"""

    def __init__(self, classifier, tools: dict):
        self.classifier = classifier
        self.tools = tools
        self.samples: List[EvalSample] = []
        self.results: List[EvalResult] = []

    def load_samples(self, path: str = "./eval_data/eval_dataset.json"):
        with open(path) as f:
            data = json.load(f)
        self.samples = [EvalSample(**s) for s in data]
        print(f"加载 {len(self.samples)} 条评测样本")

    def run(self) -> dict:
        """运行全量评测"""
        for sample in self.samples:
            import time
            start = time.time()

            route = self.classifier.route(sample.message)
            intent = route["intent"]
            llm_latency = (time.time() - start) * 1000

            tool_called = ""
            tool_success = False
            if route.get("action") == "TASK":
                tool_called = intent
                tool_success = self.tools.get(intent, lambda: False)()

            self.results.append(EvalResult(
                sample_id=sample.id,
                predicted_intent=intent,
                intent_correct=intent == sample.expected_intent,
                tool_called=tool_called,
                tool_success=tool_success,
                latency_ms=llm_latency,
            ))

        return self.report()

    def report(self) -> dict:
        """生成评测报告"""
        n = len(self.results)
        if n == 0:
            return {}
        intent_acc = sum(1 for r in self.results if r.intent_correct) / n
        tools_attempted = [r for r in self.results if r.tool_called]
        tool_success_rate = sum(1 for r in tools_attempted if r.tool_success) / max(len(tools_attempted), 1)
        avg_latency = sum(r.latency_ms for r in self.results) / n
        return {
            "date": datetime.now().date().isoformat(),
            "samples": n,
            "intent_accuracy": round(intent_acc * 100, 1),
            "tool_success_rate": round(tool_success_rate * 100, 1),
            "avg_latency_ms": round(avg_latency, 1),
            "p95_latency_ms": round(sorted(r.latency_ms for r in self.results)[int(n * 0.95)], 1),
        }

    def export_badcases(self, path: str = "./eval_data/badcases.json"):
        """导出坏例"""
        badcases = []
        for r in self.results:
            sample = next(s for s in self.samples if s.id == r.sample_id)
            if not r.intent_correct:
                badcases.append({
                    "id": r.sample_id,
                    "message": sample.message,
                    "expected": sample.expected_intent,
                    "predicted": r.predicted_intent,
                    "category": "intent_error",
                })
        with open(path, "w") as f:
            json.dump(badcases, f, ensure_ascii=False, indent=2)
        print(f"导出 {len(badcases)} 条坏例到 {path}")

// CsatCollector.java — CSAT 评分采集
@Component
public class CsatCollector {

    private final MongoTemplate mongo;
    private final MeterRegistry meterRegistry;

    public CsatCollector(MongoTemplate mongo, MeterRegistry meterRegistry) {
        this.mongo = mongo;
        this.meterRegistry = meterRegistry;
    }

    public void recordScore(String sessionId, String agentType, int score) {
        CsatRecord record = new CsatRecord(sessionId, agentType, score, Instant.now());
        mongo.save(record, "csat_scores");

        // Prometheus 指标
        meterRegistry.summary("csat.score", "agent", agentType).record(score);
        meterRegistry.counter("csat.total", "agent", agentType).increment();
    }

    public double getAverageCsat(String agentType, LocalDate since) {
        Query query = Query.query(
            Criteria.where("agentType").is(agentType)
                .and("timestamp").gte(since.atStartOfDay())
        );
        List<CsatRecord> records = mongo.find(query, CsatRecord.class, "csat_scores");
        return records.stream().mapToInt(CsatRecord::score).average().orElse(0.0);
    }
}

  • 坏例不归档，重复犯错

  • 每日自动 + 每周人工双轨评测
  • 坏例库闭环驱动迭代
  • A/B 实验验证变更效果再全量
先想 10 秒：CSAT 4.2 但任务成功率只有 70%，你信哪个？

信任务成功率。CSAT 受到多种因素影响（坐席态度、响应速度等），任务成功率直接衡量「问题是否被解决」。高 CSAT + 低任务成功率说明用户对服务态度满意但问题没解决，需要优化工具层和服务流程。

A/B 实验与坏例库

ab cs 定义 A/B 实验框架：按租户 50/50 分流，实验组用新路由策略，对照组用旧策略。观察期至少 7 天，统计显著性采用卡方检验。

badcase cs 是坏例库的核心。每条坏例记录：原始消息、预期的正确行为、实际行为、错误分类（intent_error / tool_error / reply_error）、归因（规则不完善/模型未覆盖/prompt 歧义）、修复建议、状态（待修复/已修复/已验证）。

python
# badcase_lib.py — 坏例库管理
class BadcaseEntry:
    def __init__(self, message: str, expected: str, actual: str,
                 category: str, root_cause: str, fix: str):
        self.id = hashlib.md5(message.encode()).hexdigest()[:8]
        self.message = message
        self.expected = expected
        self.actual = actual
        self.category = category  # intent_error / tool_error / reply_error
        self.root_cause = root_cause
        self.fix = fix
        self.status = "open"  # open / fixed / verified

    def to_dict(self) -> dict:
        return {k: v for k, v in self.__dict__.items()}

常见问题（FAQ）

自动评测的标注集怎么维护？
CorpAssist 的标注集分为核心集（500 条，覆盖所有意图，每周更新）和增量集（每周从生产会话中新增 50-100 条，覆盖新增场景）。标注集存储在 JSON 文件中，版本化管理。
人工抽检的样本怎么选？
分层抽样：Bot 解决会话按 5% 随机抽，转人工会话 100% 抽检，低 CSAT 会话 100% 抽检。确保抽检样本既能反映整体质量，又能覆盖高风险会话。
A/B 实验最少需要多少样本量？
取决于指标提升幅度。CorpAssist 的做法：预期提升 2pp，需每组至少 5000 条会话（约 2-3 天流量），显著性水平 α=0.05，统计功效 80%。建议 A/B 实验至少运行 7 天覆盖不同日期的用户行为差异。
指标之间相互矛盾怎么办（如 CSAT 升了但解决率降了）？
优先相信解决率。CSAT 受短期情绪影响大，解决率反映实际问题是否解决。如果解决率下降，说明产品本身有问题，CSAT 的回升可能是偶然因素（如坐席态度改善）。应该深挖解决率下降的原因，而不是庆祝 CSAT 上升。

本章结论

  • 五维指标：意图准确率 ≥92%、任务成功率 ≥85%、CSAT ≥4.2、转人工准确率 ≥80%、工具成功率 ≥95%。
  • 自动 + 人工双轨：每日 2000 条自动跑分，每周 50-100 条人工校准。
  • 坏例飞轮：低分样本归因 → 修复 → 回归验证，24 小时内可完成一轮。
  • A/B 实验：按租户 50/50 分流，7 天观察期，卡方检验验证。
  • 解决率优先：指标矛盾时优先相信任务成功率，深挖根因。

判断练习

  1. 坏例库中有一条分类错误：用户说「我要退了重买」被分为 refund（正确是 exchange）。归因是什么？如何修复？

    参考答案

    归因：规则中「退」关键词匹配了 refund，但「退了重买」实际是换货意图。修复方案：1) 规则层加负向规则——「退」+「重买」→ exchange；2) 模型层在训练集中增加「退了重买」「退货重下」等换货语料；3) 短期在 Agent 确认环节增加 intent 确认步骤。
