集成测试：模拟完整对话

java
@SpringBootTest
class CustomerServiceIntegrationTest {

    @Autowired
    private CustomerServiceAgent agent;

    @MockBean
    private OrderApiClient orderApiClient;

    @Test
    void testOrderQueryFlow() {
        // 模拟订单 API
        when(orderApiClient.query("ORD-001", "13800138000"))
            .thenReturn(new OrderResponse("ORD-001", "SHIPPED",
                "顺丰快递", "SF123456", "2026-06-05"));

        // 模拟完整对话
        StepVerifier.create(
            agent.chat("session-1", "查一下我的订单 ORD-001")
        )
        .expectNextMatches(token ->
            token.contains("ORD-001") || token.contains("顺丰"))  // 工具调用正确
        .thenCancel()
        .verify();

        // 验证工具被调用
        verify(orderApiClient, times(1)).query("ORD-001", "13800138000");
    }

    @Test
    void testReturnFlow() {
        // 模拟退货流程
        when(returnApiClient.create(any()))
            .thenReturn(new ReturnResult("RET-001", "APPROVED"));

        // 用户要求退货
        StepVerifier.create(
            agent.chat("session-2", "我要退货，订单 ORD-002")
        )
        .expectNextMatches(token ->
            token.contains("退货") || token.contains("原因"))
        .thenCancel()
        .verify();
    }
}

A/B 分流与坏例库回归

java
// A/B 分流
@Service
public class ABTestRouter {

    private final LoadBalancer loadBalancer;

    public String selectModel(String userId, String experimentName) {
        // 基于 userId 做一致性哈希，确保同一用户始终在 A/B 同一组
        int hash = Math.abs(userId.hashCode());
        boolean isTreatment = (hash % 100) < 20;  // 20% 流量到实验组

        if (isTreatment) {
            metricRegistry.counter("ab.test", "experiment", experimentName,
                "group", "treatment").increment();
            return "gpt-4o-mini";  // 新模型
        } else {
            return "qwen-plus";     // 当前模型
        }
    }
}

// 坏例库与回归
@Service
public class RegressionTestSuite {

    private final List badCaseLib = List.of(
        new TestCase("订单还没到", "应该调用 query_order"),
        new TestCase("我要退钱", "应该调用 create_return，不是 query_order"),
        new TestCase("你们客服是机器人吗", "应该友好回应，不是转人工")
    );

    public TestReport run() {
        int passed = 0;
        int failed = 0;

        for (TestCase tc : badCaseLib) {
            boolean result = runSingleTest(tc);
            if (result) passed++; else failed++;
        }

        return new TestReport(passed, failed, badCaseLib.size());
    }
}
