Spring AI 意图分类器（结构化输出）

java
public enum Intent {
    ORDER_QUERY,        // 订单查询
    RETURN_REQUEST,     // 退货申请
    REFUND_INQUIRY,     // 退款咨询
    SHIPPING_INQUIRY,   // 物流查询
    POLICY_QUERY,       // 政策查询
    COMPLAINT,          // 投诉
    GREETING,           // 问候
    UNKNOWN             // 未识别
}

public record IntentResult(
    Intent intent,
    double confidence,
    Map slots   // key: 实体名, value: 值
) {}

@Service
public class IntentClassifier {

    private final ChatClient chatClient;

    public IntentResult classify(String message) {
        return chatClient.prompt()
            .system("""
                你是客服意图分类器。请将用户消息分类到以下意图之一：
                - ORDER_QUERY: 查订单状态/物流/预计到货时间
                - RETURN_REQUEST: 退货/退换/不要了
                - REFUND_INQUIRY: 退款/退钱/费用问题
                - SHIPPING_INQUIRY: 物流/快递/配送
                - POLICY_QUERY: 保修/政策/规则/运费
                - COMPLAINT: 投诉/差评/不满/曝光
                - GREETING: 问候/你好/在吗
                - UNKNOWN: 其他/无法确定

                同时提取关键实体（slots）如：orderId, productName, date等。
                """)
            .user(message)
            .call()
            .entity(IntentResult.class);  // 结构化输出
    }
}

结构化输出通过 chatClient.call().entity(IntentResult.class) 实现——Spring AI 使用 JSON Schema 约束 LLM 输出格式，自动反序列化为 Java 对象。这比正则解析更灵活、更准确。CorpAssist 实验中结构化输出的意图识别准确率约 94%，对比基于关键词规则的 82%。

规则 + 模型双引擎路由

生产环境采用双引擎：规则引擎优先匹配精确关键词（低延迟），匹配不到时用模型分类（高准确率）。

java
@Service
public class DualEngineRouter {
    private final RuleEngine ruleEngine;
    private final IntentClassifier modelClassifier;
    private final MeterRegistry meterRegistry;

    public IntentResult route(String message) {
        // 1. 规则引擎优先
        IntentResult ruleResult = ruleEngine.match(message);
        if (ruleResult != null && ruleResult.confidence() >= 0.9) {
            meterRegistry.counter("intent.router", "engine", "rule").increment();
            return ruleResult;
        }

        // 2. 模型分类
        IntentResult modelResult = modelClassifier.classify(message);

        // 3. 模型与规则不一致时的仲裁
        if (ruleResult != null && modelResult.intent() != ruleResult.intent()) {
            // 记录不一致，用于后续分析
            meterRegistry.counter("intent.router", "engine", "conflict").increment();
            // 高置信度的一边获胜
            return modelResult.confidence() > 0.9 ? modelResult : ruleResult;
        }

        meterRegistry.counter("intent.router", "engine", "model").increment();
        return modelResult;
    }
}

@Component
class RuleEngine {
    private final List rules = List.of(
        new Rule(List.of("查订单", "物流", "快递", "到哪了"), Intent.ORDER_QUERY),
        new Rule(List.of("退货", "退换", "不要了", "取消"), Intent.RETURN_REQUEST),
        new Rule(List.of("退款", "退钱", "没收到"), Intent.REFUND_INQUIRY),
        new Rule(List.of("投诉", "差评", "曝光"), Intent.COMPLAINT)
    );

    public IntentResult match(String message) {
        for (Rule rule : rules) {
            for (String keyword : rule.keywords) {
                if (message.contains(keyword)) {
                    return new IntentResult(rule.intent, 0.95, Map.of());
                }
            }
        }
        return null;
    }
}

意图置信度阈值与兜底

java
@Service
public class IntentThresholdHandler {

    private static final double HIGH_CONFIDENCE = 0.80;
    private static final double MEDIUM_CONFIDENCE = 0.50;

    public RoutingDecision decide(IntentResult result, String originalMessage) {
        if (result.confidence() >= HIGH_CONFIDENCE) {
            return RoutingDecision.route(result.intent());
        }

        if (result.confidence() >= MEDIUM_CONFIDENCE) {
            // 中等置信度：二次确认
            return RoutingDecision.clarify(
                "您是想" + describeIntent(result.intent()) + "吗？请确认一下。"
            );
        }

        // 低置信度或 UNKNOWN：兜底
        if (result.intent() == Intent.UNKNOWN) {
            return RoutingDecision.escalate(
                "抱歉，我没有完全理解您的意思。已为您转接人工坐席。"
            );
        }

        return RoutingDecision.escalate(
            "我已将您的问题转给人工坐席处理。"
        );
    }
}
