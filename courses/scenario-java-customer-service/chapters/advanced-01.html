Spring Cloud Gateway + Resilience4j 限流

yaml
spring:
  cloud:
    gateway:
      default-filters:
        - name: RequestRateLimiter
          args:
            redis-rate-limiter:
              replenishRate: 100   # 每秒 100 个请求
              burstCapacity: 200   # 突发 200
              requestedTokens: 1
        - name: CircuitBreaker
          args:
            name: csServiceCB
            fallbackUri: forward:/fallback/cs

resilience4j:
  circuitbreaker:
    instances:
      csChatEngineCB:
        sliding-window-size: 10
        failure-rate-threshold: 50
        wait-duration-in-open-state: 30s
        permitted-number-of-calls-in-half-open-state: 3
  timelimiter:
    instances:
      csChatEngineTL:
        timeout-duration: 30s
java
@RestController
@RequestMapping("/fallback")
public class FallbackController {

    @PostMapping("/cs")
    public Mono> fallback() {
        return Mono.just(ServerSentEvent.builder()
            .event("error")
            .data("系统繁忙，请稍后再试。或直接联系人工客服电话 400-xxx-xxxx。")
            .build());
    }
}

JWT + Spring Security 鉴权

java
@Configuration
@EnableWebFluxSecurity
public class SecurityConfig {

    @Bean
    public SecurityWebFilterChain securityWebFilterChain(
            ServerHttpSecurity http) {
        return http
            .authorizeExchange(exchanges -> exchanges
                .pathMatchers("/api/v1/cs/chat/**").authenticated()
                .pathMatchers("/api/v1/cs/handoff/**").hasRole("AGENT")
                .pathMatchers("/fallback/**").permitAll()
                .anyExchange().authenticated()
            )
            .oauth2ResourceServer(oauth2 -> oauth2
                .jwt(Customizer.withDefaults())
            )
            .build();
    }

    @Bean
    public ReactiveJwtDecoder jwtDecoder() {
        return NimbusReactiveJwtDecoder
            .withJwkSetUri("https://auth.corp.example/.well-known/jwks.json")
            .build();
    }
}

SSE 全链路超时配置

yaml
# Gateway 层
spring:
  cloud:
    gateway:
      routes:
        - id: cs-chat-stream
          uri: lb://cs-chat-engine
          predicates:
            - Path=/api/v1/cs/chat/stream
          metadata:
            response-timeout: 60000  # 60 秒 SSE 超时

# 应用层
server:
  reactive:
    session:
      timeout: 30m
  netty:
    max-keep-alive-requests: 1000

# LLM 调用超时
spring:
  ai:
    openai:
      chat:
        options:
          timeout: 30s  # LLM 调用 30 秒超时

SSE 超时策略的三层配置：Gateway 层 60s -> 应用层 30s(LIM) -> LLM 层 30s。任何一层超时都会返回友好错误给用户，并记录超时事件到监控系统。

本章结论

  • Gateway + Resilience4j 限流(100 QPS) + 熔断(50% 失败率)
  • JWT + Spring Security OAuth2 鉴权，坐席角色额外权限
  • SSE 三层超时：Gateway 60s -> 应用 30s -> LLM 30s
  • 熔断降级到 FallbackController 返回友好提示
