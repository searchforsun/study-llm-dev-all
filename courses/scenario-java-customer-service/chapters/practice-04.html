置信度阈值 -> 人工转接

java
@Service
public class HandoffDecider {

    private static final double AUTO_HANDOFF_THRESHOLD = 0.30;
    private static final double CLARIFY_FAIL_THRESHOLD = 0.50;

    public HandoffDecision evaluate(String sessionId,
                                     IntentResult intent,
                                     int clarificationCount,
                                     boolean userRequestedAgent) {

        // 1. 用户主动要求转人工
        if (userRequestedAgent) {
            return HandoffDecision.immediate("用户主动要求转人工");
        }

        // 2. 置信度过低
        if (intent.confidence() < AUTO_HANDOFF_THRESHOLD) {
            return HandoffDecision.immediate("意图置信度低于阈值: " + intent.confidence());
        }

        // 3. 二次确认超过 2 次仍未澄清
        if (clarificationCount >= 2) {
            return HandoffDecision.immediate("二次确认超过 2 次未澄清");
        }

        // 4. 敏感场景
        if (intent.intent() == Intent.COMPLAINT) {
            return HandoffDecision.immediate("投诉类对话自动转人工");
        }

        return HandoffDecision.continueChat();
    }
}

转接时上下文移交

java
@Service
public class ContextHandoffService {

    private final ChatMemory chatMemory;
    private final MeterRegistry meterRegistry;

    public HandoffPackage buildHandoffPackage(String sessionId,
                                                String userId,
                                                String reason) {
        // 1. 获取完整对话历史
        List history = chatMemory.get(sessionId, 50);

        // 2. 获取意图判断和工具调用记录
        // 3. 构建坐席工作台需要的上下文
        HandoffPackage pkg = new HandoffPackage(
            sessionId,
            userId,
            reason,
            formatHistoryForAgent(history),
            Instant.now()
        );

        meterRegistry.counter("cs.handoff", "reason", reason).increment();
        return pkg;
    }

    private String formatHistoryForAgent(List history) {
        StringBuilder sb = new StringBuilder();
        sb.append("--- 对话历史 (转人工) ---\n\n");
        for (Message msg : history) {
            String role = switch (msg.getMessageType()) {
                case "user" -> "用户";
                case "assistant" -> "机器人";
                case "tool" -> "工具返回";
                default -> "系统";
            };
            sb.append("[").append(role).append("] ")
              .append(msg.getText()).append("\n\n");
        }
        return sb.toString();
    }
}

排队与坐席分配（Redis 队列）

java
@Service
public class AgentQueueService {

    private static final String HANDOFF_QUEUE = "cs:handoff:queue";
    private static final String AGENT_AVAILABLE = "cs:agent:available";
    private static final String AGENT_SESSION = "cs:agent:session:";

    public AgentAssignment assignAgent(HandoffPackage pkg) {
        // 1. 找可用坐席
        String agentId = redisTemplate.opsForSet()
            .pop(AGENT_AVAILABLE);

        if (agentId == null) {
            // 2. 没有可用坐席 -> 排队
            long position = redisTemplate.opsForList()
                .rightPush(HANDOFF_QUEUE, pkg.sessionId());
            return AgentAssignment.queued(position, pkg);
        }

        // 3. 分配坐席
        redisTemplate.opsForValue()
            .set(AGENT_SESSION + agentId, pkg.sessionId(),
                 Duration.ofHours(2));
        return AgentAssignment.assigned(agentId, pkg);
    }
}
