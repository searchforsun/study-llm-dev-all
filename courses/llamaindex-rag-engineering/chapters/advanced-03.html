缓存策略

缓存策略分两种：精确缓存（相同查询直接返回缓存答案）和语义缓存（语义相似的查询命中缓存）。LlamaIndex 不内置缓存，但可以通过装饰器或中间件实现。

flowchart LR
    Q[查询] -->|精确缓存| EX{在缓存中?}
    EX -->|是| CACHE[直接返回缓存答案
延迟: 1ms 成本: 0]
    EX -->|否| SEM{语义缓存
相似度 > 0.95?}
    SEM -->|是| SCACHE[返回缓存答案
延迟: 50ms 成本: 嵌入]
    SEM -->|否| LLM[调用 LLM 合成]
    LLM -->|写入缓存| STORE[Redis]
python
import hashlib, json, time
import redis as redis_lib
from llama_index.embeddings.openai import OpenAIEmbedding

# Redis 连接
redis_client = redis_lib.Redis(host="redis", port=6379, decode_responses=True)
embed_model = OpenAIEmbedding(model="text-embedding-3-small", dimensions=512)

# ---- 1. 精确缓存 ----
def exact_cache_key(query: str, tenant_id: str) -> str:
    """精确缓存 Key: md5(query + tenant_id)"""
    raw = f"{query}:{tenant_id}"
    return f"cache:exact:{hashlib.md5(raw.encode()).hexdigest()}"

def get_exact_cache(query: str, tenant_id: str) -> str | None:
    key = exact_cache_key(query, tenant_id)
    return redis_client.get(key)

def set_exact_cache(query: str, tenant_id: str, answer: str, ttl=3600):
    key = exact_cache_key(query, tenant_id)
    redis_client.setex(key, ttl, answer)

# ---- 2. 语义缓存 ----
class SemanticCache:
    def __init__(self, similarity_threshold=0.95):
        self.threshold = similarity_threshold
        self.embed_model = OpenAIEmbedding(model="text-embedding-3-small", dimensions=512)

    def find_similar(self, query: str, tenant_id: str) -> str | None:
        """查找语义相似的缓存条目"""
        query_vec = self.embed_model.get_text_embedding(query)
        # 从 Redis 获取该租户的所有缓存向量
        cache_entries = redis_client.hgetall(f"cache:semantic:{tenant_id}")
        best_score, best_answer = 0, None
        for cached_query, cached_data in cache_entries.items():
            data = json.loads(cached_data)
            cached_vec = data["vector"]
            # 余弦相似度
            score = sum(a*b for a,b in zip(query_vec, cached_vec)) / (
                sum(a*a for a in query_vec)**0.5 * sum(b*b for b in cached_vec)**0.5
            )
            if score > self.threshold and score > best_score:
                best_score, best_answer = score, data["answer"]
        return best_answer

    def store(self, query: str, tenant_id: str, answer: str, ttl=7200):
        query_vec = self.embed_model.get_text_embedding(query)
        data = json.dumps({"vector": query_vec, "answer": answer, "timestamp": time.time()})
        redis_client.hset(f"cache:semantic:{tenant_id}", query, data)
        redis_client.expire(f"cache:semantic:{tenant_id}", ttl)

# ---- 3. 组合缓存中间件 ----
class CacheMiddleware:
    def __init__(self):
        self.exact_ttl = 3600       # 精确缓存 1 小时
        self.semantic_ttl = 7200    # 语义缓存 2 小时
        self.semantic_cache = SemanticCache(threshold=0.95)

    async def get_or_compute(self, query: str, tenant_id: str, compute_fn):
        # 1. 精确缓存
        cached = get_exact_cache(query, tenant_id)
        if cached:
            return json.loads(cached)

        # 2. 语义缓存
        semantic = self.semantic_cache.find_similar(query, tenant_id)
        if semantic:
            return json.loads(semantic)

        # 3. 计算新答案
        result = await compute_fn()

        # 4. 写入缓存
        result_json = json.dumps(result)
        set_exact_cache(query, tenant_id, result_json, self.exact_ttl)
        self.semantic_cache.store(query, tenant_id, result_json, self.semantic_ttl)
        return result

小模型路由

小模型路由根据查询复杂度选择 LLM：简单查询（事实性、分类）用低成本模型，复杂查询（推理、比较）用高性能模型。可通过分类器或 LLM 自身判断。

from llama_index.llms.openai import OpenAI
from llama_index.llms.deepseek import DeepSeek

class ModelRouter:
    """根据查询复杂度路由到不同 LLM"""
    def __init__(self):
        self.cheap_llm = DeepSeek(model="deepseek-chat", temperature=0.1)
        self.medium_llm = OpenAI(model="gpt-4.1-mini", temperature=0.1)
        self.expensive_llm = OpenAI(model="gpt-4.1", temperature=0.1)

    def classify_query(self, query: str) -> str:
        """用简单规则 + 小模型分类查询复杂度"""
        # 简单规则快速过滤
        simple_keywords = ["什么", "多少", "谁", "哪里", "何时", "是不是", "有没有"]
        complex_keywords = ["对比", "比较", "分析", "总结", "为什么", "如何解决", "区别"]

        if any(kw in query for kw in complex_keywords):
            return "complex"
        if any(kw in query for kw in simple_keywords):
            return "simple"
        # 不确定的用小模型分类
        response = self.cheap_llm.complete(
            f"判断以下查询的复杂度，只返回 simple/medium/complex: {query}"
        )
        return str(response).strip().lower()

    def get_llm(self, query: str):
        complexity = self.classify_query(query)
        if complexity == "simple":
            return self.cheap_llm
        elif complexity == "medium":
            return self.medium_llm
        else:
            return self.expensive_llm

# 使用
router = ModelRouter()
llm = router.get_llm("年假是多少天")  # → DeepSeek（简单）
llm2 = router.get_llm("对比HR的年假和IT的VPN配置")  # → GPT-4.1（复杂）

预计算热点问答

预计算热点问答是通过分析历史查询日志，识别高频重复问题，提前计算答案并存入缓存。这样热点问题直接用预计算答案响应，零检索零生成。

python
# 1. 分析历史日志，找出热点查询
from collections import Counter
import json

def analyze_query_logs(log_path: str, top_n: int = 100):
    """分析查询日志，找出高频问题"""
    queries = []
    with open(log_path) as f:
        for line in f:
            record = json.loads(line)
            queries.append(record["query"])

    counter = Counter(queries)
    hotspots = counter.most_common(top_n)
    print(f"Top 5 热点查询:")
    for q, count in hotspots[:5]:
        print(f"  {count}次: {q}")
    return [q for q, _ in hotspots]

# 2. 预计算热点答案
async def precompute_hotspots(hotspot_queries: list[str], tenant_id: str):
    """预计算热点问题的答案并缓存"""
    engine = get_tenant_engine(tenant_id)
    results = {}
    for query in hotspot_queries:
        resp = engine.query(query)
        answer = str(resp)
        # 写入精确缓存（TTL 设为 24 小时）
        set_exact_cache(query, tenant_id, json.dumps({
            "answer": answer,
            "sources": [n.node_id for n in resp.source_nodes],
            "precomputed": True
        }), ttl=86400)
        results[query] = answer
        print(f"预计算: {query[:30]}... → {len(answer)} 字符")
    return results

# 3. 定时任务（每天凌晨 3 点执行）
async def daily_hotspot_update():

Token 成本归因

Token 成本归因将每次查询的 Token 消耗分解到检索、合成、路由各个环节，并归属到租户和部门。这是优化 ROI 的基础数据。

python
from dataclasses import dataclass
from datetime import datetime
import json

@dataclass
class CostRecord:
    query_id: str
    tenant_id: str
    query: str
    retrieval_tokens: int
    llm_model: str
    prompt_tokens: int
    completion_tokens: int
    latency_ms: int
    from_cache: bool
    created_at: str

class CostTracker:
    def __init__(self):
        self.records = []

    def record_query(self, **kwargs):
        record = CostRecord(
            query_id=kwargs['query_id'],
            tenant_id=kwargs['tenant_id'],
            query=kwargs['query'][:100],
            retrieval_tokens=kwargs.get('retrieval_tokens', 0),
            llm_model=kwargs.get('llm_model', 'none'),
            prompt_tokens=kwargs.get('prompt_tokens', 0),
            completion_tokens=kwargs.get('completion_tokens', 0),
            latency_ms=kwargs.get('latency_ms', 0),
            from_cache=kwargs.get('from_cache', False),
            created_at=datetime.utcnow().isoformat()
        )
        self.records.append(record)

    def calculate_cost(self, record: CostRecord) -> float:
        """计算单次查询的 Token 费用"""
        if record.from_cache:
            return 0.0
        rates = {
            "gpt-4.1": {"prompt": 15, "completion": 60},
            "gpt-4.1-mini": {"prompt": 1.5, "completion": 6},
            "deepseek-chat": {"prompt": 0.5, "completion": 2},
            "text-embedding-3-small": {"prompt": 0.002, "completion": 0},
        }
        model_rates = rates.get(record.llm_model, rates["gpt-4.1"])
        # 输出是百万 Token 价格，计算到分的费用
        cost = (record.prompt_tokens * model_rates["prompt"] +
                record.completion_tokens * model_rates["completion"]) / 1_000_000
        return round(cost, 6)

    def report_by_tenant(self) -> str:
        """按租户生成费用报告"""
        tenant_costs = {}
        for record in self.records:
            tenant = record.tenant_id
            if tenant not in tenant_costs:
                tenant_costs[tenant] = {"count": 0, "cost": 0, "cache_hit": 0}
            tenant_costs[tenant]["count"] += 1
            tenant_costs[tenant]["cost"] += self.calculate_cost(record)
            if record.from_cache:
                tenant_costs[tenant]["cache_hit"] += 1

        report = "租户费用报告:\n"
        for tenant, data in tenant_costs.items():
            cache_rate = data["cache_hit"] / data["count"] * 100
            report += f"  {tenant}: {data['count']}次, ${data['cost']:.2f}, 缓存率{cache_rate:.0f}%\n"
        return report

# 使用
tracker = CostTracker()
# 每次查询后记录
tracker.record_query(
    query_id="q_001", tenant_id="hr", query="年假几天",
    retrieval_tokens=128, llm_model="deepseek-chat",
    prompt_tokens=350, completion_tokens=120, latency_ms=450, from_cache=False
)
print(tracker.report_by_tenant())

面试可能问

精确缓存和语义缓存的适用场景分别是什么？语义缓存的阈值怎么确定？

精确缓存：用户完全相同的查询（"年假几天"和"年假几天"）。适合系统自动生成的问题、常见 FAQ。语义缓存：表述不同但意图相同的查询（"年假几天"和"我有多少天年假"）。阈值通过实验确定：在验证集上测试不同阈值（0.90/0.95/0.99）的精度和召回率。阈值太高（0.99）命中少，太低（0.90）可能误命中不同意图的问题。CorpAssist 经验值 0.95 是平衡点。

小模型路由分类错误（把复杂查询路由到简单模型）导致回答质量差，怎么兜底？

双层兜底：1) 快速验证：简单模型生成答案后，用另一个模型快速评估答案质量（如检查答案长度、是否包含"不确定"等信号），质量低则重新路由到高级模型；2) 用户反馈：设置用户对答案的"有用/没用"评分，评分低的查询自动记录并重新路由。用户也可以手动触发"深入回答"按钮，强制使用高级模型重新生成。CorpAssist 策略：先 DeepSeek 快速回答，用户可一键切换到 GPT-4.1 深究。

预计算热点的答案会过期（HR 政策调整后旧答案错误），怎么处理？

答案过期是预计算的核心挑战。处理方案：1) TTL 管理：热点答案缓存 TTL 设为 24 小时，每天凌晨重新预计算；2) 文档变更触发：HR 文档更新时，清除关联的热点缓存（基于 metadata 中记录的 source 文档 ID）；3) 版本标记：预计算答案打上文档版本号，查询时如果发现文档版本高于缓存版本，重新生成；4) 渐进式更新：不一次全部重新计算，而是每小时扫描 10% 的热点题目，7 小时内完成全量刷新。

判断练习

场景：语义缓存命中了一个相似但不完全相同的问题，答案有些偏差，怎么在延迟和准确性之间平衡？

方案 1：提高阈值（从 0.95 到 0.98），减少误命中但降低命中率。方案 2：缓存答案时同时缓存该答案的 source_nodes，返回缓存答案时附带来源引用让用户自判。方案 3：两阶段策略——先用缓存快速返回答案（200ms），同时后台触发完整 LLM 调用，如果新答案与缓存答案不一致（通过 LLM 判断），推送更新页面。CorpAssist 采用方案 3，体验最佳。

场景：小模型路由中，查询"帮我写一篇关于公司年假政策的 500 字报告"被误判为"simple"，怎么优化？

误判原因是规则中没有识别"写报告"类复杂请求。优化：1) 在分类规则中加入"写|创建|生成...报告|文章|邮件"等动词+名词组合的复杂查询模式；2) 查询长度也是一个信号——超过 30 个字符的查询更可能是复杂查询；3) 使用更快速的轻量分类器（如朴素贝叶斯或 small LM 分类器）代替规则。修改后"写...报告"类查询被正确路由到 GPT-4.1，分类准确率从 82% 提升到 94%。
