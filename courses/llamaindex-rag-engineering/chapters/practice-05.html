# app/main.py — CorpAssist RAG 服务主入口
from fastapi import FastAPI, UploadFile, BackgroundTasks, HTTPException
from fastapi.responses import StreamingResponse
from contextlib import asynccontextmanager
import os, uuid, json

from llama_index.core import VectorStoreIndex, SimpleDirectoryReader, Settings
from llama_index.core.ingestion import IngestionPipeline
from llama_index.core.node_parser import SentenceSplitter
from llama_index.embeddings.openai import OpenAIEmbedding
from llama_index.vector_stores.milvus import MilvusVectorStore
from pydantic import BaseModel

# --- 全局配置 ---
MILVUS_URI = os.getenv("MILVUS_URI", "http://milvus:19530")
UPLOAD_DIR = os.getenv("UPLOAD_DIR", "/data/uploads")
Settings.embed_model = OpenAIEmbedding(model="text-embedding-3-small", dimensions=1536)

# --- 生命周期管理 ---
@asynccontextmanager
async def lifespan(app: FastAPI):
    # 启动：初始化 Milvus 连接
    global vector_store, index
    vector_store = MilvusVectorStore(uri=MILVUS_URI, dim=1536)
    index = VectorStoreIndex.from_vector_store(vector_store)
    os.makedirs(UPLOAD_DIR, exist_ok=True)
    yield
    # 关闭：清理资源
    vector_store.close()

app = FastAPI(title="CorpAssist RAG Service", version="1.0.0", lifespan=lifespan)

# --- Pydantic 模型 ---
class QueryRequest(BaseModel):
    question: str
    top_k: int = 5
    tenant_id: str = "default"
    streaming: bool = False

class QueryResponse(BaseModel):
    answer: str
    sources: list[dict]
    token_usage: int

# --- API 端点 ---
@app.post("/api/v1/upload")
async def upload_document(file: UploadFile, tenant_id: str = "default", background: BackgroundTasks):
    """上传文档并异步建立索引"""
    file_path = f"{UPLOAD_DIR}/{uuid.uuid4()}_{file.filename}"
    content = await file.read()
    with open(file_path, "wb") as f:
        f.write(content)

    background.add_task(process_document, file_path, tenant_id)
    return {"status": "accepted", "file_id": file_path}

async def process_document(file_path: str, tenant_id: str):
    """后台处理文档：解析→分割→嵌入→写入 Milvus"""
    docs = SimpleDirectoryReader(input_files=[file_path]).load_data()
    for doc in docs:
        doc.metadata["tenant_id"] = tenant_id

    pipeline = IngestionPipeline(
        transformations=[
            SentenceSplitter(chunk_size=512, chunk_overlap=128),
            Settings.embed_model
        ],
        vector_store=vector_store
    )
    nodes = pipeline.run(documents=docs)
    print(f"文档处理完成: {len(nodes)} nodes from {file_path}")

FastAPI 服务化

检索和问答端点是服务的核心，支持同步和异步两种模式。

python
# app/routes/query.py — 检索和问答路由
from fastapi import APIRouter, Depends
from llama_index.core.postprocessor import SimilarityPostprocessor

router = APIRouter(prefix="/api/v1", tags=["query"])

def get_tenant_engine(tenant_id: str):
    """根据租户获取检索引擎"""
    return index.as_query_engine(
        similarity_top_k=5,
        filters=MetadataFilters(
            filters=[ExactMatchFilter(key="tenant_id", value=tenant_id)]
        ),
        node_postprocessors=[SimilarityPostprocessor(similarity_cutoff=0.5)]
    )

@router.post("/query", response_model=QueryResponse)
async def query(req: QueryRequest):
    """同步问答"""
    engine = get_tenant_engine(req.tenant_id)
    resp = engine.query(req.question)

    return QueryResponse(
        answer=str(resp),
        sources=[
            {"node_id": n.node_id, "score": n.score, "text": n.text[:200]}
            for n in resp.source_nodes
        ],
        token_usage=resp.metadata.get("token_usage", 0)
    )

@router.post("/query/stream")
async def query_stream(req: QueryRequest):
    """流式问答（SSE）"""
    engine = get_tenant_engine(req.tenant_id)
    streaming_engine = index.as_query_engine(
        streaming=True,
        similarity_top_k=req.top_k,
        filters=MetadataFilters(
            filters=[ExactMatchFilter(key="tenant_id", value=req.tenant_id)]
        )
    )

    async def event_generator():
        response = streaming_engine.query(req.question)
        async for token in response.async_response_gen():
            yield f"data: {json.dumps({'token': token})}\n\n"
        # 发送结束标记
        yield f"data: {json.dumps({'done': True, 'sources': [n.node_id for n in response.source_nodes]})}\n\n"

    return StreamingResponse(
        event_generator(),
        media_type="text/event-stream",
        headers={"Cache-Control": "no-cache", "X-Accel-Buffering": "no"}
    )

@router.delete("/documents/{doc_id}")
async def delete_document(doc_id: str):
    """删除文档及对应索引"""
    index.delete_ref_doc(doc_id, delete_from_docstore=True)
    return {"status": "deleted", "doc_id": doc_id}

同步接口阻塞

大文档检索+合成耗时 5-8 秒，客户端需等待完整响应，用户体验差。FastAPI 工作线程被占用。

流式 SSE 响应

首个 token 在 500ms 内返回（检索完成后立即开始吐字），客户端逐 token 渲染。FastAPI 使用 async generator 不阻塞工作线程。实测用户感知延迟降低 60%。

流式响应

流式响应是 CorpAssist 客服场景的核心需求。LlamaIndex 通过 streaming=True 参数启用，配合 FastAPI 的 StreamingResponse 实现 SSE（Server-Sent Events）协议。

本章结论

  • CorpAssist RAG 服务采用三层架构：FastAPI（API 层）+ LlamaIndex（引擎层）+ Milvus（存储层）
  • 文档上传通过 BackgroundTasks 异步处理，30s 内完成索引，不阻塞 API 响应
  • 流式响应基于 SSE 协议，首个 token 在 1s 内返回，显著提升用户体验
  • 多租户隔离通过 metadata filter + PermissionFilter 双层保障
  • 验收清单覆盖功能、性能、安全、可用性四个维度，是上线前的必检项

面试可能问

FastAPI 作为 RAG 服务的 Web 框架有哪些优势？对比 Flask 和 Django？

FastAPI 优势：1) 原生异步支持（async def），流式响应天然高效；2) Pydantic 模型自动生成 OpenAPI 文档，Java 侧可对接生成客户端代码；3) BackgroundTasks 适合异步文档处理。对比 Flask：同步框架，流式响应需要通过 Response 类手动实现。对比 Django：太重，RAG 服务不需要 ORM 和 Admin。CorpAssist 选择 FastAPI 是因为双栈架构中 Python 侧是轻量服务，不需要 Django 的重量级框架。

流式响应中如何传递 source_nodes 给客户端？

有两种常见策略：1) 在 SSE 流的最后发送一个特殊消息（如 {"done": true, "sources": [...]}），客户端解析最后一个消息获取来源。2) 在 SSE 流的第一个消息中发送 sources 元数据，之后逐 token 输出。第一种更常用，因为检索完成后才能确定 source_nodes。CorpAssist 使用策略 1：所有 token 输出完毕后发送 done 消息附带 sources。

文档上传后索引立即可用，但偶发"找不到刚上传内容"的问题，原因和解决方案？

原因：Async indexing 的时序问题——文档上传后立即查询，但后台任务还未完成索引写入（Index not yet ready）。解决方案：1) 上传接口返回 doc_id，查询时携带 doc_id 轮询索引状态；2) 实现索引写入确认机制——Milvus 写入后等待 flush() 完成再返回；3) 上传采用同步模式（第一次上传时阻塞等待索引完成）而不是后台任务；4) 使用 Milvus 的一致性级别（consistency_level="strong"），但会增加写入延迟。CorpAssist 采用方案 4，接受 3-5s 的写入延迟换取查询即时一致。

判断练习

场景：服务启动后第一次查询很慢（~10s），后续查询正常（~1s），怎么优化？

第一次查询慢是因为 Milvus 加载索引数据到内存（warm-up）。优化：1) 在 FastAPI lifespan 启动事件中预加载索引：vector_store.collection.load()；2) 配置 Milvus 的预加载参数 collection.load(replica_number=1)；3) 应用启动后发送一个预热查询空转。CorpAssist 方案：lifespan 中 load() 集合 + 每 5 分钟 health check 自动预热。

场景：上传了 100 个文档后，查询结果突然变差（Miss），可能的原因？

可能原因：1) IngestionPipeline 处理这些文档时 chunk_size 不合理（太大导致语义混杂，或太小导致切碎上下文）；2) 新文档的 metadata 缺少 tenant_id filter 字段，但查询时加了 filter 导致无法检索到；3) 文档内容与之前差异大（如纯英文文档混入中文索引）。排查：检查新文档的 Node.metadata 是否完整；对比新文档与旧文档的检索得分分布。修复：调整 chunk_size（从 512 调到 256），检查 metadata 注入逻辑。
