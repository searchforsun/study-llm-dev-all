带监控的 Agent 演示服务

python 
# demo_app.py — Agent 演示服务（带流式输出和监控）
from fastapi import FastAPI
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
from langchain_openai import ChatOpenAI
from langchain.agents import create_react_agent, AgentExecutor, tool
from langchain_core.prompts import PromptTemplate
from typing import AsyncGenerator
import json
import time

app = FastAPI(title="CorpAssist Agent 演示")

class DemoRequest(BaseModel):
    question: str
    max_steps: int = 10

class StepMetrics(BaseModel):
    step: int
    thought: str
    action: str
    action_input: str
    observation: str
    latency_ms: int
    tokens_used: int

# 监控数据
demo_metrics: list[StepMetrics] = []

@tool
def query_inventory(sku: str) -> str:
    """查询商品库存"""
    stock = {"SKU-001": 150, "SKU-002": 32, "SKU-003": 0}
    return json.dumps({"sku": sku, "stock": stock.get(sku, 0)})

@tool
def query_price(sku: str) -> str:
    """查询商品价格"""
    prices = {"SKU-001": 299.00, "SKU-002": 159.00, "SKU-003": 89.00}
    return json.dumps({"sku": sku, "price": prices.get(sku, 0)})

tools = [query_inventory, query_price]

def build_agent():
    llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
    prompt = PromptTemplate.from_template(
        "你是一个演示 Agent。\n工具: {tools}\n问题: {input}\n{agent_scratchpad}"
    )
    agent = create_react_agent(llm, tools, prompt)
    return AgentExecutor(
        agent=agent, tools=tools,
        max_iterations=10, return_intermediate_steps=True,
        handle_parsing_errors=True, verbose=True,
    )

agent_executor = build_agent()

async def stream_demo(question: str) -> AsyncGenerator[str, None]:
    """流式输出 Agent 执行过程"""
    start_time = time.time()
    step = 0

    async for event in agent_executor.astream_events(
        {"input": question},
        version="v2",
    ):
        kind = event["event"]
        if kind == "on_chat_model_start":
            step += 1
            yield f"data: {json.dumps({'type': 'step', 'step': step})}\n\n"
        elif kind == "on_tool_start":
            yield f"data: {json.dumps({'type': 'tool_call', 'tool': event['name'], 'input': str(event['data'].get('input', ''))})}\n\n"
        elif kind == "on_tool_end":
            yield f"data: {json.dumps({'type': 'tool_result', 'output': str(event['data'].get('output', ''))})}\n\n"
        elif kind == "on_chain_end":
            if "output" in event["data"]:
                yield f"data: {json.dumps({'type': 'final', 'output': event['data']['output']})}\n\n"

    elapsed = time.time() - start_time
    yield f"data: {json.dumps({'type': 'metrics', 'steps': step, 'latency_ms': int(elapsed * 1000)})}\n\n"

@app.post("/demo/ask")
async def demo_ask(req: DemoRequest):
    """非流式演示"""
    result = agent_executor.invoke({"input": req.question})
    steps = len(result.get("intermediate_steps", []))
    return {"answer": result["output"], "steps": steps}

@app.post("/demo/stream")
async def demo_stream(req: DemoRequest):
    """流式演示"""
    return StreamingResponse(
        stream_demo(req.question),
        media_type="text/event-stream",
    )

@app.get("/demo/metrics")
async def get_metrics():
    """获取演示监控数据"""
    return {"total_calls": len(demo_metrics), "metrics": demo_metrics[-50:]}

步数与 Token 监控

Token 预算 管理是 Agent 生产化的重要环节。演示时实时监控步数和 Token 消耗，帮助快速发现异常行为。

python 
# metrics_monitor.py — 步数和 Token 监控
from dataclasses import dataclass
from typing import List
import time

@dataclass
class AgentRunMetrics:
    """Agent 运行指标"""
    run_id: str
    question: str
    steps: int = 0
    tools_called: List[str] = None
    total_tokens: int = 0
    prompt_tokens: int = 0
    completion_tokens: int = 0
    latency_ms: int = 0
    status: str = "running"

    def __post_init__(self):
        self.tools_called = self.tools_called or []

class MetricsCollector:
    """运行指标收集器"""

    def __init__(self):
        self.runs: List[AgentRunMetrics] = []

    def start_run(self, run_id: str, question: str) -> AgentRunMetrics:
        metrics = AgentRunMetrics(run_id=run_id, question=question)
        self.runs.append(metrics)
        return metrics

    def record_step(self, run_id: str, tool_name: str, tokens: int = 0):
        for run in self.runs:
            if run.run_id == run_id:
                run.steps += 1
                run.tools_called.append(tool_name)
                run.total_tokens += tokens
                break

    def finish_run(self, run_id: str, latency_ms: int):
        for run in self.runs:
            if run.run_id == run_id:
                run.latency_ms = latency_ms
                run.status = "completed"
                break

    def summary(self) -> dict:
        if not self.runs:
            return {"avg_steps": 0, "avg_tokens": 0}
        avg_steps = sum(r.steps for r in self.runs) / len(self.runs)
        avg_tokens = sum(r.total_tokens for r in self.runs) / len(self.runs)
        return {
            "total_runs": len(self.runs),
            "avg_steps": round(avg_steps, 1),
            "avg_tokens": int(avg_tokens),
            "avg_latency_ms": int(sum(r.latency_ms for r in self.runs) / len(self.runs)),
        }

常见问题（FAQ）

Agent 演示时卡在某个步数不动了怎么办？
预设 max_iterations=10，超限后返回当前信息。同时在 UI 显示进度「步骤 3/10」，让观众知道 Agent 在思考。如果长期不动，后端设置 30 秒超时，超时后返回错误并提示「正在重试」。
现场演示时 API Key 泄露了怎么办？
使用临时 API Key（仅演示期间有效，额度上限 10 元）。不要在代码中硬编码 API Key，用环境变量或演示专用的密钥管理服务。演示结束后立即吊销。
录屏的最佳方案是什么？
推荐 OBS Studio + 本地录制。设置：1920x1080, 30fps, MP4 格式。先录一次完整的 Agent 运行过程，再用剪辑工具加速等待时间。关键步骤保留原速，工具调用过程加速 2x。在视频中加入字幕说明当前正在做什么。

本章结论

  • 展示过程 > 展示结果：流式输出 Agent 的 Thought-Action-Observation，让观众理解 Agent 如何思考。
  • 步数/Token 监控：实时展示每一步的消耗，帮助发现异常行为和优化瓶颈。
  • 预案准备：备选模型、录屏软件、离线版、PDF 备用——确保演示不翻车。
  • 透明演示：用 SSE 流式输出 + Streamlit UI 实现可交互的演示体验。
  • 录屏脚本：提前写好输入和预期路径，避免现场手忙脚乱。
