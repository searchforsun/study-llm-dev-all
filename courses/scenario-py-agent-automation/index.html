本章先记住 3 件事
  • 重试不等于可靠：不设退避的重试会打爆下游。指数退避 + 随机抖动是标准做法。
  • 熔断要快速失败：连续失败 N 次后直接返回错误，不再调用，给下游恢复时间。
  • 最终兜底是人：所有自动化方案都失效时，告警通知人工介入。

重试策略：指数退避与抖动

当工具调用失败时，直接重试往往会让问题更糟——下游系统可能已经过载。正确的做法是指数退避 + 随机抖动。

python 
# retry_strategy.py — 指数退避重试
import asyncio
import random
import time
from functools import wraps
from typing import Callable, Any

def retry_with_backoff(
    max_retries: int = 3,
    base_delay: float = 1.0,
    max_delay: float = 60.0,
    jitter: bool = True,
    retryable_exceptions: tuple = (TimeoutError, ConnectionError),
):
    """指数退避重试装饰器"""
    def decorator(func: Callable) -> Callable:
        @wraps(func)
        async def wrapper(*args, **kwargs) -> Any:
            last_exception = None
            for attempt in range(max_retries + 1):
                try:
                    return await func(*args, **kwargs)
                except retryable_exceptions as e:
                    last_exception = e
                    if attempt == max_retries:
                        raise
                    delay = min(base_delay * (2 ** attempt), max_delay)
                    if jitter:
                        delay = delay * (0.5 + random.random())
                    print(f"[重试] 第 {attempt+1}/{max_retries} 次，等待 {delay:.1f}s")
                    await asyncio.sleep(delay)
            raise last_exception
        return wrapper
    return decorator

# 使用示例
class UnreliableAPIClient:
    @retry_with_backoff(max_retries=3, base_delay=0.5)
    async def call_erp(self, query: str) -> str:
        """不稳定的 ERP 接口"""
        if random.random() < 0.3:
            raise TimeoutError("ERP 超时")
        return f"结果: {query}"

async def test():
    client = UnreliableAPIClient()
    try:
        result = await client.call_erp("SELECT * FROM orders")
        print(f"成功: {result}")
    except TimeoutError:
        print("最终失败")

# 同步版本
def retry_sync(max_retries=3, base_delay=1.0):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(max_retries + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_retries:
                        raise
                    delay = base_delay * (2 ** attempt) * (0.5 + random.random())
                    time.sleep(delay)
            return None
        return wrapper
    return decorator

if __name__ == "__main__":
    asyncio.run(test())

熔断器模式

# circuit_breaker.py — 熔断器
import asyncio
from enum import Enum
from datetime import datetime, timedelta

class CircuitState(Enum):
    CLOSED = "closed"       # 正常
    OPEN = "open"           # 熔断
    HALF_OPEN = "half_open" # 半开（尝试恢复）

class CircuitBreaker:
    """熔断器"""

    def __init__(self, failure_threshold: int = 5, recovery_timeout: int = 30):
        self.state = CircuitState.CLOSED
        self.failure_count = 0
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.last_failure_time: datetime | None = None

    async def call(self, func, *args, **kwargs):
        if self.state == CircuitState.OPEN:
            if datetime.utcnow() - self.last_failure_time > timedelta(seconds=self.recovery_timeout):
                self.state = CircuitState.HALF_OPEN
                print("[熔断器] 进入半开状态，尝试恢复")
            else:
                raise CircuitBreakerOpenError("熔断器已打开，请求被拒绝")

        try:
            result = await func(*args, **kwargs)
            if self.state == CircuitState.HALF_OPEN:
                self.state = CircuitState.CLOSED
                self.failure_count = 0
                print("[熔断器] 恢复成功，关闭熔断器")
            return result
        except Exception as e:
            self.failure_count += 1
            self.last_failure_time = datetime.utcnow()
            if self.failure_count >= self.failure_threshold:
                self.state = CircuitState.OPEN
                print(f"[熔断器] 熔断打开（连续失败 {self.failure_count} 次）")
            raise e

class CircuitBreakerOpenError(Exception):
    pass

人工兜底与告警体系

当自动化方案全部失效时，必须有人工兜底通道。CorpAssist 的告警体系：

告警级别触发条件通知方式响应要求
P0 危急 Agent 连续失败 > 10 次 / 影响核心业务 电话 + 短信 + IM 15 分钟内响应
P1 严重 熔断器打开 / 单个工具持续不可用 IM + 邮件 30 分钟内响应
P2 警告 Agent 步数超过阈值 / Token 消耗异常 IM 4 小时内处理
P3 通知 检查点写入延迟 / 偶发重试 日志 下一个工作日
可靠性实践 Checklist
  • 每个工具调用都有超时和重试（指数退避）
  • 熔断器保护关键下游（ERP、CRM）
  • Agent 有 max_iterations 和 max_execution_time
  • 降级路径可用（缓存 → 替代工具 → 人工）
  • 健康检查和自动恢复
  • P0-P3 告警配置完整
  • 人工兜底流程已演练过

常见问题（FAQ）

熔断器打开后，半开状态下如何决定是否恢复？
半开状态允许一个请求通过。如果成功，熔断器关闭；如果失败，熔断器重新打开并重启 recovery_timeout 计时。每次失败都增加 recovery_timeout（指数增长），防止频繁切换。
指数退避 + 抖动后，用户等待时间太长怎么办？
给用户即时反馈：「正在重试（第 2/3 次）」，避免用户以为系统卡死。同时在前端显示进度。如果重试超过 2 次，建议切换到降级路径而非继续等待。
人工兜底的具体流程是什么？
P0 告警触发后：1) 值班人员收到电话/短信通知；2) 查看 LangSmith Trace 定位失败原因；3) 手动执行 Agent 该步骤的操作（如查询数据库、发送邮件）；4) 将结果手动注入工作流（update_state）；5) 工作流恢复执行。全程记录在事后复盘文档中。

本章结论

  • 重试用指数退避：base_delay=1s, max_retries=3, jitter 随机化避免惊群效应。
  • 熔断快速失败：5 次连续失败打开熔断器，30s 后尝试半开恢复。
  • 四级降级：正常→限流降级→功能降级→人工兜底。每级有明确的触发条件。
  • P0-P3 告警：按严重程度分四级，每级有对应的通知方式和响应时间要求。
  • 人工兜底是最后防线：自动方案全失效时，值班人员查看 Trace 手动恢复。

# baseline_eval.py — 效果基线评估
import json
import time
from typing import List, Dict
from dataclasses import dataclass, asdict

@dataclass
class EvalResult:
    task_id: str
    question: str
    expected_answer: str
    actual_answer: str
    steps: int
    tokens_used: int
    latency_ms: int
    correct: bool
    error: str = ""

class BaselineEval:
    """评估器：建立效果基线"""

    def __init__(self, agent_executor):
        self.agent = agent_executor
        self.results: List[EvalResult] = []

    async def run_eval(self, test_cases: List[Dict]) -> dict:
        """运行评估"""
        for case in test_cases:
            start = time.time()
            try:
                result = await self.agent.ainvoke({"input": case["question"]})
                actual = result["output"]
                steps = len(result.get("intermediate_steps", []))
                correct = self._check_answer(actual, case["expected"])
            except Exception as e:
                actual = ""
                steps = 0
                correct = False

            elapsed = int((time.time() - start) * 1000)
            self.results.append(EvalResult(
                task_id=case["id"],
                question=case["question"],
                expected_answer=case["expected"],
                actual_answer=actual,
                steps=steps,
                tokens_used=0,  # 实际从 LangSmith 获取
                latency_ms=elapsed,
                correct=correct,
                error=str(e) if 'e' in dir() else "",
            ))
        return self.summarize()

    def _check_answer(self, actual: str, expected: str) -> bool:
        """检查答案是否正确（使用 LLM 评估）"""
        from langchain_openai import ChatOpenAI
        llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
        prompt = f"""判断实际答案是否与预期答案一致。
预期：{expected}
实际：{actual}
只需返回 YES 或 NO。"""
        response = llm.invoke(prompt)
        return "YES" in response.content.upper()

    def summarize(self) -> dict:
        """生成摘要报告"""
        total = len(self.results)
        correct = sum(1 for r in self.results if r.correct)
        avg_steps = sum(r.steps for r in self.results) / total if total else 0
        return {
            "total": total,
            "accuracy": round(correct / total * 100, 1) if total else 0,
            "avg_steps": round(avg_steps, 1),
            "avg_latency_ms": int(sum(r.latency_ms for r in self.results) / total),
        }

    def report(self) -> str:
        """生成可阅读的 Markdown 报告"""
        summary = self.summarize()
        report = f"""# Agent 效果评估报告

## 总览
- 测试用例数: {summary['total']}
- 准确率: {summary['accuracy']}%
- 平均步数: {summary['avg_steps']}
- 平均延迟: {summary['avg_latency_ms']}ms

## 失败案例分析
"""
        for r in self.results:
            if not r.correct:
                report += f"\n### {r.task_id}: {r.question}\n"
                report += f"- 预期: {r.expected_answer[:100]}...\n"
                report += f"- 实际: {r.actual_answer[:100]}...\n"
                report += f"- 错误: {r.error}\n"
        return report
不推荐：无基线优化
  • 上线前没跑评估，不知道 Agent 实际准确率
  • 优化靠感觉改 Prompt，无效果验证
  • 出问题后无法量化影响范围
  • 向领导汇报只能说「效果不错」
推荐：基线驱动优化
  • 上线前跑 200 条测试用例建立基线
  • 每次优化后重新评估，对比基线看提升
  • 准确率从 60% → 85% 的叙事有数据支撑
  • 汇报时说「准确率提升 25 个百分点，步数降低 40%」

class CostCalculator:
    """Agent 运营成本计算"""

    INPUT_TOKEN_PRICE = 0.15 / 1_000_000   # gpt-4o-mini 输入
    OUTPUT_TOKEN_PRICE = 0.60 / 1_000_000  # gpt-4o-mini 输出

    def estimate_monthly(self, tasks_per_month: int, avg_input_tokens: int, avg_output_tokens: int) -> dict:
        token_cost = tasks_per_month * (avg_input_tokens * self.INPUT_TOKEN_PRICE + avg_output_tokens * self.OUTPUT_TOKEN_PRICE)
        vector_store = 500  # Chroma 月费

            "token_cost": round(token_cost, 2),
            "vector_store": vector_store,
            "infrastructure": infrastructure,
            "human_oversight": round(human_oversight, 2),
            "total_monthly": round(token_cost + vector_store + infrastructure + human_oversight, 2),
        }

常见问题（FAQ）

Agent 效果评估用什么做 ground truth？
三种来源：1) 人工标注——让业务方标注 200 条数据作为标准答案；2) 历史数据——已有的正确的处理结果；3) LLM 自评——用更强模型评估弱模型的结果。CorpAssist 使用人工标注 + LLM 自评双重校验，不一致时人工裁定。
Agent 的准确率和人工的准确率怎么公平对比？
用同一组测试集（至少 200 条），分别让人工和 Agent 处理，对比结果。注意：人工的「准确率」不是 100%——人工也会犯错（CorpAssist 统计人工发票处理的错误率约 8%）。Agent 的目标是超越人工基线，而不是追求 100%。
SLA 定 95% 还是 99%？
取决于成本。从 95% 提升到 99% 可能需要 2x 的成本（更贵的模型、更多的验证步骤、更多的人工复核）。建议先定 90%，稳定运行 1 个月后再评估是否提升到 95%。不要一开始就追求 99%——成本可能无法承受。

本章结论

  • 指标五维度：准确率、步数、Token、TTFT、P99。上线前先建立基线。
  • 基线驱动优化：每次修改后重新评估，对比基线数据看效果。没用基线优化的都是玄学。
  • 成本透明：Token + 向量存储 + 基础设施 + 人工兜底，每月运营成本可精确计算。
  • SLA 事前约定：准确率、延迟、可用性目标在上线前和业务方对齐。
  • 效果叙事：从 X 到 Y 的数据叙事比「效果不错」有说服力 100 倍。

# compliance.py — CorpAssist 合规检查器
from pydantic import BaseModel, Field
from typing import Optional
import re

class ComplianceChecker:
    """合规检查：PII 检测 + 操作白名单 + 审计日志写入"""

    SENSITIVE_PATTERNS = [
        r'\d{18}(?:\d|[Xx])',       # 身份证号
        r'\d{16,19}',               # 银行卡号
        r'1[3-9]\d{9}',             # 手机号
    ]

    WRITE_OPERATIONS = {
        "erp_write", "payment_execute", "data_delete",
        "permission_change", "email_send",
    }

    BANNED_OPERATIONS = {
        "payment_execute", "data_delete", "permission_change",
    }

    @classmethod
    def check_pii(cls, text: str) -> list[dict]:
        """检测文本中的敏感信息"""
        found = []
        for pattern in cls.SENSITIVE_PATTERNS:
            matches = re.finditer(pattern, text)
            for m in matches:
                found.append({
                    "type": "PII",
                    "pattern": pattern,
                    "position": m.span(),
                    "preview": text[max(0, m.start()-10):m.end()+10],
                })
        return found

    @classmethod
    def check_operation(cls, tool_name: str) -> dict:
        """检查工具调用是否合规"""
        if tool_name in cls.BANNED_OPERATIONS:
            return {"allowed": False, "reason": f"{tool_name} 是禁止操作，需人工审批"}
        if tool_name in cls.WRITE_OPERATIONS:
            return {"allowed": True, "requires_approval": True}
        return {"allowed": True, "requires_approval": False}

    @classmethod
    def mask_pii(cls, text: str) -> str:
        """脱敏处理"""
        for pattern in cls.SENSITIVE_PATTERNS:
            text = re.sub(pattern, lambda m: m.group()[:4] + "****" + m.group()[-4:], text)
        return text

# 使用示例
if __name__ == "__main__":
    text = "请将款项转入6228480012345678，用户身份证号 110101199001011234"
    pii = ComplianceChecker.check_pii(text)
    print(f"检测到 {len(pii)} 处敏感信息")
    masked = ComplianceChecker.mask_pii(text)
    print(f"脱敏后: {masked}")

    result = ComplianceChecker.check_operation("payment_execute")
    print(f"操作检查: {result}")

常见问题（FAQ）

Agent 自动化和传统的 RPA 有什么区别？
RPA（机器人流程自动化）基于规则和屏幕抓取，适合固定步骤、结构化操作。Agent 自动化基于 LLM 推理，能处理非结构化输入（发票图片、自由文本）和动态决策（异常路由、多步规划）。两者互补：RPA 做确定性操作，Agent 做需要判断的环节。CorpAssist 常将 Agent 作为 RPA 的"大脑"来驱动 RPA 执行。
Agent 出错时谁担责？合规上怎么设计？
企业部署 Agent 必须建立人机协作的责任边界：高风险操作（付款、删除数据）必须经人工审批，Agent 不能独立执行。低风险操作（信息提取、格式转换）Agent 可独立执行但需审计日志。推荐在 SLA 中明确「Agent 辅助决策，人类负最终责任」。
部署 Agent 需要企业 IT 做哪些前置准备？
三件事：1) API 网关暴露内部系统接口（ERP、OA、CRM），支持 OAuth2 或 API Key 鉴权；2) 确定数据合规边界——哪些字段可以发给云端 LLM，哪些必须本地处理；3) 准备测试数据集——至少 100 条真实脱敏数据用于验证 Agent 准确率。
小公司没有 ERP API，能做 Agent 自动化吗？
可以。CorpAssist 的 多工具协作 模式支持通过邮件、Excel、Web 界面作为输入/输出。例如，Agent 从邮件附件提取数据 → 写入 Google Sheets → 发送摘要邮件。不需要 ERP 也能跑通端到端流程。

本章结论

  • 办公自动化四大场景：文档处理、数据搬运、审批催办、报告生成，覆盖 70% 的企业重复操作。
  • 合规三条红线：PII 不出境、操作不绕过审批、审计日志不可篡改。每条红线都有对应的架构应对方案。
  • ROI 要算全口径：LLM API + 基础设施 + 人工兜底 + 开发摊销 vs 人力节省 + 错误减少收益。Agent 准确率不足 90% 时 ROI 可能为负。

为什么要学本章

ReAct (Reasoning + Acting) 是 Agent 最核心的设计模式——它让 LLM 不只是回答问题，而是能思考-行动-观察地完成任务。CorpAssist 的发票处理、数据同步、审批催办等 Agent 都基于 ReAct 模式构建。不理解 ReAct，就无法理解 Agent。

学完你能

  • 画出 ReAct 循环的完整流程图，标注 Thought → Action → Observation → ... → Final Answer
  • 配置 Agent 的 max_iterations、early_stopping_method 等关键参数
  • 编写自定义工具函数并通过 tool decorator 注册到 Agent
  • 理解 ReAct 的局限：幻觉、循环、工具选择错误及其应对
本章先记住 3 件事
  • ReAct = 推理 + 行动：LLM 先生成思考（Thought），再决定调用什么工具（Action），看到结果后继续推理。
  • 工具是 Agent 的手脚：Agent 本身只做推理，所有对外操作都通过注册的工具函数完成。
  • 无限循环是最大风险：必须设置 max_iterations 和 early_stopping，否则 Agent 可能在工具调用中死循环。

ReAct 模式原理：思考-行动-观察循环

ReAct 模式由 Shunyu Yao 在 2022 年提出，核心思想是将推理（Reasoning）和行动（Acting）交织进行。LLM 在每一步生成结构化的思维链，决定是否需要调用外部工具，观察结果后续推，最终给出答案。

CorpAssist · 数据同步 Agent 某零售企业需要跨系统同步商品信息：从 ERP 系统查询商品价格 → 从仓储系统查库存 → 从物流系统查配送状态 → 汇总到一张报表。Agent 需要分多步执行，每步依赖上一步的结果。

ReAct 循环中，LLM 交替生成 Thought 和 Action，工具返回 Observation，直到 LLM 判断可以给出 Final Answer。

ReAct Agent 的最小实现

python 
# react_agent.py — CorpAssist ReAct Agent 最小实现
from langchain_openai import ChatOpenAI
from langchain.agents import create_react_agent, AgentExecutor
from langchain.agents import tool
from langchain_core.prompts import PromptTemplate

# 1. 定义工具函数
@tool
def query_price(sku: str) -> str:
    """查询商品价格，输入 SKU 编码，返回价格信息"""
    price_map = {"SKU-001": "299.00", "SKU-002": "159.00", "SKU-003": "89.00"}
    return price_map.get(sku, "未知商品")

@tool
def query_stock(sku: str) -> str:
    """查询商品库存，输入 SKU 编码，返回库存数量"""
    stock_map = {"SKU-001": "150", "SKU-002": "32", "SKU-003": "0"}
    return stock_map.get(sku, "未知商品")

# 2. 构建 ReAct Prompt
REACT_PROMPT = PromptTemplate.from_template(
    "你是一个企业数据助手 CorpAssist。请使用可用工具查询信息并回答问题。\n\n"
    "可用工具：\n{tools}\n\n"
    "工具名称：{tool_names}\n"
    "用户问题：{input}\n\n"
    "请按以下格式回答：\n"
    "Thought: 下一步要做什么\n"
    "Action: 工具名称\n"
    "Action Input: 工具输入参数\n"
    "Observation: 工具返回结果\n"
    "...（重复 Thought/Action/Observation）\n"
    "Thought: 我已收集到足够信息\n"
    "Final Answer: 最终答案\n\n"
    "历史记录：\n{agent_scratchpad}"
)

# 3. 创建 Agent
def create_corpassist_agent():
    llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
    tools = [query_price, query_stock]
    agent = create_react_agent(llm=llm, tools=tools, prompt=REACT_PROMPT)
    agent_executor = AgentExecutor(
        agent=agent, tools=tools,
        max_iterations=5, early_stopping_method="force",
        handle_parsing_errors=True, verbose=True,
    )
    return agent_executor

if __name__ == "__main__":
    agent = create_corpassist_agent()
    result = agent.invoke({"input": "SKU-001 的商品价格和库存是多少？"})
    print(f"答案：{result['output']}")
不推荐
  • 不给 Agent 设置 max_iterations，默认无限循环
  • 工具函数返回空字符串或不规范格式
  • 用字符串拼接构造 Prompt，无结构化模板
  • 不处理 parsing_errors，Agent 解析失败时静默崩溃
推荐
  • 设置 max_iterations=5~15，配合 early_stopping_method
  • 工具函数返回格式化的 JSON 字符串，含错误信息
  • 使用 PromptTemplate 结构化模板，维护方便
  • handle_parsing_errors=True，输出格式错误时自动重试
先想 10 秒：max_iterations=5 时如果 Agent 在第 5 步还没收集全信息会怎样？

AgentExecutor 会强制停止并返回当前收集到的信息作为 Final Answer。生产环境推荐 "force" 避免无限循环。

生产级 Agent 配置

python 
# advanced_agent.py — 生产级 Agent 配置
from langchain_openai import ChatOpenAI
from langchain.agents import create_react_agent, AgentExecutor, tool
from langchain_core.prompts import PromptTemplate
from langchain_core.callbacks import BaseCallbackHandler

class AgentMonitorCallback(BaseCallbackHandler):
    """Agent 运行时监控回调"""
    def on_agent_action(self, action, **kwargs) -> None:
        print(f"[Agent Action] {action.tool}({action.tool_input})")
    def on_agent_finish(self, finish, **kwargs) -> None:
        print(f"[Agent Finish] {finish.return_values['output'][:100]}...")

def create_production_agent():
    llm = ChatOpenAI(model="gpt-4o-mini", temperature=0, max_tokens=4096)
    @tool
    def query_erp(sql: str) -> str:
        """执行 ERP 只读查询，输入 SQL 返回 JSON 结果"""
        if sql.strip().upper().startswith(("SELECT", "WITH")):
            return '[{"id": 1, "name": "商品A", "price": 299.00}]'
        return '{"error": "仅支持只读查询"}'
    @tool
    def send_notification(recipient: str, message: str) -> str:
        """发送系统通知"""
        return f"通知已发送给 {recipient}"
    tools = [query_erp, send_notification]
    agent = create_react_agent(llm, tools, PromptTemplate.from_template(
        "你是一个 CorpAssist 企业数据助手。\n工具：{tools}\n工具名称：{tool_names}\n问题：{input}\n\n{agent_scratchpad}"
    ))
    executor = AgentExecutor(
        agent=agent, tools=tools,
        max_iterations=10, max_execution_time=120,
        early_stopping_method="force", handle_parsing_errors=True,
        return_intermediate_steps=True, trim_intermediate_steps=50,
        callbacks=[AgentMonitorCallback()],
    )
    return executor

常见问题（FAQ）

ReAct Agent 的 Action 输出格式必须是 JSON 吗？
LangChain 的 create_react_agent 期望的 Action 格式是 Action: 工具名称\nAction Input: 输入参数——纯文本格式。工具函数的输入输出推荐用 JSON，方便 LLM 解析。如果使用 OpenAI function calling，底层走的是 Tool 调用协议，格式由 API 自动处理。
为什么我的 Agent 总是在第 3 步就停止，明明还有工具需要调用？
检查三个配置：1) max_iterations 是否过小；2) Prompt 中有没有「尽快结束」的隐式指令；3) 工具返回的 Observation 是否让 LLM 误以为任务已完成。推荐在 Prompt 中加入「思考完成后，请逐一调用所有需要的工具」。
ReAct 能并行调用多个工具吗？
标准的 ReAct 是串行的——一次只能调用一个工具。如果需要并行（如同时查价格和库存），需要在工具函数内部做批量操作，或者使用 LangGraph 的 parallel 节点（见 practice-02 多工具协作）。
如何防止 Agent 调用不该调用的工具？
在工具层实现权限检查。不要在 Agent 层做权限控制——Prompt 可能被注入绕过。在工具函数内部调用 工具沙箱 做权限校验，同时将高度敏感的操作从工具列表中彻底移除。

本章结论

  • ReAct = 推理 + 行动交织：Agent 交替进行 Thought → Action → Observation 循环，直到收集足够信息给出 Final Answer。
  • 工具是 Agent 的边界：Agent 只能通过注册的工具函数与外部交互，工具描述质量决定 Agent 行为准确性。
  • 生产必须设限：max_iterations、max_execution_time、handle_parsing_errors 是防止 Agent 失控的三大护栏。
  • ReAct 适合中短流程：步数 < 20 时效果最佳，复杂长流程推荐 LangGraph。
  • 工具描述要精确：包含输入输出格式、单位、边界条件、错误示例。

本章先记住 3 件事

规划策略：CoT 与 Plan-and-Solve

任务规划的核心问题是：如何将用户的一句话需求拆解为可执行的子任务序列。两种主流策略：

策略思路适用场景步数效率
Chain-of-Thought (CoT) 在 ReAct 的 Thought 中逐步推理，边想边做 步数少（< 10）、步间依赖紧密 中
Plan-and-Solve (P&S) 先一次性产出完整计划，再逐个执行 步数多（10~50）、子任务相对独立 高
ReAct + 重规划 边执行边调整计划，每 N 步重新评估 动态环境、外部输入不可预测 低但鲁棒

  • Agent 边走边想，每次调用都损失上下文
  • 无法识别可并行步骤，串行执行浪费时间

  • 可预估总 Token 消耗和完成时间

执行-观察-反思循环

执行-观察-反思（Execute-Observe-Reflect） 是提升规划质量的关键机制。Agent 在执行过程中定期暂停并反思：当前进度是否符合预期？是否需要调整计划？

    Check -->|否| Reflect[反思原因]
    Reflect --> Adjust[调整计划]
    Adjust --> Execute
    Next --> Done{所有任务完成?}
    Done -->|否| Execute
    Done -->|是| Final[输出最终结果]

带反思的规划 Agent 实现

python 
# reflect_agent.py — 带反思的规划 Agent
from langchain_openai import ChatOpenAI
from pydantic import BaseModel, Field
from typing import List, Optional, Literal
import json

class SubTask(BaseModel):
    """子任务定义"""
    step_id: int
    description: str
    tool: str
    depends_on: List[int] = Field(default_factory=list)
    status: Literal["pending", "running", "done", "failed"] = "pending"

class TaskPlan(BaseModel):
    """完整任务计划"""
    goal: str
    subtasks: List[SubTask]

class ReflectAgent:
    """带执行-观察-反思的规划 Agent"""

    def __init__(self, model: str = "gpt-4o-mini"):
        self.llm = ChatOpenAI(model=model, temperature=0)
        self.plan: Optional[TaskPlan] = None
        self.results: dict[int, str] = {}

    def plan_task(self, goal: str) -> TaskPlan:
        """第一步：LLM 生成任务计划"""
        prompt = f"""将以下目标分解为子任务列表，以 JSON 格式返回：
目标：{goal}
要求：
- 每个子任务包含 step_id, description, tool, depends_on
- depends_on 是依赖的上一步 step_id 列表
- 无依赖的任务列在 depends_on: []
- 返回格式：{{"goal": "...", "subtasks": [...]}}"""
        response = self.llm.invoke(prompt)
        data = json.loads(response.content)
        self.plan = TaskPlan(**data)
        return self.plan

    def execute_step(self, step: SubTask) -> str:
        """执行单个子任务（模拟）"""
        print(f"[执行] 步骤 {step.step_id}: {step.description}")
        # 实际工程中调用对应工具
        return f"步骤 {step.step_id} 完成: {step.description}"

    def reflect(self, step: SubTask, result: str) -> Optional[str]:
        """反思步骤：检查结果是否需要调整计划"""
        prompt = f"""你正在执行以下任务计划：
目标：{self.plan.goal}
当前步骤：{step.step_id} - {step.description}
执行结果：{result}

请反思：
1. 结果是否与预期一致？
2. 是否需要调整后续计划？
3. 是否有新的依赖关系出现？
回答格式：{{"consistent": bool, "adjustment": "调整内容或null"}}"""
        response = self.llm.invoke(prompt)
        data = json.loads(response.content)
        return data.get("adjustment")

    def run(self, goal: str) -> dict[int, str]:
        """执行完整规划+执行+反思流程"""
        print(f"[规划] 目标: {goal}")
        self.plan_task(goal)
        print(f"[规划] 共 {len(self.plan.subtasks)} 个子任务")

        for step in self.plan.subtasks:
            result = self.execute_step(step)
            self.results[step.step_id] = result

            adjustment = self.reflect(step, result)
            if adjustment:
                print(f"[反思] 计划调整: {adjustment}")
                # 此处可修改 self.plan.subtasks
        return self.results

if __name__ == "__main__":
    agent = ReflectAgent()
    results = agent.run("查询 Q2 销售数据，生成 PPT 报告并邮件发送")
    for step_id, result in results.items():
        print(f"  步骤 {step_id}: {result[:60]}...")
先想 10 秒：如果子任务 2 执行失败，子任务 3 依赖它的结果，你怎么设计恢复策略？

三种策略：1) 重试：子任务 2 最多重试 3 次，每次间隔指数退避；2) 降级：子任务 3 使用子任务 2 的部分结果或默认值继续；3) 重规划：通知规划器重新调整任务分解。生产环境推荐组合使用：先重试，失败则降级，降级不满足业务需求时触发重规划。

Token 预算与步数控制

规划 Agent 的 Token 预算 管理是生产化的核心问题。每步 Thought/Action/Observation 都会增加 Token 消耗。

策略Token 消耗/步10 步总消耗优缺点
CoT (边想边做) ~500 ~5,000 灵活但消耗大
Plan-and-Solve ~200 ~2,000 高效但不够灵活
ReAct + 重规划 ~350 ~3,500 灵活性和效率折中
CorpAssist 实践建议

对于 5~15 步的中等复杂度任务，推荐 Plan-and-Solve + 每 5 步触发一次反思。这样既能保持效率，又能应对执行中的偏差。设置 max_total_tokens=8000 作为安全上限。

常见问题（FAQ）

Plan-and-Solve 和 LangGraph 是什么关系？
Plan-and-Solve 是策略层概念——它描述 Agent 先规划再执行的行为模式。LangGraph 是实现层框架——它提供了定义有状态工作流的图结构。你可以用 LangGraph 实现 Plan-and-Solve：规划阶段用一个 LLM 节点，执行阶段用一串工具节点。实际上，下一章的 practice-01 正是用 LangGraph 实现可检查点的任务规划工作流。
如果 LLM 生成的计划有 20 步，但实际只需要 5 步，怎么办？
这是「过度分解」问题。解决方法：1) 在规划 Prompt 中给出步数范围（「请分解为 3-8 个子任务」）；2) 在反思阶段检查是否有子任务可以合并；3) 使用动态规划——先分解为粗粒度子任务，执行时再进一步细分。CorpAssist 的实践经验是：粗规划 + 细执行 = 最佳平衡。
Agent 规划错了步骤（比如第三步其实应该放在第一步），怎么处理？
分两种场景：1) 执行前发现——可以用另一轮 LLM 调用做「计划验证」，让第二个 LLM 检查计划正确性；2) 执行后失败才发现——依赖检测机制。当子任务 B 依赖子任务 A 的结果，但 A 还没执行时，B 的 Observation 会报错「缺少依赖数据」，触发重规划。推荐在工具函数中做前置条件检查。

本章结论

  • 任务规划三策略：CoT（边想边做）、Plan-and-Solve（先计划再执行）、ReAct+重规划（动态调整），按场景复杂度选择。
  • DAG 是规划核心：子任务的依赖关系用有向无环图表示，决定串行/并行边界。
  • 执行-观察-反思：定期暂停并检查进度，发现偏差时重规划，提升鲁棒性。
  • Token 预算管理：CoT 最耗 Token，Plan-and-Solve 最省。设置 max_iterations 和 max_total_tokens 防止失控。
  • 过度分解要避免：粗规划 + 细执行，用反思阶段合并可合并的子任务。

from langchain.memory import (
    ConversationBufferMemory,
    ConversationSummaryMemory,
    VectorStoreRetrieverMemory,
)
from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings
from langchain.schema import Document
from typing import List, Optional
import sqlite3
import json

class MemoryManager:
    """三层记忆管理器"""

    def __init__(self, user_id: str, session_id: str):
        self.user_id = user_id
        self.session_id = session_id

        # 第一层：短期缓冲记忆 (最后 2000 tokens)
        self.buffer_memory = ConversationBufferMemory(
            memory_key="chat_history",
            return_messages=True,
            max_token_limit=2000,
        )

        # 第二层：会话摘要记忆
        self.summary_memory = ConversationSummaryMemory(
            llm=ChatOpenAI(model="gpt-4o-mini", temperature=0),
            memory_key="summary",
            return_messages=True,
        )

        # 第三层：长期向量记忆
        embeddings = OpenAIEmbeddings(model="text-embedding-3-small", dimensions=512)
        vector_store = Chroma(
            collection_name=f"user_memory_{user_id}",
            embedding_function=embeddings,
            persist_directory="./data/user_memory",
        )
        self.long_term_memory = VectorStoreRetrieverMemory(
            retriever=vector_store.as_retriever(search_kwargs={"k": 3}),
            memory_key="long_term",
            return_docs=True,
        )

    def save_context(self, user_input: str, agent_response: str):
        """保存对话上下文到所有记忆层"""
        # 短期记忆
        self.buffer_memory.save_context(
            {"input": user_input}, {"output": agent_response}
        )
        # 会话摘要（当缓冲溢出时自动压缩）
        self.summary_memory.save_context(
            {"input": user_input}, {"output": agent_response}
        )
        # 长期记忆（提取关键信息存入向量库）
        self._extract_and_save_entities(user_input, agent_response)

    def _extract_and_save_entities(self, user_input: str, response: str):
        """提取关键实体并存入长期记忆"""
        # 实际工程中用 LLM 或 NER 模型提取实体
        extract_prompt = f"从以下对话中提取关键实体（姓名、编号、偏好等），返回 JSON：\n用户：{user_input}\n助手：{response}"
        # 模拟提取结果
        entities = {"name": "张三", "customer_id": "CUST-001"}
        doc = Document(
            page_content=json.dumps(entities, ensure_ascii=False),
            metadata={"user_id": self.user_id, "session": self.session_id},
        )
        self.long_term_memory.save_context(
            {"input": str(entities)}, {"output": "saved"}
        )

    def load_memory(self) -> dict:
        """加载所有层级的记忆"""
        buffer = self.buffer_memory.load_memory_variables({})
        summary = self.summary_memory.load_memory_variables({})
        long_term = self.long_term_memory.load_memory_variables({"input": ""})
        return {
            "short_term": buffer.get("chat_history", []),
            "summary": summary.get("summary", ""),
            "long_term": long_term.get("long_term", []),
        }

    def clear_session(self):
        """清除会话级记忆但保留长期记忆"""
        self.buffer_memory.clear()
        self.summary_memory.clear()
不推荐：无限制追加记忆
  • 每轮对话全部追加到 Prompt，Token 快速增长
  • 超过上下文窗口后直接截断，丢失早期关键信息
  • 不区分会话内和跨会话记忆
  • 每次重连丢失所有上下文
推荐：三层分层管理
  • Buffer 窗口 2000 tokens，溢出后压缩为摘要
  • 摘要+向量检索组合提供跨会话记忆
  • 实体记忆持久化用户偏好和关键信息
  • 会话结束后短期记忆释放，长期记忆保留

工具生态：注册、发现与错误处理

工具是 Agent 的能力边界。一个好的工具生态需要解决三个问题：注册（如何新增工具）、发现（Agent 如何找到合适工具）、容错（工具失败时怎么办）。

工具注册与发现系统

python 
# tool_registry.py — 工具注册表
from langchain.agents import tool
from pydantic import BaseModel, Field
from typing import Callable, Dict, List, Optional, Any
import time
import functools

class ToolSpec(BaseModel):
    """工具规格定义"""
    name: str
    description: str
    parameters: Dict[str, str]
    requires_approval: bool = False
    timeout: int = 30
    retry_count: int = 2
    fallback: Optional[str] = None  # 降级时使用的工具名称

class ToolRegistry:
    """可扩展的工具注册表"""

    def __init__(self):
        self._tools: Dict[str, ToolSpec] = {}
        self._functions: Dict[str, Callable] = {}

    def register(self, spec: ToolSpec):
        """注册工具"""
        self._tools[spec.name] = spec
        print(f"[注册] 工具 {spec.name}: {spec.description[:50]}...")

    def register_function(self, name: str, func: Callable):
        """注册工具函数实现"""
        self._functions[name] = func

    def get_tool_list(self) -> List[ToolSpec]:
        """获取所有可用工具列表"""
        return list(self._tools.values())

    def execute(self, tool_name: str, **kwargs) -> dict:
        """执行工具，包含重试和降级"""
        spec = self._tools.get(tool_name)
        if not spec:
            return {"success": False, "error": f"工具 {tool_name} 不存在"}

        func = self._functions.get(tool_name)
        if not func:
            return {"success": False, "error": f"工具 {tool_name} 未实现"}

        # 重试逻辑
        last_error = None
        for attempt in range(spec.retry_count + 1):
            try:
                start = time.time()
                result = func(**kwargs)
                elapsed = time.time() - start
                return {"success": True, "result": result, "elapsed": elapsed}
            except Exception as e:
                last_error = str(e)
                if attempt < spec.retry_count:
                    time.sleep(1 * (attempt + 1))  # 指数退避
                    continue

        # 降级逻辑
        if spec.fallback:
            print(f"[降级] {tool_name} 失败，降级到 {spec.fallback}")
            return self.execute(spec.fallback, **kwargs)

        return {"success": False, "error": last_error}

Agent 降级策略

Agent 降级 是工具不可用时的兜底方案。CorpAssist 使用三级降级：

降级级别触发条件行为
L1 - 缓存降级 主 API 超时或返回 5xx 从本地缓存读取最近一次成功的结果
L2 - 替代工具 主工具完全不可用 调用注册表中指定的 fallback 工具
L3 - 告知用户 所有降级路径都失败 Agent 返回「该功能暂时不可用，请稍后再试」
python 
# fallback_handler.py — 降级处理器
from langchain.agents import AgentExecutor, tool
from langchain_openai import ChatOpenAI
from functools import wraps
import time

def with_fallback(fallback_func=None, max_retries=2):
    """工具降级装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(max_retries + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt < max_retries:
                        time.sleep(1 * (attempt + 1))
                        continue
                    if fallback_func:
                        print(f"[降级] {func.__name__} -> {fallback_func.__name__}")
                        return fallback_func(*args, **kwargs)
                    return f"错误: {str(e)}"
        return wrapper
    return decorator

# 使用降级装饰器
@with_fallback(max_retries=2)
@tool
def query_erp(sql: str) -> str:
    """执行 ERP 查询"""
    # 模拟故障
    import random
    if random.random() < 0.3:
        raise TimeoutError("ERP 系统超时")
    return f"[ERP结果] {sql[:50]}..."

@with_fallback
@tool
def query_erp_fallback(sql: str) -> str:
    """ERP 查询降级：从本地快照读取"""
    return f"[缓存] {sql[:50]}...(上次成功结果)"

常见问题（FAQ）

ConversationSummaryMemory 会不会丢失重要细节？
会。摘要记忆的压缩过程本质是有损的——LLM 可能会忽略某些细节。CorpAssist 的应对方案是「摘要+关键词」双通道：摘要记忆保存压缩上下文，同时用实体记忆保存关键数据（订单号、金额、日期等）。用户问细节时从实体记忆检索，用户问整体时用摘要。详见 记忆分层 策略。
工具越来越多，Agent 选择工具的时间会不会变长？
会。LLM 在大量工具中选择时，准确率会下降。推荐：1) 工具数量控制在 20 个以内；2) 用工具分组（Group）管理，Agent 先选组再选工具；3) 使用 多工具协作 的层次化调度。CorpAssist 最多 15 个活跃工具，非活跃工具从 Prompt 中移除。
降级工具返回了过期数据，会不会误导用户？
需要在返回结果中标注数据来源和时效。例如使用缓存降级时，在结果中附加 {"source": "cache", "timestamp": "2026-05-30T10:00:00Z"}，并在生成回答时告知用户「以下数据来自缓存...」。Agent 的 Prompt 应指示：如果数据有延迟标注，需在回答中说明。

本章结论

  • 三层记忆架构：短期缓冲 + 会话摘要 + 长期向量持久化，平衡 Token 消耗和信息保留。
  • 工具注册表：可插拔设计，新增工具无需修改 Agent 代码。YAML 配置 + CI/CD 部署。
  • 三级降级策略：缓存降级 → 替代工具 → 告知用户。每级都有明确的触发条件和行为。
  • 记忆+工具协同：Agent 先查记忆识别人物偏好，再调用工具获取实时数据，最后更新记忆。
  • 工具数量控制：每组不超过 20 个，用分组+层次化调度优化选择准确率。

本章先记住 3 件事
  • 图 = 节点 + 边：节点是处理函数，边是状态转移条件。LangGraph 的核心就是定义一个 StateGraph。
  • 检查点 = 断点续传：工作流每步自动持久化状态，崩溃后可从最后检查点恢复。
  • 人工审批 = 中断等待：在审批节点设置中断，有人确认后才继续执行。

LangGraph 图定义与状态管理

LangGraph 的核心抽象是 StateGraph——一个有向图，其中：

  • State：工作流的全局状态（一个 TypedDict 或 Pydantic BaseModel），在所有节点间传递
  • Node：处理函数，接收 state 返回更新的 state（或部分更新）
  • Edge：节点间的连接，可以是有条件的（condition）
CorpAssist · 报销审批工作流 员工提交报销单后，Agent 需要：① 提取发票关键信息 → ② 验证发票真伪 → ③ 检查预算余额 → ④ 部门经理审批 → ⑤ 财务打款。其中步骤④需要人工审批中断，其他步骤可自动执行。

# expense_workflow.py — LangGraph 报销审批工作流
from typing import TypedDict, Dict, Any, Optional, Literal
from langgraph.graph import StateGraph, END
from langgraph.checkpoint import MemorySaver
from langchain_openai import ChatOpenAI
import json

# 1. 定义状态
class ExpenseState(TypedDict):
    """报销工作流状态"""
    expense_id: str
    employee_id: str
    amount: float
    invoice_text: str          # 原始发票文本
    extracted_info: dict       # LLM 提取的发票信息
    validation_result: dict    # 真伪验证结果
    budget_check: dict         # 预算检查结果
    approval_status: Optional[str]  # None / approved / rejected
    approval_comment: str      # 审批意见
    payment_result: str        # 打款结果
    error: Optional[str]       # 错误信息

# 2. 定义节点函数
def extract_invoice(state: ExpenseState) -> dict:
    """节点1: LLM 提取发票信息"""
    llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
    prompt = f"""从以下发票文本中提取信息，返回 JSON：
发票文本：{state['invoice_text']}
返回字段：invoice_no, date, amount, vendor_name, tax_id"""
    result = llm.invoke(prompt)
    extracted = json.loads(result.content)
    return {"extracted_info": extracted}

def validate_invoice(state: ExpenseState) -> dict:
    """节点2: 验证发票真伪（模拟税务API）"""
    invoice = state["extracted_info"]
    # 真实场景调用国家税务总局 API
    valid = invoice.get("invoice_no", "").startswith("INV")
    return {
        "validation_result": {
            "valid": valid,
            "message": "发票验证通过" if valid else "发票号码异常",
        }
    }

def check_budget(state: ExpenseState) -> dict:
    """节点3: 检查预算余额"""
    # 模拟 ERP 预算检查
    dept_budget_remaining = 50000.0
    if state["amount"] <= dept_budget_remaining:
        return {"budget_check": {"sufficient": True, "remaining": dept_budget_remaining - state["amount"]}}
    return {"budget_check": {"sufficient": False, "remaining": dept_budget_remaining}}

def process_payment(state: ExpenseState) -> dict:
    """节点5: 执行打款"""
    return {"payment_result": f"已向员工 {state['employee_id']} 打款 {state['amount']} 元"}

def process_rejection(state: ExpenseState) -> dict:
    """节点6: 驳回处理"""
    return {"payment_result": f"报销被驳回: {state.get('approval_comment', '无理由')}"}

# 3. 条件边：根据审批状态选择路径
def route_after_approval(state: ExpenseState) -> Literal["process_payment", "process_rejection", "wait"]:
    if state.get("approval_status") == "approved":
        return "process_payment"
    elif state.get("approval_status") == "rejected":
        return "process_rejection"
    return "wait"  # 尚未审批

# 4. 构建图
def build_expense_graph():
    workflow = StateGraph(ExpenseState)

    # 添加节点
    workflow.add_node("extract_invoice", extract_invoice)
    workflow.add_node("validate_invoice", validate_invoice)
    workflow.add_node("check_budget", check_budget)
    workflow.add_node("process_payment", process_payment)
    workflow.add_node("process_rejection", process_rejection)

    # 添加边
    workflow.set_entry_point("extract_invoice")
    workflow.add_edge("extract_invoice", "validate_invoice")
    workflow.add_edge("validate_invoice", "check_budget")
    workflow.add_conditional_edges(
        "check_budget",
        lambda s: "process_payment" if s["budget_check"]["sufficient"] else "process_rejection",
    )
    workflow.add_edge("process_payment", END)
    workflow.add_edge("process_rejection", END)

    # 编译：启用检查点
    checkpointer = MemorySaver()
    app = workflow.compile(checkpointer=checkpointer)
    return app

if __name__ == "__main__":

    # 初始状态

检查点持久化与中断恢复

LangGraph 的检查点（Checkpoint）机制在工作流的每个节点执行后自动持久化状态。如果进程崩溃，可以从最后一个检查点恢复执行。CorpAssist 检查点使用场景：

  • 崩溃恢复：Agent 在第 5 步崩溃，重启后从第 5 步继续
  • 人工审批：工作流在审批节点暂停，等待人工回复后继续
  • 审计追踪：每个检查点记录完整的状态快照，用于事后审计
python 
# checkpoint_demo.py — 检查点持久化与恢复
from langgraph.checkpoint import MemorySaver, SqliteSaver
from langgraph.graph import StateGraph, END
from typing import TypedDict
import sqlite3

class SimpleState(TypedDict):
    messages: list
    step: int

def step_processor(state: SimpleState) -> dict:
    """模拟处理步骤"""
    current = state.get("step", 0) + 1
    return {
        "messages": state["messages"] + [f"完成步骤 {current}"],
        "step": current,
    }

# 使用 SQLite 持久化检查点（生产环境）
conn = sqlite3.connect("checkpoints.db", check_same_thread=False)
checkpointer = SqliteSaver(conn)

# 构建工作流
builder = StateGraph(SimpleState)
builder.add_node("process", step_processor)
builder.set_entry_point("process")
builder.add_conditional_edges(
    "process",
    lambda s: END if s["step"] >= 5 else "process",  # 5 步后结束
)

app = builder.compile(checkpointer=checkpointer)

# 第一次运行
thread_id = "thread-001"
config = {"configurable": {"thread_id": thread_id}}
initial = {"messages": ["开始"], "step": 0}

for event in app.stream(initial, config=config):
    print(event)

# 模拟崩溃后恢复：从检查点继续
print("\n[恢复执行] 从检查点继续...")
for event in app.stream(None, config=config):  # None 表示从上次继续
    print(event)

# 获取历史状态
state_history = list(app.get_state_history(config))
print(f"\n检查点数量: {len(state_history)}")

  • 每次调用都是独立会话，崩溃后丢失全部上下文
  • 无法在中间步骤暂停等待

  • 人工审批只能通过外部系统模拟
推荐：LangGraph 有状态工作流
  • 每步自动持久化，崩溃可恢复

  • 检查点历史作为审计日志

# human_approval.py — 人工审批节点
from langgraph.graph import StateGraph, END
from typing import TypedDict, Optional

class ApprovalState(TypedDict):
    request_id: str
    amount: float
    status: Optional[str]       # pending / approved / rejected
    reviewer: Optional[str]
    comment: str

def request_approval(state: ApprovalState) -> dict:
    """人工审批节点：暂停等待"""
    # 状态设为 pending，LangGraph 在此中断
    print(f"[审批请求] #{state['request_id']} 金额: {state['amount']} 等待审批")
    return {"status": "pending"}

def after_approval(state: ApprovalState) -> dict:
    """审批后的处理"""
    if state["status"] == "approved":
        print(f"[审批通过] 审批人: {state.get('reviewer')}, 意见: {state.get('comment')}")
    else:
        print(f"[审批驳回] 审批人: {state.get('reviewer')}, 原因: {state.get('comment')}")
    return state

# 在外部恢复执行时设置审批结果
def resume_with_approval(app, thread_id: str, approved: bool, reviewer: str, comment: str = ""):
    """恢复工作流，设置审批结果"""
    config = {"configurable": {"thread_id": thread_id}}
    app.update_state(
        config,
        {"status": "approved" if approved else "rejected", "reviewer": reviewer, "comment": comment},
    )
    # 继续执行
    for event in app.stream(None, config):
        print(f"事件: {event}")

常见问题（FAQ）

LangGraph 和直接写 Python 状态机有什么区别？
LangGraph 提供了状态持久化、中断恢复、流式输出、LangSmith 追踪等基础设施。自己写状态机需要手撸：状态序列化/反序列化、崩溃恢复、并发控制、审计日志——这些 LangGraph 都已内置。而且 LangGraph 与 LangChain 生态（工具、记忆、LLM 调用）无缝集成。
检查点存在哪里？生产环境推荐什么存储？
LangGraph 支持 MemorySaver（内存，开发用）、SqliteSaver（SQLite，单机）、PostgresSaver（PostgreSQL，生产）。生产环境推荐 PostgreSQL + 连接池，支持 ACID 事务和水平扩展。检查点记录了每次 state 变更，注意设置生命周期避免存储膨胀。
一个 LangGraph 工作流可以嵌套另一个工作流吗？
可以。LangGraph 支持节点内调用子图（Subgraph）。你在一个节点函数中调用另一个编译好的 StateGraph，子图有自己的状态空间和检查点。这在多 Agent 协作场景中非常有用——每个 Agent 是一个子图，由编排 Agent 统一调度。

本章结论

  • StateGraph 是核心：用 TypedDict 定义状态，用节点函数处理逻辑，用边定义流转条件。
  • 检查点是生产保障：自动持久化每步状态，支持崩溃恢复和中断恢复。生产用 PostgreSQL 存储。
  • 人工审批节点：工作流在审批点中断，等待外部输入后恢复，通过 update_state 注入审批结果。
  • LangGraph vs ReAct：ReAct 适合简单工具调用链，LangGraph 适合需要状态管理、中断恢复的复杂流程。
  • 审计与可观测：检查点历史天然就是审计日志，配合 LangSmith 实现全流程追踪。

# parallel_tools.py — 并行工具编排
import asyncio
import time
from typing import Dict, Any, List
from datetime import datetime, timedelta

class ToolExecutor:
    """并行工具执行器"""

    async def execute_parallel(self, tool_calls: List[Dict[str, Any]]) -> Dict[str, Any]:
        """并行执行多个工具调用"""
        async def run_single(call: Dict[str, Any]):
            tool_name = call["tool"]
            params = call.get("params", {})
            timeout = call.get("timeout", 10)
            try:
                result = await asyncio.wait_for(
                    self._invoke_tool(tool_name, params),
                    timeout=timeout,
                )
                return {"tool": tool_name, "success": True, "result": result}
            except asyncio.TimeoutError:
                return {"tool": tool_name, "success": False, "error": "超时"}
            except Exception as e:
                return {"tool": tool_name, "success": False, "error": str(e)}

        tasks = [run_single(call) for call in tool_calls]
        results = await asyncio.gather(*tasks, return_exceptions=False)
        return {"parallel_results": results}

    async def _invoke_tool(self, tool_name: str, params: dict) -> Any:
        """实际调用工具（模拟）"""
        if tool_name == "check_calendar":
            await asyncio.sleep(0.5)
            return {"free_slots": ["14:00-15:00", "15:00-16:00"]}
        elif tool_name == "check_room":
            await asyncio.sleep(0.3)
            return {"available_rooms": ["A201", "B305"]}
        elif tool_name == "create_event":
            await asyncio.sleep(0.5)
            return {"event_id": "EVT-001", "status": "created"}
        elif tool_name == "send_email":
            await asyncio.sleep(1.0)
            return {"message_id": "MSG-001", "status": "sent"}
        raise ValueError(f"未知工具: {tool_name}")

    def sync_execute_parallel(self, tool_calls: List[Dict]) -> Dict:
        """同步版本的并行执行"""
        return asyncio.run(self.execute_parallel(tool_calls))

# 使用示例
if __name__ == "__main__":
    executor = ToolExecutor()

    # 并行查询日历和会议室
    parallel_calls = [
        {"tool": "check_calendar", "params": {"date": "2026-06-03"}, "timeout": 5},
        {"tool": "check_room", "params": {"capacity": 10}, "timeout": 5},
    ]
    results = executor.sync_execute_parallel(parallel_calls)
    print(f"并行结果: {results}")

  • 后一个工具不需要前一个的结果，却串行等待
  • 总耗时 = 各工具耗时之和
推荐：并行执行独立工具
  • 查日历和查会议室同时开始
  • 总耗时 = max(0.5s, 0.3s) = 0.5s
  • 节省 37.5% 的时间

串行编排与错误处理

当工具调用有数据依赖时，必须串行执行。串行编排的关键是错误传播和幂等性。

python 
# serial_pipeline.py — 串行工具编排与错误处理
from dataclasses import dataclass
from typing import Optional, Callable
import time

@dataclass
class PipelineStep:
    name: str
    func: Callable
    retry_count: int = 2
    timeout: int = 30
    critical: bool = True  # 是否关键步骤

class ToolPipeline:
    """串行工具管线"""

    def __init__(self, steps: list[PipelineStep]):
        self.steps = steps

    def execute(self, context: dict) -> dict:
        """顺序执行所有步骤，context 在步骤间传递"""
        for step in self.steps:
            print(f"[{step.name}] 开始执行")
            result = self._execute_with_retry(step, context)
            if result["success"]:
                context[step.name] = result["data"]
            elif step.critical:
                raise RuntimeError(f"关键步骤 {step.name} 失败: {result['error']}")
            else:
                context[step.name] = {"error": result["error"]}
        return context

    def _execute_with_retry(self, step: PipelineStep, context: dict) -> dict:
        for attempt in range(step.retry_count + 1):
            try:
                start = time.time()
                data = step.func(context)
                elapsed = time.time() - start
                return {"success": True, "data": data, "elapsed": elapsed}
            except Exception as e:
                if attempt < step.retry_count:
                    wait = 2 ** attempt
                    print(f"[{step.name}] 重试 {attempt+1}/{step.retry_count} (等待 {wait}s)")
                    time.sleep(wait)
                else:
                    return {"success": False, "error": str(e)}

# 步骤函数
def create_calendar_event(ctx: dict) -> dict:
    """创建日历事件"""
    free_slots = ctx.get("check_calendar", {}).get("free_slots", [])
    room = ctx.get("check_room", {})
    return {"event_id": "EVT-001", "time": free_slots[0], "room": room.get("available_rooms", [None])[0]}

def send_notification(ctx: dict) -> str:
    """发送通知"""
    event = ctx.get("create_calendar_event", {})
    return f"会议通知已发送，事件ID: {event.get('event_id')}"

多工具编排中的幂等性

多次调用同一工具不能产生副作用。CorpAssist 为每个工具调用生成幂等键（idempotency key），服务端检测到相同幂等键时直接返回上次结果。

工具类型幂等策略示例
查询类 天然幂等（多次查询结果不变） 查日历、查库存
创建类 幂等键去重 创建工单、创建事件
通知类 消息ID去重 发送邮件、发送短信

常见问题（FAQ）

并行执行多个工具时，Rate Limit 怎么控制？
使用信号量（Semaphore）控制并发数。CorpAssist 为每个 API 设置独立的最大并发限制（如日历 API 最大 5 并发），用 asyncio.Semaphore 实现。超过限制的任务排队等待。
工具 A 和工具 B 都需要调用同一个下游 API，如何避免重复调用？
在编排层做结果缓存。同一个请求周期内（同一 thread_id），相同入参的工具调用直接返回缓存结果。使用 context 共享缓存字典。
部分成功场景怎么处理？比如邮件发送成功但创建事件失败？
需要补偿事务。CorpAssist 维护工具调用的「待补偿列表」：如果后续步骤失败，自动调用各工具的回滚函数（如取消事件、撤回邮件）。LangGraph 的条件边可以实现补偿流程。

本章结论

  • 编排三模式：并行（无依赖）、串行（有依赖）、混合（先并行后串行）。
  • 并行用 asyncio：asyncio.gather + 独立超时控制，总耗时 = max(各工具耗时)。
  • 串行要幂等：幂等键防止重复调用导致数据错乱。
  • 错误隔离：非关键步骤失败不影响全局，关键步骤失败触发回滚。
  • Rate Limit：Semaphore 控制并发数，避免触发 API 限流。

# multi_agent.py — CorpAssist 多 Agent 协作系统
from pydantic import BaseModel, Field
from typing import List, Optional, Literal
from langchain_openai import ChatOpenAI
from enum import Enum
import asyncio
import json

# ---- 通信协议 ----
class MessageRole(str, Enum):
    SUPERVISOR = "supervisor"
    FINANCE = "finance_agent"
    PROCUREMENT = "procurement_agent"
    LEGAL = "legal_agent"

class AgentMessage(BaseModel):
    """Agent 间通信消息"""
    message_id: str
    sender: MessageRole
    receiver: MessageRole
    content: str
    timestamp: str
    metadata: dict = Field(default_factory=dict)

class TaskResult(BaseModel):
    """Agent 任务结果"""
    agent: MessageRole
    status: Literal["success", "failed", "blocked"]
    summary: str
    details: dict
    confidence: float  # 0~1

# ---- 基础 Agent 类 ----
class BaseSpecialistAgent:
    """专业 Agent 基类"""

    def __init__(self, role: MessageRole, system_prompt: str):
        self.role = role
        self.llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)
        self.system_prompt = system_prompt

    async def process_task(self, task: AgentMessage) -> TaskResult:
        """处理任务并返回结果"""
        prompt = f"{self.system_prompt}\n\n任务内容：{task.content}"
        response = self.llm.invoke(prompt)
        # 解析 LLM 输出为结构化结果
        return TaskResult(
            agent=self.role,
            status="success",
            summary=response.content[:200],
            details={"raw_output": response.content},
            confidence=0.9,
        )

# ---- 具体 Agent 实现 ----
class FinanceAgent(BaseSpecialistAgent):
    def __init__(self):
        super().__init__(
            role=MessageRole.FINANCE,
            system_prompt="你是一个企业财务 Agent。负责：1) 检查预算余额；2) 分析采购成本；3) 评估财务风险。返回 JSON 格式结果。",
        )

class ProcurementAgent(BaseSpecialistAgent):
    def __init__(self):
        super().__init__(
            role=MessageRole.PROCUREMENT,
            system_prompt="你是一个采购 Agent。负责：1) 评估供应商资质；2) 比较报价；3) 检查历史合作记录。返回 JSON 格式。",
        )

class LegalAgent(BaseSpecialistAgent):
    def __init__(self):
        super().__init__(
            role=MessageRole.LEGAL,
            system_prompt="你是一个法务 Agent。负责：1) 审核合同条款合规性；2) 检查法律风险；3) 提供修改建议。返回 JSON 格式。",
        )

# ---- Supervisor Agent ----
class SupervisorAgent:
    """Supervisor: 负责任务分发、结果汇总、冲突裁决"""

    def __init__(self):
        self.llm = ChatOpenAI(model="gpt-4o", temperature=0)
        self.specialists: dict[MessageRole, BaseSpecialistAgent] = {
            MessageRole.FINANCE: FinanceAgent(),
            MessageRole.PROCUREMENT: ProcurementAgent(),
            MessageRole.LEGAL: LegalAgent(),
        }

    async def process_request(self, user_request: str) -> dict:
        """处理用户请求：分发任务 → 收集结果 → 综合决策"""
        # 步骤1: 任务规划
        plan = await self._plan_tasks(user_request)
        print(f"[Supervisor] 规划完成: {len(plan['agents'])} 个 Agent 需参与")

        # 步骤2: 并行分发任务
        tasks = []
        for agent_role in plan["agents"]:
            agent = self.specialists[agent_role]
            msg = AgentMessage(
                message_id=f"msg-{agent_role.value}",
                sender=MessageRole.SUPERVISOR,
                receiver=agent_role,
                content=plan["task_description"],
                timestamp="2026-05-30T10:00:00Z",
            )
            tasks.append(agent.process_task(msg))

        results = await asyncio.gather(*tasks)
        print(f"[Supervisor] 收到 {len(results)} 个 Agent 结果")

        # 步骤3: 结果分析与冲突裁决
        final_decision = await self._synthesize(user_request, results)
        return final_decision

    async def _plan_tasks(self, user_request: str) -> dict:
        """LLM 规划需要哪些 Agent 参与"""
        prompt = f"分析以下请求，决定需要哪些 Agent 参与（finance/procurement/legal）：\n{user_request}"
        response = self.llm.invoke(prompt)
        # 模拟规划结果
        return {"agents": [MessageRole.FINANCE, MessageRole.PROCUREMENT, MessageRole.LEGAL], "task_description": user_request}

    async def _synthesize(self, user_request: str, results: List[TaskResult]) -> dict:
        """综合所有 Agent 结果并裁决冲突"""
        # 检查是否有冲突
        confidences = [r.confidence for r in results]
        avg_conf = sum(confidences) / len(confidences)

        # 如果有 Agent 意见矛盾，让 LLM 裁决
        if any(r.status == "failed" for r in results):
            return {"decision": "blocked", "reason": "部分 Agent 任务失败，需要人工介入"}
        if avg_conf < 0.7:
            return {"decision": "clarify", "reason": "置信度不足，需要追问用户"}

        return {"decision": "approved", "reason": "所有 Agent 均通过", "details": [r.summary for r in results]}

# 运行示例
async def main():
    supervisor = SupervisorAgent()
    result = await supervisor.process_request("申请采购 100 台服务器，预算 50 万，供应商 某某科技")
    print(f"最终决策: {json.dumps(result, ensure_ascii=False, indent=2)}")

if __name__ == "__main__":
    asyncio.run(main())

冲突解决与消息总线

多 Agent 协作中，冲突解决 是核心挑战。当财务 Agent 说「预算充足」但采购 Agent 说「供应商资质存疑」时，Supervisor 必须裁决。

python 
# conflict_resolver.py — 冲突裁决器
from pydantic import BaseModel
from typing import List, Optional

class ConflictRecord(BaseModel):
    """冲突记录"""
    topic: str
    agents_involved: List[str]
    positions: dict  # agent -> 立场
    resolution: Optional[str] = None
    resolver: str = "supervisor"

class ConflictResolver:
    """基于 LLM 的冲突裁决器"""

    def __init__(self):
        self.llm = ChatOpenAI(model="gpt-4o", temperature=0)
        self.history: List[ConflictRecord] = []

    def detect_conflicts(self, results: List[TaskResult]) -> List[ConflictRecord]:
        """检测不同 Agent 结果之间的矛盾"""
        conflicts = []
        # 检查财务和采购的结果是否一致
        finance = next((r for r in results if r.agent == MessageRole.FINANCE), None)
        procurement = next((r for r in results if r.agent == MessageRole.PROCUREMENT), None)
        if finance and procurement:
            if "budget_ok" in finance.details and "supplier_risk" in procurement.details:
                conflicts.append(ConflictRecord(
                    topic="预算 vs 供应商风险",
                    agents_involved=["finance", "procurement"],
                    positions={"finance": "预算充足", "procurement": "供应商有风险"},
                ))
        return conflicts

    def resolve(self, conflicts: List[ConflictRecord]) -> List[ConflictRecord]:
        """裁决冲突"""
        for conflict in conflicts:
            prompt = f"""以下有两个 Agent 的结论存在冲突，请裁决：
冲突主题：{conflict.topic}
Agent 1 ({conflict.agents_involved[0]}): {conflict.positions[conflict.agents_involved[0]]}
Agent 2 ({conflict.agents_involved[1]}): {conflict.positions[conflict.agents_involved[1]]}
请决定采纳哪个意见或返回折中方案："""
            response = self.llm.invoke(prompt)
            conflict.resolution = response.content
            self.history.append(conflict)
        return conflicts
CorpAssist 实践建议

消息总线推荐使用 Redis Pub/Sub 或 RabbitMQ。每个 Agent 监听自己的消息队列，Supervisor 将任务发布到对应队列。这种异步架构的好处是：Agent 可以独立扩缩容、一个 Agent 的崩溃不影响其他 Agent、消息可持久化（崩溃恢复）。LangGraph 的 Subgraph 也可以实现类似效果，但消息总线的扩展性更好。

常见问题（FAQ）

一个 Supervisor 管理多少个 Specialist 比较合适？
CorpAssist 的经验是 3-8 个。少于 3 个不需要多 Agent（单 Agent 即可），多于 8 个时 Supervisor 的 Prompt 会过长，且任务分发效率下降。如果需要更多 Agent，采用分级架构——Supervisor 下面有 Sub-Supervisor。
Specialist Agent 之间可以直接通信吗？
可以，但不推荐。直接通信会导致消息拓扑复杂化，难以追踪。推荐所有通信经过 Supervisor——Specialist 把结果发给 Supervisor，Supervisor 决定是否需要让其他 Specialist 知晓。这种星型拓扑最易于调试和审计。
如果三个 Agent 给出三个不同结论，Supervisor 怎么选？
三种策略：1) 投票制（多数胜出）；2) 权重制（不同 Agent 的权重不同，如法务的否决权高于采购）；3) LLM 裁决（把三个结论发给 LLM 做综合判断）。CorpAssist 默认用策略 3，因为最灵活。

本章结论

  • Supervisor + Specialist 模式：Supervisor 负责任务分发和结果汇总，Specialist 各司其职。
  • 统一通信协议：AgentMessage Schema 包含 sender/receiver/content/timestamp，确保消息可追踪。
  • 冲突裁决：LLM 裁决 > 权重制 > 投票制。Supervisor 的裁决能力决定多 Agent 系统的质量上限。
  • 异步消息总线：Redis/RabbitMQ 解耦 Agent 通信，支持独立扩缩容和故障隔离。
  • 3-8 个 Specialist：超过 8 个时使用分级 Supervior 架构。

    API->>W: 恢复执行信号
    W->>DB: 读取审批结果
    W->>W: 继续执行后续节点
    W->>DB: 记录完整审批链路

工作流在审批节点暂停，将请求持久化到数据库。审批人通过 API 查看和操作，完成后触发工作流恢复。

# approval_node.py — 人工审批节点完整实现
from pydantic import BaseModel, Field
from typing import Optional, Literal
from datetime import datetime
from fastapi import FastAPI, HTTPException
import uuid
import asyncio

# ---- 数据模型 ----
class ApprovalRequest(BaseModel):
    """审批请求"""
    request_id: str = Field(default_factory=lambda: f"APP-{uuid.uuid4().hex[:8]}")
    workflow_id: str
    thread_id: str
    title: str
    description: str
    amount: Optional[float] = None
    requester: str
    status: Literal["pending", "approved", "rejected", "timeout"] = "pending"
    assigned_to: str
    created_at: str = Field(default_factory=lambda: datetime.utcnow().isoformat())
    decided_at: Optional[str] = None
    comment: str = ""

class ApprovalDecision(BaseModel):
    """审批决定"""
    request_id: str
    decision: Literal["approved", "rejected"]
    reviewer: str
    comment: str = ""

# ---- 审批存储 ----
class ApprovalStore:
    """审批请求存储（生产用 PostgreSQL）"""
    def __init__(self):
        self._store: dict[str, ApprovalRequest] = {}
        self._pending_events: dict[str, asyncio.Event] = {}

    async def create_request(self, req: ApprovalRequest) -> ApprovalRequest:
        self._store[req.request_id] = req
        self._pending_events[req.request_id] = asyncio.Event()
        return req

    async def get_request(self, request_id: str) -> Optional[ApprovalRequest]:
        return self._store.get(request_id)

    async def decide(self, decision: ApprovalDecision) -> ApprovalRequest:
        req = self._store.get(decision.request_id)
        if not req:
            raise HTTPException(404, "审批请求不存在")
        req.status = decision.decision
        req.decided_at = datetime.utcnow().isoformat()
        req.comment = decision.comment
        # 通知等待的工作流
        event = self._pending_events.get(decision.request_id)
        if event:
            event.set()
        return req

    async def wait_for_decision(self, request_id: str, timeout: int = 3600) -> ApprovalRequest:
        """等待审批结果（带超时）"""
        event = self._pending_events.get(request_id)
        if not event:
            raise ValueError(f"未知审批请求: {request_id}")
        try:
            await asyncio.wait_for(event.wait(), timeout=timeout)
        except asyncio.TimeoutError:
            req = self._store[request_id]
            req.status = "timeout"
            return req
        return self._store[request_id]

超时策略与备选审批人

审批超时是最容易被忽视的生产坑。CorpAssist 使用三级超时策略：

级别超时时间行为
L1 提醒 30 分钟 IM/邮件提醒审批人「有待审批请求」
L2 升级 2 小时 通知备选审批人（审批人的上级）
L3 超时 24 小时 标记为超时拒绝，通知申请人和所有审批链路上的人

# approval_timeout.py — 审批超时处理
import asyncio

class TimeoutManager:
    """审批超时管理器"""

    def __init__(self, store: ApprovalStore, notifier):
        self.store = store
        self.notifier = notifier  # 通知服务（邮件/IM）

    async def monitor(self, request_id: str):
        """监控审批超时，执行升级策略"""
        await asyncio.sleep(30 * 60)  # 30 分钟
        req = await self.store.get_request(request_id)
        if req and req.status == "pending":
            await self.notifier.send(req.assigned_to, f"【提醒】审批 {req.request_id} 待处理")

        await asyncio.sleep(90 * 60)  # 再等 1.5 小时（共 2 小时）
        req = await self.store.get_request(request_id)
        if req and req.status == "pending":
            backup_approver = self._get_backup(req.assigned_to)
            await self.notifier.send(backup_approver, f"【升级】审批 {req.request_id} 已超时 2 小时，请处理")

        await asyncio.sleep(22 * 3600)  # 再等 22 小时（共 24 小时）
        req = await self.store.get_request(request_id)
        if req and req.status == "pending":
            req.status = "timeout"
            await self.notifier.send(req.requester, f"【超时】审批 {req.request_id} 已超时 24 小时，自动拒绝")

    def _get_backup(self, approver: str) -> str:
        """获取备选审批人（实际工程中查组织架构）"""
        backup_map = {"manager": "director", "director": "vp"}
        return backup_map.get(approver, "admin")

常见问题（FAQ）

审批中断时，工作流占用的内存和连接怎么释放？
LangGraph 的检查点机制会将状态持久化到数据库，然后工作流进程可以安全退出。恢复时从检查点重新加载状态。这意味审批中断期间不消耗内存或连接资源。这是 LangGraph 相比自己写 while True 等待的最大优势。
如果审批人在审批时看到了过期数据怎么办？
审批请求中应该包含数据的时间戳和有效期。如果审批时发现数据已过期（如库存已变），Agent 应触发重新评估流程。推荐在审批界面显示「数据截止时间：xxx」，并在恢复执行时做数据新鲜度检查。
同一个审批人可以同时审批多个请求吗？
可以。每个审批请求是独立的 request_id，互不干扰。审批 Dashboard 显示所有待审批列表，点开每个查看详情。需要注意并发安全性——同一个审批人对同一请求的两次操作，以第一次为准（幂等）。

本章结论

  • 中断 != 失败：LangGraph 检查点持久化状态，工作流进程可安全退出，恢复时重新加载。
  • 异步审批 API：FastAPI 提供审批 REST 接口，asyncio.Event 实现非阻塞等待。
  • 三级超时策略：30min 提醒 → 2h 升级 → 24h 超时拒绝。每个级别有明确的业务行为。
  • 完整审计：谁审批、何时、看了哪些数据、结果如何——全程记录不可篡改。
  • 数据新鲜度：审批时检查数据有效期，过期数据触发重新评估。

async def stream_demo(question: str) -> AsyncGenerator[str, None]:
    """流式输出 Agent 执行过程"""
    start_time = time.time()
    step = 0

    async for event in agent_executor.astream_events(
        {"input": question},
        version="v2",
    ):
        kind = event["event"]
        if kind == "on_chat_model_start":
            step += 1
            yield f"data: {json.dumps({'type': 'step', 'step': step})}\n\n"
        elif kind == "on_tool_start":
            yield f"data: {json.dumps({'type': 'tool_call', 'tool': event['name'], 'input': str(event['data'].get('input', ''))})}\n\n"
        elif kind == "on_tool_end":
            yield f"data: {json.dumps({'type': 'tool_result', 'output': str(event['data'].get('output', ''))})}\n\n"
        elif kind == "on_chain_end":
            if "output" in event["data"]:
                yield f"data: {json.dumps({'type': 'final', 'output': event['data']['output']})}\n\n"

    elapsed = time.time() - start_time
    yield f"data: {json.dumps({'type': 'metrics', 'steps': step, 'latency_ms': int(elapsed * 1000)})}\n\n"

步数与 Token 监控

Token 预算 管理是 Agent 生产化的重要环节。演示时实时监控步数和 Token 消耗，帮助快速发现异常行为。

python 
# metrics_monitor.py — 步数和 Token 监控
from dataclasses import dataclass
from typing import List
import time

@dataclass
class AgentRunMetrics:
    """Agent 运行指标"""
    run_id: str
    question: str
    steps: int = 0
    tools_called: List[str] = None
    total_tokens: int = 0
    prompt_tokens: int = 0
    completion_tokens: int = 0
    latency_ms: int = 0
    status: str = "running"

    def __post_init__(self):
        self.tools_called = self.tools_called or []

class MetricsCollector:
    """运行指标收集器"""

    def __init__(self):
        self.runs: List[AgentRunMetrics] = []

    def start_run(self, run_id: str, question: str) -> AgentRunMetrics:
        metrics = AgentRunMetrics(run_id=run_id, question=question)
        self.runs.append(metrics)
        return metrics

    def record_step(self, run_id: str, tool_name: str, tokens: int = 0):
        for run in self.runs:
            if run.run_id == run_id:
                run.steps += 1
                run.tools_called.append(tool_name)
                run.total_tokens += tokens
                break

    def finish_run(self, run_id: str, latency_ms: int):
        for run in self.runs:
            if run.run_id == run_id:
                run.latency_ms = latency_ms
                run.status = "completed"
                break

    def summary(self) -> dict:
        if not self.runs:
            return {"avg_steps": 0, "avg_tokens": 0}
        avg_steps = sum(r.steps for r in self.runs) / len(self.runs)
        avg_tokens = sum(r.total_tokens for r in self.runs) / len(self.runs)
        return {
            "total_runs": len(self.runs),
            "avg_steps": round(avg_steps, 1),
            "avg_tokens": int(avg_tokens),
            "avg_latency_ms": int(sum(r.latency_ms for r in self.runs) / len(self.runs)),
        }
