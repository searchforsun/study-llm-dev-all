SSE 端到端链路

SSE（Server-Sent Events）流式输出客服回答，需要从前端到 Nginx 到 FastAPI 全链路支持。

QPS 限流与削峰

python
"""限流与削峰"""

from fastapi import FastAPI, HTTPException
import asyncio
from collections import deque
import time

app = FastAPI()

class RateLimiter:
    def __init__(self, max_qps: int = 50):
        self.max_qps = max_qps
        self.requests = deque()

    async def acquire(self):
        now = time.monotonic()
        # 清除 1 秒前的记录
        while self.requests and self.requests[0] < now - 1:
            self.requests.popleft()
        if len(self.requests) >= self.max_qps:
            raise HTTPException(status_code=429, detail="Too Many Requests")
        self.requests.append(now)

class PeakShavingQueue:
    """削峰队列"""
    def __init__(self, max_queue: int = 200):
        self.queue = asyncio.Queue(maxsize=max_queue)

    async def enqueue(self, request):
        try:
            await asyncio.wait_for(self.queue.put(request), timeout=5)
        except asyncio.TimeoutError:
            raise HTTPException(status_code=503, detail="Service Busy")

用户鉴权（JWT + SSO）

python
"""JWT 鉴权中间件"""

from fastapi import Request, HTTPException
from jose import jwt, JWTError
from datetime import datetime, timedelta
from typing import Optional

SECRET_KEY = "corp-assist-secret-key-2026"
ALGORITHM = "HS256"

def create_access_token(data: dict, expires_delta: timedelta = timedelta(minutes=15)) -> str:
    to_encode = data.copy()
    to_encode.update({"exp": datetime.utcnow() + expires_delta})
    return jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)

def verify_token(token: str) -> Optional[dict]:
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        return payload
    except JWTError:
        return None

async def auth_middleware(request: Request, call_next):
    if request.url.path not in ["/health", "/metrics"]:
        auth = request.headers.get("Authorization", "")
        if not auth.startswith("Bearer "):
            raise HTTPException(status_code=401, detail="Unauthorized")
        payload = verify_token(auth[7:])
        if not payload:
            raise HTTPException(status_code=401, detail="Token expired")
        request.state.user = payload
    return await call_next(request)

SSO 集成与企业 LDAP

企业客户通常使用 SSO（单点登录）和 LDAP（轻量级目录访问协议）管理员工身份。CorpAssist 的智能客服需要与企业现有的身份系统集成。

python
"""SSO 集成（OIDC 协议）"""

from fastapi import Request, HTTPException
from authlib.integrations.starlette_client import OAuth
from starlette.config import Config

config = Config(".env")  # OIDC 配置
oauth = OAuth(config)
oauth.register(
    name="corp_sso",
    server_metadata_url="https://sso.corp.com/.well-known/openid-configuration",
    client_kwargs={"scope": "openid profile email"},
)

async def sso_login(request: Request):
    """SSO 登录重定向"""
    redirect_uri = request.url_for("sso_callback")
    return await oauth.corp_sso.authorize_redirect(request, redirect_uri)

async def sso_callback(request: Request):
    """SSO 回调处理"""
    token = await oauth.corp_sso.authorize_access_token(request)
    user_info = token.get("userinfo", {})
    # 生成内部 JWT Token
    internal_token = create_access_token({
        "sub": user_info.get("sub"),
        "name": user_info.get("name"),
        "roles": user_info.get("roles", ["全员可见"]),
    })
    return {"access_token": internal_token, "token_type": "bearer"}

集成要点：

  • 企业 SSO 支持 OIDC（推荐）和 SAML 2.0（兼容）两种协议。
  • LDAP 同步员工组织架构和角色信息，每日增量同步一次。
  • 用户角色（如「HR_可见」「经理_可见」）从 SSO 的 group/role 声明中提取。
  • 统一登录入口：CorpAssist 门户 → SSO 登录 → 重定向回门户 → 携带 JWT Token。

本章结论

  • SSE 需要 Nginx+FastAPI 全链路配置，关键在 X-Accel-Buffering: no。
  • 令牌桶限流 50 QPS + 队列削峰，超出队列返回 503。
  • JWT Token 15 分钟 + Refresh Token 7 天，SSO 集成企业 LDAP。
  • P0 告警 15 分钟响应，覆盖 P99/错误率/转接率三个关键指标。
  • 7x24 值班方案分 P0/P1/P2 三级，确保严重故障及时响应。
