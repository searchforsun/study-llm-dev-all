规则+模型双校验 DST

对话状态管理（DST）在并发场景下会出现双写冲突。两个用户消息同时到达，Redis 的两次读写之间出现竞态条件。

python
"""分布式锁解决 DST 冲突"""

import asyncio
from redis import asyncio as aioredis
from typing import Optional

class LockedStateManager:
    """带分布式锁的状态管理器"""

    def __init__(self, redis_url: str = "redis://redis:6379/1"):
        self.redis = aioredis.from_url(redis_url)
        self.lock_timeout = 5

    async def with_lock(self, session_id: str, func, *args, **kwargs):
        """带锁执行状态操作"""
        lock_key = f"lock:session:{session_id}"
        lock = self.redis.lock(lock_key, timeout=self.lock_timeout)
        async with lock:
            return await func(session_id, *args, **kwargs)

    async def safe_append_history(
        self, session_id: str, role: str, content: str
    ):
        """安全追加对话历史"""
        async def _append(sid):
            state_data = await self.redis.hgetall(f"session:{sid}")
            history = state_data.get(b"history", b"[]")
            import json
            hist = json.loads(history)
            hist.append({"role": role, "content": content})
            await self.redis.hset(f"session:{sid}", "history", json.dumps(hist))
        await self.with_lock(session_id, _append)

确定性工具调用的工程实践

LLM 在生成工具调用时存在「非确定性」问题——同一条 user query 两次调用，LLM 可能抽取不同的参数，甚至编造不存在的参数。

python
"""确定性工具调用保证"""

class DeterministicToolCaller:
    """确定性工具调用——相同输入始终得到相同输出"""

    def __init__(self):
        self._call_cache = {}

    def call_with_cache(self, tool_name: str, params: dict) -> dict:
        """带缓存的工具调用"""
        import json
        cache_key = f"{tool_name}:{json.dumps(params, sort_keys=True)}"
        if cache_key in self._call_cache:
            return self._call_cache[cache_key]
        result = self._execute_tool(tool_name, params)
        self._call_cache[cache_key] = result
        return result

    def _execute_tool(self, tool_name: str, params: dict) -> dict:
        """实际执行工具（仅查询类可缓存）"""
        if tool_name in ["query_ticket", "query_order", "query_balance"]:
            # 查询类可以缓存结果（TTL=30s）
            from src.tools import get_tool
            return get_tool(tool_name).run(params)
        # 修改类操作不缓存
        from src.tools import get_tool
        return get_tool(tool_name).run(params)

确定性还需要考虑 LLM 的参数抽取一致性。CorpAssist 的做法是：不依赖 LLM 从对话中抽取参数，而是用规则引擎优先抽取（如正则匹配工单编号模式 TKT-\d+），只有规则无法抽取时才使用 LLM。这样保证了常见参数（工单号、订单号）的抽取 100% 确定，只有非常见参数才走 LLM。

意图→API 不回写幻觉

LLM 可能在工具调用时「发明」API 返回结果——这是客服场景独有的幻觉。比如 LLM 假装成功调用了密码重置 API 并返回了新密码，但实际上 API 根本没被调用。

python确保工具调用结果来自真实 API 而非 LLM 臆想
"""工具调用验证器"""

class ToolCallValidator:
    """验证工具调用是否真实执行"""

    def validate_tool_call(self, tool_name: str, params: dict, llm_claimed_result: str) -> bool:
        """验证 LLM 声称的结果是否与真实 API 返回一致"""
        # 重新调用工具获取真实结果
        from src.tools import get_tool
        tool = get_tool(tool_name)
        real_result = tool.run(params)
        # 比较真实结果和 LLM 声称的结果
        return real_result == llm_claimed_result or "失败" in real_result

class SafeToolExecutor:
    """安全工具执行器——防止不回写幻觉"""

    def execute(self, tool_name: str, params: dict) -> dict:
        """执行工具并验证结果"""
        # 1. 调用工具
        from src.tools import get_tool
        tool = get_tool(tool_name)
        real_result = tool.run(params)
        # 2. 记录真实结果到上下文（而非让 LLM 自己写结果）
        return {"tool_name": tool_name, "real_result": real_result, "status": "executed"}
        # LLM 只负责解释 real_result，不能自己编造结果

高并发优化

客服系统在高峰时段可能达到 50 QPS，SSE 长连接 + LLM 调用对资源消耗巨大。以下是 CorpAssist 在实战中踩过的坑。

坑 1：SSE 连接泄漏。上线初期我们发现服务端内存每 4 小时翻倍，排查发现客户端的 SSE 连接断开后，服务端的异步生成器没有正确终止。修复方案：在 SSE 生成器中加入 try/finally 块，确保生成器被 close 时 LLM 的 stream 迭代器也被终止。同时加入连接超时机制（120 秒无数据传输自动断开）。

坑 2：Redis 热点键。高峰期所有对话都写入同一个 Redis 实例的同一个 session hash 键附近，导致 Redis 响应时间从 2ms 飙升至 200ms。修复方案：按 session_id 的前缀做 Redis 集群分片，key 分布到不同节点。同时将高频操作（读取最近 3 轮历史）从 Redis 移至本地内存缓存，减少 Redis 读取频率。

SSE 连接池500+ 并发连接 OOM连接池上限 200, 超出排队OOM 归零

"""QPS 限流与缓存"""

from fastapi import HTTPException
import time
from collections import deque

class TokenBucket:
    """令牌桶限流器"""

    def __init__(self, rate: float = 50, capacity: int = 100):
        self.rate = rate
        self.capacity = capacity
        self.tokens = capacity
        self.last_refill = time.monotonic()

    def consume(self, tokens: int = 1) -> bool:
        now = time.monotonic()
        elapsed = now - self.last_refill
        self.tokens = min(self.capacity, self.tokens + elapsed * self.rate)
        self.last_refill = now
        if self.tokens >= tokens:
            self.tokens -= tokens
            return True
        return False

class ResponseCache:
    """问答缓存"""

    def __init__(self, redis_client):
        self.redis = redis_client
        self.ttl = 3600

    async def get(self, key: str) -> Optional[str]:
        data = await self.redis.get(f"cache:qa:{key}")
        return data.decode() if data else None

    async def set(self, key: str, value: str):
        await self.redis.setex(f"cache:qa:{key}", self.ttl, value)

    async def get_or_compute(self, key: str, compute_func) -> str:
        cached = await self.get(key)
        if cached:
            return cached
        result = await compute_func()
        await self.set(key, result)
        return result

Redis 缓存命中率优化实战

缓存是降低客服推理成本最立竿见影的手段。但「随便缓存」和「系统化缓存」的效果差异巨大。

python
"""Redis 缓存命中率优化"""

from collections import Counter
from typing import List

class CacheOptimizer:
    """缓存命中率分析器"""

    def __init__(self):
        self.hit_counter = Counter()
        self.miss_counter = Counter()

    def analyze_queries(self, queries: List[str], cache, top_k: int = 20):
        """分析查询分布来优化缓存策略"""
        # 1. 分析高频问题
        query_freq = Counter(queries)
        top_queries = query_freq.most_common(top_k)
        print(f"Top-{top_k} 问题占总查询量的 "
              f"{sum(f for _, f in top_queries) / len(queries) * 100:.1f}%")

        # 2. 归一化（去除终止符、空格、标点差异）
        normalized = {}
        for q in queries:
            norm = q.strip().rstrip("。！？").lower()
            normalized[norm] = normalized.get(norm, 0) + 1

        # 3. 估算归一化后的理论缓存命中率
        cache_hits_est = sum(1 for n, f in normalized.items() if f > 1)
        print(f"归一化后可缓存的问题: {cache_hits_est}/{len(normalized)} "
              f"({cache_hits_est/len(normalized)*100:.1f}%)")

缓存优化层次：CorpAssist 采用三层缓存策略。第一层精确匹配缓存（命中率 35%）：相同的 query 直接返回。第二层归一化缓存（命中率 +8%）：去除标点停用词后的语义相同查询。第三层语义相似缓存（命中率 +9%）：对 embedding 相似度 > 0.95 的查询复用结果。三层叠加总命中率 52%，日均减少 2600 次 LLM 调用，年节省 $7,800。
