学完后能做什么

本章要点速记

1) 自动评测指标：FCR/转人工率/CSAT/平均处理时长/准确率；2) 人工抽检用 Rubric 评分表（1-5 分），覆盖正确性/礼貌/效率/合规；3) A/B 测试需运行 7 天，样本量 > 2000 对话才有统计意义；4) 坏例库按根因分类维护，每次发版前跑回归；5) 解决率的计算方式（用户确认 vs 系统判断）会导致数值差异 10-15%。

"""对话级评估"""

from typing import List, Dict

class DialogLevelEvaluator:
    """对话级评估"""

    def evaluate_turn_efficiency(self, dialogs: List[Dict]) -> dict:
        """评估多轮效率"""
        turns = [d.get("turn_count", 0) for d in dialogs]
        resolved_turns = [
            d.get("turn_count", 0) for d in dialogs if d.get("resolved")
        ]
        return {
            "avg_turns_total": sum(turns) / len(turns) if turns else 0,
            "avg_turns_resolved": sum(resolved_turns) / len(resolved_turns) if resolved_turns else 0,
            "one_turn_resolution_rate": sum(1 for t in resolved_turns if t <= 2) / len(resolved_turns) if resolved_turns else 0,
        }

    def evaluate_sentiment_trend(self, dialogs: List[Dict]) -> float:
        """评估情感保持度：结束情感 ≥ 开始情感的对话比例"""
        count = 0
        for d in dialogs:
            sentiment_history = d.get("sentiment_history", [])
            if len(sentiment_history) >= 2:
                if sentiment_history[-1] >= sentiment_history[0]:
                    count += 1
        return count / len(dialogs) if dialogs else 0

自动评测

自动评测从对话日志中提取指标，覆盖效率和效果两个维度。

python
"""自动评测指标计算"""

from dataclasses import dataclass
from typing import List
from datetime import datetime, timedelta

@dataclass
class DialogRecord:
    session_id: str
    started_at: datetime
    ended_at: datetime
    resolved: bool  # 用户确认解决
    transferred: bool
    csat_score: float  # 1-5
    intent_accuracy: bool
    turn_count: int

class AutoEvaluator:
    """自动评测器"""

    def evaluate(self, dialogs: List[DialogRecord]) -> dict:
        total = len(dialogs)
        if total == 0:
            return {}
        return {
            "fcr": sum(1 for d in dialogs if d.resolved) / total,
            "transfer_rate": sum(1 for d in dialogs if d.transferred) / total,
            "avg_csat": sum(d.csat_score for d in dialogs) / total,
            "avg_duration_s": sum(
                (d.ended_at - d.started_at).total_seconds() for d in dialogs
            ) / total,
            "intent_accuracy": sum(1 for d in dialogs if d.intent_accuracy) / total,
            "avg_turns": sum(d.turn_count for d in dialogs) / total,
        }

def compute_daily_report(dialogs: List[DialogRecord]) -> str:
    evaluator = AutoEvaluator()
    metrics = evaluator.evaluate(dialogs)
    report = "日客服效果报告:\n" + "\n".join(
        f"  {k}: {v:.1%}" if isinstance(v, float) else f"  {k}: {v:.2f}"
        for k, v in metrics.items()
    )
    return report

人工抽检 Rubric 评分标尺

python
"""Rubric 评分标尺"""

from pydantic import BaseModel
from typing import List, Optional

class RubricScore(BaseModel):
    """Rubric 评分"""
    correctness: int  # 1-5: 回答是否正确
    politeness: int   # 1-5: 语气是否礼貌
    efficiency: int   # 1-5: 是否尽快解决问题
    compliance: int   # 1-5: 是否符合政策
    notes: Optional[str] = None

RUBRIC_GUIDELINES = {
    "correctness": {
        1: "完全错误", 2: "部分错误", 3: "基本正确但有遗漏",
        4: "正确且完整", 5: "正确+额外有用信息",
    },
    "politeness": {
        1: "语气生硬", 2: "略显生硬", 3: "中性",
        4: "礼貌", 5: "非常礼貌+个性化",
    },
    "efficiency": {
        1: "需要>5轮", 2: "3-5轮", 3: "2轮解决",
        4: "1轮解决", 5: "1轮解决+预判用户需求",
    },
    "compliance": {
        1: "违反政策", 2: "打擦边球", 3: "合规",
        4: "合规+引用政策原文", 5: "合规+引用+解释",
    },
}

class HumanInspector:
    """人工抽检"""

    def sample(self, dialogs: List[DialogRecord], rate: float = 0.05) -> List[DialogRecord]:
        """按比例抽样"""
        import random
        n = max(1, int(len(dialogs) * rate))
        return random.sample(dialogs, n)

    def score_dialog(self, dialog: DialogRecord, scores: RubricScore) -> dict:
        avg = (scores.correctness + scores.politeness + scores.efficiency + scores.compliance) / 4
        return {"dialog_id": dialog.session_id, "scores": scores, "average": avg}

A/B 实验

python
"""A/B 实验框架"""

from scipy import stats
from typing import Tuple

class ABExperiment:
    """A/B 实验"""

    def run(
        self,
        control_metrics: list,
        treatment_metrics: list,
        metric_name: str,
    ) -> Tuple[float, float, bool]:
        """运行 A/B 实验"""
        t_stat, p_value = stats.ttest_ind(control_metrics, treatment_metrics)
        control_mean = sum(control_metrics) / len(control_metrics)
        treatment_mean = sum(treatment_metrics) / len(treatment_metrics)
        significant = p_value < 0.05
        return control_mean, treatment_mean, significant

    def plan_experiment(
        self, min_effect: float = 0.02, power: float = 0.8, alpha: float = 0.05
    ) -> int:
        """计算所需样本量"""
        from statsmodels.stats.power import TTestIndPower
        analysis = TTestIndPower()
        sample_size = analysis.solve_power(
            effect_size=min_effect / 0.15,  # 假设标准差 0.15
            power=power, alpha=alpha,
        )
        return int(sample_size)

if __name__ == "__main__":
    exp = ABExperiment()
    control = [0.7 + i/1000 for i in range(1000)]
    treatment = [0.75 + i/1000 for i in range(1000)]
    c_mean, t_mean, sig = exp.run(control, treatment, "FCR")
    print(f"Control FCR: {c_mean:.3f}, Treatment FCR: {t_mean:.3f}, Significant: {sig}")
    # 输出: Control FCR: 0.720, Treatment FCR: 0.770, Significant: True

  • 自动评测覆盖 FCR/转人工率/CSAT/响应时长/准确率 5 个指标。

评测自动化与持续集成

评测不能是「上线前跑一次」的手动活动。CorpAssist 将评测流程集成到 CI/CD 管线中，每次代码变更自动触发。

yaml
# .github/workflows/eval.yml
name: 客服评测
on:
  pull_request:
    branches: [main]
jobs:
  evaluate:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: 运行自动评测
        run: |
          python -m pytest tests/test_eval.py -v
          python -m src.evaluation.auto_eval --dataset test_suite_v2
      - name: 检查指标
        run: |
          python scripts/check_metrics.py \
            --fcr-threshold 0.80 \
            --transfer-threshold 0.32 \
            --csat-threshold 3.8
      - name: 坏例回归
        run: |
          python scripts/regression_test.py \
            --badcase-file data/bad_cases.jsonl
      - name: 生成报告
        run: python scripts/generate_eval_report.py

CI 门禁规则：FCR < 80% 或转人工率 > 32% 时阻断合并请求。这条规则在 2026 年阻止了 8 次可能导致性能回退的合并。其中一次是意图分类器的 ONNX 模型文件被误替换成未训练的版本——CI 评测立即检测到准确率从 95% 降到 42%，阻止了这个严重故障上线。
