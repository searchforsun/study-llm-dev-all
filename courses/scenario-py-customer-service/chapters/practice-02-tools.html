LangChain @tool 装饰器与工具定义

LangChain 的 `@tool` 装饰器是定义 Agent 工具最简洁的方式——它自动从函数签名和 docstring 生成 JSON Schema，让 LLM 知道这个工具能做什么、需要什么参数。CorpAssist 定义了三个核心工具：OrderTool（查询订单）、RefundTool（提交退款）、AddressTool（修改地址）。每个工具都遵循相同的设计模式——Pydantic v2 参数模型 + 自定义错误消息 + 幂等性检查 + 三级降级。

"""LangChain @tool 装饰器定义工具"""
from langchain_core.tools import tool
from pydantic import BaseModel, Field, field_validator, model_validator
from typing import Optional, List, Dict, Any
from datetime import datetime, timedelta
import hashlib, json, time, asyncio
import httpx
from redis import Redis

redis_client = Redis(host="redis.internal", port=6379, decode_responses=True)

# ========== 参数模型 ==========
class OrderQueryParams(BaseModel):
    order_id: str = Field(..., description="订单号, 格式 ORD-8位数字")
    user_id: str = Field(..., description="用户 ID")

    @field_validator('order_id')
    @classmethod
    def validate_order_id(cls, v: str) -> str:
        if not v.startswith("ORD-") or len(v) != 12:
            raise ValueError(f"订单号格式错误: {v}, 预期格式 ORD-XXXXXXXX")
        return v

class RefundSubmitParams(BaseModel):
    order_id: str = Field(..., description="订单号")
    amount: float = Field(gt=0, le=50000, description="退款金额(元)")
    reason: str = Field(min_length=5, max_length=500, description="退款原因")

    @model_validator(mode='after')
    def check_amount_threshold(self):
        if self.amount > 5000 and "人工审核" not in self.reason:
            import warnings
            warnings.warn(f"大额退款 {self.amount} 元需要人工审核")
        return self

class AddressUpdateParams(BaseModel):
    order_id: str
    new_address: str = Field(..., min_length=10, max_length=200)
    phone: Optional[str] = None

    @field_validator('new_address')
    @classmethod
    def validate_address(cls, v: str) -> str:
        if len(v) < 10:
            raise ValueError(f"地址太短: {v}, 至少 10 个字符")
        if not any(c.isdigit() for c in v):
            raise ValueError("地址必须包含门牌号")
        return v

# ========== 工具实现 ==========
@tool("query_order", args_schema=OrderQueryParams)
async def query_order(order_id: str, user_id: str) -> str:
    """查询订单状态和物流信息"""
    # 幂等性检查：查询类操作天然幂等
    cache_key = f"order:{order_id}:{user_id}"
    cached = redis_client.get(cache_key)
    if cached:
        return cached

    # 模拟第三方 API 调用
    async with httpx.AsyncClient() as client:
        resp = await client.get(
            f"https://api.internal/orders/{order_id}",
            params={"user_id": user_id},
            timeout=5.0
        )
        data = resp.json()

        # 缓存 60 秒
        redis_client.setex(cache_key, 60, json.dumps(data))
        return json.dumps(data)

@tool("submit_refund", args_schema=RefundSubmitParams)
async def submit_refund(order_id: str, amount: float, reason: str) -> str:
    """提交退款申请。注意: 金额超过 5000 元需要说明已了解人工审核流程"""
    idempotency_key = f"refund:{order_id}:{hashlib.md5(json.dumps(
        {"amount": amount, "reason": reason}, sort_keys=True
    ).encode()).hexdigest()}"

    # 幂等性检查
    existing = redis_client.get(idempotency_key)
    if existing:
        return f"退款已提交 (幂等命中): {existing}"

    # 分布式锁 (SET NX)
    lock = redis_client.set(
        f"lock:refund:{order_id}", "locked",
        nx=True, ex=30  # 30s 自动释放
    )
    if not lock:
        return "该订单正在处理退款, 请稍后再试"

    try:
        async with httpx.AsyncClient() as client:
            resp = await client.post(
                "https://api.internal/refunds",
                json={"order_id": order_id, "amount": amount, "reason": reason},
                timeout=10.0
            )
            result = resp.text
            # 记录幂等 Key
            redis_client.setex(idempotency_key, 86400, result)
            return f"退款已提交: {result}"
    finally:
        redis_client.delete(f"lock:refund:{order_id}")

@tool("update_address", args_schema=AddressUpdateParams)
async def update_address(order_id: str, new_address: str,
                          phone: Optional[str] = None) -> str:
    """修改订单收货地址（仅未发货订单可修改）"""
    idempotency_key = f"addr:{order_id}:{hashlib.md5(new_address.encode()).hexdigest()[:16]}"
    existing = redis_client.get(idempotency_key)
    if existing:
        return f"地址已修改 (幂等命中)"

    async with httpx.AsyncClient() as client:
        resp = await client.put(
            f"https://api.internal/orders/{order_id}/address",
            json={"address": new_address, "phone": phone},
            timeout=5.0
        )
        result = resp.text
        redis_client.setex(idempotency_key, 86400, result)
        return f"地址已修改: {result}"

Pydantic v2 参数校验与自定义错误消息

工具调用的第一道关是参数校验——LLM 生成的参数经常不正确：订单号格式错了、退款金额超限了、地址太短了。Pydantic v2 的 `field_validator` 和 `model_validator` 可以捕获这些问题并返回清晰的自定义错误消息，让 LLM 能根据错误消息自动修正参数后重试。这个机制被称为「Tool Calling Self-Correction」——LLM 调用工具→收到参数错误→根据错误消息修正参数→再次调用。

python
"""参数校验与错误处理"""
from pydantic import ValidationError
from typing import Tuple, Optional, Any

class ToolParamValidator:
    """工具参数校验器"""

    @staticmethod
    def validate(params: BaseModel) -> Tuple[bool, Optional[str], Optional[BaseModel]]:
        """
        校验参数
        返回: (成功?, 错误消息, 修正后的参数)
        """
        try:
            validated = params.model_validate(params.model_dump())
            return True, None, validated
        except ValidationError as e:
            # 提取用户友好的错误消息
            errors = []
            for err in e.errors():
                field = ".".join(str(loc) for loc in err["loc"])
                msg = err["msg"]
                errors.append(f"字段 [{field}]: {msg}")
            error_msg = "; ".join(errors)

            # 尝试自动修正常见错误
            corrected = ToolParamValidator._auto_fix(params, e)
            if corrected:
                return True, f"参数已自动修正: {error_msg}", corrected
            return False, error_msg, None

    @staticmethod
    def _auto_fix(params: BaseModel, error: ValidationError) -> Optional[BaseModel]:
        """自动修正常见参数错误"""
        fixed_data = params.model_dump()
        for err in error.errors():
            field = err["loc"][0] if err["loc"] else ""
            if field == "order_id" and "格式" in err["msg"]:
                val = fixed_data.get(field, "")
                # 自动补 ORD- 前缀
                if val and not val.startswith("ORD-"):
                    fixed_data[field] = f"ORD-{val}"
            elif field == "amount" and "less_than_equal" in err["type"]:
                fixed_data[field] = 50000  # 截断到最大值
        try:
            return params.__class__(**fixed_data)
        except:
            return None

CorpAssist 工具调用场景

用户说「帮我退一下 ORD-2026001 这个订单，质量有问题」。LLM 调用 submit_refund(order_id="ORD-2026001", amount=0, reason="质量有问题")——amount=0 触发 Pydantic 校验错误「金额必须大于 0」。LLM 收到错误后意识到需要补全额，重新调用 submit_refund(order_id="ORD-2026001", amount=2999, reason="商品质量问题")，这次校验通过。从错误到修正的完整过程耗时约 1.5s，比直接让 LLM 自己猜金额的准确率从 45% 提升到 92%。

Redis 分布式锁 + 幂等性控制

工具调用最严重的问题不是失败而是「重复执行」——用户退款请求因为网络超时被重复提交、Agent 的重试机制导致同个操作执行两次、用户在前端点了两次按钮。CorpAssist 的幂等性控制分两层：幂等性 Token（Idempotency Key）防止重复提交，分布式锁（Redis SET NX）防止并发冲突。

python复制
"""Redis 分布式锁与幂等性实现"""
import hashlib, json, time, uuid
from redis import Redis
from typing import Optional, Callable, Awaitable, Any
from contextlib import asynccontextmanager

class IdempotencyManager:
    """幂等性管理器"""

    def __init__(self, redis: Redis, default_ttl: int = 86400):
        self.redis = redis
        self.default_ttl = default_ttl

    def make_key(self, session_id: str, tool_name: str,
                 params: dict) -> str:
        """生成幂等 Key"""
        param_hash = hashlib.sha256(
            json.dumps(params, sort_keys=True).encode()
        ).hexdigest()[:16]
        return f"idempotent:{session_id}:{tool_name}:{param_hash}"

    async def execute_once(self, key: str, ttl: int,
                           fn: Callable[[], Awaitable[Any]]
                           ) -> tuple[bool, Any]:
        """确保函数最多执行一次"""
        # 先检查是否已执行
        existing = self.redis.get(key)
        if existing:
            return True, json.loads(existing)

        # SET NX 锁，防止并发
        lock_value = str(uuid.uuid4())
        acquired = self.redis.set(
            f"lock:{key}", lock_value,
            nx=True, ex=30
        )
        if not acquired:
            return False, "操作正在处理中，请稍后"

        try:
            result = await fn()
            self.redis.setex(key, ttl, json.dumps({
                "result": str(result),
                "timestamp": time.time()
            }))
            return True, result
        except Exception as e:
            self.redis.delete(f"lock:{key}")
            raise
        finally:
            # 只删除自己加的锁
            current = self.redis.get(f"lock:{key}")
            if current and current.decode() == lock_value:
                self.redis.delete(f"lock:{key}")

# 基于 Redlock 的分布式锁
class DistributedLock:
    """分布式锁 (Redlock 简化版)"""

    def __init__(self, redis_nodes: list[Redis], ttl: int = 30):
        self.nodes = redis_nodes
        self.ttl = ttl
        self.quorum = len(redis_nodes) // 2 + 1

    @asynccontextmanager
    async def acquire(self, resource: str):
        lock_id = str(uuid.uuid4())
        acquired = 0
        start = time.time()

        for node in self.nodes:
            try:
                if node.set(f"redlock:{resource}", lock_id,
                            nx=True, ex=self.ttl):
                    acquired += 1
            except:
                pass

        if acquired >= self.quorum and (time.time() - start) < self.ttl:
            try:
                yield True
            finally:
                for node in self.nodes:
                    try:
                        if node.get(f"redlock:{resource}") == lock_id:
                            node.delete(f"redlock:{resource}")
                    except:
                        pass
        else:
            yield False
            # 回滚已获取的锁
            for node in self.nodes:
                try:
                    if node.get(f"redlock:{resource}") == lock_id:
                        node.delete(f"redlock:{resource}")
                except:
                    pass

idempotent = IdempotencyManager(redis_client)

async def submit_refund_safe(order_id: str, amount: float):
    key = idempotent.make_key("SID-38291", "submit_refund",
                                {"order_id": order_id, "amount": amount})
    async def _do_refund():
        async with httpx.AsyncClient() as client:
            resp = await client.post("https://api.internal/refunds",
                json={"order_id": order_id, "amount": amount})
            return resp.text

    ok, result = await idempotent.execute_once(key, 86400, _do_refund)
    if ok:
        return result
    else:
        return f"操作冲突: {result}"
方案实现适用场景优势局限性
幂等性 Token`idempotent:{session}:{tool}:{param_hash}` 存储执行结果同一参数的重复请求简单可靠，TTL 可控不同参数无法去重
SET NX 锁`lock:{resource}` = uuid, ex=30s单实例并发控制原子操作，性能好单点故障，不支持分布式
Redlock多数 Redis 节点同时加锁分布式环境互斥高可用，自动故障转移实现复杂，需要奇数节点
数据库乐观锁版本号检查 `WHERE version=N`修改类操作与业务数据一致性需要数据库支持

三级降级：重试→简化查询→人工确认

工具调用不可能 100% 成功——第三方 API 可能超时、参数可能始终无法通过校验、业务规则可能不被满足。CorpAssist 的三级降级策略确保工具调用在遇到问题时不是简单报错，而是逐级尝试替代方案。每一级降级都记录日志用于后续分析。

python复制
"""三级降级策略"""
from typing import Callable, Awaitable, Any, Optional
from dataclasses import dataclass
import asyncio, logging

logger = logging.getLogger(__name__)

@dataclass
class DegradationResult:
    success: bool
    data: Any = None
    degradation_level: int = 0   # 0=原始, 1=重试, 2=简化, 3=人工
    error: Optional[str] = None

class ToolDegradationManager:
    """工具降级管理器"""

    def __init__(self, max_retries: int = 2):
        self.max_retries = max_retries

    async def execute_with_degradation(
        self,
        primary_fn: Callable[[], Awaitable[Any]],
        simplified_fn: Optional[Callable[[], Awaitable[Any]]] = None,
        human_confirmation_fn: Optional[Callable[[str], Awaitable[Any]]] = None,
        context: str = "",
    ) -> DegradationResult:
        """执行带三级降级的工具调用"""
        # 第一级：原始调用 + 重试
        last_error = None
        for attempt in range(1 + self.max_retries):
            try:
                if attempt > 0:
                    wait = 2 ** attempt  # 指数退避 2s, 4s
                    logger.info(f"重试第 {attempt} 次, 等待 {wait}s")
                    await asyncio.sleep(wait)
                result = await primary_fn()
                return DegradationResult(
                    success=True, data=result,
                    degradation_level=0
                )
            except httpx.TimeoutException as e:
                last_error = str(e)
                logger.warning(f"超时 (attempt {attempt + 1}): {e}")
            except httpx.HTTPStatusError as e:
                if e.response.status_code in (502, 503, 504):
                    last_error = str(e)
                    continue  # 服务端错误可以重试
                return DegradationResult(
                    success=False, error=str(e),
                    degradation_level=0
                )
            except Exception as e:
                last_error = str(e)
                break

        # 第二级：简化查询
        if simplified_fn:
            try:
                result = await simplified_fn()
                logger.info(f"降级到简化查询: {context}")
                return DegradationResult(
                    success=True, data=result,
                    degradation_level=1, error=last_error
                )
            except Exception as e:
                last_error = str(e)

        # 第三级：人工确认
        if human_confirmation_fn:
            try:
                result = await human_confirmation_fn(
                    f"工具调用失败，请手动处理。上下文: {context}, "
                    f"错误: {last_error}"
                )
                return DegradationResult(
                    success=True, data=result,
                    degradation_level=2, error=last_error
                )
            except Exception as e:
                last_error = str(e)

        return DegradationResult(
            success=False, error=last_error,
            degradation_level=3
        )

# 使用示例
async def demo_degradation():
    manager = ToolDegradationManager(max_retries=2)

    async def primary():
        raise httpx.TimeoutException("工单 API 超时")

    async def simplified():
        return "模糊查询结果: 工单 TKT-001 状态为处理中"

    result = await manager.execute_with_degradation(
        primary,
        simplified,
        context="query_order(order_id=TKT-001)"
    )
    print(f"降级级别: {result.degradation_level}, 数据: {result.data}")
    # 输出: 降级级别: 1, 数据: 模糊查询结果: 工单 TKT-001 状态为处理中
降级级别触发条件操作延迟用户体验成功率
0 - 正常API 正常响应直接返回结果< 1s正常92%
1 - 重试超时 / 5xx 错误指数退避重试（最多 2 次）< 7s短暂等待后成功提升 4%（→96%）
2 - 简化精确查询失败改用模糊查询或缓存数据< 2s信息可能不完全精确提升 2.5%（→98.5%）
3 - 人工前两级都失败生成工单转人工处理30s+等待人工提升 0.7%（→99.2%）

FastAPI 工具执行器端点

工具层暴露为独立的 FastAPI 微服务，负责接收对话引擎的 Tool Call 请求，执行参数校验、幂等性检查、分布式锁、HTTP 调用和降级处理。所有调用都记录审计日志用于事后分析。

python复制
"""FastAPI 工具执行器服务"""
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import time, logging

app = FastAPI(title="CorpAssist 工具执行器", version="1.0.0")
idempotent = IdempotencyManager(redis_client)
degradation = ToolDegradationManager()

class ToolExecuteRequest(BaseModel):
    session_id: str
    tool_name: str     # query_order / submit_refund / update_address
    params: dict
    idempotency_key: str = ""

tool_registry = {
    "query_order": query_order,
    "submit_refund": submit_refund,
    "update_address": update_address,
}

@app.post("/tool/execute")
async def execute_tool(req: ToolExecuteRequest):
    start = time.time()
    tool_fn = tool_registry.get(req.tool_name)
    if not tool_fn:
        raise HTTPException(status_code=404, detail=f"工具 {req.tool_name} 未注册")

    # 执行（含幂等和降级）
    result = await degradation.execute_with_degradation(
        primary=lambda: tool_fn(**req.params),
        context=f"{req.tool_name}({req.params})"
    )

    latency = (time.time() - start) * 1000
    # 记录审计日志
    logger.info(
        f"tool_call session={req.session_id} tool={req.tool_name} "
        f"success={result.success} level={result.degradation_level} "
        f"latency={latency:.0f}ms"
    )

    return {
        "success": result.success,
        "data": result.data,
        "degradation_level": result.degradation_level,
        "latency_ms": round(latency, 1),
        "error": result.error
    }

@app.get("/health")
async def health():
    return {"status": "ok", "service": "tool-executor"}

线上效果数据：CorpAssist 工具层日均执行约 2800 次工具调用。P99 延迟 850ms（在 <1.0s 目标内），其中 submit_refund 的 P99 最高（950ms，因涉及写操作和锁）。三级降级使总成功率从 92% 提升到 99.2%。幂等性控制拦截了约 3.2% 的重复请求（日均 90 次）。分布式锁在退款操作上的冲突率约 0.5%（日均 14 次），冲突时返回「请稍后重试」，用户通常 3-5 秒后重试成功。

本章结论

  • LangChain @tool 装饰器配合 Pydantic v2 的 Field/field_validator/model_validator 实现了自动 Schema 生成 + 参数校验 + 自修正三位一体
  • Redis 幂等性 Token（SET NX + TTL=24h + 参数 Hash）确保修改类操作最多执行一次，拦截 3.2% 的重复请求
  • 三级降级（重试→简化→人工）将工具调用成功率从 92% 提升到 99.2%，降级到人工的仅占 0.7%
  • 工具执行器作为独立微服务，P99 延迟 850ms，熔断器在错误率 >30% 时自动打开 30s
  • 参数校验的自修正机制让 LLM 的工具调用首次成功率从 45% 提升到 92%，大幅减少无效重试

面试可能问

Q1: 工具调用怎么防止重复操作？用户点了两次退款按钮怎么办？

防止重复操作需要前端和后端协同。前端方面：按钮点击后立即 disable 并显示「提交中」，防止用户连点。后端方面：CorpAssist 使用两层防护。第一层是幂等性 Token（Idempotency Key）——每个工具调用请求携带一个从参数 + session_id 派生的 Key，Redis 中存储执行结果，重复 Key 直接返回缓存结果，不执行实际操作。第二层是分布式锁——使用 Redis SET NX 命令对资源加锁（比如 `lock:refund:ORD-2026001`），锁 TTL=30 秒自动释放，防止同一资源的并发操作。对于「用户点了两次退款按钮」的场景：第一次点击生成退款请求，幂等 Key 写入 Redis；第二次点击同样的 Key 被检测到已存在，直接返回「退款已提交，无需重复操作」。线上数据显示幂等性控制日均拦截约 90 次重复请求（占总请求 3.2%），分布式锁的冲突率约 0.5%。如果第三方系统支持 Idempotency-Key Header 则优先使用，否则在工具层自己做去重。

Q2: 第三方 API 超时或报错怎么办？怎么保证工具调用不堵死整个客服系统？

第三方 API 超时是客服系统最常见的故障模式。CorpAssist 的防护策略分三个层次。第一层是超时控制——每个工具调用设置 timeout_ms 参数（默认 5000ms，submit_refund 设 10000ms），使用 httpx.AsyncClient 的 timeout 参数控制，超时时抛出 TimeoutException 而不是无限等待。第二层是熔断器——工具执行器维护一个每个工具的指标计数器：过去 60 秒内如果错误率 >30% 或 P99 延迟 >2s，熔断器打开，30 秒内所有对该工具的请求直接返回「服务暂不可用」，不实际调用 API。熔断器使用 Python `pybreaker` 库实现，支持半开状态自动恢复。第三层是线程池隔离——每个工具有独立的 asyncio semaphore（信号量），默认最大并发数 10 个。当一个工具的调用占满所有 Slot 时，新请求排队等待而不是耗尽全局资源。CorpAssist 曾有一次工单系统故障导致所有工具调用都在等待——如果没有熔断器，工具执行器的 50 个工作线程会被全部占满，影响所有对话。熔断器打开后，30 秒内只有 5% 的请求受影响（通过熔断快速失败），其余 95% 的使用缓存数据或降级方案正常处理。

Q3: LLM 调用工具时参数经常填错怎么办？怎么让 LLM 自动修正？

LLM 调用工具时参数错误是常见问题——它可能传空字符串、错误格式的订单号、超限额的退款金额。CorpAssist 用「错误消息驱动自修正」机制处理：Pydantic v2 的 field_validator 和 model_validator 在校验失败时返回格式化的错误消息，这些消息被返回给 LLM 作为 Tool Call Error。LLM 读取错误消息后，会根据错误描述修正参数并重新调用。例如：LLM 调用 submit_refund(amount=0)，Pydantic 返回「字段 [amount]: Input should be greater than 0」。LLM 理解到「金额需要填实际值」，重新调用 submit_refund(amount=2999)，校验通过。这个自修正循环一般需要 1-2 次重试，增加约 1.5s 延迟。为了让 LLM 更容易自修正，我们在每个工具参数的 Field 中写了详细的 description（如「订单号格式 ORD-8位数字」）和自定义错误消息，LLM 在调用前就能理解参数规范。同时，系统在 Prompt 中注明了常见错误案例——「注意：退款金额 amount 必须 >0 且 <=50000，不要传 0 或负数」。这些措施使 LLM 工具调用的首次参数正确率从 45% 提升到 92%。

判断练习

如果 query_order 工具连续 3 次都超时，但用户查的是昨天查过的订单，应该怎么处理？

这个场景触发了三级降级。第一级重试（2 次指数退避 2s/4s）已经用完了，第二级简化查询可以考虑使用缓存——如果该订单在 60 秒内被查询过（Redis 缓存命中），直接返回缓存数据，虽然可能不是最新状态但总比报错好。第三级人工生成一条「工具调用降级通知」给前端，让对话引擎回复「系统暂时无法查询工单状态，建议您稍后重试或联系人工客服」，同时后台生成异步重试任务（Celery 任务，5 分钟后自动重试）。如果自动重试后成功查询到结果，通过 WebSocket 推送给用户的客服对话界面。这个策略让工具层的最终成功率（含降级和异步重试）达到了 99.8%。关键原则是「不要让用户干等，给一个可行方案比不给方案好」。

同一个用户同时从 Web 端和手机端发起退款请求，怎么防止重复退款？

这是典型的分布式并发问题。即使前端禁用了按钮，用户仍然可以同时打开两个页面发起同样的操作。CorpAssist 的防护机制是分层防御。第一层：幂等性 Token 基于参数 Hash——两个请求的参数完全相同（相同订单号、相同金额），生成的 Idempotency Key 也相同，后者直接返回「退款已提交，无需重复操作」，不执行实际退款。第二层：分布式锁基于 Redis SET NX——如果幂等性没有拦截（比如参数略有不同——一个传了备注另一个没传），锁 `lock:refund:ORD-2026001` 确保同一订单的退款操作串行执行。第三层：第三方 API 自身的幂等性——工单系统的退款接口支持 Idempotency-Key Header，即使前两层都失效，第三方也会拒绝重复的退款请求。这三层防御使 CorpAssist 在 6 个月的运行中实现了零次重复退款事故。如果用户确实需要修改退款（比如从部分退款改成全额退款），需要先取消前一个退款请求，再提交新的——这个过程需要人工审核，不走自动工具调用。
