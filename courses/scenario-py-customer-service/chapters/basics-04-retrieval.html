政策库 RAG

"""政策库 RAG 检索"""

from datetime import date
from typing import List, Dict

class PolicyRetriever:
    """政策文档检索器（带版本过滤）"""

    def __init__(self, es_client, milvus_client):
        self.es = es_client
        self.milvus = milvus_client

    def search(
        self, query: str, top_k: int = 5
    ) -> List[Dict]:
        """检索当前生效的政策文档"""
        today = date.today()
        # 构建过滤条件
        filter_dsl = {
            "bool": {
                "must": [
                    {"multi_match": {"query": query, "fields": ["title^3", "content"]}},
                ],
                "filter": [
                    {"term": {"status": "active"}},
                    {"range": {"effective_from": {"lte": today.isoformat()}}},
                ],
            }
        }
        resp = self.es.search(index="policies", query=filter_dsl, size=top_k)
        return [
            {"doc_id": h["_id"], "title": h["_source"].get("title"),
             "content": h["_source"].get("content"),
             "version": h["_source"].get("version")}
            for h in resp["hits"]["hits"]
        ]

    def get_policy_version(self, policy_id: str) -> Dict:
        """获取政策版本信息"""
        resp = self.es.get(index="policies", id=policy_id)
        src = resp["_source"]
        return {
            "title": src["title"],
            "version": src["version"],
            "effective_from": src["effective_from"],
            "effective_to": src.get("effective_to"),
            "status": src["status"],
        }

话术模板与变量注入

标准话术确保了客服回答的一致性。CorpAssist 使用 Jinja2 模板引擎管理话术。

python
"""话术模板引擎"""

from jinja2 import Template
from typing import Dict

# 话术模板库
TEMPLATES = {
    "password_reset": Template(
        "您好，{{ user_name }}。您的 {{ app_name }} 密码已重置成功。"
        "新密码已发送至您的注册手机 {{ phone_tail }}，有效期 24 小时。"
        "如有问题请随时联系。"
    ),
    "ticket_status": Template(
        "工单 {{ ticket_id }} 当前状态为「{{ status }}」。"
        "当前处理人：{{ assignee }}。预计完成时间：{{ eta }}。"
        "如需加急请回复「加急」。"
    ),
    "refund_policy": Template(
        "根据《退款政策》v{{ policy_version }}，您的退款申请 "
        "{{ '符合' if eligible else '不符合' }}条件。"
        "详情请参考：{{ policy_link }}"
    ),
}

class ScriptEngine:
    """话术引擎"""

    def render(self, template_key: str, params: Dict) -> str:
        template = TEMPLATES.get(template_key)
        if not template:
            return "抱歉，我暂时无法回答这个问题。"
        return template.render(**params)

if __name__ == "__main__":
    engine = ScriptEngine()
    msg = engine.render("ticket_status", {
        "ticket_id": "TKT-001", "status": "财务审批中",
        "assignee": "张三", "eta": "本周五",
    })
    print(msg)
    # 工单 TKT-001 当前状态为「财务审批中」...

输出内容安全审核

客服回答必须经过审核后才能发给用户。CorpAssist 的输出审核管线分三步。

python
"""输出内容安全审核"""

import re
from typing import Tuple

class OutputCensor:
    """输出内容安全审核器"""

    SENSITIVE_PATTERNS = [
        r"手机[号]?[:：]?\d{11}",
        r"身份证[:：]?\d{17}[\dXx]",
        r"银行卡[:：]?\d{16,19}",
        r"密码[:：]?\S{6,}",
    ]

    POLICY_KEYWORDS = {
        "退款": ["退款政策", "退货政策"],
        "投诉": ["投诉处理流程"],
        "赔偿": ["赔偿标准"],
    }

    def __init__(self):
        self.sensitive_regex = [
            re.compile(p) for p in self.SENSITIVE_PATTERNS
        ]

    def check_sensitive_data(self, text: str) -> Tuple[bool, str]:
        """检查是否泄漏敏感信息"""
        for regex in self.sensitive_regex:
            match = regex.search(text)
            if match:
                return False, f"泄漏敏感信息: {match.group()[:20]}..."
        return True, ""

    def check_policy_compliance(self, text: str, policy_docs: list) -> Tuple[bool, str]:
        """检查回答是否与政策一致"""
        for keyword, related_policies in self.POLICY_KEYWORDS.items():
            if keyword in text:
                # 检查是否引用了相关政策的文档
                has_ref = any(
                    p["title"] in text for p in policy_docs
                    for title in related_policies
                )
                if not has_ref:
                    return False, f"提到'{keyword}'但未引用相关政策"
        return True, ""

    def censor(self, text: str, policy_docs: list) -> Tuple[bool, str]:
        """完整审核链"""
        ok, reason = self.check_sensitive_data(text)
        if not ok:
            return False, reason
        ok, reason = self.check_policy_compliance(text, policy_docs)
        if not ok:
            return False, reason
        return True, "通过"

if __name__ == "__main__":
    censor = OutputCensor()
    print(censor.censor("退款需在7天内申请", [{"title": "退款政策"}]))
    # (True, '通过')
    print(censor.censor("你的密码是abc123", []))
    # (False, '泄漏敏感信息: 密码:abc123...')

面试可能问

Q1: 如果 AI 客服回答错了政策内容，造成了业务损失，怎么溯源和追责？

全链路审计日志记录了每次客服回答的完整信息：用户问题、检索到的政策文档（含版本号）、LLM 生成的完整回答、审核结果。如果出现错误，可以快速定位是「检索到过期政策」「LLM 错误理解」「还是审核漏过」。基于审计日志可以做追责——但更重要的是建立预防机制：政策版本强制过滤（不让过期政策被检索到）、回答必须引用政策文号才发出、敏感场景必须走人工复核。上线以来，政策相关的错误回答率为 0.3%。

Q2: 话术模板和 LLM 自由生成怎么平衡？

用「模板优先，LLM 兜底」策略。对于高频标准化问题（查工单/重置密码/查余额），直接调用话术模板——准确率 100%，延迟 < 10ms。对于模板未覆盖或需要个性化解释的问题，启用 LLM 生成，但生成结果必须经过政策 RAG 检索和输出审核。在我们的生产数据中，约 65% 的客服问题由模板覆盖，35% 由 LLM 生成。用户对两者的满意度分别为 4.7/5 和 4.1/5。
