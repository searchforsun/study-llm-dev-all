Python Agent 执行引擎

python agent run 使用 LangGraph 构建有状态的 Agent 执行引擎。每个 Agent 实例由一个 thread_id 标识，LangGraph 的 checkpointer 负责保存中间状态。

LangGraph Agent 实现

python
# Python LangGraph Agent - agent_executor.py
from typing import Literal, Annotated
from typing_extensions import TypedDict
from langgraph.graph import StateGraph, MessageGraph
from langgraph.checkpoint.memory import MemorySaver
from langchain_openai import ChatOpenAI
from langchain_core.messages import BaseMessage, HumanMessage
import json

class AgentState(TypedDict):
    messages: list[BaseMessage]
    session_id: str
    tenant_id: str
    user_id: str
    next_action: str

# 工具函数
async def query_order(state: AgentState) -> AgentState:
    """查询订单状态"""
    # 调用企业订单 API
    order_api = OrderAPI(state["tenant_id"])
    query = state["messages"][-1].content
    result = await order_api.query(query)
    state["messages"].append(
        AIMessage(content=json.dumps(result, ensure_ascii=False)))
    state["next_action"] = "continue"
    return state

async def query_ticket(state: AgentState) -> AgentState:
    """查询工单状态"""
    ticket_api = TicketAPI(state["tenant_id"])
    result = await ticket_api.query(state["messages"][-1].content)
    state["messages"].append(
        AIMessage(content=json.dumps(result, ensure_ascii=False)))
    state["next_action"] = "continue"
    return state

async def handoff_human(state: AgentState) -> AgentState:
    """转人工"""
    state["next_action"] = "handoff"
    state["messages"].append(
        AIMessage(content="我已将您的问题转接人工客服，请稍候。"))
    return state

# 编译 Agent 图
def create_agent():
    workflow = StateGraph(AgentState)

    workflow.add_node("agent", agent_node)
    workflow.add_node("query_order", query_order)
    workflow.add_node("query_ticket", query_ticket)
    workflow.add_node("handoff", handoff_human)

    workflow.set_entry_point("agent")

    workflow.add_conditional_edges(
        "agent",
        decide_next_action,
        {
            "query_order": "query_order",
            "query_ticket": "query_ticket",
            "handoff": "handoff",
            "end": "__end__",
        },
    )

    return workflow.compile(checkpointer=MemorySaver())

app = FastAPI()

@app.post("/v1/agent/run")
async def agent_run(
    request: AgentRunRequest,
    x_session_id: str = Header(None),
):
    """执行 Agent 推理（SSE 流式输出）"""
    agent = create_agent()
    thread_id = x_session_id or str(uuid.uuid4())

    config = {
        "configurable": {"thread_id": thread_id},
    }

    async def event_generator():
        async for event in agent.astream_events(
            {"messages": [HumanMessage(content=request.query)]},
            config,
            version="v2",
        ):
            if event["event"] == "on_chat_model_stream":
                yield f"data: {json.dumps({'token': event['data']['chunk'].content})}\n\n"

    return StreamingResponse(event_generator(), media_type="text/event-stream")

Spring 网关与会话管理

gateway agent 是 Spring 治理面中负责 Agent 会话管理的组件。

Spring Agent 会话管理

java
// Spring BFF - Agent 会话管理与 SSE 代理
@RestController
@RequestMapping("/api/v1/corpassist")
public class AgentController {

    private final WebClient pythonAgentClient;
    private final RedisTemplate<String, String> redis;

    @PostMapping(value = "/agent/run", produces = MediaType.TEXT_EVENT_STREAM_VALUE)
    public Flux<ServerSentEvent<?>> agentRun(
            @RequestBody AgentRunRequest request,
            @RequestHeader("X-User-Id") String userId,
            @RequestHeader("X-Tenant-Id") String tenantId) {

        // 1. 获取或创建会话
        String sessionId = resolveSessionId(request, userId, tenantId);

        // 2. 记录会话开始
        recordSessionStart(sessionId, userId, tenantId, request);

        // 3. 调用 Python Agent（SSE 代理）
        return pythonAgentClient.post()
            .uri("/v1/agent/run")
            .header("X-Session-Id", sessionId)
            .header("X-User-Id", userId)
            .header("X-Tenant-Id", tenantId)
            .bodyValue(request)
            .retrieve()
            .bodyToFlux(new ParameterizedTypeReference<ServerSentEvent<String>>() {})
            .doOnNext(event -> recordAgentEvent(sessionId, event))
            .doOnComplete(() -> recordSessionEnd(sessionId));
    }

    private String resolveSessionId(AgentRunRequest req,
                                     String userId, String tenantId) {
        if (req.sessionId() != null && !req.sessionId().isBlank()) {
            return req.sessionId();
        }
        return "session:" + tenantId + ":" + userId + ":" + UUID.randomUUID();
    }

    private void recordSessionStart(String sessionId, String userId,
                                     String tenantId, AgentRunRequest req) {
        Map<String, String> session = Map.of(
            "user_id", userId,
            "tenant_id", tenantId,
            "query", req.query(),
            "started_at", Instant.now().toString());
        redis.opsForHash().putAll("agent:session:" + sessionId, session);
        redis.expire("agent:session:" + sessionId, Duration.ofHours(24));
    }

    private void recordSessionEnd(String sessionId) {
        redis.opsForHash().put(
            "agent:session:" + sessionId,
            "ended_at", Instant.now().toString());
    }
}

转人工策略

s2 s3 bridge 指 Agent 会话从 S2（AI 自动处理）升级到 S3（人工处理）的桥接策略。

转人工的触发条件：

  • Agent 主动触发：Agent 判断超出能力范围，在 LangGraph 中发出 handoff 信号
  • 用户要求：用户输入「转人工」「找客服」等关键词
  • 超时/失败：Agent 超过最大推理步数或连续失败 N 次
python
# Agent 转人工逻辑
class HandoffSignal(Exception):
    """转人工信号"""
    def __init__(self, reason: str, summary: str):
        self.reason = reason
        self.summary = summary

def decide_next_action(state: AgentState) -> str:
    """决定下一步动作"""
    messages = state["messages"]

    # 检查是否触发了转人工
    if len(messages) > 20:  # 超过 10 轮对话
        return "handoff"

    last_message = messages[-1].content if messages else ""

    # 用户要求转人工
    if any(kw in last_message for kw in ["转人工", "人工客服", "投诉"]):
        return "handoff"

    # LLM 判断是否需要查询工具
    if "需要查订单" in last_message or "查询订单" in last_message:
        return "query_order"
    if "查询工单" in last_message or "工单进度" in last_message:
        return "query_ticket"

    return "end"

常见问题（FAQ）

Spring 为什么不做 Agent 推理而交给 Python？
Agent 推理的核心是 LangGraph 的图执行引擎、checkpointer 状态管理、工具调用编排。这些功能在 Python 生态（LangChain/LangGraph）中非常成熟，Java 端的 Spring AI 虽然支持 Agent，但在复杂工作流、图状态管理、流式输出等方面还不如 Python 完善。CorpAssist 将推理执行放在 Python，Spring 做治理，各取所长。
Agent 执行时间很长（30秒+），如何处理超时？
BFF 对 Python Agent 的调用设置 120 秒 的 read-timeout。同时，Agent 本身需要实现 streaming（SSE），让用户能实时看到推理过程。如果 Agent 超时，BFF 返回部分结果并提示用户「处理时间过长，请稍后查看结果」。如果 Agent 超过最大步数（如 20 步），LangGraph 自动终止并返回已收集的信息。
如何保证 Agent 工具调用的安全性和可审计性？
tool audit 通过 Spring AOP 或过滤器实现：BFF 记录每次 Agent 请求中调用的工具名称、参数、返回值和耗时。Python 的每个工具函数也需要在调用前后记录日志。Redis 中存储每个 session 的工具调用链，方便事后审计。
Agent 会话如何跨请求保持状态？
LangGraph 的 checkpointer 在每个步骤后自动保存状态到 MemorySaver（内存/REDIS 后端）。Spring BFF 在下一次请求时通过 X-Session-Id 头传递给 Python Agent，LangGraph 从 checkpointer 恢复图状态。会话元数据（用户信息、开始/结束时间）由 Spring 存储在 Redis 中。
