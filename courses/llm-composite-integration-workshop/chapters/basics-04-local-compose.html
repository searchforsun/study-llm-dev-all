为什么要学本章

本地一键起 Spring + Python + Redis + 向量库是双栈联调的前提。如果每个开发者都要手动启动 6 个服务、配置网络、等依赖就绪，开发效率会非常低。Docker Compose 将整个 local stack 定义为可重复的声明式配置。本章基于 docker compose 搭建 CorpAssist 完整的本地开发栈。

学完你能

  • 设计 service network 实现服务间通信
  • 配置 env config 使用 .env 文件管理环境变量

CorpAssist 本地开发栈包含 9 个服务，涵盖 Spring 治理面、Python 能力面和基础设施。

完整 docker-compose.yml

yaml
version: "3.8"

services:
  # ============================================================
  # 基础设施层
  # ============================================================
  redis:
    image: redis:7-alpine
    ports: ["6379:6379"]
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 5s
      timeout: 3s
      retries: 5
    volumes:
      - redis_data:/data
    networks: [corpassist-net]

  milvus:
    image: milvusdb/milvus:2.4-latest
    ports: ["19530:19530", "9091:9091"]
    environment:
      ETCD_ENDPOINTS: etcd:2379
      MINIO_ADDRESS: minio:9000
    depends_on:
      etcd: { condition: service_healthy }
      minio: { condition: service_healthy }
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:9091/health"]
      interval: 10s
      timeout: 5s
      retries: 10
    networks: [corpassist-net]

  etcd:
    image: quay.io/coreos/etcd:v3.5.9
    environment:
      ETCD_AUTO_COMPACTION_MODE: revision
      ETCD_AUTO_COMPACTION_RETENTION: "1000"
      ETCD_QUOTA_BACKEND_BYTES: "4294967296"
    healthcheck:
      test: ["CMD", "etcdctl", "endpoint", "health"]
      interval: 30s
      timeout: 10s
    networks: [corpassist-net]

  minio:
    image: minio/minio:latest
    ports: ["9000:9000", "9001:9001"]
    environment:
      MINIO_ROOT_USER: minioadmin
      MINIO_ROOT_PASSWORD: minioadmin
    command: server /data --console-address ":9001"
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:9000/minio/health/live"]
      interval: 30s
      timeout: 10s
      retries: 5
    networks: [corpassist-net]

  postgres:
    image: postgres:16-alpine
    ports: ["5432:5432"]
    environment:
      POSTGRES_DB: corpassist
      POSTGRES_USER: corpassist
      POSTGRES_PASSWORD: corpassist_dev
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U corpassist"]
      interval: 5s
      timeout: 3s
      retries: 5
    volumes:
      - pg_data:/var/lib/postgresql/data
    networks: [corpassist-net]

  rabbitmq:
    image: rabbitmq:3-management-alpine
    ports: ["5672:5672", "15672:15672"]
    environment:
      RABBITMQ_DEFAULT_USER: corpassist
      RABBITMQ_DEFAULT_PASS: corpassist_dev
    healthcheck:
      test: ["CMD", "rabbitmqctl", "status"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks: [corpassist-net]

  # ============================================================
  # Python 能力面
  # ============================================================
  python-rag:
    build:
      context: ./python-services
      dockerfile: Dockerfile.rag
    ports: ["8001:8001"]
    environment:
      - MILVUS_HOST=milvus
      - REDIS_HOST=redis
      - OPENAI_API_KEY=${OPENAI_API_KEY}
      - LOG_LEVEL=DEBUG
    env_file:
      - .env
    depends_on:
      milvus: { condition: service_healthy }
      redis: { condition: service_healthy }
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8001/health"]
      interval: 10s
      timeout: 5s
      retries: 10
      start_period: 30s
    networks: [corpassist-net]
    volumes:
      - ./python-services:/app  # 热重载

  python-agent:
    build:
      context: ./python-services
      dockerfile: Dockerfile.agent
    ports: ["8002:8002"]
    environment:
      - MILVUS_HOST=milvus
      - REDIS_HOST=redis
      - OPENAI_API_KEY=${OPENAI_API_KEY}
    env_file:
      - .env
    depends_on:
      milvus: { condition: service_healthy }
      redis: { condition: service_healthy }
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8002/health"]
      interval: 10s
      timeout: 5s
      retries: 10
      start_period: 30s
    networks: [corpassist-net]
    volumes:
      - ./python-services:/app

  # ============================================================
  # Spring 治理面
  # ============================================================
  auth-service:
    build:
      context: ./java-services/auth-service
    ports: ["8082:8082"]
    environment:
      - SPRING_PROFILES_ACTIVE=dev
      - DB_URL=jdbc:postgresql://postgres:5432/corpassist
      - DB_USERNAME=corpassist
      - DB_PASSWORD=corpassist_dev
    depends_on:
      postgres: { condition: service_healthy }
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8082/actuator/health"]
      interval: 10s
      timeout: 5s
      retries: 10
      start_period: 45s
    networks: [corpassist-net]

  spring-bff:
    build:
      context: ./java-services/spring-bff
    ports: ["8081:8081"]
    environment:
      - SPRING_PROFILES_ACTIVE=dev
      - PYTHON_RAG_URL=http://python-rag:8001
      - PYTHON_AGENT_URL=http://python-agent:8002
      - REDIS_HOST=redis
      - AUTH_SERVICE_URL=http://auth-service:8082
    depends_on:
      python-rag: { condition: service_healthy }
      redis: { condition: service_healthy }
      auth-service: { condition: service_healthy }
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8081/actuator/health"]
      interval: 10s
      timeout: 5s
      retries: 10
      start_period: 60s
    networks: [corpassist-net]

  spring-gateway:
    build:
      context: ./java-services/spring-gateway
    ports: ["8080:8080"]
    environment:
      - SPRING_PROFILES_ACTIVE=dev
      - BFF_URL=http://spring-bff:8081
    depends_on:
      spring-bff: { condition: service_healthy }
      auth-service: { condition: service_healthy }
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8080/actuator/health"]
      interval: 10s
      timeout: 5s
      retries: 10
      start_period: 60s
    networks: [corpassist-net]

volumes:
  redis_data:
  pg_data:

networks:
  corpassist-net:
    driver: bridge

启动顺序与健康检查

startup order 通过 depends_on + condition: service_healthy 控制。启动层级：

  1. 第 1 层：Redis, PostgreSQL, etcd, minio（基础中间件）
  2. 第 2 层：Milvus（依赖 etcd + minio）
  3. 第 3 层：Python RAG/Agent（依赖 Milvus + Redis）
  4. 第 4 层：Auth Service（依赖 PostgreSQL）
  5. 第 5 层：Spring BFF（依赖 Python + Redis + Auth）
  6. 第 6 层：Spring Gateway（依赖 BFF + Auth）

.env 环境变量文件

env config 通过 .env 文件集中管理环境变量，避免敏感信息硬编码：

bash
# .env - CorpAssist 本地开发环境变量
# 模型
OPENAI_API_KEY=sk-your-key-here
OPENAI_MODEL=gpt-4o

# 数据库
POSTGRES_DB=corpassist
POSTGRES_USER=corpassist
POSTGRES_PASSWORD=corpassist_dev

# Redis
REDIS_HOST=redis
REDIS_PORT=6379

MILVUS_PORT=19530

常用命令

bash
# 启动全部服务
docker compose --profile dev up -d

# 仅启动基础设施（开发 Python 时）
docker compose --profile infra up -d

# 查看所有服务日志
docker compose logs -f

# 查看特定服务日志
docker compose logs -f python-rag

# 重启单个服务（修改代码后）
docker compose restart python-rag

# 重新构建并启动（修改依赖后）
docker compose up -d --build spring-bff

# 停止全部
docker compose down

# 停止并清除数据卷
docker compose down -v

# 查看服务状态
docker compose ps

常见问题（FAQ）

首次启动 Milvus 需要很长时间，怎么办？
Milvus 首次启动需要初始化 etcd 和 MinIO，通常需要 1-3 分钟。建议：1) 使用 start_period 给 Milvus 更长的启动时间；2) 不要同时启动所有服务，先启动基础设施再启动应用服务；3) 使用卷持久化数据，避免每次重新初始化。
Python 热重载在容器中不生效？
确保：1) Dockerfile 中使用 CMD ["uvicorn", "main:app", "--reload", "--host", "0.0.0.0"]；2) Volume 挂载了正确的源码路径；3) 文件修改后容器确实能感知到（Windows 上可能需要设置文件共享）。如果依然不生效，可以使用 docker compose restart python-rag 手动重启。
如何调试 Spring 服务中的远程调用？
在 docker-compose 中为 Java 服务添加 JVM 调试参数：JAVA_TOOL_OPTIONS: "-agentlib:jdwp=transport=dt_socket,server=y,suspend=n,address=*:5005"。然后映射调试端口（如 5005:5005），使用 IDE 的 Remote JVM Debug 配置连接到 localhost:5005。
本地开发时需要模拟外部 OAuth2 提供者？
可以使用 testcontainers（Java 集成测试）或 Keycloak 开发容器。在 docker-compose 中添加 auth-service 时，可以配置 Keycloak 的 dev 模式，导入预定义的 realm 和用户。BFF 和 Gateway 的 dev profile 配置指向本地 Keycloak。

动手练习

基于 CorpAssist 场景编写完整的本地开发栈配置（约 30 分钟）。

操作步骤

  1. 编写 docker-compose.yml，包含基础设施/Spring/Python 共 9 个服务
  2. 为每个关键服务配置 healthcheck 和 start_period
  3. 编写 .env 文件管理环境变量
  4. 使用 docker compose up -d 启动所有服务并验证健康状态

判断练习

  1. 启动后发现 python-rag 一直重启，healthcheck 失败，如何排查？

    参考答案

    1) docker compose logs python-rag 查看错误日志；2) 检查 Milvus 是否健康；3) 检查 .env 中的 API Key 是否配置；4) 检查 Dockerfile 是否正确安装了所有依赖。

  2. 本地开发时不想每次都启动全部 9 个服务，有什么策略？

    参考答案

    1) 使用 Docker Compose profiles 将服务分组（如 infra/py/java/all）；2) 开发 Python 时仅启动 infra + python 服务；3) 使用 docker compose up <service> 只启动特定服务及其依赖。
