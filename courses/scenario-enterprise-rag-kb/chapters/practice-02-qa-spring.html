Spring AI 核心配置

java 
// CorpAssistApplication.java — Spring AI 问答服务
@SpringBootApplication
public class CorpAssistApplication {

    public static void main(String[] args) {
        SpringApplication.run(CorpAssistApplication.class, args);
    }

    @Bean
    public VectorStore milvusStore() {
        return new MilvusVectorStore(MilvusVectorStoreConfig.builder()
            .withUri("http://milvus:19530")
            .withCollectionName("corpassist_kb")
            .withIndexType(IndexType.IVF_FLAT)
            .withMetricType(MetricType.IP)
            .build());
    }

    @Bean
    public QuestionAnswerAdvisor qaAdvisor(VectorStore vectorStore) {
        return QuestionAnswerAdvisor.builder(vectorStore)
            .searchRequest(SearchRequest.builder()
                .topK(5)
                .similarityThreshold(0.5)
                .filterExpression("compliance_level == 'public'")
                .build())
            .build();
    }

    @Bean
    public ChatClient chatClient(ChatModel chatModel, QuestionAnswerAdvisor advisor) {
        return ChatClient.builder(chatModel)
            .defaultAdvisors(advisor)
            .build();
    }
}

引用溯源实现

citation source 是 CorpAssist 的生产强制要求。每个答案必须标注信息来源，格式为 [doc_id#chunk_index]。Spring AI 通过自定义 Advisor 和 Prompt 模板实现。

java 
// CustomCitationAdvisor.java — 强制引用溯源
@Component
public class CustomCitationAdvisor implements Advisor {

    @Override
    public AdvisedResponse advise(AdvisedRequest request) {
        // 从上下文中获取检索到的文档
        List documents = request.retrievedDocuments();
        if (documents == null || documents.isEmpty()) {
            return new AdvisedResponse(
                "抱歉，我在知识库中未找到相关信息。",
                request.params()
            );
        }

        // 构建引用标注
        StringBuilder citationBlock = new StringBuilder("\n\n**参考资料：**\n");
        for (int i = 0; i < documents.size(); i++) {
            Document doc = documents.get(i);
            String docId = doc.getMetadata().get("doc_id").toString();
            int chunkIdx = (int) doc.getMetadata().get("chunk_index");
            String source = doc.getMetadata().getOrDefault("source_file", "未知来源").toString();
            citationBlock.append(String.format(
                "[%d] %s#chunk-%d (%s)\n",
                i + 1, docId, chunkIdx, source
            ));
        }

        // 修改 prompt，要求 LLM 在答案中标注引用
        String enhancedPrompt = request.userText() + "\n\n"
            + "注意：对于答案中的每个事实，请在括号中标注来源文档编号，"
            + "例如「根据规定（来源：HR-001#chunk-3），年假为15天」。";

        return new AdvisedResponse(
            enhancedPrompt + citationBlock.toString(),
            request.params()
        );
    }
}

Python 端验证引用完整性

python 
# validate_citations.py — 验证引用完整性
import re
from typing import List

def validate_citation(answer: str, expected_doc_ids: List[str]) -> dict:
    """验证答案中的引用是否完整"""
    # 提取所有引用
    citations = re.findall(r'\[([A-Za-z0-9\-]+)#chunk-(\d+)\]', answer)

    if not citations:
        return {"valid": False, "reason": "答案中无任何引用标注"}

    cited_docs = set(c[0] for c in citations)
    expected_set = set(expected_doc_ids)

    missing_docs = expected_set - cited_docs
    if missing_docs:
        return {
            "valid": False,
            "reason": f"以下文档未被引用: {missing_docs}",
            "citations": citations,
        }

    return {"valid": True, "citations": citations, "count": len(citations)}


# 测试
answer = """根据公司规定（[HR-001#chunk-3]），员工年假为15天。
需提前3个工作日申请（[HR-001#chunk-5]）。"""
result = validate_citation(answer, ["HR-001"])
print(result)  # {"valid": True, ...}

SSE 流式问答

stream qa 是企业级 RAG 的体验标配。Spring AI 原生支持 Flux（Reactive Streams）流式输出。

流式问答时序
sequenceDiagram
    participant U as 用户
    participant API as Controller
    participant S as Spring AI
    participant LLM as LLM

    U->>API: GET /ask/stream?q=年假几天
    API->>S: stream() 返回 Flux
    S->>S: 检索文档 (同步, ~150ms)
    Note over S: 检索期间无输出
    S->>LLM: 发送 prompt + context
    LLM-->>S: token 0 (流式)
    S-->>API: data: {"token": "根据"}
    API-->>U: SSE: data: {"token": "根据"}
    LLM-->>S: token 1
    S-->>API: data: {"token": "公司"}
    API-->>U: SSE: data: {"token": "公司"}
    LLM-->>S: token N
    S-->>API: data: {"token": "。"}
    API-->>U: SSE: data: [DONE]

    Note over U: 首 token 延迟 ~350ms

检索阶段不可流式（需等完整结果），LLM 生成阶段逐 token SSE 推送。

Spring AI 流式控制器

java 
// QAController.java — 流式问答端点
@RestController
@RequestMapping("/api/qa")
public class QAController {

    private final ChatClient chatClient;

    public QAController(ChatClient chatClient) {
        this.chatClient = chatClient;
    }

    @GetMapping(value = "/stream", produces = MediaType.TEXT_EVENT_STREAM_VALUE)
    public Flux> streamAnswer(@RequestParam String question) {
        return chatClient.prompt()
            .user(question)
            .advisors(advisor -> advisor
                .param("tenant_id", getCurrentTenant())
                .param("user_role", getCurrentUserRole()))
            .stream()
            .content()
            .map(token -> ServerSentEvent.builder()
                .data(token)
                .event("token")
                .build())
            .concatWithValues(ServerSentEvent.builder()
                .data("[DONE]")
                .event("done")
                .build())
            .doOnError(e -> log.error("Stream error: {}", e.getMessage()));
    }

    @PostMapping("/ask")
    public Mono ask(@RequestBody QuestionRequest req) {
        return Mono.fromFuture(chatClient.prompt()
            .user(req.question())
            .advisors(advisor -> advisor
                .param("tenant_id", req.tenantId())
                .param("user_role", req.userRole()))
            .call()
            .content()
            .thenApply(content -> new AnswerResponse(content)));
    }

    private String getCurrentTenant() {
        // 从 SecurityContext 获取
        return SecurityContextHolder.getContext().getAuthentication()
            .getDetails().toString();
    }
}

# sse_client.py — SSE 流式客户端
import requests
import json

def stream_ask(url: str, question: str):
    """调用 CorpAssist 流式问答端点"""
    with requests.get(
        f"{url}/api/qa/stream",
        params={"question": question},
        stream=True,
        headers={"Accept": "text/event-stream"},
    ) as response:
        for line in response.iter_lines():
            if line:
                decoded = line.decode("utf-8")
                if decoded.startswith("data: "):
                    data = decoded[6:]
                    if data == "[DONE]":
                        break
                    print(data, end="", flush=True)

常见问题（FAQ）

Spring AI 和 LangChain 可以混用吗？
CorpAssist 架构中二者各司其职：LangChain（Python）做离线索引，Spring AI（Java）做在线问答。不混用在同一个进程中。通过向量库（Milvus）作为共享存储实现数据层面的协作。这是经过生产验证的 dual stack s1 模式。
流式输出时，引用怎么处理？
CorpAssist 采用「检索后置引用」方案：先流式输出答案文本，答案结束后推送一个引用块。这样首 Token 延迟不受影响，同时用户可以看到完整的引用信息。实现方式是在 Advisor 链中增加一个 PostGenerationAdvisor。
FilterExpression 支持复杂条件吗？
支持。Spring AI 的 FilterExpression 支持 AND/OR 组合、IN、CONTAINS、比较操作符。CorpAssist 常用组合：(compliance_level in ['public','internal']) AND (allowed_roles CONTAINS 'employee')。详见 权限、审计与多租户。
如果 LLM 调用失败，Spring AI 有重试机制吗？
有。Spring AI 的 ChatModel 内置重试机制，可通过 spring.ai.retry.max-attempts 和 spring.ai.retry.backoff 配置。CorpAssist 配置为最大 2 次重试，指数退避（初始 1s，最大 10s）。

本章结论

  • Spring AI 编排 RAG 流程：QuestionAnswerAdvisor 自动完成检索→增强→生成，Advisor 链可灵活扩展。
  • 引用溯源是安全基线：自定义 CitationAdvisor 强制答案标注来源，Python 端验证工具确保合规。
  • 流式是 A 类体验要求：Flux + SSE 实现首 Token 延迟 < 500ms，显著改善用户体验。
  • FilterExpression 实现 ACL：在检索层过滤用户的可见文档，确保数据安全。

判断练习

  1. 流式端点首 Token 延迟达到 4s，可能的原因和解决方案是什么？

    参考答案

    可能原因： 1) 检索阶段慢（向量库跨网络延迟 + BM25 RRF + Rerank）；2) Rerank 用 CPU 推理过慢；3) LLM 的 streaming 未正确开启。解决方案：将 Rerank 改用 GPU；确认 ChatModel 设置 streaming=true；检索阶段设置超时控制。

  2. 答案中的引用格式是 [HR-001#chunk-3]，但该 chunk 在索引中已经删除。前端应该如何展示？

    参考答案

    理想做法： 在前端展示时，后端同时返回引用 chunk 的 metadata（doc_id、标题、摘要）。如果 chunk 已删除，后端标记该引用为「已过期」。CorpAssist 的引用服务会对每个引用做存在性检查。
