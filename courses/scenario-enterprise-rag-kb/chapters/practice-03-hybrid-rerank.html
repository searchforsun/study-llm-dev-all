Spring AI 混合检索实现（双路查询）

java 
// HybridSearchConfig.java — Spring AI 混合检索配置
@Configuration
public class HybridSearchConfig {

    @Bean
    @Primary
    public VectorStore milvusVectorStore() {
        return new MilvusVectorStore(MilvusVectorStoreConfig.builder()
            .withUri("http://milvus:19530")
            .withCollectionName("corpassist_kb")
            .build());
    }

    @Bean
    public ElasticsearchVectorStore esVectorStore() {
        return new ElasticsearchVectorStore(
            ElasticsearchVectorStoreConfig.builder()
                .withIndexName("corpassist_bm25")
                .withUri("http://elasticsearch:9200")
                .build()
        );
    }

    @Bean
    public HybridRetriever hybridRetriever(
            VectorStore milvusStore,
            ElasticsearchVectorStore esStore) {
        return new HybridRetriever(milvusStore, esStore);
    }
}

// HybridRetriever.java — RRF 融合检索
@Component
public class HybridRetriever {
    private final VectorStore vectorStore;
    private final ElasticsearchVectorStore esStore;

    public List hybridSearch(String query, int topK) {
        // 向量检索
        var vectorResults = vectorStore.similaritySearch(
            SearchRequest.query(query).withTopK(topK * 2));

        // BM25 检索
        var bm25Results = esStore.similaritySearch(
            SearchRequest.query(query).withTopK(topK * 2));

        // RRF 融合
        return rrfMerge(vectorResults, bm25Results, topK, 60);
    }

    private List rrfMerge(List vec, List bm25,
                                     int topK, int k) {
        Map scores = new HashMap<>();
        indexDocs(vec, scores, k);
        indexDocs(bm25, scores, k);

        return scores.entrySet().stream()
            .sorted(Map.Entry.comparingByValue().reversed())
            .limit(topK)
            .map(entry -> findDocById(entry.getKey(), vec, bm25))
            .collect(Collectors.toList());
    }
}

Ablation 实验 Python 脚本

python 
# ablation.py — CorpAssist 消融实验
from dataclasses import dataclass
from typing import List, Callable
import time

@dataclass
class AblationConfig:
    use_vector: bool = True
    use_bm25: bool = True
    use_rrf: bool = True
    use_reranker: bool = True

def run_ablation(
    test_queries: List[dict],  # [{"q": "..", "expected_chunk_ids": [...]}]
    config: AblationConfig,
) -> dict:
    """运行一组 ablation 配置，返回 MRR 和延迟"""
    start = time.time()
    total_mrr = 0.0

    for tq in test_queries:
        candidates = []
        if config.use_vector:
            vec_results = vector_store.similarity_search(tq["q"], k=20)
            candidates.extend(vec_results)
        if config.use_bm25:
            bm25_results = bm25_retrieve(tq["q"], k=20)
            candidates.extend(bm25_results)

        if config.use_rrf:
            candidates = rrf_merge(candidates, [], top_n=10)
        else:
            candidates = candidates[:10]

        if config.use_reranker:
            reranked = reranker.rerank(tq["q"], candidates, top_k=5)
            candidates = [c for c, s in reranked]

        # 计算 MRR
        for rank, doc in enumerate(candidates[:5]):
            if doc.metadata.get("chunk_id") in tq["expected_chunk_ids"]:
                total_mrr += 1.0 / (rank + 1)
                break

    elapsed = time.time() - start
    mrr = total_mrr / len(test_queries)
    return {"mrr": round(mrr, 3), "latency_ms": round(elapsed * 1000)}

# 执行 ablation
configs = [
    ("纯向量", AblationConfig(use_vector=True, use_bm25=False, use_rrf=False, use_reranker=False)),
    ("向量+BM25", AblationConfig(use_vector=True, use_bm25=True, use_rrf=True, use_reranker=False)),
    ("全量(含Rerank)", AblationConfig()),
]
for name, cfg in configs:
    result = run_ablation(test_queries, cfg)
    print(f"{name}: MRR={result['mrr']}, 延迟={result['latency_ms']}ms")

Spring AI 延迟监控

java 
// LatencyMonitor.java — 延迟分解监控
@Component
public class LatencyMonitor {
    private final MeterRegistry meterRegistry;

    public void recordRetrievalLatency(long ms) {
        meterRegistry.timer("rag.retrieval.latency").record(Duration.ofMillis(ms));
    }

    public void recordRerankLatency(long ms) {
        meterRegistry.timer("rag.rerank.latency").record(Duration.ofMillis(ms));
    }

    public void recordLlmTtft(long ms) {
        meterRegistry.timer("rag.llm.ttft").record(Duration.ofMillis(ms));
    }

    @EventListener
    public void onRequestComplete(RequestCompleteEvent event) {
        long total = event.getDuration().toMillis();
        meterRegistry.timer("rag.total.latency").record(Duration.ofMillis(total));

        if (total > 2000) {
            log.warn("SLA breached: total latency={}ms, breakdown={}",
                total, event.getBreakdown());
        }
    }
}

常见问题（FAQ）

Ablation 实验应该在 Python 端还是 Spring 端做？
两端都要做。Python 端做索引参数的 ablation（chunk_size、overlap、embedding 模型），Spring 端做检索策略的 ablation（BM25 on/off、Rerank on/off、RRF k 值）。CorpAssist 的实践是：先 Python 端固定索引，再 Spring 端调检索策略。
Python 和 Spring 两端的 embedding 模型必须一致吗？
必须一致。Python 端用 text-embedding-3-small 向量化，Spring 端查询时也要用同样的模型。否则向量空间的分布不同，检索结果无意义。CorpAssist 在 Spring AI 配置中指定 spring.ai.openai.embedding.model=text-embedding-3-small。
延迟预算中 Rerank 占了 200ms，如何优化？
三种方案：1) 减少 Rerank 候选数（从 top-30 降到 top-20）；2) 使用更快的 Reranker 模型（bge-reranker-v2-m3 比 v1 快 30%）；3) 使用 GPU 推理（NVIDIA T4 约 150ms / 30 候选，CPU 约 600ms）。详见 效果数据与优化复盘。
联调时发现 Python 索引的 Recall@5 是 0.85，但 Spring 端只有 0.76，可能原因？
常见原因：1) Spring 端的 embedding 模型配置与 Python 端不同；2) Spring 端检索的 FilterExpression 额外过滤了部分文档；3) 两端的相似度阈值不一致（Python 端可能没设 threshold 但 Spring 端设了 0.7）。CorpAssist 的 joint debug 工具可以逐层对比两端的检索中间结果。

本章结论

  • 共享索引解耦双栈：Python 写入索引，Spring 读取，通过相同 collection 协作。索引 Schema 一致是前提。
  • Ablation 实验量化每步贡献：BM25 混合 +0.12 MRR，Rerank +0.07 MRR，决策哪些组件必须保留。
  • 延迟预算分阶段控制：向量检索 100ms + BM25 100ms + Rerank 200ms + LLM 500ms，P95 控制在 2s 以内。
  • 联合调优需要两端工具链：Python ablation 框架 + Spring 延迟监控，定位瓶颈时两端日志关联分析。

判断练习

  1. Ablation 结果显示 BM25 只提升了 0.02 MRR，是否应该去掉 BM25 以节省架构复杂度？

    参考答案

    不建议去掉。 MRR 是整体指标，但 BM25 对精确匹配场景（制度编号、产品型号）有不可替代的作用。建议分析 BM25 对「制度编号类」子集的单独 MRR。如果子集提升大（如 +0.15），即使整体提升小也值得保留。

  2. 延迟预算中 LLM 首 Token 用了 800ms（预算 500ms）。做哪些优化？

    参考答案

    1) 检查 LLM 是否使用了 streaming=true；2) 考虑换更快的模型（gpt-4o-mini → gpt-4o-mini 本身已是最快，考虑用更小的模型）；3) 减少 prompt 中的 context 长度（如 top-5 → top-3）；4) 如果是 API 调用，检查网络延迟，考虑在相同区域部署。
