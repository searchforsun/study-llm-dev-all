Controller 设计

java
// 仓库问答 Controller
@RestController
@RequestMapping("/api/code")
public class CodeQueryController {

    private final CodeRetriever codeRetriever;
    private final CodeGenerator codeGenerator;
    private final TokenBudgetManager tokenBudget;

    @PostMapping("/search")
    public ResponseEntity<List<SearchResult>> search(
            @RequestBody SearchRequest request) {
        // 检索代码切片
        List<SearchResult> results = codeRetriever.search(
            request.getProjectId(),
            request.getQuery(),
            request.getTopK()
        );
        return ResponseEntity.ok(results);
    }

    @PostMapping("/ask")
    public ResponseEntity<AskResponse> ask(
            @RequestBody AskRequest request) {
        // 1. 检索
        List<SearchResult> contexts = codeRetriever.search(
            request.getProjectId(), request.getQuery(), 5);

        // 2. 拼装上下文（Token 预算管理）
        String codeContext = tokenBudget.formatContext(contexts);

        // 3. 生成回答
        String answer = codeGenerator.generate(
            request.getQuery(),
            request.getLanguage(),
            codeContext,
            contexts  // 用于引用
        );

        return ResponseEntity.ok(new AskResponse(answer, contexts));
    }

    // SSE 流式问答
    @PostMapping(value = "/ask/stream", produces = MediaType.TEXT_EVENT_STREAM_VALUE)
    public SseEmitter askStream(@RequestBody AskRequest request) {
        SseEmitter emitter = new SseEmitter(30_000L);  // 30 秒超时

        executor.execute(() -> {
            try {
                // 先发送检索结果（快速）
                List<SearchResult> contexts = codeRetriever
                    .search(request.getProjectId(), request.getQuery(), 5);
                emitter.send(SseEmitter.event()
                    .name("context")
                    .data(contexts));

                // 再流式生成回答
                Flux<String> stream = codeGenerator
                    .generateStream(request.getQuery(), request.getLanguage(),
                        tokenBudget.formatContext(contexts), contexts);
                stream.subscribe(
                    chunk -> emitter.send(SseEmitter.event()
                        .name("token").data(chunk)),
                    emitter::completeWithError,
                    emitter::complete
                );
            } catch (Exception e) {
                emitter.completeWithError(e);
            }
        });

        return emitter;
    }
}

引用代码块与 SSE 流式

// 引用代码块格式化
@Component
public class CitationFormatter {

    public String formatCitations(List<SearchResult> results) {
        StringBuilder sb = new StringBuilder("\n\n**引用代码：**\n");
        int idx = 1;
        for (SearchResult r : results) {
            sb.append(String.format(
                "\n[%d] %s:%d-%d\n```%s\n%s\n```\n",
                idx++, r.getFilePath(), r.getStartLine(),
                r.getEndLine(), r.getLanguage(), r.getCodeSnippet()
            ));
        }
        return sb.toString();
    }
}

// SSE 流式配置
@Configuration
public class SseConfig {

    @Bean
    public CodeGenerator codeGenerator(ChatClient.Builder builder) {
        return new CodeGenerator(builder.build());
    }
}

@Component
public class CodeGenerator {

    private final ChatClient chatClient;

    public Flux<String> generateStream(String query, String language,
                                        String context,
                                        List<SearchResult> citations) {
        return chatClient.prompt()
            .user(u -> u.text("""
                请回答以下代码问题。

                相关代码上下文：
                {context}

                问题：{query}
                编程语言：{language}

                回答要求：
                - 引用具体代码行
                - 给出代码示例
                - 不确定时说明"不确定"
                """)
                .param("context", context)
                .param("query", query)
                .param("language", language)
            )
            .stream()
            .content()
            .concatWithValues( // 追加引用
                Flux.just(formatCitations(citations)).toString()
            );
    }
}

本章结论

  • Controller 提供 search/ask/ask-stream 三个 API 端点
  • RAG Pipeline 三阶段：向量检索(800ms) + 上下文拼装(200ms) + LLM 生成(2-5s)
  • 引用代码块包含文件路径+行号+代码片段，SSE 流式减少首字节延迟
  • SseEmitter 30 秒超时 + Flux 流式输出
