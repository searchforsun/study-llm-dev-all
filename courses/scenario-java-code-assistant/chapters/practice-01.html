Git 增量索引

使用 JGit 库获取仓库的提交历史，分析变更文件，只重新索引修改过的文件。

flowchart TD
    A[定时器触发: 每5分钟] --> B[获取上次索引的 commitId]
    B --> C[JGit: 获取 recent commits]
    C --> D[对比 files changed]
    D --> E{文件类型?}
    E -->|Java/Python/TS等| F[重新索引]
    E -->|非代码文件| G[跳过]
    F --> H[删除 Milvus 中旧数据]
    H --> I[写入新切片到 Milvus]
    I --> J[更新 lastIndexedCommit]
java
// Git 增量索引实现
@Component
public class GitIncrementalIndexer {

    @Value("${code-index.repo.path}")
    private String repoPath;

    private final Map<String, String> lastIndexedCommits = new ConcurrentHashMap<>();

    public List<GitChange> detectChanges(String projectId) {
        try (Git git = Git.open(new File(repoPath + "/" + projectId))) {
            // 获取上次索引的 commit
            String lastCommit = lastIndexedCommits
                .getOrDefault(projectId, getInitialCommit(git));

            // 获取从上次索引后的所有新提交
            Iterable<RevCommit> commits = git.log()
                .addRange(
                    git.resolve(lastCommit + "^"),
                    git.resolve("HEAD"))
                .call();

            // 汇总所有变更文件
            Set<String> changedFiles = new LinkedHashSet<>();
            for (RevCommit commit : commits) {
                try (RevWalk walk = new RevWalk(git.getRepository())) {
                    RevCommit revCommit = walk.parseCommit(commit.getId());
                    if (revCommit.getParentCount() > 0) {
                        RevTree parentTree = revCommit.getParent(0).getTree();
                        RevTree thisTree = revCommit.getTree();
                        // 对比两棵树的差异
                        diffTrees(git, parentTree, thisTree, changedFiles);
                    }
                }
            }

            // 记录最新的 commitId
            RevCommit latest = git.log().setMaxCount(1).call().iterator().next();
            lastIndexedCommits.put(projectId, latest.getName());

            return changedFiles.stream()
                .filter(this::isCodeFile)
                .map(f -> new GitChange(projectId, f, GitChange.Type.MODIFY))
                .toList();
        } catch (Exception e) {
            throw new RuntimeException("Git diff failed", e);
        }
    }
}

Spring Scheduling 调度

java
// 索引调度配置
@Configuration
@EnableScheduling
public class IndexSchedulerConfig {

    @Bean
    public TaskScheduler indexTaskScheduler() {
        ThreadPoolTaskScheduler scheduler = new ThreadPoolTaskScheduler();
        scheduler.setPoolSize(3);          // 支持 3 个仓库并行索引
        scheduler.setThreadNamePrefix("code-index-");
        scheduler.setWaitForTasksToCompleteOnShutdown(true);
        return scheduler;
    }
}

@Component
public class IndexScheduler {

    private final GitIncrementalIndexer gitIndexer;
    private final CodeIndexBuilder indexBuilder;

    @Scheduled(cron = "0 0 2 * * ?")  // 每天凌晨 2 点全量
    public void fullIndexAll() {
        log.info("开始全量索引所有项目");
        List<String> projects = gitIndexer.listProjects();
        for (String project : projects) {
            try {
                indexBuilder.fullIndex(project);
                log.info("全量索引完成: {}", project);
            } catch (Exception e) {
                log.error("全量索引失败: {}", project, e);
                alertService.sendAlert("索引失败: " + project);
            }
        }
    }

    @Scheduled(fixedRate = 300000)  // 每 5 分钟增量
    public void incrementalIndex() {
        List<String> projects = gitIndexer.listProjects();
        for (String project : projects) {
            List<GitChange> changes = gitIndexer.detectChanges(project);
            if (!changes.isEmpty()) {
                log.info("检测到 {} 个文件变更: {}", changes.size(), project);
                indexBuilder.incrementalIndex(changes);
            }
        }
    }
}

// Milvus 集合管理
@Component
public class MilvusCollectionManager {

    private final MilvusClient milvusClient;

    public void createCollection(String projectId) {
        String collectionName = "code_" + projectId.replace("-", "_");

        // 检查集合是否存在
        if (milvusClient.hasCollection(collectionName)) {
            return;
        }

        // 创建集合 Schema
        FieldType idField = FieldType.builder()
            .name("id").dataType(DataType.Int64).primaryKey(true)
            .autoID(true).build();

        FieldType embeddingField = FieldType.builder()
            .name("embedding").dataType(DataType.FloatVector)
            .dimension(1024).build();

        FieldType filePathField = FieldType.builder()
            .name("filePath").dataType(DataType.VarChar)
            .maxLength(500).build();

        FieldType languageField = FieldType.builder()
            .name("language").dataType(DataType.VarChar)
            .maxLength(20).build();

        FieldType symbolNameField = FieldType.builder()
            .name("symbolName").dataType(DataType.VarChar)
            .maxLength(200).build();

        CollectionSchema schema = CollectionSchema.builder()
            .addField(idField).addField(embeddingField)
            .addField(filePathField).addField(languageField)
            .addField(symbolNameField)
            .build();

        milvusClient.createCollection(
            CreateCollectionReq.builder()
                .collectionName(collectionName)
                .collectionSchema(schema)
                .build()
        );

        // 创建索引
        milvusClient.createIndex(
            CreateIndexReq.builder()
                .collectionName(collectionName)
                .fieldName("embedding")
                .indexType(IndexType.IVF_FLAT)
                .metricType(MetricType.IP)
                .extraParam("{\"nlist\":1024}")
                .build()
        );

        log.info("Milvus 集合创建完成: {}", collectionName);
    }
}
