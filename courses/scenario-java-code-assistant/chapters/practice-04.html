语义缓存（Spring Cache）

java
// 语义缓存配置
@Configuration
@EnableCaching
public class SemanticCacheConfig {

    @Bean
    public CacheManager semanticCacheManager(RedisConnectionFactory cf) {
        RedisCacheConfiguration config = RedisCacheConfiguration.defaultCacheConfig()
            .entryTtl(Duration.ofMinutes(5))   // 缓存 5 分钟
            .prefixName("semantic_cache:")
            .disableCachingNullValues();

        return RedisCacheManager.builder(cf)
            .cacheDefaults(config)
            .build();
    }
}

@Component
public class SemanticCacheService {

    @Cacheable(value = "completion_cache",
               key = "#cacheKey",
               unless = "#result == null")
    public String getCachedCompletion(String prefix, String suffix,
                                       String language) {
        return null;  // 缓存未命中，调用方执行实际查询
    }

    public String buildCacheKey(String prefix, String suffix, String language) {
        // 语义哈希：相同语义的代码产生相同 Key
        String normalized = normalizeCode(prefix) + "|||"
            + normalizeCode(suffix) + "|||" + language;
        return DigestUtils.md5DigestAsHex(normalized.getBytes());
    }

    private String normalizeCode(String code) {
        // 标准化：去掉空格、换行、注释，只保留语义结构
        return code.replaceAll("\\s+", " ")
            .replaceAll("/\\*.*?\\*/", "")
            .replaceAll("//.*", "")
            .trim();
    }
}

// 缓存命中统计
@Component
public class CacheMetrics {

    private final MeterRegistry registry;
    private final Counter hitCounter;
    private final Counter missCounter;

    public CacheMetrics(MeterRegistry registry) {
        this.registry = registry;
        this.hitCounter = Counter.builder("cache.hits")
            .register(registry);
        this.missCounter = Counter.builder("cache.misses")
            .register(registry);
    }

    public void recordHit() {
        hitCounter.increment();
    }

    public void recordMiss() {
        missCounter.increment();
    }

    public double getHitRate() {
        double hits = hitCounter.count();
        double total = hits + missCounter.count();
        return total > 0 ? hits / total * 100 : 0;
    }
}

// 模型路由器
@Component
public class ModelRouter {

    private final ChatClient turboClient;   // 简单场景
    private final ChatClient plusClient;    // 中等场景
    private final ChatClient maxClient;     // 复杂场景

    public ChatClient route(String prefix, String suffix, String language) {
        int complexity = assessComplexity(prefix, suffix);
        if (complexity <= 3) {
            return turboClient;   // 简单：getter/setter/import
        } else if (complexity <= 7) {
            return plusClient;    // 中等：方法逻辑/循环
        } else {
            return maxClient;     // 复杂：算法/多文件
        }
    }

    private int assessComplexity(String prefix, String suffix) {
        int score = 0;
        // 关键词检测
        if (containsKeyword(prefix, "for|while|if|switch")) score += 2;
        if (containsKeyword(prefix, "try|catch|throw")) score += 2;
        if (containsKeyword(prefix, "class|interface|enum")) score += 3;
        if (containsKeyword(prefix, "import|package")) score += 1;
        // 长度评估
        int lines = prefix.split("\n").length;
        if (lines > 50) score += 2;
        if (lines > 100) score += 2;
        return score;
    }

    // 路由分配统计
    // | 模型 | 分配比例 | 平均延迟 | 采纳率 |
    // | qwen-turbo | 45% | 180ms | 32% |
    // | qwen-plus  | 40% | 450ms | 28% |
    // | qwen-max   | 15% | 1.2s  | 22% |
}

// vLLM 客户端配置
@Configuration
public class VllmConfig {

    @Bean
    @ConditionalOnProperty(name = "model.provider", havingValue = "vllm")
    public ChatClient vllmChatClient() {
        return ChatClient.builder()
            .baseUrl("http://localhost:8000/v1")  // vLLM 服务地址
            .defaultRequest(r -> r
                .model("codeqwen-7b")
                .parameters(Map.of(
                    "max_tokens", 256,
                    "temperature", 0.2,
                    "top_p", 0.95,
                    "presence_penalty", 0.0,
                    "frequency_penalty", 0.0,
                    "stop", List.of("\n\n", "<|endoftext|>")
                ))
            )
            .build();
    }
}
