python 
# 记忆子系统评测
class MemoryEvalSuite:
    def __init__(self, memory_api_client):
        self.client = memory_api_client

    def eval_consistency(self, scenario: str, expected_steps: int) -> dict:
        """评测多轮一致性"""
        # 加载标准评测场景
        turns = self._load_scenario(scenario)
        inconsistencies = []

        for i, turn in enumerate(turns):
            response = self.client.chat(turn["user_input"])
            for required in turn["must_remember"]:
                if required not in response:
                    inconsistencies.append({
                        "turn": i,
                        "required": required,
                        "missing": True
                    })

        consistency = 1 - len(inconsistencies) / expected_steps
        return {
            "consistency": consistency,
            "inconsistencies": inconsistencies
        }

    def eval_recall_accuracy(self, test_facts: list) -> dict:
        """评测记忆召回准确率"""
        correct = 0
        for item in test_facts:
            recalled = self.client.recall_facts(
                item["user_id"], item["query"])
            if any(item["fact_id"] == f["fact_id"] for f in recalled):
                correct += 1

        return {
            "accuracy": correct / len(test_facts),
            "correct": correct,
            "total": len(test_facts)
        }

// 记忆评测 - Java 实现
@RestController
@RequestMapping("/api/eval")
public class MemoryEvalController {

    @PostMapping("/consistency")
    public EvalResult evalConsistency(@RequestBody EvalRequest request) {
        int totalChecks = 0;
        int passed = 0;

        for (EvalTurn turn : request.turns()) {
            String response = chatService.chat(turn.userInput());
            totalChecks += turn.mustRemember().size();
            passed += turn.mustRemember().stream()
                .filter(response::contains)
                .count();
        }

        double consistency = (double) passed / totalChecks;
        return new EvalResult("consistency", consistency, passed, totalChecks);
    }
}

端到端验收时建议固定三类用例：多轮约束保持（语言/格式）、跨会话事实召回（长期记忆）、人在回路纠正（审批后写入）。每类至少 5 条 badcase 入库，作为后续压缩策略与路由规则的回归集。

先想 10 秒：为什么评测要覆盖「纠正后」场景？

用户纠正往往只更新会话层；若评测只测首轮写入，会高估长期记忆一致性，上线后仍会出现「用户已否认、模型仍引用旧事实」。

CorpAssist · 记忆子系统发布门禁 发布前 consistency 评测 ≥0.85、P95 记忆召回延迟 <120ms，且双栈对照表已签字（Python 记忆服务 + Java BFF 职责无重叠写入）。

双栈对照不仅是 API 映射表，还要约定失败语义：Python 记忆服务不可用时 Java BFF 是降级为无长期记忆，还是返回可重试错误码，避免静默丢事实。

端到端演示环境应预置「纠正—审批—回放」脚本，让产品、法务能在发布评审会上复现记忆写入全链路，而不只依赖单元测试报告。

Capstone 验收时把「记忆子系统」作为独立模块打标签，便于与 RAG、Agent 课的集成问题单区分归属。

integration mem 是双栈协作的关键：Python 负责 AI 密集型任务（事实提取、向量搜索），Java 负责事务型任务（审批、审计、Advisor 集成）。两者通过统一的 mem api 通信。

双栈协作最佳实践

Python 侧做快速原型和事实提取，Java 侧做生产级 BFF 和审批集成。两者共享同一 Redis 和 Milvus 实例，通过 session key 和 fact_id 关联数据。

handover mem 文档记录了双栈之间的 API 契约和数据格式，供后续课程使用。

本章结论
