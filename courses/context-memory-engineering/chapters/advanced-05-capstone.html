python 
# 集成测试套件
class IntegrationTestSuite:
    def __init__(self, memory_api_url: str):
        self.api = memory_api_url

    def test_session_continuity(self) -> dict:
        """RAG 集成测试：会话连续性"""
        # 模拟 5 轮连续对话
        session = []
        for i, msg in enumerate(["查订单", "取消", "退款"]):
            resp = requests.post(f"{self.api}/v1/session/messages", json={
                "tenant": "test", "user_id": "user1", "content": msg
            })
            session.append(resp.json())

        # 验证上下文包含之前的意图
        context = session[-1].get("context", "")
        has_history = "订单" in context or "取消" in context
        return {"test": "session_continuity", "passed": has_history}

    def test_agent_checkpoint(self) -> dict:
        """Agent 集成测试：Checkpoint 恢复"""
        # 模拟 Agent 中断恢复
        agent_id = f"agent:{uuid4()}"
        session_id = f"session:{uuid4()}"

        # 写入 Checkpoint
        requests.post(f"{self.api}/v1/agent/checkpoint", json={
            "agent_id": agent_id,
            "session_id": session_id,
            "state": {"step": 3, "context": "test"}
        })

        # 恢复
        resp = requests.get(
            f"{self.api}/v1/agent/checkpoint/{agent_id}/{session_id}")
        restored = resp.json()

        return {
            "test": "agent_checkpoint",
            "passed": restored.get("state", {}).get("step") == 3
        }
java 
// 集成验收 - Java 测试
@SpringBootTest
class MemoryIntegrationTest {

    @Autowired
    private MemoryApiClient memoryClient;

    @Test
    void testMultiTurnConsistency() {
        // 3 轮对话后检查状态一致性
        memoryClient.sendMessage("tenant1", "user1", "查订单 20240501");
        memoryClient.sendMessage("tenant1", "user1", "现在能取消吗");

        var context = memoryClient.getSessionContext("tenant1", "user1");
        assertThat(context.getHistory())
            .anyMatch(msg -> msg.contains("20240501"));
    }

    @Test
    void testRecallLatency() {
        // P95 延迟 <= 200ms
        var results = new ArrayList<Long>();
        for (int i = 0; i < 100; i++) {
            long start = System.currentTimeMillis();
            memoryClient.recallFacts("user1", "test query", 5);
            results.add(System.currentTimeMillis() - start);
        }
        Collections.sort(results);
        long p95 = results.get((int)(results.size() * 0.95));
        assertThat(p95).isLessThanOrEqualTo(200);
    }
}

# CorpAssist 记忆子系统交接文档

## 1. 系统概述
记忆子系统提供三层记忆：工作/会话/长期。
部署拓扑：Python 记忆服务 + Java BFF + Redis/Milvus/PostgreSQL。

## 2. API 端点
| 端点 | 方法 | 用途 | 调用方 |
|------|------|------|--------|
| /v1/session/{t}/{u}/messages | GET/POST | 会话消息 | Agent |
| /v1/memory/{u}/recall | POST | 长期召回 | Agent |
| /v1/memory/{u}/facts | POST | 写入事实 | RAG |
| /v1/memory/{u} | DELETE | GDPR 删除 | 合规 |

## 3. Redis Key 约定
- 会话: `corpassist:session:{tenant}:{user}`
- 黑板书: `corpassist:blackboard:{session_id}`
- Checkpoint: `corpassist:checkpoint:{thread_id}`

## 4. 验收指标
- 多轮一致性 ≥ 95%
- P95 召回延迟 ≤ 200ms
- API 可用性 ≥ 99.9%

## 5. 常见问题
见 handover 文档目录下的 FAQ.md
不推荐
  • 无 API 文档，靠口口相传
  • 验收指标不明确，靠感觉判断
  • 无集成测试，上线后才发现问题
  • 无交接文档，新团队无法接入
推荐
  • OpenAPI 规范定义接口
  • 验收指标量化到 CI/CD
  • 集成测试覆盖所有跨系统场景
  • 交接文档包含 API、Key 约定、FAQ
检查理解：为什么 API 契约要先于实现定义？

三组团队（记忆/RAG/Agent）并行开发时，如果各自按自己的理解定义接口，集成时必然出现参数类型、数据格式的 mismatch。先定义 OpenAPI 契约，团队可以并行 mock 测试。

本章结论

  • API 契约：OpenAPI 规范定义的记忆子系统接口，供 RAG 和 Agent 课程消费。
  • 验收指标：多轮一致性 ≥ 95%、P95 延迟 ≤ 200ms、API 可用性 ≥ 99.9%。
  • 集成测试：覆盖会话连续性、Checkpoint 恢复、GDPR 删除等跨系统场景。
  • 交接文档：包含 API 端点、Redis Key 约定、部署拓扑和 FAQ。
  • Metric Bind：验收指标绑定到 CI/CD 管道，每次部署自动验证。
