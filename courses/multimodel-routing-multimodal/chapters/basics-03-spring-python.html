dual stack route 架构：Spring 和 Python 从同一配置源加载路由策略。

java 
// Spring Router — 多模型注册与动态选择
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.openai.OpenAiChatModel;
import org.springframework.ai.openai.OpenAiChatOptions;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

@Configuration
public class DualStackRouterConfig {

    @Bean
    public OpenAiChatModel turboChatModel() {
        return OpenAiChatModel.builder()
            .baseUrl("https://dashscope.aliyuncs.com/compatible-mode/v1")
            .apiKey(System.getenv("DASHSCOPE_API_KEY"))
            .defaultOptions(OpenAiChatOptions.builder()
                .model("qwen-turbo").maxTokens(2048).build())
            .build();
    }

    @Bean
    public OpenAiChatModel plusChatModel() {
        return OpenAiChatModel.builder()
            .baseUrl("https://dashscope.aliyuncs.com/compatible-mode/v1")
            .apiKey(System.getenv("DASHSCOPE_API_KEY"))
            .defaultOptions(OpenAiChatOptions.builder()
                .model("qwen-plus").maxTokens(4096).build())
            .build();
    }

    @Bean
    public OpenAiChatModel vlChatModel() {
        return OpenAiChatModel.builder()
            .baseUrl("https://dashscope.aliyuncs.com/compatible-mode/v1")
            .apiKey(System.getenv("DASHSCOPE_API_KEY"))
            .defaultOptions(OpenAiChatOptions.builder()
                .model("qwen-vl-plus").maxTokens(2048).build())
            .build();
    }
}

// 路由服务
@Service
class ModelRouterService {
    private final Map models;

    ModelRouterService(
            OpenAiChatModel turbo, OpenAiChatModel plus, OpenAiChatModel vl) {
        models = Map.of("turbo", turbo, "plus", plus, "vl", vl);
    }

    public String route(String tier, String message) {
        var model = models.getOrDefault(tier, models.get("turbo"));
        return ChatClient.builder(model).build()
            .prompt().user(message).call().content();
    }
}
先想 10 秒：Spring 和 Python 的路由规则不一致时，哪个影响更大？

成本和质量风险都大。Spring BFF 处理用户实时请求（对话/创作），Python 处理后台任务（批量/分析）。如果 Spring 走了 vl-plus 而 Python 走了 turbo，用户体验和成本都会出现问题。

Python 路由微服务

Python Router 基于 FastAPI 提供统一路由决策：

python 
# route_service.py — Python 统一路由服务
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from openai import OpenAI
import yaml

app = FastAPI(title="S5 路由服务")

# 加载共享配置
with open("routing-rules.yaml") as f:
    CONFIG = yaml.safe_load(f)

# DashScope 客户端
client = OpenAI(
    base_url="https://dashscope.aliyuncs.com/compatible-mode/v1",
    api_key="YOUR_DASHSCOPE_KEY",
)

class RouteRequest(BaseModel):
    intent: str
    message: str
    has_attachment: bool = False

class RouteResponse(BaseModel):
    model: str
    content: str

@app.post("/v1/route", response_model=RouteResponse)
def route(req: RouteRequest):
    rules = CONFIG["routing"]["rules"]
    selected = next(
        (r for r in rules if r["intent"] == req.intent),
        rules[-1]  # 默认
    )
    model = selected["model"]
    try:
        resp = client.chat.completions.create(
            model=model,
            messages=[{"role": "user", "content": req.message}],
        )
        return RouteResponse(model=model, content=resp.choices[0].message.content)
    except Exception:
        for fb in selected.get("fallback", []):
            try:
                resp = client.chat.completions.create(
                    model=fb, messages=[{"role": "user", "content": req.message}],
                )
                return RouteResponse(model=fb, content=resp.choices[0].message.content)
            except Exception:
                continue
        raise HTTPException(503, "所有模型不可用")

共享配置同步

Config sync 保证双栈使用相同策略：

python 
# config_watcher.py — 热加载路由规则
import time
import yaml
from pathlib import Path

class RouteConfig:
    """热重载路由配置"""
    def __init__(self, path="routing-rules.yaml"):
        self.path = Path(path)
        self._mtime = 0
        self._rules = self._load()

    def _load(self):
        mtime = self.path.stat().st_mtime
        if mtime > self._mtime:
            with open(self.path) as f:
                self._rules = yaml.safe_load(f)
                self._mtime = mtime
        return self._rules

    def get_model(self, intent: str) -> str:
        self._load()
        for rule in self._rules["routing"]["rules"]:
            if rule["intent"] == intent:
                return rule["model"]
        return "qwen-turbo"

config = RouteConfig()  # 全局单例

常见问题（FAQ）

双栈路由如何保证路由规则同步？
推荐将 routing-rules.yaml 放在共享存储（如 NFS、Git 子模块），双栈定时轮询或监听文件变更。更成熟的方案是使用配置中心（Nacos/Etcd），双栈通过 SDK 监听变更。
Python 路由服务挂了对 Spring 有影响吗？
Spring 应缓存路由规则，路由服务不可用时使用缓存规则降级。Python 路由服务本身应多副本部署，避免单点故障。
路由决策应该放在客户端还是服务端？
服务端。客户端（IDE 插件、前端）不应做路由决策，因为它们可能被篡改。路由决策在 BFF 层完成，客户端只透传意图标识。
双栈路由如何监控和审计？
每个路由请求打印日志：{intent, model_selected, cost_estimate, latency}。双栈日志汇聚到同一 ELK/Prometheus，按 model 和 intent 维度聚合。详见 路由治理。

  • Spring Router：Spring Router 通过 ChatClient 动态切换模型。
  • Python Router：Python Router 作为独立微服务或 SDK 集成。
