# incremental_indexer.py — CorpAssist Git 增量索引器
import subprocess
import hashlib
import ast
from pathlib import Path
from typing import Set, List, Dict, Optional
from pymilvus import Collection, CollectionSchema, FieldSchema, DataType, connections
from openai import OpenAI

class GitChangeDetector:
    """检测 Git 仓库的变更文件"""

    SUPPORTED_EXTS = {".py", ".java", ".js", ".ts", ".go", ".rs", ".sql"}

    def __init__(self, repo_path: str):
        self.repo = Path(repo_path)
        if not (self.repo / ".git").exists():
            raise ValueError(f"不是 Git 仓库: {repo_path}")

    def get_changed_files(self, since_commit: str = "HEAD~1") -> Dict[str, List[str]]:
        """获取指定 commit 以来的变更文件列表"""
        result = subprocess.run(
            ["git", "diff", "--name-status", since_commit],
            cwd=self.repo, capture_output=True, text=True, check=True,
        )
        added, modified, deleted = [], [], []
        for line in result.stdout.strip().split("\n"):
            if not line.strip():
                continue
            parts = line.split("\t")
            if len(parts) < 2:
                continue
            status, filepath = parts[0], parts[1]
            ext = Path(filepath).suffix
            if ext not in self.SUPPORTED_EXTS:
                continue
            if status == "A":
                added.append(filepath)
            elif status == "M":
                modified.append(filepath)
            elif status == "D":
                deleted.append(filepath)
            elif status in ("R", "R100"):
                parts = filepath.split("\t")
                deleted.append(parts[0])
                added.append(parts[1])
        return {"added": added, "modified": modified, "deleted": deleted}

    def full_scan(self) -> List[str]:
        """全量扫描所有文件"""
        result = subprocess.run(
            ["git", "ls-files"], cwd=self.repo, capture_output=True, text=True, check=True,
        )
        files = []
        for f in result.stdout.strip().split("\n"):
            if f and Path(f).suffix in self.SUPPORTED_EXTS:
                files.append(f)
        return files


class CodeChunker:
    """AST 驱动的代码切分器，按函数/类切分"""

    def __init__(self, max_chunk_lines: int = 100):
        self.max_chunk_lines = max_chunk_lines

    def chunk_file(self, file_path: str, code: str) -> List[dict]:
        """将代码文件切分为函数/类级别的块"""
        ext = Path(file_path).suffix
        if ext == ".py":
            return self._chunk_python(file_path, code)
        # 其他语言用行数切分
        return self._chunk_by_lines(file_path, code)

    def _chunk_python(self, file_path: str, code: str) -> List[dict]:
        chunks = []
        lines = code.splitlines()
        try:
            tree = ast.parse(code)
            for node in ast.walk(tree):
                if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef, ast.ClassDef)):
                    chunk = "\n".join(lines[node.lineno - 1:node.end_lineno])
                    if len(chunk.splitlines()) <= self.max_chunk_lines:
                        chunks.append({
                            "chunk_id": hashlib.md5(f"{file_path}:{node.name}".encode()).hexdigest()[:12],
                            "file_path": file_path,
                            "symbol": node.name,
                            "symbol_type": "function" if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)) else "class",
                            "start_line": node.lineno,
                            "end_line": node.end_lineno,
                            "code": chunk,
                        })
        except SyntaxError:
            pass
        if not chunks:
            chunks = self._chunk_by_lines(file_path, code)
        return chunks

    def _chunk_by_lines(self, file_path: str, code: str) -> List[dict]:
        lines = code.splitlines()
        chunks = []
        chunk_size = 50
        for i in range(0, len(lines), chunk_size):
            chunk = "\n".join(lines[i:i + chunk_size])
            chunks.append({
                "chunk_id": hashlib.md5(f"{file_path}:{i}".encode()).hexdigest()[:12],
                "file_path": file_path,
                "symbol": Path(file_path).stem,
                "symbol_type": "file",
                "start_line": i + 1,
                "end_line": min(i + chunk_size, len(lines)),
                "code": chunk,
            })
        return chunks


class MilvusIndexer:
    """Milvus 向量索引管理器"""

    def __init__(self, collection_name: str = "code_index", dim: int = 512):
        self.collection_name = collection_name
        self.dim = dim
        self.embeddings = OpenAI(
            base_url="https://dashscope.aliyuncs.com/compatible-mode/v1",
        )
        connections.connect(host="localhost", port=19530)
        self.collection = self._get_or_create_collection()

    def _get_or_create_collection(self) -> Collection:
        if self.collection_name in connections.list_collections():
            return Collection(self.collection_name)
        schema = CollectionSchema([
            FieldSchema("chunk_id", DataType.VARCHAR, max_length=12, is_primary=True),
            FieldSchema("file_path", DataType.VARCHAR, max_length=256),
            FieldSchema("symbol", DataType.VARCHAR, max_length=128),
            FieldSchema("symbol_type", DataType.VARCHAR, max_length=32),
            FieldSchema("start_line", DataType.INT64),
            FieldSchema("end_line", DataType.INT64),
            FieldSchema("code", DataType.VARCHAR, max_length=8192),
            FieldSchema("repo", DataType.VARCHAR, max_length=64),
            FieldSchema("embedding", DataType.FLOAT_VECTOR, dim=self.dim),
        ])
        collection = Collection(self.collection_name, schema)
        collection.create_index("embedding", {"index_type": "IVF_FLAT", "metric_type": "COSINE", "params": {"nlist": 128}})
        collection.load()
        return collection

    def index_chunks(self, chunks: List[dict], repo: str):
        """将代码块索引到 Milvus"""
        if not chunks:
            return
        texts = [c["code"] for c in chunks]
        vectors = self.embeddings.embeddings.create(input=texts, model="text-embedding-3-small", dimensions=self.dim).data
        entities = [
            [c["chunk_id"] for c in chunks],
            [c["file_path"] for c in chunks],
            [c["symbol"] for c in chunks],
            [c["symbol_type"] for c in chunks],
            [c["start_line"] for c in chunks],
            [c["end_line"] for c in chunks],
            [c["code"] for c in chunks],
            [repo] * len(chunks),
            [v.embedding for v in vectors],
        ]
        self.collection.insert(entities)
        self.collection.flush()
        print(f"[Milvus] 索引 {len(chunks)} 个块到 {repo}")

    def delete_file(self, file_path: str):
        expr = f'file_path == "{file_path}"'
        self.collection.delete(expr)
        print(f"[Milvus] 删除 {file_path}")


if __name__ == "__main__":
    detector = GitChangeDetector("./my_repo")
    chunker = CodeChunker()
    indexer = MilvusIndexer()

    changes = detector.get_changed_files()
    print(f"新增: {len(changes['added'])} 修改: {len(changes['modified'])} 删除: {len(changes['deleted'])}")

    for fp in changes["deleted"]:
        indexer.delete_file(fp)

    for fp in changes["added"] + changes["modified"]:
        full_path = Path("./my_repo") / fp
        if not full_path.exists():
            continue
        code = full_path.read_text(encoding="utf-8")
        chunks = chunker.chunk_file(str(full_path), code)
        indexer.delete_file(fp)
        indexer.index_chunks(chunks, "my_repo")

// SymbolIndexer.java — 符号索引管理
@Component
public class SymbolIndexer {

    private final JdbcTemplate jdbc;

    public SymbolIndexer(JdbcTemplate jdbc) { this.jdbc = jdbc; }

    @Transactional
    public void indexSymbols(String repo, String filePath, List<SymbolEntry> symbols) {
        deleteSymbols(repo, filePath);
        for (SymbolEntry s : symbols) {
            jdbc.update("INSERT INTO code_symbols (repo, file_path, symbol, symbol_type, start_line, end_line) VALUES (?,?,?,?,?,?)",
                repo, filePath, s.name(), s.type(), s.startLine(), s.endLine());
        }
    }

    public void deleteSymbols(String repo, String filePath) {
        jdbc.update("DELETE FROM code_symbols WHERE repo=? AND file_path=?", repo, filePath);
    }

    public List<SymbolEntry> searchBySymbol(String repo, String symbol) {
        return jdbc.query("SELECT * FROM code_symbols WHERE repo=? AND symbol LIKE ?",
            (rs, row) -> new SymbolEntry(rs.getString("symbol"), rs.getString("symbol_type"),
                rs.getInt("start_line"), rs.getInt("end_line")),
            repo, "%" + symbol + "%");
    }
}

不推荐
  • 每次全量重索引全部文件（耗时数十分钟）
  • 按固定行数切分（50行一刀切，切断函数体）
  • 不处理文件删除（向量库中残留旧数据）
  • 只有向量索引，没有符号索引
推荐
  • Git diff 增量更新（平均 45s vs 15min 全量）
  • AST 驱动函数级切分（语义完整）
  • 先删后插，保持索引一致性
  • 向量 + 符号双通道（语义 + 精确匹配）
先想 10 秒：增量索引过程中如果又有一个新的提交，如何处理？

需要在索引器外部加锁或版本号。CorpAssist 的做法：1) 索引器启动时记录当前 HEAD commit hash；2) 索引期间新提交不影响当前批次；3) 下一轮增量索引从上一轮的 HEAD 开始 diff。极端情况下某文件在一轮索引中被修改两次，后一轮索引会覆盖前一轮的结果（先删后插保证幂等）。

# code_qa_api.py — CorpAssist 代码问答 API
from fastapi import FastAPI, HTTPException
from fastapi.responses import StreamingResponse
from pydantic import BaseModel, Field
from openai import OpenAI
from pymilvus import Collection, connections
from typing import AsyncGenerator, List, Optional
import json, time

app = FastAPI(title="CorpAssist 代码问答 API", version="1.0.0")

class QARequest(BaseModel):
    question: str = Field(..., min_length=2, max_length=2000, description="关于代码库的问题")
    repo: str = "default"
    top_k: int = Field(default=5, ge=1, le=20)
    score_threshold: float = Field(default=0.0, ge=0.0, le=1.0)
    stream: bool = False

class CodeReference(BaseModel):
    file_path: str
    start_line: int
    end_line: int
    symbol: str = ""
    score: float = 0.0

class QAResponse(BaseModel):
    answer: str
    references: List[CodeReference]
    latency_ms: float

class CodeRetriever:
    """代码检索器：Mivus 向量检索 + 符号检索"""

    def __init__(self):
        connections.connect(host="localhost", port=19530)
        self.collection = Collection("code_index")
        self.collection.load()
        self.embeddings = OpenAI(
            base_url="https://dashscope.aliyuncs.com/compatible-mode/v1",
        )

    def retrieve(self, question: str, repo: str, top_k: int = 5,
                 score_threshold: float = 0.0) -> List[dict]:
        """双通道检索：向量 + 符号"""
        start = time.time()

        # 向量检索
        vector = self.embeddings.embeddings.create(
            input=[question], model="text-embedding-3-small", dimensions=512
        ).data[0].embedding

        expr = f'repo == "{repo}"' if repo != "default" else ""

        results = self.collection.search(
            data=[vector],
            anns_field="embedding",
            param={"metric_type": "COSINE", "params": {"nprobe": 10}},
            limit=top_k,
            expr=expr,
            output_fields=["file_path", "symbol", "code", "start_line", "end_line", "repo"],
        )

        docs = []
        for hits in results:
            for hit in hits:
                if hit.score >= score_threshold:
                    docs.append({
                        "file_path": hit.entity.get("file_path"),
                        "symbol": hit.entity.get("symbol", ""),
                        "code": hit.entity.get("code", ""),
                        "score": round(hit.score, 4),
                        "start_line": hit.entity.get("start_line"),
                        "end_line": hit.entity.get("end_line"),
                    })

        elapsed = (time.time() - start) * 1000
        print(f"[检索] {question[:30]}... → {len(docs)} 条结果, {elapsed:.0f}ms")
        return docs

# latency_optimizer.py — CorpAssist 延迟优化器
import hashlib
import time
import asyncio
from typing import Dict, Optional, Tuple, Callable
from collections import OrderedDict
from openai import OpenAI
import numpy as np

class SemanticCache:
    """语义缓存：通过向量相似度判断缓存命中"""

    def __init__(self, max_size: int = 2000, similarity_threshold: float = 0.92):
        self.max_size = max_size
        self.threshold = similarity_threshold
        self.cache: OrderedDict[str, Tuple[str, list, str]] = OrderedDict()  # key -> (completion, embedding, model)
        self.embeddings = OpenAI(base_url="https://dashscope.aliyuncs.com/compatible-mode/v1")

    def _embed(self, text: str) -> list:
        resp = self.embeddings.embeddings.create(input=[text], model="text-embedding-3-small", dimensions=256)
        return resp.data[0].embedding

    def get(self, prefix: str, suffix: str, language: str = "") -> Optional[str]:
        query = f"{prefix}:{suffix}:{language}"
        q_vec = self._embed(query[:200])  # 取前 200 字符做语义
        best_key, best_sim = None, 0.0
        for key, (completion, emb, _) in self.cache.items():
            sim = np.dot(q_vec, emb) / (np.linalg.norm(q_vec) * np.linalg.norm(emb) + 1e-10)
            if sim > best_sim:
                best_sim = sim
                best_key = key
        if best_sim >= self.threshold and best_key:
            self.cache.move_to_end(best_key)
            print(f"[语义缓存] 命中: sim={best_sim:.3f}")
            return self.cache[best_key][0]
        return None

    def set(self, prefix: str, suffix: str, completion: str, language: str = ""):
        key = f"{prefix[:100]}:{suffix[:50]}:{language}"
        emb = self._embed(prefix[:200])
        self.cache[key] = (completion, emb, "gpt-4o-mini")
        if len(self.cache) > self.max_size:
            self.cache.popitem(last=False)


class ModelRouter:
    """模型路由：按请求复杂度选择模型"""

    def __init__(self):
        self.small = OpenAI(base_url="https://dashscope.aliyuncs.com/compatible-mode/v1")
        self.large = OpenAI(base_url="https://dashscope.aliyuncs.com/compatible-mode/v1")
        # 小模型：qwen-turbo，大模型：gpt-4o-mini
        self.small_model = "qwen-turbo"
        self.large_model = "gpt-4o-mini"

    def route(self, prefix: str, suffix: str) -> Tuple[str, str]:
        """返回 (model_name, client)"""
        total_len = len(prefix) + len(suffix)
        if total_len < 300 and "def " not in prefix and "class " not in prefix:
            return self.small_model, self.small  # 简单：小模型 ~50ms
        return self.large_model, self.large  # 复杂：大模型 ~200ms


class LatencyOptimizer:
    """延迟优化器：缓存 → 路由 → 推理"""

    def __init__(self):
        self.cache = SemanticCache()
        self.router = ModelRouter()
        self._latencies = []
        self.metrics = {"total": 0, "cache_hit": 0, "small_model": 0, "large_model": 0}

    async def complete(self, prefix: str, suffix: str, language: str = "",
                       style: str = "") -> Tuple[str, float]:
        start = time.time()

        # L1: 语义缓存
        cached = self.cache.get(prefix, suffix, language)
        if cached:
            self.metrics["cache_hit"] += 1
            return cached, (time.time() - start) * 1000

        # L2: 模型路由
        model, client = self.router.route(prefix, suffix)
        if model == "qwen-turbo":
            self.metrics["small_model"] += 1
        else:
            self.metrics["large_model"] += 1

        prompt = f"Complete the code at cursor:\nPrefix:{prefix}\nSuffix:{suffix}\nLanguage:{language}\n"
        resp = client.chat.completions.create(
            model=model,
            messages=[{"role": "user", "content": prompt}],
            temperature=0.2,
            max_tokens=128,
        )
        completion = resp.choices[0].message.content.strip()

        # 写入缓存
        self.cache.set(prefix, suffix, completion, language)

        elapsed = (time.time() - start) * 1000
        self._latencies.append(elapsed)
        self.metrics["total"] += 1
        return completion, elapsed

    def report(self) -> dict:
        n = len(self._latencies)
        sorted_lats = sorted(self._latencies)
        return {
            "total_requests": self.metrics["total"] + self.metrics["cache_hit"],
            "cache_hit_rate": round(self.metrics["cache_hit"] / max(self.metrics["total"] + self.metrics["cache_hit"], 1) * 100, 1),
            "small_model_pct": round(self.metrics["small_model"] / max(self.metrics["total"], 1) * 100, 1),
            "p50_ms": round(sorted_lats[int(n * 0.5)], 1) if n > 0 else 0,
            "p95_ms": round(sorted_lats[int(n * 0.95)], 1) if n > 0 else 0,
            "p99_ms": round(sorted_lats[int(n * 0.99)], 1) if n > 0 else 0,
        }

# ide_bridge.py — CorpAssist IDE 本地代理
import hashlib
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel, Field
from typing import Optional, List
import httpx
import ast

app = FastAPI(title="CorpAssist IDE 本地代理")

class LSPCompletionParams(BaseModel):
    uri: str = Field(..., description="file:// 格式的文件路径")
    position_line: int = Field(..., ge=0)
    position_character: int = Field(..., ge=0)
    context_before: str = Field(..., description="光标前 2000 字符")
    context_after: str = Field(default="", description="光标后 500 字符")
    language_id: str = Field(default="python")
    file_hash: Optional[str] = Field(None, description="文件内容的 SHA256 前缀(16位)")

class LSPCompletionItem(BaseModel):
    label: str
    text: str
    kind: int = 14  # Snippet
    detail: str = ""
    doc_string: str = ""

class IDEConnector:
    """IDE 连接器：本地代理逻辑"""

    SENSITIVE_PATTERNS = ["password", "secret", "token", "api_key", "private_key", "credential"]
    SENSITIVE_FILES = ["payment", "auth", "security", "crypto"]

    def __init__(self, cloud_api: str = "http://cloud-api:8000"):
        self.cloud_api = cloud_api
        self.cache = {}
        self.client = httpx.AsyncClient(timeout=10.0)

    def is_sensitive(self, uri: str, code: str) -> bool:
        """判断是否敏感文件"""
        uri_lower = uri.lower()
        for pattern in self.SENSITIVE_FILES:
            if pattern in uri_lower:
                return True
        for pattern in self.SENSITIVE_PATTERNS:
            if pattern in code.lower():
                return True
        return False

    def privacy_filter(self, code: str) -> str:
        """脱敏处理（替换敏感信息）"""
        import re
        code = re.sub(r'(password["\s:=]+)[^\s"\']+', r'\1***REDACTED***', code)
        code = re.sub(r'(api_key["\s:=]+)[^\s"\']+', r'\1***REDACTED***', code)
        code = re.sub(r'(secret["\s:=]+)[^\s"\']+', r'\1***REDACTED***', code)
        return code

    def check_syntax(self, code: str, language: str = "python") -> bool:
        """语法检查"""
        if language == "python":
            try:
                ast.parse(code)
                return True
            except SyntaxError:
                return False
        return True

    async def get_completion(self, params: LSPCompletionParams) -> List[LSPCompletionItem]:
        cache_key = hashlib.md5(f"{params.uri}:{params.position_line}:{params.context_before[:50]}".encode()).hexdigest()[:16]

        # 缓存命中
        if cache_key in self.cache:
            cached = self.cache[cache_key]
            return [LSPCompletionItem(label=cached[:30], text=cached, detail="(cache)")]

        # 隐私过滤 + 敏感检测
        code = params.context_before + params.context_after
        if self.is_sensitive(params.uri, code):
            code = self.privacy_filter(code)
            # 敏感文件走本地模型（此处模拟）
            completion = "# 敏感文件：本地模型处理\n# 实现逻辑略"
            if self.check_syntax(completion):
                self.cache[cache_key] = completion
                return [LSPCompletionItem(label=completion[:30], text=completion, detail="(local)")]

        # 普通文件走云端 API
        try:
            resp = await self.client.post(
                f"{self.cloud_api}/v1/completions",
                json={"prefix": params.context_before[-2000:], "suffix": params.context_after[:500],
                      "language": params.language_id, "max_tokens": 128, "temperature": 0.2},
                timeout=5.0,
            )
            data = resp.json()
            completion = data.get("completion", "")
            if self.check_syntax(completion):
                self.cache[cache_key] = completion
                return [LSPCompletionItem(label=completion[:50], text=completion, detail=params.uri)]
        except (httpx.TimeoutException, httpx.RequestError) as e:
            raise HTTPException(503, f"云端 API 不可用: {e}")

        return []

    async def close(self):
        await self.client.aclose()

connector = IDEConnector()

@app.post("/v1/completions", response_model=List[LSPCompletionItem])
async def complete(params: LSPCompletionParams):
    return await connector.get_completion(params)

@app.get("/health")
async def health():
    return {"status": "ok", "cache_size": len(connector.cache)}

    uvicorn.run(app, host="127.0.0.1", port=8100)

面试官：「代码助手的补全功能和问答功能，分别用什么技术方案？为什么？」

候选人：「补全用微调，问答用 RAG。补全要求延迟 <200ms，微调模型（如 CodeLlama 7B 做 FIM 微调）用 vLLM 部署，单次推理 50-100ms。问答需要检索最新代码库内容，必须用 RAG——代码库每天都有新提交，微调跟不上更新频率。组合方案效果最好：微调模型保证补全的生成质量，RAG 提供最新的代码上下文。」

面试官：「微调模型会不会遗忘旧知识（灾难性遗忘）？」

候选人：「会。CorpAssist 的做法是混合训练：每次微调时保留 20% 的通用代码补全数据（来自开源代码），防止模型只记住企业特定模式而遗忘通用编程能力。同时设置学习率 1e-5，低学习率减少遗忘。每季度做一次通用能力的回归评测。」

场景模拟：评估指标冲突

面试官：「采纳率从 40% 降到 35%，但用户满意度评分反而从 4.0 升到 4.2，你怎么分析？」

候选人：「采纳率下降说明用户更少接受我们的补全建议，但满意度上升说明用户对展示的补全质量更满意。可能的原因是我们提高了质量网关的阈值，拦截了更多低质量补全（采纳率分母变了）。我会看两组数据：一是质量网关的拦截率是否升高（从 8% 到 15%）；二是被采纳的补全中，修改后采纳的比例是否下降（说明展示的补全质量确实提高了）。如果确实是质量网关优化导致，这是好现象——用户看到更少但更准确的补全。」

先想 10 秒：面试官问「代码补全和代码问答为什么用不同的模型架构」？

代码补全需要低延迟（<200ms）和高吞吐，使用 FIM 微调的因果语言模型（如 CodeLlama），用 vLLM 部署。代码问答需要检索能力，使用 RAG 架构的 chat 模型（如 gpt-4o-mini），首 token 延迟 500ms 可以接受。两者对延迟和检索的要求不同，所以模型架构不同。

# eval_framework.py — CorpAssist 准确率评测框架
from pydantic import BaseModel, Field
from typing import List, Optional, Callable
import json
import ast
from datetime import datetime
from openai import OpenAI

class EvalCase(BaseModel):
    """评测用例"""
    id: str
    prefix: str
    suffix: str = ""
    expected: str = Field(..., description="预期的正确补全")
    language: str = "python"
    category: str = Field(default="logic", description="logic/api/boilerplate/bug_fix")

class EvalResult(BaseModel):
    """单条评测结果"""
    case_id: str
    actual_completion: str
    syntax_correct: bool
    semantics_correct: bool
    security_safe: bool
    quality_score: float
    latency_ms: float

class CodeEvalPipeline:
    """代码补全评估管线"""

    def __init__(self, completion_fn: Callable):
        self.completion_fn = completion_fn
        self.llm_judge = OpenAI(base_url="https://dashscope.aliyuncs.com/compatible-mode/v1")
        self.cases: List[EvalCase] = []
        self.results: List[EvalResult] = []

    def load_dataset(self, path: str):
        with open(path) as f:
            data = json.load(f)
        self.cases = [EvalCase(**c) for c in data]
        print(f"加载 {len(self.cases)} 条评测用例")

    def run(self) -> dict:
        """全量评测"""
        import time
        for case in self.cases:
            start = time.time()
            completion = self.completion_fn(case.prefix, case.suffix, case.language)

            syntax_ok = self._check_syntax(completion, case.language)
            semantics_ok = self._judge_semantics(case.prefix + completion, case.expected)
            security_ok = self._check_security(completion)
            # 质量评分
            issues = (0 if syntax_ok else 1) + (0 if semantics_ok else 1) + (0 if security_ok else 1)
            score = max(0.0, 1.0 - issues * 0.2)

            self.results.append(EvalResult(
                case_id=case.id, actual_completion=completion,
                syntax_correct=syntax_ok, semantics_correct=semantics_ok,
                security_safe=security_ok, quality_score=round(score, 2),
                latency_ms=(time.time() - start) * 1000,
            ))

        return self._report()

    def _check_syntax(self, code: str, language: str) -> bool:
        if language != "python":
            return True
        try:
            ast.parse(code)
            return True
        except SyntaxError:
            return False

    def _check_security(self, code: str) -> bool:
        import re
        dangerous = [r"\beval\s*\(", r"\bexec\s*\(", r"subprocess.*shell=True"]
        return not any(re.search(p, code) for p in dangerous)

    def _judge_semantics(self, full_code: str, expected: str) -> bool:
        """LLM 评判语义是否一致"""
        resp = self.llm_judge.chat.completions.create(
            model="qwen-turbo",
            messages=[{"role": "system", "content": "判断两段代码是否语义相同（实现逻辑一致）。仅返回 YES 或 NO。"},
                      {"role": "user", "content": f"生成代码:\n{full_code}\n\n预期代码:\n{expected}"}],
            temperature=0,
            max_tokens=10,
        )
        return "YES" in resp.choices[0].message.content.upper()

    def _report(self) -> dict:
        n = len(self.results)
        if n == 0:
            return {}
        syntax_rate = sum(1 for r in self.results if r.syntax_correct) / n
        semantics_rate = sum(1 for r in self.results if r.semantics_correct) / n
        security_rate = sum(1 for r in self.results if r.security_safe) / n
        avg_score = sum(r.quality_score for r in self.results) / n
        avg_latency = sum(r.latency_ms for r in self.results) / n
        return {
            "date": datetime.now().isoformat(),
            "total_cases": n,
            "syntax_accuracy": round(syntax_rate * 100, 1),
            "semantics_accuracy": round(semantics_rate * 100, 1),
            "security_safe_rate": round(security_rate * 100, 1),
            "avg_quality_score": round(avg_score, 2),
            "avg_latency_ms": round(avg_latency, 1),
            "security_risk_count": sum(1 for r in self.results if not r.security_safe),
        }

采纳率追踪实现

# metrics_tracker.py — 采纳率和性能指标追踪
from pydantic import BaseModel
from datetime import datetime
from typing import Optional
import time

class CompletionEvent(BaseModel):
    """代码补全事件"""
    event_id: str
    user_id: str
    project: str
    file_path: str
    language: str
    cursor_line: int
    context_chars: int           # 上下文长度
    completion_length: int       # 补全的代码长度
    ttft_ms: int                 # 首 Token 延迟
    total_latency_ms: int        # 总延迟
    accepted: Optional[bool]     # 是否被采纳
    timestamp: str = ""

class MetricsAggregator:
    """指标聚合器"""

    def __init__(self):
        self.events: list[CompletionEvent] = []

    def record_event(self, event: CompletionEvent):
        self.events.append(event)

    def acceptance_rate(self) -> float:
        """计算采纳率"""
        total = len([e for e in self.events if e.accepted is not None])
        accepted = sum(1 for e in self.events if e.accepted)
        return accepted / total if total > 0 else 0.0

    def p99_latency(self) -> int:
        """计算 P99 延迟"""
        latencies = sorted(e.total_latency_ms for e in self.events)
        if not latencies:
            return 0
        idx = int(len(latencies) * 0.99)
        return latencies[idx]

    def avg_ttft(self) -> float:
        """平均首 Token 延迟"""
        values = [e.ttft_ms for e in self.events]
        return sum(values) / len(values) if values else 0.0

    def report(self) -> dict:
        return {
            "total_completions": len(self.events),
            "acceptance_rate": round(self.acceptance_rate() * 100, 1),
            "avg_ttft_ms": round(self.avg_ttft(), 1),
            "p99_latency_ms": self.p99_latency(),
            "avg_completion_length": int(sum(e.completion_length for e in self.events) / len(self.events)) if self.events else 0,
        }

延迟对开发体验的影响

代码补全的延迟直接影响开发者的心流。研究表明：当补全延迟超过 300ms 时，开发者会感觉「卡顿」；超过 1s 时，大部分开发者会关闭自动补全功能。

TTFT 范围用户体验预估采纳率优化重点
< 100ms即时反馈，无感知35-45%保持
100-200ms流畅，偶尔微卡30-35%优化模型推理
200-500ms明显等待20-30%缓存+小模型
> 500ms打断思路< 15%降级方案
CorpAssist 实践

代码补全的 TTFT 优化是核心工程挑战。我们采用三级加速：L1 缓存命中（< 30ms）、L2 小模型推理（< 100ms）、L3 大模型推理（< 300ms）。超过 300ms 的请求返回「补全不可用」，避免打断开发者。代码问答则放宽到 5s，因为开发者可以等待更完整的答案。

常见问题（FAQ）

代码补全采纳率 30% 算好吗？
算及格。GitHub Copilot 公开数据显示采纳率约 25-35%。但不同类型代码差异很大：样板代码（getter/setter）采纳率可达 60%+，算法逻辑代码可能 < 20%。建议按文件类型和代码复杂度分层统计采纳率。
代码问答的准确率怎么评估？
使用人工标注的问答对（至少 200 条）。每条包含：问题、预期答案、预期引用的代码行。评估指标：答案正确率 + 引用准确率（引用的代码行是否正确）。CorpAssist 用 gpt-4o 作为评估模型，人工抽检 20% 确保评估质量。
代码审查的检出率和误报率怎么平衡？
检出率（召回问题的比例）和误报率（报告的问题中实际不是问题的比例）是 trade-off。推荐先追求低误报率（< 20%），再逐步提升检出率。因为高误报率会让开发者忽略所有审查意见。CorpAssist 的策略：高风险问题（安全漏洞）优先检出，风格问题可选择性报告。

本章结论

  • 三大场景：代码补全（低延迟）、代码问答（中延迟、高准确率）、代码审查（全量分析）。
  • 核心指标：采纳率、TTFT、P99 延迟、准确率。每个指标有明确的业务意义。
  • 延迟影响采纳率：TTFT > 200ms 时采纳率明显下降，200ms 是代码补全的生命线。
  • 安全是红线：代码助手绝不能生成有安全漏洞的代码，需要独立的审核流水线。
  • 基线驱动：上线前先建立指标基线，每次优化后对比。

    Budget --> Check{超出预算?}
    Check -->|否| Build[构建 FIM Prompt]
    Check -->|是| Drop[丢弃最低优先级]
    Drop --> L5
    Drop --> L4
    Drop --> L3

预算优先保障 L1-L3，超出时从 L5 开始丢弃。L1 永远不被丢弃。

# context_builder.py — CorpAssist 五层上下文构建器
import ast
import re
from pathlib import Path
from typing import List, Optional, Dict
import tiktoken

class CodeContextBuilder:
    """层次化代码上下文构建器"""

    LAYER_CONFIG = {
        "nearby": {"tokens": 2000, "priority": 5},
        "function": {"tokens": 1000, "priority": 4},
        "file_structure": {"tokens": 2000, "priority": 3},
        "related": {"tokens": 2000, "priority": 2},
        "project": {"tokens": 1000, "priority": 1},
    }

    def __init__(self, max_tokens: int = 8000):
        self.max_tokens = max_tokens
        self.encoder = tiktoken.get_encoding("cl100k_base")

    def _count_tokens(self, text: str) -> int:
        return len(self.encoder.encode(text))

    def build_context(self, file_path: str, cursor_line: int, repo_root: str) -> str:
        with open(file_path, encoding="utf-8") as f:
            lines = f.readlines()

        layers = []

        # L1: 光标附近
        nearby_start = max(0, cursor_line - 100)
        nearby_end = min(len(lines), cursor_line + 50)
        nearby_code = "".join(lines[nearby_start:nearby_end])
        layers.append(("nearby", f"// == 光标附近 ({nearby_start+1}-{nearby_end}) ==\n{nearby_code}"))

        # L2: 当前函数
        func_code = self._extract_function(lines, cursor_line)
        if func_code:
            layers.append(("function", f"// == 当前函数 ==\n{func_code}"))

        # L3: 文件结构（import 和符号列表）
        structure = self._get_structure(lines)
        layers.append(("file_structure", f"// == 文件结构 ==\n{structure}"))

        # L4: 相关文件（被 import 的文件摘要）
        imports = self._parse_imports(lines)
        related = self._get_related_summaries(imports, repo_root)
        if related:
            layers.append(("related", f"// == 相关文件 ==\n{related}"))

        # L5: 项目结构
        project_structure = self._get_project_structure(repo_root)
        layers.append(("project", f"// == 项目结构 ==\n{project_structure}"))

        # 按优先级排序，执行 Token 预算分配
        layers.sort(key=lambda x: self.LAYER_CONFIG[x[0]]["priority"], reverse=True)

        result = []
        used_tokens = 0
        for layer_name, content in layers:
            tokens = self._count_tokens(content)
            max_layer_tokens = self.LAYER_CONFIG[layer_name]["tokens"]
            # 截断到该层上限
            if tokens > max_layer_tokens:
                chars_per_token = len(content) / max(tokens, 1)
                content = content[:int(max_layer_tokens * chars_per_token)]
                tokens = self._count_tokens(content)
            if used_tokens + tokens <= self.max_tokens:
                result.append(content)
                used_tokens += tokens
            else:
                remaining = self.max_tokens - used_tokens
                if remaining > 100:
                    content = content[:int(remaining * (len(content) / max(tokens, 1)))]
                    result.append(content)
                break

        return "\n\n".join(result)

    def _extract_function(self, lines: List[str], cursor_line: int) -> str:
        try:
            tree = ast.parse("".join(lines))
            for node in ast.walk(tree):
                if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef, ast.ClassDef)):
                    if node.lineno <= cursor_line + 1 <= node.end_lineno:
                        return "".join(lines[node.lineno - 1:node.end_lineno])
        except SyntaxError:
            pass
        return ""

    def _get_structure(self, lines: List[str]) -> str:
        structure_lines = []
        for line in lines[:100]:
            stripped = line.strip()
            if stripped.startswith(("import ", "from ")) or \
               stripped.startswith(("def ", "class ", "async def ")):
                structure_lines.append(stripped)
        return "\n".join(structure_lines[:50])

    def _parse_imports(self, lines: List[str]) -> List[str]:
        imports = []
        for line in lines[:50]:
            if line.strip().startswith("import ") or line.strip().startswith("from "):
                parts = line.strip().split()
                if len(parts) > 1:
                    imports.append(parts[1].split(".")[0])
        return list(set(imports))

    def _get_related_summaries(self, imports: List[str], repo_root: str) -> str:
        summaries = []
        for imp in imports:
            for ext in [".py", ".java", ".js", ".ts"]:
                path = Path(repo_root) / f"{imp}{ext}"
                if path.exists():
                    with open(path) as f:
                        first_lines = "".join(f.readlines()[:30])
                    summaries.append(f"// {path.name} (摘要)\n{first_lines[:500]}")
                    break
        return "\n".join(summaries[:3])

    def _get_project_structure(self, repo_root: str) -> str:
        return f"项目根目录: {repo_root}"

超长函数（超过 200 行）应该进一步截断。CorpAssist 的做法：对超长函数只取光标所在的行块（前后 50 行），而不是整个函数。同时从 L1 和 L2 的预算中合并分配 3K（不分 2K + 1K），灵活调度。极端情况优先保障 L1（光标附近），L2 降级为函数签名 + 前 30 行。

不推荐
  • 直接塞入整个文件甚至整个仓库的所有代码
  • 无 Token 预算管理，超长上下文被 LLM 截断
  • 不区分代码类型，import 和业务代码混在一起
  • 不提取函数定义，光标在函数内也不知道上下文
推荐
  • 五层层次化选择，按优先级分配 Token 预算
  • 超出预算从低优先级丢弃，L1 永不丢失
  • AST 提取函数定义，关联 import 文件摘要
  • FIM 格式精确描述补全位置

本章结论

  • 五层上下文：光标附近 → 当前函数 → 文件结构 → 相关文件 → 项目结构。
  • Token 预算：8K 上限，L1（2K）+ L2（1K）+ L3（2K）+ L4（2K）+ L5（1K），超出丢弃低优先级。
  • AST 提取：精准定位光标所在的函数/类定义，提取完整代码。
  • FIM 格式：prefix + suffix + context 描述补全位置，模型精确填充。
  • 优先级保障：L1（光标附近）永不丢弃，其他层级按优先级截断。

# code_indexer.py — 代码索引器
import ast
import hashlib
from pathlib import Path
from typing import List, Dict, Generator
from pydantic import BaseModel

class CodeChunk(BaseModel):
    """代码块"""
    chunk_id: str
    file_path: str
    language: str
    chunk_type: str  # function / class / module
    name: str
    start_line: int
    end_line: int
    code: str
    docstring: str
    imports: List[str]

class AstChunker:
    """AST 代码切分器"""

    def __init__(self, repo_root: str):
        self.repo_root = Path(repo_root)

    def chunk_file(self, file_path: str) -> List[CodeChunk]:
        """将文件按函数/类/模块切分"""
        full_path = self.repo_root / file_path
        if not full_path.exists():
            return []
        with open(full_path) as f:
            source = f.read()
        try:
            tree = ast.parse(source)
        except SyntaxError:
            return [self._make_module_chunk(file_path, source)]

        chunks = []
        for node in ast.iter_child_nodes(tree):
            if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                chunk = CodeChunk(
                    chunk_id=hashlib.md5(f"{file_path}:{node.name}".encode()).hexdigest()[:12],
                    file_path=file_path,
                    language=self._detect_language(file_path),
                    chunk_type="function",
                    name=node.name,
                    start_line=node.lineno,
                    end_line=node.end_lineno,
                    code=ast.get_source_segment(source, node) or "",
                    docstring=ast.get_docstring(node) or "",
                    imports=self._extract_imports(node),
                )
                chunks.append(chunk)
            elif isinstance(node, ast.ClassDef):
                for item in node.body:
                    if isinstance(item, (ast.FunctionDef, ast.AsyncFunctionDef)):
                        name = f"{node.name}.{item.name}"
                        chunk = CodeChunk(
                            chunk_id=hashlib.md5(f"{file_path}:{name}".encode()).hexdigest()[:12],
                            file_path=file_path,
                            language=self._detect_language(file_path),
                            chunk_type="function",
                            name=name,
                            start_line=item.lineno,
                            end_line=item.end_lineno,
                            code=ast.get_source_segment(source, item) or "",
                            docstring=ast.get_docstring(item) or "",
                            imports=self._extract_imports(item),
                        )
                        chunks.append(chunk)
        return chunks

    def _detect_language(self, path: str) -> str:
        ext = Path(path).suffix
        return {"py": "python", "java": "java", "js": "javascript", "ts": "typescript"}.get(ext, "unknown")

    def _extract_imports(self, node: ast.AST) -> List[str]:
        imports = []
        for child in ast.walk(node):
            if isinstance(child, ast.Call) and hasattr(child.func, "id"):
                imports.append(child.func.id)
        return imports

    def _make_module_chunk(self, file_path: str, source: str) -> CodeChunk:
        return CodeChunk(
            chunk_id=hashlib.md5(file_path.encode()).hexdigest()[:12],
            file_path=file_path, language=self._detect_language(file_path),
            chunk_type="module", name=Path(file_path).stem,
            start_line=1, end_line=len(source.splitlines()),
            code=source[:2000], docstring="", imports=[],
        )

    def chunk_all(self, glob_pattern: str = "**/*.py") -> Generator[CodeChunk, None, None]:
        for f in self.repo_root.glob(glob_pattern):
            rel = f.relative_to(self.repo_root)
            yield from self.chunk_file(str(rel))

if __name__ == "__main__":

from langchain_openai import OpenAIEmbeddings
from pymilvus import connections, Collection, FieldSchema, CollectionSchema, DataType, utility
from typing import List

class CodeVectorIndex:
    """代码向量索引（Milvus）"""

    def __init__(self, host: str = "localhost", port: str = "19530"):
        connections.connect(host=host, port=port)
        self.collection_name = "code_index"
        self.embeddings = OpenAIEmbeddings(model="text-embedding-3-small", dimensions=512)
        self._ensure_collection()

    def _ensure_collection(self):
        if utility.has_collection(self.collection_name):
            self.collection = Collection(self.collection_name)
            return
        fields = [
            FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
            FieldSchema(name="chunk_id", dtype=DataType.VARCHAR, max_length=64),
            FieldSchema(name="file_path", dtype=DataType.VARCHAR, max_length=512),
            FieldSchema(name="function_name", dtype=DataType.VARCHAR, max_length=256),
            FieldSchema(name="code", dtype=DataType.VARCHAR, max_length=65535),
            FieldSchema(name="embedding", dtype=DataType.FLOAT_VECTOR, dim=512),
        ]
        schema = CollectionSchema(fields, "代码索引集合")
        self.collection = Collection(self.collection_name, schema)
        index_params = {"index_type": "IVF_FLAT", "metric_type": "COSINE", "params": {"nlist": 128}}
        self.collection.create_index("embedding", index_params)

    def insert_chunks(self, chunks: List[dict]):
        """插入代码块到 Milvus"""
        texts = [c["code"] for c in chunks]
        vectors = self.embeddings.embed_documents(texts)
        entities = [
            [c["chunk_id"] for c in chunks],
            [c["file_path"] for c in chunks],
            [c.get("function_name", "") for c in chunks],
            texts,
            vectors,
        ]
        self.collection.insert(entities)
        self.collection.flush()

    def search(self, query: str, top_k: int = 5) -> List[dict]:
        """向量检索"""
        self.collection.load()
        vector = self.embeddings.embed_query(query)
        results = self.collection.search(
            data=[vector], anns_field="embedding",
            param={"metric_type": "COSINE", "params": {"nprobe": 10}},
            limit=top_k, output_fields=["chunk_id", "file_path", "function_name", "code"],
        )
        hits = []
        for hits_batch in results:
            for hit in hits_batch:
                hits.append({"chunk_id": hit.entity.get("chunk_id"), "score": hit.score, "code": hit.entity.get("code")[:200]})
        return hits

        if language != "python":
            return True, []
        try:
            ast.parse(code)
            return True, []
        except SyntaxError as e:
            return False, [{"line": e.lineno or 0, "msg": e.msg, "text": e.text}]

    def check_security(self, code: str) -> Tuple[bool, List[Dict]]:
        """安全扫描：检测常见漏洞模式"""
        findings = []
        for pattern, cwe, msg in self.DANGEROUS_PATTERNS:
            matches = re.finditer(pattern, code, re.MULTILINE)
            for m in matches:
                line_no = code[:m.start()].count("\n") + 1
                # 检查是否在注释中
                line = code.split("\n")[line_no - 1].strip()
                if line.startswith("#") or line.startswith("//") or line.startswith("/*"):
                    continue
                findings.append({"cwe": cwe, "message": msg, "line": line_no, "pattern": pattern})
        return len(findings) == 0, findings

    def check_style(self, code: str, language: str = "python") -> Tuple[bool, List[Dict]]:
        """风格检查（简化版）"""
        issues = []
        lines = code.split("\n")
        for i, line in enumerate(lines):
            if len(line) > 100:
                issues.append({"line": i + 1, "msg": "行超过 100 字符", "severity": "warning"})
            if line.strip().startswith("print("):
                issues.append({"line": i + 1, "msg": "生产代码避免使用 print()，用 logging 代替", "severity": "warning"})
        return len(issues) < 3, issues[:5]

    def run_all(self, code: str, language: str = "python") -> QualityReport:
        """运行所有质量检查"""
        syntax_ok, syntax_errors = self.check_syntax(code, language)
        security_ok, security_findings = self.check_security(code)
        style_ok, style_issues = self.check_style(code, language)

        total_issues = len(syntax_errors) + len(security_findings) + len(style_issues)
        score = max(0.0, 1.0 - total_issues * 0.15)
        passed = syntax_ok and security_ok and style_ok

        if not syntax_ok:
            print(f"[质量网关] 语法错误: {syntax_errors}")
        if not security_ok:
            print(f"[质量网关] 安全违规: {security_findings}")
        if not style_ok:
            print(f"[质量网关] 风格问题: {style_issues}")

        return QualityReport(passed=passed, syntax_ok=syntax_ok, syntax_errors=syntax_errors,
                             security_ok=security_ok, security_findings=security_findings,
                             style_ok=style_ok, style_issues=style_issues, quality_score=round(score, 2))

    def quality_label(self, score: float) -> str:
        if score >= 0.9: return "优秀"
        if score >= 0.7: return "良好"
        if score >= 0.5: return "一般"
        return "差"

// SecurityScanner.java — 安全扫描器（集成 Semgrep）
@Component
public class SecurityScanner {

    private static final List<Pattern> DANGEROUS_PATTERNS = List.of(
        Pattern.compile("eval\\s*\\("),
        Pattern.compile("exec\\s*\\("),
        Pattern.compile("Runtime\\.getRuntime\\(\\)\\.exec"),
        Pattern.compile("ProcessBuilder")
    );

    public SecurityReport scan(String code, String language) {
        List<Finding> findings = new ArrayList<>();
        for (Pattern p : DANGEROUS_PATTERNS) {
            Matcher m = p.matcher(code);
            while (m.find()) {
                int line = code.substring(0, m.start()).split("\n").length;
                findings.add(new Finding("CWE-78", "Command Injection Risk", line));
            }
        }
        return new SecurityReport(findings.isEmpty(), findings);
    }
}

record Finding(String cwe, String message, int line) {}
record SecurityReport(boolean passed, List<Finding> findings) {}
不推荐
  • 不检查语法，用户看到编译错误的代码
  • 不安全扫描，生成包含 eval/exec 的代码
  • 无风格检查，生成代码和项目风格不一致
  • 展示所有生成结果，没有过滤机制
推荐
  • AST 语法检查拦截错误，补全前预验证
  • 8 种 CWE 安全规则扫描
  • 风格校验 + 质量评分（≥0.7 才展示）
  • 未通过检查的补全直接丢弃，不展示

本章结论

  • 三步过滤：语法检查（AST）→ 安全扫描（8 种 CWE 模式）→ 风格校验（行长度/print）。
  • 质量评分：0-1 分，≥0.7 才展示给开发者，总体拦截率 ~8%。
  • 安全规则：覆盖 eval/exec/shell 注入/SQL 注入/反序列化等 CWE 标准模式。
  • 质量标签：优秀(≥0.9)、良好(≥0.7)、一般(≥0.5)、差(<0.5)。
