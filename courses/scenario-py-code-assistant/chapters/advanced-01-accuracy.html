# eval_framework.py — CorpAssist 准确率评测框架
from pydantic import BaseModel, Field
from typing import List, Optional, Callable
import json
import ast
from datetime import datetime
from openai import OpenAI

class EvalCase(BaseModel):
    """评测用例"""
    id: str
    prefix: str
    suffix: str = ""
    expected: str = Field(..., description="预期的正确补全")
    language: str = "python"
    category: str = Field(default="logic", description="logic/api/boilerplate/bug_fix")

class EvalResult(BaseModel):
    """单条评测结果"""
    case_id: str
    actual_completion: str
    syntax_correct: bool
    semantics_correct: bool
    security_safe: bool
    quality_score: float
    latency_ms: float

class CodeEvalPipeline:
    """代码补全评估管线"""

    def __init__(self, completion_fn: Callable):
        self.completion_fn = completion_fn
        self.llm_judge = OpenAI(base_url="https://dashscope.aliyuncs.com/compatible-mode/v1")
        self.cases: List[EvalCase] = []
        self.results: List[EvalResult] = []

    def load_dataset(self, path: str):
        with open(path) as f:
            data = json.load(f)
        self.cases = [EvalCase(**c) for c in data]
        print(f"加载 {len(self.cases)} 条评测用例")

    def run(self) -> dict:
        """全量评测"""
        import time
        for case in self.cases:
            start = time.time()
            completion = self.completion_fn(case.prefix, case.suffix, case.language)

            syntax_ok = self._check_syntax(completion, case.language)
            semantics_ok = self._judge_semantics(case.prefix + completion, case.expected)
            security_ok = self._check_security(completion)
            # 质量评分
            issues = (0 if syntax_ok else 1) + (0 if semantics_ok else 1) + (0 if security_ok else 1)
            score = max(0.0, 1.0 - issues * 0.2)

            self.results.append(EvalResult(
                case_id=case.id, actual_completion=completion,
                syntax_correct=syntax_ok, semantics_correct=semantics_ok,
                security_safe=security_ok, quality_score=round(score, 2),
                latency_ms=(time.time() - start) * 1000,
            ))

        return self._report()

    def _check_syntax(self, code: str, language: str) -> bool:
        if language != "python":
            return True
        try:
            ast.parse(code)
            return True
        except SyntaxError:
            return False

    def _check_security(self, code: str) -> bool:
        import re
        dangerous = [r"\beval\s*\(", r"\bexec\s*\(", r"subprocess.*shell=True"]
        return not any(re.search(p, code) for p in dangerous)

    def _judge_semantics(self, full_code: str, expected: str) -> bool:
        """LLM 评判语义是否一致"""
        resp = self.llm_judge.chat.completions.create(
            model="qwen-turbo",
            messages=[{"role": "system", "content": "判断两段代码是否语义相同（实现逻辑一致）。仅返回 YES 或 NO。"},
                      {"role": "user", "content": f"生成代码:\n{full_code}\n\n预期代码:\n{expected}"}],
            temperature=0,
            max_tokens=10,
        )
        return "YES" in resp.choices[0].message.content.upper()

    def _report(self) -> dict:
        n = len(self.results)
        if n == 0:
            return {}
        syntax_rate = sum(1 for r in self.results if r.syntax_correct) / n
        semantics_rate = sum(1 for r in self.results if r.semantics_correct) / n
        security_rate = sum(1 for r in self.results if r.security_safe) / n
        avg_score = sum(r.quality_score for r in self.results) / n
        avg_latency = sum(r.latency_ms for r in self.results) / n
        return {
            "date": datetime.now().isoformat(),
            "total_cases": n,
            "syntax_accuracy": round(syntax_rate * 100, 1),
            "semantics_accuracy": round(semantics_rate * 100, 1),
            "security_safe_rate": round(security_rate * 100, 1),
            "avg_quality_score": round(avg_score, 2),
            "avg_latency_ms": round(avg_latency, 1),
            "security_risk_count": sum(1 for r in self.results if not r.security_safe),
        }


# 数据集构建
def create_sample_dataset():
    dataset = [
        EvalCase(id="py-001", prefix="def add(a, b):\n    ", expected="return a + b",
                 language="python", category="logic"),
        EvalCase(id="py-002", prefix="from typing import List\n\ndef sum_list(items: List[int]) -> int:\n    ", expected="return sum(items)",
                 language="python", category="logic"),
        EvalCase(id="py-003", prefix="import json\n\ndef parse_config(path: str) -> dict:\n    with open(path) as f:\n        ",
                 expected="return json.load(f)", language="python", category="api"),
        EvalCase(id="py-004", prefix="def get_user(user_id: int) -> dict:\n    # TODO: 从数据库查询用户",
                 expected="return db.query(User).filter(User.id == user_id).first()",
                 language="python", category="api"),
    ]
    with open("./eval_dataset.json", "w") as f:
        json.dump([c.model_dump() for c in dataset], f, ensure_ascii=False, indent=2)
    print(f"数据集已创建: {len(dataset)} 条")

if __name__ == "__main__":
    create_sample_dataset()
    # 模拟评测
    def mock_completion(prefix, suffix, lang):
        return "return a + b"
    pipeline = CodeEvalPipeline(mock_completion)
    pipeline.load_dataset("./eval_dataset.json")
    report = pipeline.run()
    print(json.dumps(report, indent=2))
采纳率跟踪

采纳率通过 IDE 插件埋点获取。每次补全展示后记录：展示时间、补全 ID、是否被采纳（用户按 Tab 或 Enter 接受）、修改后采纳（用户编辑后接受）、拒绝（用户继续输入或按 Esc）。典型的采纳率基线：初始 28%，优化到 45%。每周统计并对比变更前后的采纳率变化。

优化记录与迭代管理

迭代变更内容语法准确率语义准确率采纳率
基线初始 Prompt92%65%28%
V2FIM 格式优化 + 上下文截断94%72%35%
V3风格约束参数 + 后处理修复96%78%40%
V4质量网关集成（拦截率 8%）98%82%45%

每次迭代必须同时观察所有四个指标。如果某个指标明显退步（如采纳率上升但安全违规率也上升），说明变更方向有问题，需要回滚或调整。CorpAssist 的实践：每次变更先在 10% 流量灰度 3 天，对比基线数据后再全量。

本章结论

  • 四维指标：语法准确率 ≥95%、语义准确率 ≥80%、采纳率 ≥40%、安全违规率 ≤3%。
  • 评测数据集：至少 100 条覆盖 logic/api/boilerplate/bug_fix 四类场景。
  • LLM as Judge：语义准确率用 LLM 评判（qwen-turbo），比人工高效。
  • 采纳率是北极星：其他指标服务于采纳率，最终看用户是否接受补全。
