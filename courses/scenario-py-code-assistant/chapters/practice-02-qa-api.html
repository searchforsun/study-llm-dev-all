本章先记住 3 件事

  • 引用必须精确：每条答案都需标注来自哪个文件的哪几行。
  • 流式是标配：SSE 流式输出首 Token < 500ms，非流式 > 5s。

# code_qa_api.py — CorpAssist 代码问答 API
from fastapi import FastAPI, HTTPException
from fastapi.responses import StreamingResponse
from pydantic import BaseModel, Field
from openai import OpenAI
from pymilvus import Collection, connections
from typing import AsyncGenerator, List, Optional
import json, time

app = FastAPI(title="CorpAssist 代码问答 API", version="1.0.0")

class QARequest(BaseModel):
    question: str = Field(..., min_length=2, max_length=2000, description="关于代码库的问题")
    repo: str = "default"
    top_k: int = Field(default=5, ge=1, le=20)
    score_threshold: float = Field(default=0.0, ge=0.0, le=1.0)
    stream: bool = False

class CodeReference(BaseModel):
    file_path: str
    start_line: int
    end_line: int
    symbol: str = ""
    score: float = 0.0

class QAResponse(BaseModel):
    answer: str
    references: List[CodeReference]
    latency_ms: float

class CodeRetriever:
    """代码检索器：Mivus 向量检索 + 符号检索"""

    def __init__(self):
        connections.connect(host="localhost", port=19530)
        self.collection = Collection("code_index")
        self.collection.load()
        self.embeddings = OpenAI(
            base_url="https://dashscope.aliyuncs.com/compatible-mode/v1",
        )

    def retrieve(self, question: str, repo: str, top_k: int = 5,
                 score_threshold: float = 0.0) -> List[dict]:
        """双通道检索：向量 + 符号"""
        start = time.time()

        # 向量检索
        vector = self.embeddings.embeddings.create(
            input=[question], model="text-embedding-3-small", dimensions=512
        ).data[0].embedding

        expr = f'repo == "{repo}"' if repo != "default" else ""

        results = self.collection.search(
            data=[vector],
            anns_field="embedding",
            param={"metric_type": "COSINE", "params": {"nprobe": 10}},
            limit=top_k,
            expr=expr,
            output_fields=["file_path", "symbol", "code", "start_line", "end_line", "repo"],
        )

        docs = []
        for hits in results:
            for hit in hits:
                if hit.score >= score_threshold:
                    docs.append({
                        "file_path": hit.entity.get("file_path"),
                        "symbol": hit.entity.get("symbol", ""),
                        "code": hit.entity.get("code", ""),
                        "score": round(hit.score, 4),
                        "start_line": hit.entity.get("start_line"),
                        "end_line": hit.entity.get("end_line"),
                    })

        elapsed = (time.time() - start) * 1000
        print(f"[检索] {question[:30]}... → {len(docs)} 条结果, {elapsed:.0f}ms")
        return docs

# Prompt 模板
QA_SYSTEM = """你是一个专业的代码助手 CorpAssist。基于以下代码上下文回答开发者的问题。

规则：
1. 只回答代码上下文中包含的信息，不编造 API 或行为
2. 引用具体的 file_path 和行号，格式：`文件路径:行号`
3. 如果上下文中没有足够信息，回复「代码库中未找到相关信息」
4. 给出代码示例时标注来源文件

代码上下文（按相关性排序）：
{context}
"""

qa_prompt = lambda ctx, q: [
    {"role": "system", "content": QA_SYSTEM.format(context=ctx)},
    {"role": "user", "content": q},
]

def format_context(docs: List[dict]) -> str:
    parts = []
    for i, d in enumerate(docs):
        header = f"[{i+1}] {d['file_path']}:{d['start_line']}-{d['end_line']}"
        if d.get("symbol"):
            header += f" ({d['symbol']})"
        parts.append(f"{header}\n```python\n{d['code'][:1500]}\n```")
    return "\n\n".join(parts)

retriever = CodeRetriever()
llm = OpenAI(base_url="https://dashscope.aliyuncs.com/compatible-mode/v1")

@app.post("/qa", response_model=QAResponse)
async def qa(req: QARequest):
    start = time.time()
    docs = retriever.retrieve(req.question, req.repo, req.top_k, req.score_threshold)
    if not docs:
        raise HTTPException(404, "代码库中未找到相关信息")

    context = format_context(docs)
    resp = llm.chat.completions.create(
        model="gpt-4o-mini",
        messages=qa_prompt(context, req.question),
        temperature=0.1,
        max_tokens=1024,
    )

    references = [CodeReference(file_path=d["file_path"], start_line=d["start_line"],
                                end_line=d["end_line"], symbol=d.get("symbol",""), score=d["score"])
                  for d in docs[:5]]
    return QAResponse(answer=resp.choices[0].message.content, references=references,
                      latency_ms=(time.time() - start) * 1000)

@app.post("/qa/stream")
async def qa_stream(req: QARequest):
    docs = retriever.retrieve(req.question, req.repo, req.top_k, req.score_threshold)
    context = format_context(docs)

    async def generate():
        yield f"data: {json.dumps({'type': 'references', 'data': [{'file': d['file_path'], 'lines': f'{d[\"start_line\"]}-{d[\"end_line\"]}'} for d in docs[:5]]})}\n\n"
        resp = llm.chat.completions.create(
            model="gpt-4o-mini",
            messages=qa_prompt(context, req.question),
            temperature=0.1, max_tokens=1024, stream=True,
        )
        for chunk in resp:
            if chunk.choices[0].delta.content:
                yield f"data: {json.dumps({'type': 'token', 'content': chunk.choices[0].delta.content})}\n\n"
        yield "data: {\"type\": \"done\"}\n\n"

    return StreamingResponse(generate(), media_type="text/event-stream",
                             headers={"Cache-Control": "no-cache", "X-Accel-Buffering": "no"})

@app.get("/health")
async def health():
    return {"status": "ok", "vector_store_ready": True}

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
不推荐
  • 只检索不引用——用户不知道答案来源
  • 无流式输出——用户等待 5s+ 才看到结果
  • top_k 固定（如 3）——简单问题不够，复杂问题超预算
  • 无 score_threshold——低质量结果也被送入 LLM
推荐
  • 每条答案附带文件路径 + 行号引用
  • SSE 流式输出，首 Token < 500ms
  • top_k 动态：简单 3 条，复杂 10 条
  • score_threshold 过滤低质量匹配

检索参数调优

参数推荐值影响
top_k5（可调 3-10）越大召回越全，但 token 成本增加
score_threshold0.3-0.5过滤低质量匹配；太高可能漏掉相关代码
embedding modeltext-embedding-3-small512 维平衡精度与性能
chunk_size函数级（AST）语义完整，避免切断逻辑

本章结论

  • 检索 + 生成管道：向量检索召回相关代码 → LLM 生成带引用的答案。
  • 引用格式：每个答案段落标注来源 file_path:start_line-end_line。
  • SSE 流式：先推引用 metadata，再流式输出 token，首 Token < 500ms。
  • 参数调优：top_k=5、score_threshold=0.3、AST 函数级切分。
