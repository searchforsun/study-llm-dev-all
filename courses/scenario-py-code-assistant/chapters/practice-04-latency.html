# latency_optimizer.py — CorpAssist 延迟优化器
import hashlib
import time
import asyncio
from typing import Dict, Optional, Tuple, Callable
from collections import OrderedDict
from openai import OpenAI
import numpy as np

class SemanticCache:
    """语义缓存：通过向量相似度判断缓存命中"""

    def __init__(self, max_size: int = 2000, similarity_threshold: float = 0.92):
        self.max_size = max_size
        self.threshold = similarity_threshold
        self.cache: OrderedDict[str, Tuple[str, list, str]] = OrderedDict()  # key -> (completion, embedding, model)
        self.embeddings = OpenAI(base_url="https://dashscope.aliyuncs.com/compatible-mode/v1")

    def _embed(self, text: str) -> list:
        resp = self.embeddings.embeddings.create(input=[text], model="text-embedding-3-small", dimensions=256)
        return resp.data[0].embedding

    def get(self, prefix: str, suffix: str, language: str = "") -> Optional[str]:
        query = f"{prefix}:{suffix}:{language}"
        q_vec = self._embed(query[:200])  # 取前 200 字符做语义
        best_key, best_sim = None, 0.0
        for key, (completion, emb, _) in self.cache.items():
            sim = np.dot(q_vec, emb) / (np.linalg.norm(q_vec) * np.linalg.norm(emb) + 1e-10)
            if sim > best_sim:
                best_sim = sim
                best_key = key
        if best_sim >= self.threshold and best_key:
            self.cache.move_to_end(best_key)
            print(f"[语义缓存] 命中: sim={best_sim:.3f}")
            return self.cache[best_key][0]
        return None

    def set(self, prefix: str, suffix: str, completion: str, language: str = ""):
        key = f"{prefix[:100]}:{suffix[:50]}:{language}"
        emb = self._embed(prefix[:200])
        self.cache[key] = (completion, emb, "gpt-4o-mini")
        if len(self.cache) > self.max_size:
            self.cache.popitem(last=False)


class ModelRouter:
    """模型路由：按请求复杂度选择模型"""

    def __init__(self):
        self.small = OpenAI(base_url="https://dashscope.aliyuncs.com/compatible-mode/v1")
        self.large = OpenAI(base_url="https://dashscope.aliyuncs.com/compatible-mode/v1")
        # 小模型：qwen-turbo，大模型：gpt-4o-mini
        self.small_model = "qwen-turbo"
        self.large_model = "gpt-4o-mini"

    def route(self, prefix: str, suffix: str) -> Tuple[str, str]:
        """返回 (model_name, client)"""
        total_len = len(prefix) + len(suffix)
        if total_len < 300 and "def " not in prefix and "class " not in prefix:
            return self.small_model, self.small  # 简单：小模型 ~50ms
        return self.large_model, self.large  # 复杂：大模型 ~200ms


class LatencyOptimizer:
    """延迟优化器：缓存 → 路由 → 推理"""

    def __init__(self):
        self.cache = SemanticCache()
        self.router = ModelRouter()
        self._latencies = []
        self.metrics = {"total": 0, "cache_hit": 0, "small_model": 0, "large_model": 0}

    async def complete(self, prefix: str, suffix: str, language: str = "",
                       style: str = "") -> Tuple[str, float]:
        start = time.time()

        # L1: 语义缓存
        cached = self.cache.get(prefix, suffix, language)
        if cached:
            self.metrics["cache_hit"] += 1
            return cached, (time.time() - start) * 1000

        # L2: 模型路由
        model, client = self.router.route(prefix, suffix)
        if model == "qwen-turbo":
            self.metrics["small_model"] += 1
        else:
            self.metrics["large_model"] += 1

        prompt = f"Complete the code at cursor:\nPrefix:{prefix}\nSuffix:{suffix}\nLanguage:{language}\n"
        resp = client.chat.completions.create(
            model=model,
            messages=[{"role": "user", "content": prompt}],
            temperature=0.2,
            max_tokens=128,
        )
        completion = resp.choices[0].message.content.strip()

        # 写入缓存
        self.cache.set(prefix, suffix, completion, language)

        elapsed = (time.time() - start) * 1000
        self._latencies.append(elapsed)
        self.metrics["total"] += 1
        return completion, elapsed

    def report(self) -> dict:
        n = len(self._latencies)
        sorted_lats = sorted(self._latencies)
        return {
            "total_requests": self.metrics["total"] + self.metrics["cache_hit"],
            "cache_hit_rate": round(self.metrics["cache_hit"] / max(self.metrics["total"] + self.metrics["cache_hit"], 1) * 100, 1),
            "small_model_pct": round(self.metrics["small_model"] / max(self.metrics["total"], 1) * 100, 1),
            "p50_ms": round(sorted_lats[int(n * 0.5)], 1) if n > 0 else 0,
            "p95_ms": round(sorted_lats[int(n * 0.95)], 1) if n > 0 else 0,
            "p99_ms": round(sorted_lats[int(n * 0.99)], 1) if n > 0 else 0,
        }

if __name__ == "__main__":
    opt = LatencyOptimizer()
    loop = asyncio.get_event_loop()
    for i in range(10):
        result, ms = loop.run_until_complete(
            opt.complete("def hello(name):\n    print(f\"Hello, {", ")", "python")
        )
        print(f"第{i+1}次: {ms:.0f}ms")
    print(opt.report())
不推荐
  • 所有请求都用同一个大模型（延迟 200-500ms）
  • 不做缓存，每次都是完整推理
  • 不用 vLLM，用原始 HuggingFace 推理（吞吐低 10x）
  • 不监控 P99，不知道延迟恶化
推荐
  • 语义缓存 30% + 小模型 40% + 大模型 30%
  • 缓存命中 <5ms，小模型 ~50ms，大模型 ~200ms
  • vLLM FlashAttention + 连续批处理，吞吐 10x
  • P99 延迟 < 500ms 监控看板

本章结论

  • 三级缓存：语义缓存命中率 30%，命中延迟 <5ms。
  • 模型路由：简单请求用 qwen-turbo（~50ms），复杂用 gpt-4o-mini（~200ms）。
  • vLLM 加速：FlashAttention + 连续批处理 + prefix caching，吞吐 10x。
  • P99 监控：目标 P99 < 500ms，每天看板监控。
