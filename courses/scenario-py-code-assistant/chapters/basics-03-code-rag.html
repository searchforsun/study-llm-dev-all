为什么要学本章

代码问答和代码审查依赖代码 RAG——从代码仓库中检索相关代码片段。与文档 RAG 不同，代码有结构化信息（函数、类、继承关系）和符号引用。本章讲解如何为代码构建专门的 RAG 管线，包括代码切分、向量化、符号索引、混合检索。

  • 设计代码索引的双通道架构（符号索引 + 向量索引）
  • 实现基于 AST 的代码切分策略
  • 选择适合代码的 Embedding 模型并配置 Milvus 集合
  • 理解代码 RAG 的检索优化：函数级 vs 行级检索
本章先记住 3 件事
  • 代码不是文本：代码有语法结构、依赖关系、符号定义——这些信息不能通过纯文本 Embedding 捕获。
  • 双通道检索：符号索引（精准匹配）+ 向量索引（语义匹配），互补提升召回率。
  • 函数是最小单元：按函数/类切分代码，保留完整上下文，比按行切分效果好 2x。

# code_indexer.py — 代码索引器
import ast
import hashlib
from pathlib import Path
from typing import List, Dict, Generator
from pydantic import BaseModel

class CodeChunk(BaseModel):
    """代码块"""
    chunk_id: str
    file_path: str
    language: str
    chunk_type: str  # function / class / module
    name: str
    start_line: int
    end_line: int
    code: str
    docstring: str
    imports: List[str]

class AstChunker:
    """AST 代码切分器"""

    def __init__(self, repo_root: str):
        self.repo_root = Path(repo_root)

    def chunk_file(self, file_path: str) -> List[CodeChunk]:
        """将文件按函数/类/模块切分"""
        full_path = self.repo_root / file_path
        if not full_path.exists():
            return []
        with open(full_path) as f:
            source = f.read()
        try:
            tree = ast.parse(source)
        except SyntaxError:
            return [self._make_module_chunk(file_path, source)]

        chunks = []
        for node in ast.iter_child_nodes(tree):
            if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                chunk = CodeChunk(
                    chunk_id=hashlib.md5(f"{file_path}:{node.name}".encode()).hexdigest()[:12],
                    file_path=file_path,
                    language=self._detect_language(file_path),
                    chunk_type="function",
                    name=node.name,
                    start_line=node.lineno,
                    end_line=node.end_lineno,
                    code=ast.get_source_segment(source, node) or "",
                    docstring=ast.get_docstring(node) or "",
                    imports=self._extract_imports(node),
                )
                chunks.append(chunk)
            elif isinstance(node, ast.ClassDef):
                for item in node.body:
                    if isinstance(item, (ast.FunctionDef, ast.AsyncFunctionDef)):
                        name = f"{node.name}.{item.name}"
                        chunk = CodeChunk(
                            chunk_id=hashlib.md5(f"{file_path}:{name}".encode()).hexdigest()[:12],
                            file_path=file_path,
                            language=self._detect_language(file_path),
                            chunk_type="function",
                            name=name,
                            start_line=item.lineno,
                            end_line=item.end_lineno,
                            code=ast.get_source_segment(source, item) or "",
                            docstring=ast.get_docstring(item) or "",
                            imports=self._extract_imports(item),
                        )
                        chunks.append(chunk)
        return chunks

    def _detect_language(self, path: str) -> str:
        ext = Path(path).suffix
        return {"py": "python", "java": "java", "js": "javascript", "ts": "typescript"}.get(ext, "unknown")

    def _extract_imports(self, node: ast.AST) -> List[str]:
        imports = []
        for child in ast.walk(node):
            if isinstance(child, ast.Call) and hasattr(child.func, "id"):
                imports.append(child.func.id)
        return imports

    def _make_module_chunk(self, file_path: str, source: str) -> CodeChunk:
        return CodeChunk(
            chunk_id=hashlib.md5(file_path.encode()).hexdigest()[:12],
            file_path=file_path, language=self._detect_language(file_path),
            chunk_type="module", name=Path(file_path).stem,
            start_line=1, end_line=len(source.splitlines()),
            code=source[:2000], docstring="", imports=[],
        )

    def chunk_all(self, glob_pattern: str = "**/*.py") -> Generator[CodeChunk, None, None]:
        for f in self.repo_root.glob(glob_pattern):
            rel = f.relative_to(self.repo_root)
            yield from self.chunk_file(str(rel))

if __name__ == "__main__":
    chunker = AstChunker("./my_repo")
    for chunk in chunker.chunk_all("**/*.py"):
        print(f"{chunk.chunk_type}: {chunk.name} ({chunk.start_line}-{chunk.end_line})")

# vector_index.py — 代码向量索引
from langchain_openai import OpenAIEmbeddings
from pymilvus import connections, Collection, FieldSchema, CollectionSchema, DataType, utility
from typing import List

class CodeVectorIndex:
    """代码向量索引（Milvus）"""

    def __init__(self, host: str = "localhost", port: str = "19530"):
        connections.connect(host=host, port=port)
        self.collection_name = "code_index"
        self.embeddings = OpenAIEmbeddings(model="text-embedding-3-small", dimensions=512)
        self._ensure_collection()

    def _ensure_collection(self):
        if utility.has_collection(self.collection_name):
            self.collection = Collection(self.collection_name)
            return
        fields = [
            FieldSchema(name="id", dtype=DataType.INT64, is_primary=True, auto_id=True),
            FieldSchema(name="chunk_id", dtype=DataType.VARCHAR, max_length=64),
            FieldSchema(name="file_path", dtype=DataType.VARCHAR, max_length=512),
            FieldSchema(name="function_name", dtype=DataType.VARCHAR, max_length=256),
            FieldSchema(name="code", dtype=DataType.VARCHAR, max_length=65535),
            FieldSchema(name="embedding", dtype=DataType.FLOAT_VECTOR, dim=512),
        ]
        schema = CollectionSchema(fields, "代码索引集合")
        self.collection = Collection(self.collection_name, schema)
        index_params = {"index_type": "IVF_FLAT", "metric_type": "COSINE", "params": {"nlist": 128}}
        self.collection.create_index("embedding", index_params)

    def insert_chunks(self, chunks: List[dict]):
        """插入代码块到 Milvus"""
        texts = [c["code"] for c in chunks]
        vectors = self.embeddings.embed_documents(texts)
        entities = [
            [c["chunk_id"] for c in chunks],
            [c["file_path"] for c in chunks],
            [c.get("function_name", "") for c in chunks],
            texts,
            vectors,
        ]
        self.collection.insert(entities)
        self.collection.flush()

    def search(self, query: str, top_k: int = 5) -> List[dict]:
        """向量检索"""
        self.collection.load()
        vector = self.embeddings.embed_query(query)
        results = self.collection.search(
            data=[vector], anns_field="embedding",
            param={"metric_type": "COSINE", "params": {"nprobe": 10}},
            limit=top_k, output_fields=["chunk_id", "file_path", "function_name", "code"],
        )
        hits = []
        for hits_batch in results:
            for hit in hits_batch:
                hits.append({"chunk_id": hit.entity.get("chunk_id"), "score": hit.score, "code": hit.entity.get("code")[:200]})
        return hits
