为什么要学本章

代码补全和代码问答需要理解大量上下文。LLM 的上下文窗口在变大，但如何高效地选择和组织上下文，远比直接塞入所有代码更关键。CorpAssist 的实践：4K token 预算内，层次化上下文选择比「全量塞入」的准确率高出 15-20%。

本章先记住 3 件事
  • 上下文不是越多越好：4K-8K token 预算，有选择地构建上下文。
  • 优先给光标附近：L1 光标附近（2000 tokens）最有效，L5 项目结构（1000 tokens）最弱。
  • FIM 格式让 model 知道补全位置：不是「生成下文」而是「填充中间」。

五层上下文选择策略

CorpAssist 使用五层上下文选择策略，按优先级分配 Token 预算。当总 Token 超过预算时，从低优先级开始丢弃。

层级内容Token 预算优先级
L1 光标附近光标前 100 行 + 后 50 行~2000最高
L2 当前函数光标所在函数完整代码（AST 提取）~1000高
L3 当前文件import 和类/函数签名列表~2000中
L4 相关文件被 import 文件的函数摘要~2000低
L5 项目结构目录结构和关键配置~1000最低
CorpAssist · 上下文组装场景 开发者在 order_service.py 的 create_order 函数中编写代码。上下文组装器提取光标附近 150 行（L1）+ create_order 函数（L2）+ 文件顶部的 import（L3）+ order.py 模型的摘要（L4）+ 项目结构（L5）。预算 8K token，L1-L3 占了 5K，L4 和 L5 被截断。

预算优先保障 L1-L3，超出时从 L5 开始丢弃。L1 永远不被丢弃。

Python 上下文构建器

python
# context_builder.py — CorpAssist 五层上下文构建器
import ast
import re
from pathlib import Path
from typing import List, Optional, Dict
import tiktoken

class CodeContextBuilder:
    """层次化代码上下文构建器"""

    LAYER_CONFIG = {
        "nearby": {"tokens": 2000, "priority": 5},
        "function": {"tokens": 1000, "priority": 4},
        "file_structure": {"tokens": 2000, "priority": 3},
        "related": {"tokens": 2000, "priority": 2},
        "project": {"tokens": 1000, "priority": 1},
    }

    def __init__(self, max_tokens: int = 8000):
        self.max_tokens = max_tokens
        self.encoder = tiktoken.get_encoding("cl100k_base")

    def _count_tokens(self, text: str) -> int:
        return len(self.encoder.encode(text))

    def build_context(self, file_path: str, cursor_line: int, repo_root: str) -> str:
        with open(file_path, encoding="utf-8") as f:
            lines = f.readlines()

        layers = []

        # L1: 光标附近
        nearby_start = max(0, cursor_line - 100)
        nearby_end = min(len(lines), cursor_line + 50)
        nearby_code = "".join(lines[nearby_start:nearby_end])
        layers.append(("nearby", f"// == 光标附近 ({nearby_start+1}-{nearby_end}) ==\n{nearby_code}"))

        # L2: 当前函数
        func_code = self._extract_function(lines, cursor_line)
        if func_code:
            layers.append(("function", f"// == 当前函数 ==\n{func_code}"))

        # L3: 文件结构（import 和符号列表）
        structure = self._get_structure(lines)
        layers.append(("file_structure", f"// == 文件结构 ==\n{structure}"))

        # L4: 相关文件（被 import 的文件摘要）
        imports = self._parse_imports(lines)
        related = self._get_related_summaries(imports, repo_root)
        if related:
            layers.append(("related", f"// == 相关文件 ==\n{related}"))

        # L5: 项目结构
        project_structure = self._get_project_structure(repo_root)
        layers.append(("project", f"// == 项目结构 ==\n{project_structure}"))

        # 按优先级排序，执行 Token 预算分配
        layers.sort(key=lambda x: self.LAYER_CONFIG[x[0]]["priority"], reverse=True)

        result = []
        used_tokens = 0
        for layer_name, content in layers:
            tokens = self._count_tokens(content)
            max_layer_tokens = self.LAYER_CONFIG[layer_name]["tokens"]
            # 截断到该层上限
            if tokens > max_layer_tokens:
                chars_per_token = len(content) / max(tokens, 1)
                content = content[:int(max_layer_tokens * chars_per_token)]
                tokens = self._count_tokens(content)
            if used_tokens + tokens <= self.max_tokens:
                result.append(content)
                used_tokens += tokens
            else:
                remaining = self.max_tokens - used_tokens
                if remaining > 100:
                    content = content[:int(remaining * (len(content) / max(tokens, 1)))]
                    result.append(content)
                break

        return "\n\n".join(result)

    def _extract_function(self, lines: List[str], cursor_line: int) -> str:
        try:
            tree = ast.parse("".join(lines))
            for node in ast.walk(tree):
                if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef, ast.ClassDef)):
                    if node.lineno <= cursor_line + 1 <= node.end_lineno:
                        return "".join(lines[node.lineno - 1:node.end_lineno])
        except SyntaxError:
            pass
        return ""

    def _get_structure(self, lines: List[str]) -> str:
        structure_lines = []
        for line in lines[:100]:
            stripped = line.strip()
            if stripped.startswith(("import ", "from ")) or \
               stripped.startswith(("def ", "class ", "async def ")):
                structure_lines.append(stripped)
        return "\n".join(structure_lines[:50])

    def _parse_imports(self, lines: List[str]) -> List[str]:
        imports = []
        for line in lines[:50]:
            if line.strip().startswith("import ") or line.strip().startswith("from "):
                parts = line.strip().split()
                if len(parts) > 1:
                    imports.append(parts[1].split(".")[0])
        return list(set(imports))

    def _get_related_summaries(self, imports: List[str], repo_root: str) -> str:
        summaries = []
        for imp in imports:
            for ext in [".py", ".java", ".js", ".ts"]:
                path = Path(repo_root) / f"{imp}{ext}"
                if path.exists():
                    with open(path) as f:
                        first_lines = "".join(f.readlines()[:30])
                    summaries.append(f"// {path.name} (摘要)\n{first_lines[:500]}")
                    break
        return "\n".join(summaries[:3])

    def _get_project_structure(self, repo_root: str) -> str:
        return f"项目根目录: {repo_root}"


# FIM 格式构建
class FIMFormatter:
    """FIM (Fill-in-the-Middle) Prompt 构建器"""

    @classmethod
    def format_fim(cls, prefix: str, suffix: str, context: str = "") -> str:
        """构建 FIM 格式的 Prompt"""
        parts = []
        if context:
            parts.append(f"\n{context}")
        parts.append(f"\n{prefix}")
        if suffix:
            parts.append(f"\n{suffix}")
        parts.append("\n")
        return "\n\n".join(parts)

    @classmethod
    def format_qa(cls, question: str, context: str) -> str:
        """构建代码问答的 Prompt"""
        return f"""基于以下代码上下文回答问题。

代码上下文：
{context}

问题：{question}
回答（引用具体文件路径和行号）："""


if __name__ == "__main__":
    builder = CodeContextBuilder(max_tokens=4000)
    context = builder.build_context("./example.py", cursor_line=50, repo_root="./")
    print(f"上下文 token: {builder._count_tokens(context)}")
    print(context[:500])
先想 10 秒：如果光标在一个 500 行的超长函数中，L2 吃掉全部 4K 预算怎么办？

超长函数（超过 200 行）应该进一步截断。CorpAssist 的做法：对超长函数只取光标所在的行块（前后 50 行），而不是整个函数。同时从 L1 和 L2 的预算中合并分配 3K（不分 2K + 1K），灵活调度。极端情况优先保障 L1（光标附近），L2 降级为函数签名 + 前 30 行。

不推荐
  • 直接塞入整个文件甚至整个仓库的所有代码
  • 无 Token 预算管理，超长上下文被 LLM 截断
  • 不区分代码类型，import 和业务代码混在一起
  • 不提取函数定义，光标在函数内也不知道上下文
推荐
  • 五层层次化选择，按优先级分配 Token 预算
  • 超出预算从低优先级丢弃，L1 永不丢失
  • AST 提取函数定义，关联 import 文件摘要
  • FIM 格式精确描述补全位置

本章结论

  • 五层上下文：光标附近 → 当前函数 → 文件结构 → 相关文件 → 项目结构。
  • Token 预算：8K 上限，L1（2K）+ L2（1K）+ L3（2K）+ L4（2K）+ L5（1K），超出丢弃低优先级。
  • AST 提取：精准定位光标所在的函数/类定义，提取完整代码。
  • FIM 格式：prefix + suffix + context 描述补全位置，模型精确填充。
  • 优先级保障：L1（光标附近）永不丢弃，其他层级按优先级截断。
