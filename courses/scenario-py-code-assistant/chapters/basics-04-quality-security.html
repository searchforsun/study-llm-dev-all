为什么要学本章

代码助手生成的代码可能存在语法错误、安全漏洞（SQL 注入、XSS）、逻辑缺陷。直接展示给开发者会造成严重后果。CorpAssist 在展示代码建议前，必须经过语法检查、安全扫描、风格校验三步过滤。未通过的补全直接丢弃，不展示给开发者。

学完你能
  • 实现代码补全的实时语法检查（AST + pylint）
  • 集成安全扫描工具（Bandit/Semgrep）到代码助手中
  • 设计代码补全质量的自动评分机制
  • 建立代码安全检测的流水线：检查 → 拦截 → 告警
本章先记住 3 件事
  • 语法检查是第一关：AST 解析通不过的代码直接丢弃。
  • 安全扫描不可跳过：检测 eval/exec/shell 注入等 CWE 模式。
  • 展示前必须验证：所有代码建议展示给开发者前必须经过质量网关。

代码质量网关

CorpAssist · 质量网关拦截率 CorpAssist 的质量网关日均检查 2 万+ 次代码建议，拦截率约 8%（其中语法错误 5%、安全违规 2%、风格违规 1%）。被拦截的代码不会展示给开发者，避免「代码助手生成了一段有 bug 的代码」的负面体验。

Python 质量网关实现

python
# quality_gate.py — CorpAssist 代码质量网关
import ast
import subprocess
import re
from typing import List, Dict, Tuple, Optional
from dataclasses import dataclass, field

@dataclass
class QualityReport:
    passed: bool
    syntax_ok: bool = True
    syntax_errors: List[Dict] = field(default_factory=list)
    security_ok: bool = True
    security_findings: List[Dict] = field(default_factory=list)
    style_ok: bool = True
    style_issues: List[Dict] = field(default_factory=list)
    quality_score: float = 1.0

class CodeQualityGate:
    """代码质量网关：语法 + 安全 + 风格三步检查"""

    # 危险模式检测（安全扫描的核心规则）
    DANGEROUS_PATTERNS = [
        (r"\beval\s*\(", "CWE-95", "避免使用 eval()，可能导致代码注入"),
        (r"\bexec\s*\(", "CWE-95", "避免使用 exec()，可能导致代码注入"),
        (r"__import__\s*\(", "CWE-95", "避免动态导入"),
        (r"subprocess\.(call|Popen|run)\s*\(.*shell=True", "CWE-78", "shell=True 可能导致命令注入"),
        (r"pickle\.loads?\s*\(", "CWE-502", "避免反序列化不可信数据"),
        (r"os\.system\s*\(", "CWE-78", "避免使用 os.system()"),
        (r"sqlite3\.execute\s*\(.*f['\"]", "CWE-89", "f-string 拼接 SQL 可能导致注入"),
        (r"requests\.(get|post)\s*\(.*verify=False", "CWE-295", "禁用 SSL 验证不安全"),
    ]

    def check_syntax(self, code: str, language: str = "python") -> Tuple[bool, List[Dict]]:
        """语法检查"""
        if language != "python":
            return True, []
        try:
            ast.parse(code)
            return True, []
        except SyntaxError as e:
            return False, [{"line": e.lineno or 0, "msg": e.msg, "text": e.text}]

    def check_security(self, code: str) -> Tuple[bool, List[Dict]]:
        """安全扫描：检测常见漏洞模式"""
        findings = []
        for pattern, cwe, msg in self.DANGEROUS_PATTERNS:
            matches = re.finditer(pattern, code, re.MULTILINE)
            for m in matches:
                line_no = code[:m.start()].count("\n") + 1
                # 检查是否在注释中
                line = code.split("\n")[line_no - 1].strip()
                if line.startswith("#") or line.startswith("//") or line.startswith("/*"):
                    continue
                findings.append({"cwe": cwe, "message": msg, "line": line_no, "pattern": pattern})
        return len(findings) == 0, findings

    def check_style(self, code: str, language: str = "python") -> Tuple[bool, List[Dict]]:
        """风格检查（简化版）"""
        issues = []
        lines = code.split("\n")
        for i, line in enumerate(lines):
            if len(line) > 100:
                issues.append({"line": i + 1, "msg": "行超过 100 字符", "severity": "warning"})
            if line.strip().startswith("print("):
                issues.append({"line": i + 1, "msg": "生产代码避免使用 print()，用 logging 代替", "severity": "warning"})
        return len(issues) < 3, issues[:5]

    def run_all(self, code: str, language: str = "python") -> QualityReport:
        """运行所有质量检查"""
        syntax_ok, syntax_errors = self.check_syntax(code, language)
        security_ok, security_findings = self.check_security(code)
        style_ok, style_issues = self.check_style(code, language)

        total_issues = len(syntax_errors) + len(security_findings) + len(style_issues)
        score = max(0.0, 1.0 - total_issues * 0.15)
        passed = syntax_ok and security_ok and style_ok

        if not syntax_ok:
            print(f"[质量网关] 语法错误: {syntax_errors}")
        if not security_ok:
            print(f"[质量网关] 安全违规: {security_findings}")
        if not style_ok:
            print(f"[质量网关] 风格问题: {style_issues}")

        return QualityReport(passed=passed, syntax_ok=syntax_ok, syntax_errors=syntax_errors,
                             security_ok=security_ok, security_findings=security_findings,
                             style_ok=style_ok, style_issues=style_issues, quality_score=round(score, 2))

    def quality_label(self, score: float) -> str:
        if score >= 0.9: return "优秀"
        if score >= 0.7: return "良好"
        if score >= 0.5: return "一般"
        return "差"


gate = CodeQualityGate()

# 测试
test_cases = [
    ("def add(a, b): return a + b", "安全的代码"),
    ("def process(data): return eval(data)", "包含 eval 的安全风险"),
    ("def broken( : syntax error", "语法错误"),
]

for code, desc in test_cases:
    report = gate.run_all(code)
    print(f"[{desc}] 通过={report.passed} 分数={report.quality_score} {gate.quality_label(report.quality_score)}")

# 生产环境集成到补全服务
def quality_filter(completion: str, language: str = "python") -> Optional[str]:
    """质量过滤器：检查通过返回代码，否则返回 None"""
    report = gate.run_all(completion, language)
    if report.passed and report.quality_score >= 0.7:
        return completion
    print(f"[质量过滤器] 拦截补全: 分数={report.quality_score}")
    return None

Java 侧：安全扫描集成

java
// SecurityScanner.java — 安全扫描器（集成 Semgrep）
@Component
public class SecurityScanner {

    private static final List<Pattern> DANGEROUS_PATTERNS = List.of(
        Pattern.compile("eval\\s*\\("),
        Pattern.compile("exec\\s*\\("),
        Pattern.compile("Runtime\\.getRuntime\\(\\)\\.exec"),
        Pattern.compile("ProcessBuilder")
    );

    public SecurityReport scan(String code, String language) {
        List<Finding> findings = new ArrayList<>();
        for (Pattern p : DANGEROUS_PATTERNS) {
            Matcher m = p.matcher(code);
            while (m.find()) {
                int line = code.substring(0, m.start()).split("\n").length;
                findings.add(new Finding("CWE-78", "Command Injection Risk", line));
            }
        }
        return new SecurityReport(findings.isEmpty(), findings);
    }
}

record Finding(String cwe, String message, int line) {}
record SecurityReport(boolean passed, List<Finding> findings) {}
不推荐
  • 不检查语法，用户看到编译错误的代码
  • 不安全扫描，生成包含 eval/exec 的代码
  • 无风格检查，生成代码和项目风格不一致
  • 展示所有生成结果，没有过滤机制
推荐
  • AST 语法检查拦截错误，补全前预验证
  • 8 种 CWE 安全规则扫描
  • 风格校验 + 质量评分（≥0.7 才展示）
  • 未通过检查的补全直接丢弃，不展示

本章结论

  • 三步过滤：语法检查（AST）→ 安全扫描（8 种 CWE 模式）→ 风格校验（行长度/print）。
  • 质量评分：0-1 分，≥0.7 才展示给开发者，总体拦截率 ~8%。
  • 安全规则：覆盖 eval/exec/shell 注入/SQL 注入/反序列化等 CWE 标准模式。
  • 质量标签：优秀(≥0.9)、良好(≥0.7)、一般(≥0.5)、差(<0.5)。
