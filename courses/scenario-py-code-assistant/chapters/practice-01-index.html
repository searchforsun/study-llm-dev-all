为什么要学本章

代码 RAG 的基础是索引流水线——将仓库代码解析、切分、向量化后存入向量库。企业代码库每天都有 Git 提交，索引必须是增量的代码索引，每次全量重索引不现实。CorpAssist 管理着 50+ 个仓库、日均 200+ 次提交，本章教你构建支持 Git 增量更新的代码索引流水线。

学完你能

  • 用 AST 解析代码做函数级切分，配置 Milvus 集合 Schema
  • 搭建向量索引 + 符号索引双通道检索
  • 管理多仓库的索引权限隔离
本章先记住 3 件事
  • 增量索引：只索引变更的文件，全量重索引作为兜底。
  • 函数级切分：按 AST 的函数/类定义切分，比按行数切分更语义完整。
  • 双通道索引：向量索引（语义相似）+ 符号索引（精确匹配函数名/类名）。

Git 增量索引实现

python
# incremental_indexer.py — CorpAssist Git 增量索引器
import subprocess
import hashlib
import ast
from pathlib import Path
from typing import Set, List, Dict, Optional
from pymilvus import Collection, CollectionSchema, FieldSchema, DataType, connections
from openai import OpenAI

class GitChangeDetector:
    """检测 Git 仓库的变更文件"""

    SUPPORTED_EXTS = {".py", ".java", ".js", ".ts", ".go", ".rs", ".sql"}

    def __init__(self, repo_path: str):
        self.repo = Path(repo_path)
        if not (self.repo / ".git").exists():
            raise ValueError(f"不是 Git 仓库: {repo_path}")

    def get_changed_files(self, since_commit: str = "HEAD~1") -> Dict[str, List[str]]:
        """获取指定 commit 以来的变更文件列表"""
        result = subprocess.run(
            ["git", "diff", "--name-status", since_commit],
            cwd=self.repo, capture_output=True, text=True, check=True,
        )
        added, modified, deleted = [], [], []
        for line in result.stdout.strip().split("\n"):
            if not line.strip():
                continue
            parts = line.split("\t")
            if len(parts) < 2:
                continue
            status, filepath = parts[0], parts[1]
            ext = Path(filepath).suffix
            if ext not in self.SUPPORTED_EXTS:
                continue
            if status == "A":
                added.append(filepath)
            elif status == "M":
                modified.append(filepath)
            elif status == "D":
                deleted.append(filepath)
            elif status in ("R", "R100"):
                parts = filepath.split("\t")
                deleted.append(parts[0])
                added.append(parts[1])
        return {"added": added, "modified": modified, "deleted": deleted}

    def full_scan(self) -> List[str]:
        """全量扫描所有文件"""
        result = subprocess.run(
            ["git", "ls-files"], cwd=self.repo, capture_output=True, text=True, check=True,
        )
        files = []
        for f in result.stdout.strip().split("\n"):
            if f and Path(f).suffix in self.SUPPORTED_EXTS:
                files.append(f)
        return files


class CodeChunker:
    """AST 驱动的代码切分器，按函数/类切分"""

    def __init__(self, max_chunk_lines: int = 100):
        self.max_chunk_lines = max_chunk_lines

    def chunk_file(self, file_path: str, code: str) -> List[dict]:
        """将代码文件切分为函数/类级别的块"""
        ext = Path(file_path).suffix
        if ext == ".py":
            return self._chunk_python(file_path, code)
        # 其他语言用行数切分
        return self._chunk_by_lines(file_path, code)

    def _chunk_python(self, file_path: str, code: str) -> List[dict]:
        chunks = []
        lines = code.splitlines()
        try:
            tree = ast.parse(code)
            for node in ast.walk(tree):
                if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef, ast.ClassDef)):
                    chunk = "\n".join(lines[node.lineno - 1:node.end_lineno])
                    if len(chunk.splitlines()) <= self.max_chunk_lines:
                        chunks.append({
                            "chunk_id": hashlib.md5(f"{file_path}:{node.name}".encode()).hexdigest()[:12],
                            "file_path": file_path,
                            "symbol": node.name,
                            "symbol_type": "function" if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)) else "class",
                            "start_line": node.lineno,
                            "end_line": node.end_lineno,
                            "code": chunk,
                        })
        except SyntaxError:
            pass
        if not chunks:
            chunks = self._chunk_by_lines(file_path, code)
        return chunks

    def _chunk_by_lines(self, file_path: str, code: str) -> List[dict]:
        lines = code.splitlines()
        chunks = []
        chunk_size = 50
        for i in range(0, len(lines), chunk_size):
            chunk = "\n".join(lines[i:i + chunk_size])
            chunks.append({
                "chunk_id": hashlib.md5(f"{file_path}:{i}".encode()).hexdigest()[:12],
                "file_path": file_path,
                "symbol": Path(file_path).stem,
                "symbol_type": "file",
                "start_line": i + 1,
                "end_line": min(i + chunk_size, len(lines)),
                "code": chunk,
            })
        return chunks


class MilvusIndexer:
    """Milvus 向量索引管理器"""

    def __init__(self, collection_name: str = "code_index", dim: int = 512):
        self.collection_name = collection_name
        self.dim = dim
        self.embeddings = OpenAI(
            base_url="https://dashscope.aliyuncs.com/compatible-mode/v1",
        )
        connections.connect(host="localhost", port=19530)
        self.collection = self._get_or_create_collection()

    def _get_or_create_collection(self) -> Collection:
        if self.collection_name in connections.list_collections():
            return Collection(self.collection_name)
        schema = CollectionSchema([
            FieldSchema("chunk_id", DataType.VARCHAR, max_length=12, is_primary=True),
            FieldSchema("file_path", DataType.VARCHAR, max_length=256),
            FieldSchema("symbol", DataType.VARCHAR, max_length=128),
            FieldSchema("symbol_type", DataType.VARCHAR, max_length=32),
            FieldSchema("start_line", DataType.INT64),
            FieldSchema("end_line", DataType.INT64),
            FieldSchema("code", DataType.VARCHAR, max_length=8192),
            FieldSchema("repo", DataType.VARCHAR, max_length=64),
            FieldSchema("embedding", DataType.FLOAT_VECTOR, dim=self.dim),
        ])
        collection = Collection(self.collection_name, schema)
        collection.create_index("embedding", {"index_type": "IVF_FLAT", "metric_type": "COSINE", "params": {"nlist": 128}})
        collection.load()
        return collection

    def index_chunks(self, chunks: List[dict], repo: str):
        """将代码块索引到 Milvus"""
        if not chunks:
            return
        texts = [c["code"] for c in chunks]
        vectors = self.embeddings.embeddings.create(input=texts, model="text-embedding-3-small", dimensions=self.dim).data
        entities = [
            [c["chunk_id"] for c in chunks],
            [c["file_path"] for c in chunks],
            [c["symbol"] for c in chunks],
            [c["symbol_type"] for c in chunks],
            [c["start_line"] for c in chunks],
            [c["end_line"] for c in chunks],
            [c["code"] for c in chunks],
            [repo] * len(chunks),
            [v.embedding for v in vectors],
        ]
        self.collection.insert(entities)
        self.collection.flush()
        print(f"[Milvus] 索引 {len(chunks)} 个块到 {repo}")

    def delete_file(self, file_path: str):
        expr = f'file_path == "{file_path}"'
        self.collection.delete(expr)
        print(f"[Milvus] 删除 {file_path}")


if __name__ == "__main__":
    detector = GitChangeDetector("./my_repo")
    chunker = CodeChunker()
    indexer = MilvusIndexer()

    changes = detector.get_changed_files()
    print(f"新增: {len(changes['added'])} 修改: {len(changes['modified'])} 删除: {len(changes['deleted'])}")

    for fp in changes["deleted"]:
        indexer.delete_file(fp)

    for fp in changes["added"] + changes["modified"]:
        full_path = Path("./my_repo") / fp
        if not full_path.exists():
            continue
        code = full_path.read_text(encoding="utf-8")
        chunks = chunker.chunk_file(str(full_path), code)
        indexer.delete_file(fp)
        indexer.index_chunks(chunks, "my_repo")

// SymbolIndexer.java — 符号索引管理
@Component
public class SymbolIndexer {

    private final JdbcTemplate jdbc;

    public SymbolIndexer(JdbcTemplate jdbc) { this.jdbc = jdbc; }

    @Transactional
    public void indexSymbols(String repo, String filePath, List<SymbolEntry> symbols) {
        deleteSymbols(repo, filePath);
        for (SymbolEntry s : symbols) {
            jdbc.update("INSERT INTO code_symbols (repo, file_path, symbol, symbol_type, start_line, end_line) VALUES (?,?,?,?,?,?)",
                repo, filePath, s.name(), s.type(), s.startLine(), s.endLine());
        }
    }

    public void deleteSymbols(String repo, String filePath) {
        jdbc.update("DELETE FROM code_symbols WHERE repo=? AND file_path=?", repo, filePath);
    }

    public List<SymbolEntry> searchBySymbol(String repo, String symbol) {
        return jdbc.query("SELECT * FROM code_symbols WHERE repo=? AND symbol LIKE ?",
            (rs, row) -> new SymbolEntry(rs.getString("symbol"), rs.getString("symbol_type"),
                rs.getInt("start_line"), rs.getInt("end_line")),
            repo, "%" + symbol + "%");
    }
}

record SymbolEntry(String name, String type, int startLine, int endLine) {}
不推荐
  • 每次全量重索引全部文件（耗时数十分钟）
  • 按固定行数切分（50行一刀切，切断函数体）
  • 不处理文件删除（向量库中残留旧数据）
  • 只有向量索引，没有符号索引
推荐
  • Git diff 增量更新（平均 45s vs 15min 全量）
  • AST 驱动函数级切分（语义完整）
  • 先删后插，保持索引一致性
  • 向量 + 符号双通道（语义 + 精确匹配）
先想 10 秒：增量索引过程中如果又有一个新的提交，如何处理？

需要在索引器外部加锁或版本号。CorpAssist 的做法：1) 索引器启动时记录当前 HEAD commit hash；2) 索引期间新提交不影响当前批次；3) 下一轮增量索引从上一轮的 HEAD 开始 diff。极端情况下某文件在一轮索引中被修改两次，后一轮索引会覆盖前一轮的结果（先删后插保证幂等）。

多仓库管理与权限隔离

CorpAssist 管理 50+ 仓库，每个仓库的索引在 Milvus 中用 repo 字段隔离。检索时传入用户有权限的仓库列表，Milvus 的检索请求携带 expr 过滤条件。

本章结论

  • Git 增量索引：git diff 检测变更，先删后插，平均 45s 完成。
  • AST 函数级切分：Python 用 ast.parse 提取函数/类定义，语义完整不割裂。
  • Milvus 向量索引：512 维 text-embedding-3-small，IVF_FLAT 索引，COSINE 度量。
  • 符号索引兜底：PostgreSQL 存储函数名/类名，精确匹配用于检索辅助。
  • 多仓库权限隔离：repo 字段过滤，检索时按用户权限限制仓库范围。
