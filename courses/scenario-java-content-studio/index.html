// 敏感词过滤器
@Component
public class SensitiveWordFilter {

    // 三层过滤
    private final Set<String> exactBlacklist;       // 精确匹配词库
    private final List<Pattern> patternBlacklist;    // 正则匹配模式
    private final ChatClient aiDetector;             // AI 变体检测

    public FilterResult filter(String content) {
        List<FilterMatch> matches = new ArrayList<>();

        // 1. 精确匹配
        for (String word : exactBlacklist) {
            int idx = content.indexOf(word);
            if (idx >= 0) {
                matches.add(new FilterMatch("EXACT", word, idx, 1.0));
            }
        }

        // 2. 正则匹配（变体/谐音）
        for (Pattern pattern : patternBlacklist) {
            Matcher matcher = pattern.matcher(content);
            while (matcher.find()) {
                matches.add(new FilterMatch("PATTERN",
                    matcher.group(), matcher.start(), 0.9));
            }
        }

        // 3. AI 变体检测（拼音/拆字/变形）
        if (matches.isEmpty()) {
            String aiResult = aiDetector.prompt()
                .user(u -> u.text("""
                    检测以下内容是否包含敏感信息的变体表达。
                    关注：拼音替代、拆字、谐音、隐晦表达。
                    内容：{content}
                    输出：{"sensitive":false,"reason":"","riskScore":0}
                    """)
                    .param("content", content)
                )
                .call()
                .content();
            // 解析 AI 检测结果
            AiDetectionResult ai = parseAiResult(aiResult);
            if (ai.isSensitive()) {
                matches.add(new FilterMatch("AI_DETECT",
                    ai.getReason(), 0, ai.getRiskScore()));
            }
        }

        return new FilterResult(!matches.isEmpty(), matches);
    }
}

// 行业规则引擎
@Component
public class IndustryRuleEngine {

    private final Map<String, List<IndustryRule>> industryRules = Map.of(
        "finance", List.of(
            new IndustryRule("保本宣传", "禁止使用'保本'、'稳赚'等承诺性表述",
                Pattern.compile("保本|稳赚|无风险| guaranteed")),
            new IndustryRule("收益承诺", "禁止承诺具体收益率",
                Pattern.compile("年化[^，。]*\\d+%|收益率[^，。]*\\d+"))
        ),
        "medical", List.of(
            new IndustryRule("疗效承诺", "禁止'包治百病'等绝对化表述",
                Pattern.compile("包治|根治|彻底治愈|特效")),
            new IndustryRule("患者案例", "使用患者案例需脱敏授权",
                Pattern.compile("患者[^，。]*治疗[^，。]*效果"))
        ),
        "education", List.of(
            new IndustryRule("效果承诺", "禁止保证提分/包过",
                Pattern.compile("包过|保证提分|保录取| guaranteed score"))
        )
    );

    public List<IndustryRuleViolation> check(String content, String industry) {
        List<IndustryRule> rules = industryRules.get(industry);
        if (rules == null) return List.of();
        return rules.stream()
            .filter(r -> r.getPattern().matcher(content).find())
            .map(r -> new IndustryRuleViolation(r.getName(), r.getDescription()))
            .toList();
    }
}

Q1: 如何控制 AI 生成内容的风格？

考点：风格控制不仅仅是"写正式一点"的指令，而是系统化的参数体系和模板管理。

好回答

五维语调参数 + Few-shot 示例 + 模板版本管理。语调参数（正式度/情感度/专业度/简洁度/幽默度）控制风格方向；Few-shot 示例提供 2-3 个"好的"例子让 LLM 模仿；模板用 Git 管理，每次修改有 PR Review 和版本 diff。这样即使换模型（从 qwen-max 到 qwen-plus），风格仍然一致。

差回答

在 Prompt 里写"请用正式语气"就行。

@Component
public class ContentStudioRoiCalculator {

    private static final double AI_COST_PER_WORD = 0.00002;  // $0.02/1000 words

    public RoiResult calculate(String scenario, int monthlyVolume,
                                double manualHours, double aiMinutes) {
        double manualMonthlyHours = monthlyVolume * manualHours;
        double manualMonthlyCost = manualMonthlyHours * 35.0; // $35/h

        double aiMonthlyHours = monthlyVolume * (aiMinutes / 60.0);
        double avgWords = estimateWords(scenario);
        double aiApiCost = monthlyVolume * avgWords * AI_COST_PER_WORD;
        double aiTotalCost = aiMonthlyHours * 35.0 + aiApiCost;

        double monthlySavings = manualMonthlyCost - aiTotalCost;
        double annualSavings = monthlySavings * 12;
        double roi = (annualSavings / (aiTotalCost * 12)) * 100;

        return new RoiResult(scenario, monthlyVolume, manualMonthlyCost,
            aiTotalCost, monthlySavings, annualSavings, roi);
    }

    record RoiResult(String scenario, int volume, double manualCost,
                     double aiCost, double monthlySaving,
                     double annualSaving, double roi) {}
}

@Component
public class LongFormWriter {

    private final ChatClient outlineClient;
    private final ChatClient segmentClient;

    public String write(String topic, BrandTone tone, int targetWords) {
        // 1. 生成大纲
        String outline = outlineClient.prompt()
            .user(u -> u.text("""
                为「{topic}」生成写作大纲，输出 JSON 格式。
                总字数约 {words} 字。
                大纲包含 5-8 个章节，每个章节标注重点内容和字数分配。
                JSON: [{"title":"引言","keyPoints":["背景","意义"],"words":300}]
                """)
                .param("topic", topic)
                .param("words", String.valueOf(targetWords))
            )
            .call()
            .content();

        // 2. 分段生成
        List<OutlineItem> items = parseOutline(outline);
        StringBuilder fullText = new StringBuilder();
        String previousSummary = "";

        for (int i = 0; i < items.size(); i++) {
            OutlineItem item = items.get(i);
            String segment = segmentClient.prompt()
                .user(u -> u.text("""
                    写作主题：{topic}
                    当前章节：{title}
                    章节重点：{keyPoints}
                    目标字数：{words}字
                    前章摘要：{prevSummary}
                    这是第 {idx}/{total} 章。

                    请生成该章节内容，保持与前章的连贯性。
                    """)
                    .param("topic", topic)
                    .param("title", item.title())
                    .param("keyPoints", String.join(",", item.keyPoints()))
                    .param("words", String.valueOf(item.words()))
                    .param("prevSummary", previousSummary)
                    .param("idx", String.valueOf(i + 1))
                    .param("total", String.valueOf(items.size()))
                )
                .call()
                .content();

            fullText.append("\n\n## ").append(item.title()).append("\n\n")
                .append(segment);
            previousSummary = summarize(segment, 100);
        }

        // 3. 拼接+检测
        return assembleAndCheck(fullText.toString());
    }
}

// 连贯性检测器
@Component
public class CoherenceChecker {

    public List<CoherenceIssue> check(String fullText) {
        List<CoherenceIssue> issues = new ArrayList<>();
        String[] paragraphs = fullText.split("\n\n");

        // 1. 相邻段落主题漂移检测
        for (int i = 1; i < paragraphs.length; i++) {
            double similarity = cosineSimilarity(
                embed(paragraphs[i-1]), embed(paragraphs[i]));
            if (similarity < 0.3) {
                issues.add(new CoherenceIssue(i-1, i,
                    "主题漂移", similarity));
            }
        }

        // 2. 人称一致性检测
        Set<String> persons = Set.of("我们", "我", "您", "你", "他们");
        for (String p : persons) {
            boolean found = false;
            for (String para : paragraphs) {
                if (para.contains(p)) {
                    if (found) continue;
                    found = true;
                }
                if (found && !para.contains(p)) {
                    issues.add(new CoherenceIssue(
                        "人称不一致: '" + p + "' 在段落中突然消失"));
                    break;
                }
            }
        }

        return issues;
    }
}

// 段落去重器
@Component
public class Deduplicator {

    public String deduplicate(String text) {
        String[] paragraphs = text.split("\n\n");
        List<String> unique = new ArrayList<>();

        for (String para : paragraphs) {
            boolean duplicate = false;
            for (String existing : unique) {
                double sim = jaccardSimilarity(para, existing);
                if (sim > 0.7) {
                    duplicate = true;
                    break;
                }
            }
            if (!duplicate) {
                unique.add(para);
            }
        }
        return String.join("\n\n", unique);
    }

    private double jaccardSimilarity(String a, String b) {
        Set<String> setA = new HashSet<>(List.of(a.split("\\s+")));
        Set<String> setB = new HashSet<>(List.of(b.split("\\s+")));
        Set<String> intersection = new HashSet<>(setA);
        intersection.retainAll(setB);
        Set<String> union = new HashSet<>(setA);
        union.addAll(setB);
        return (double) intersection.size() / union.size();
    }
}

为什么学这章

企业内容生成往往需要批量产出（一次生成 50 条产品描述、100 条客户回访文案）。Spring Batch 提供事务性批处理能力，结合 MQ 任务队列实现可靠的大规模内容生成。

学完后能做什么

本章要点速记

1) Spring Batch Job = Step(Reader→Processor→Writer)，Processor 中调用 LLM 生成；2) MQ 任务队列实现任务的拆分和调度（如 100 条 split 成 10 个 batch 各 10 条）；3) 进度追踪通过 JobExecution 的 ExecutionContext 存储每批的生成结果；4) 失败重试：Step 级别最多重试 3 次，跳过不可恢复的错误。

// Spring Batch 配置
@Configuration
public class BatchContentJobConfig {

    @Bean
    public Job batchContentJob(JobRepository jobRepository,
                                Step generateStep, Step summaryStep) {
        return new JobBuilder("batchContentJob", jobRepository)
            .start(generateStep)
            .next(summaryStep)
            .build();
    }

    @Bean
    public Step generateStep(JobRepository jobRepository,
                              PlatformTransactionManager transactionManager) {
        return new StepBuilder("generateStep", jobRepository)
            .<ContentInput, ContentOutput>chunk(10, transactionManager)
            .reader(contentReader())
            .processor(contentProcessor())
            .writer(contentWriter())
            .faultTolerant()
            .retryLimit(3)
            .retry(TimeoutException.class)
            .skipLimit(5)
            .skip(IllegalArgumentException.class)
            .build();
    }

    @Bean
    public ItemProcessor<ContentInput, ContentOutput> contentProcessor() {
        return input -> {
            // 模板填充 + LLM 调用
            String prompt = fillTemplate(input.getTemplateId(), input.getVariables());
            String content = llmService.generate(prompt);
            return new ContentOutput(input.getId(), content, "SUCCESS");
        };
    }

    @Bean
    public ItemWriter<ContentOutput> contentWriter() {
        return items -> {
            for (ContentOutput item : items) {
                contentRepository.save(item);
                progressTracker.increment(item.getId());
            }
        };
    }
}

// MQ 任务拆分
@Component
public class ContentTaskDispatcher {

    private final RabbitTemplate rabbitTemplate;

    public void dispatchBatch(List<ContentInput> inputs) {
        // 拆分为每个 10 条的子任务
        List<List<ContentInput>> batches = partition(inputs, 10);
        String batchId = UUID.randomUUID().toString();

        for (int i = 0; i < batches.size(); i++) {
            ContentTask task = new ContentTask(
                batchId, i, batches.get(i));
            rabbitTemplate.convertAndSend(
                "content.generate.queue", task);
        }
    }

    @RabbitListener(queues = "content.generate.queue")
    public void handleTask(ContentTask task) {
        try {
            for (ContentInput input : task.getInputs()) {
                String content = llmService.generate(
                    fillTemplate(input.getTemplateId(), input.getVariables()));
                contentRepository.save(new ContentOutput(
                    input.getId(), content, "SUCCESS"));
                progressTracker.update(task.getBatchId(),
                    task.getTaskIndex(), 1, 0);
            }
        } catch (Exception e) {
            progressTracker.update(task.getBatchId(),
                task.getTaskIndex(), 0, task.getInputs().size());
        }
    }
}

// 进度追踪
@Component
public class ProgressTracker {

    private final RedisTemplate<String, String> redis;

    public void initBatch(String batchId, int total) {
        redis.opsForValue().set("batch:" + batchId + ":total",
            String.valueOf(total));
        redis.opsForValue().set("batch:" + batchId + ":completed", "0");
        redis.opsForValue().set("batch:" + batchId + ":failed", "0");
    }

    public void update(String batchId, int taskIdx, int success, int fail) {
        redis.opsForValue().increment("batch:" + batchId + ":completed", success);
        redis.opsForValue().increment("batch:" + batchId + ":failed", fail);
    }

    public Progress getProgress(String batchId) {
        int total = Integer.parseInt(
            redis.opsForValue().get("batch:" + batchId + ":total"));
        int completed = Integer.parseInt(
            redis.opsForValue().get("batch:" + batchId + ":completed"));
        int failed = Integer.parseInt(
            redis.opsForValue().get("batch:" + batchId + ":failed"));
        return new Progress(total, completed, failed,
            (double) (completed + failed) / total * 100);
    }
}

@Component
public class SensitiveWordFilter implements Advisor {

    private final Set<String> sensitiveWords = Set.of(
        "敏感词A", "敏感词B", "敏感词C");

    @Override
    public AdviceResponse before(AdviceRequest request) {
        String content = request.getContent();
        for (String word : sensitiveWords) {
            if (content.contains(word)) {
                return new AdviceResponse(false,
                    "内容包含敏感词: " + word);
            }
        }
        return new AdviceResponse(true);
    }
}

        return instance.getId();
    }

    public void approve(String taskId, String comment) {
        Map<String, Object> variables = new HashMap<>();
        variables.put("status", "APPROVED");
        variables.put("comment", comment);
        taskService.complete(taskId, variables);
    }

        Map<String, Object> variables = new HashMap<>();
        variables.put("status", "REJECTED");
        variables.put("comment", comment);
        taskService.complete(taskId, variables);
    }
}

@Component
public class AbExperimentManager {

    private final Map<String, Experiment> experiments = new ConcurrentHashMap<>();

    public String assignVariant(String userId, String experimentId) {
        Experiment exp = experiments.get(experimentId);
        if (exp == null) return "control";

        // 基于用户 ID 哈希分组
        int group = Math.abs(userId.hashCode()) % 100;
        if (group < exp.getControlPercent() * 100) {
            return "control";
        }
        return "treatment";
    }

    public void recordResult(String userId, String experimentId,
                              String variant, String metric, double value) {
        String key = "abtest:" + experimentId + ":" + variant;
        redisTemplate.opsForHash().increment(key, "count", 1);
        redisTemplate.opsForHash().increment(key, metric + "_sum",
            Double.doubleToRawLongBits(value));
    }

// LLM-as-Judge 自动打分
@Component
public class AutoScorer {

    private final ChatClient judgeClient;

    public ScoreResult score(String content, String scenario,
                              BrandTone expectedTone) {
        String result = judgeClient.prompt()
            .user(u -> u.text("""
                你是一个内容质量评审专家。请从以下 5 个维度评分（1-5分）。

                内容：
                {content}

                场景：{scenario}
                期望语调：{tone}

                评分维度：
                1. 相关性(relevance) - 内容是否符合场景需求
                2. 准确性(accuracy) - 事实是否准确、逻辑是否自洽
                3. 流畅度(fluency) - 语言是否流畅、易于理解
                4. 风格匹配(style) - 是否符合品牌语调要求
                5. 合规性(compliance) - 是否违反安全规范

                输出 JSON：
                {"relevance":4,"accuracy":3,"fluency":5,
                 "style":4,"compliance":5,
                 "total":4.2,
                 "strengths":["语言流畅","品牌语调准确"],
                 "weaknesses":["部分事实不够准确"],
                 "suggestion":"建议核对数据来源"}
                """)
                .param("content", content)
                .param("scenario", scenario)
                .param("tone", expectedTone.toPromptInstruction())
            )
            .call()
            .content();

        return parseResult(result);
    }

    public record ScoreResult(double relevance, double accuracy,
                               double fluency, double style,
                               double compliance, double total,
                               List<String> strengths,
                               List<String> weaknesses,
                               String suggestion) {}
}

@RestController
@RequestMapping("/api/content")
public class ContentController {

    private final ContentGenerator generator;
    private final BatchContentService batchService;

    @PostMapping("/generate")
    public ResponseEntity<GenerateResponse> generate(
            @RequestBody GenerateRequest request) {
        String content = generator.generate(request);
        return ResponseEntity.ok(new GenerateResponse(content));
    }

    @PostMapping(value = "/generate-stream",
                 produces = MediaType.TEXT_EVENT_STREAM_VALUE)
    public SseEmitter generateStream(@RequestBody GenerateRequest request) {
        SseEmitter emitter = new SseEmitter(60_000L);

        executor.execute(() -> {
            try {
                // 流式生成 Token
                Flux<String> stream = generator.generateStream(request);
                stream.subscribe(
                    token -> emitter.send(SseEmitter.event()
                        .name("token").data(token)),
                    err -> {
                        emitter.send(SseEmitter.event()
                            .name("error").data(err.getMessage()));
                        emitter.complete();
                    },
                    () -> {
                        emitter.send(SseEmitter.event()
                            .name("done").data(""));
                        emitter.complete();
                    }
                );
            } catch (Exception e) {
                emitter.completeWithError(e);
            }
        });
        return emitter;
    }

    @PostMapping("/batch")
    public ResponseEntity<BatchResponse> batchGenerate(
            @RequestBody BatchRequest request) {
        String batchId = batchService.startBatch(
            request.getTemplateId(),
            request.getVariablesList(),
            request.getCallbackUrl()  // 生成完成后的回调
        );
        return ResponseEntity.accepted()
            .body(new BatchResponse(batchId, "IN_PROGRESS"));
    }

    @GetMapping("/batch/{batchId}/progress")
    public ResponseEntity<BatchProgress> getProgress(
            @PathVariable String batchId) {
        return ResponseEntity.ok(progressTracker.getProgress(batchId));
    }

// 批量生成回调处理
@Component
public class BatchCallbackHandler {

    private final RestTemplate restTemplate;

    @Async
    public void notifyCompletion(String batchId, String callbackUrl) {
        BatchResult result = batchService.getResult(batchId);
        CallbackPayload payload = new CallbackPayload(
            batchId, "COMPLETED",
            result.getTotalCount(), result.getSuccessCount(),
            result.getFailedCount(), result.getResults()
        );

        try {
            restTemplate.postForEntity(callbackUrl, payload, String.class);
            log.info("Batch {} callback sent to {}", batchId, callbackUrl);
        } catch (Exception e) {
            log.error("Batch {} callback failed: {}", batchId, e.getMessage());
            // 重试 3 次
            retryCallback(batchId, callbackUrl, 0);
        }
    }

    private void retryCallback(String batchId, String url, int attempt) {
        if (attempt >= 3) return;
        ScheduledExecutorService scheduler = Executors.newScheduledThreadPool(1);
        scheduler.schedule(() -> {
            try {
                notifyCompletion(batchId, url);
            } catch (Exception e) {
                retryCallback(batchId, url, attempt + 1);
            }
        }, 5 * (attempt + 1), TimeUnit.SECONDS);
    }
}
