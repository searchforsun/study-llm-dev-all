自动打分

java
// LLM-as-Judge 自动打分
@Component
public class AutoScorer {

    private final ChatClient judgeClient;

    public ScoreResult score(String content, String scenario,
                              BrandTone expectedTone) {
        String result = judgeClient.prompt()
            .user(u -> u.text("""
                你是一个内容质量评审专家。请从以下 5 个维度评分（1-5分）。

                内容：
                {content}

                场景：{scenario}
                期望语调：{tone}

                评分维度：
                1. 相关性(relevance) - 内容是否符合场景需求
                2. 准确性(accuracy) - 事实是否准确、逻辑是否自洽
                3. 流畅度(fluency) - 语言是否流畅、易于理解
                4. 风格匹配(style) - 是否符合品牌语调要求
                5. 合规性(compliance) - 是否违反安全规范

                输出 JSON：
                {"relevance":4,"accuracy":3,"fluency":5,
                 "style":4,"compliance":5,
                 "total":4.2,
                 "strengths":["语言流畅","品牌语调准确"],
                 "weaknesses":["部分事实不够准确"],
                 "suggestion":"建议核对数据来源"}
                """)
                .param("content", content)
                .param("scenario", scenario)
                .param("tone", expectedTone.toPromptInstruction())
            )
            .call()
            .content();

        return parseResult(result);
    }

    public record ScoreResult(double relevance, double accuracy,
                               double fluency, double style,
                               double compliance, double total,
                               List<String> strengths,
                               List<String> weaknesses,
                               String suggestion) {}
}

人工 Rubric 与坏例库

// 人工 Rubric 评估表
@Component
public class HumanRubricService {

    public RubricForm getRubricForm(String scenario) {
        return new RubricForm(scenario, List.of(
            new RubricItem("内容完整度", "是否覆盖所有必要信息",
                List.of("严重缺失", "缺失较多", "基本完整", "比较完整", "非常完整")),
            new RubricItem("数据准确性", "引用的数据和事实是否准确",
                List.of("多处错误", "少量错误", "基本准确", "比较准确", "完全准确")),
            new RubricItem("结构清晰度", "文章结构是否清晰",
                List.of("混乱", "不够清晰", "基本清晰", "比较清晰", "非常清晰")),
            new RubricItem("品牌一致性", "是否符合品牌语调",
                List.of("完全不符", "部分不符", "基本符合", "比较符合", "完全符合")),
            new RubricItem("行动号召", "CTA 是否清晰有力",
                List.of("无CTA", "CTA不清晰", "CTA一般", "CTA较好", "CTA优秀"))
        ));
    }

    public RubricResult submitEvaluation(String contentId,
                                          Map<String, Integer> scores,
                                          String comment) {
        RubricResult result = new RubricResult();
        result.setContentId(contentId);
        result.setScores(scores);
        result.setTotal(scores.values().stream()
            .mapToInt(Integer::intValue).average().orElse(0));
        result.setComment(comment);
        result.setEvaluatedAt(Instant.now());
        return rubricRepository.save(result);
    }
}

// 坏例库
@Component
public class BadCaseLibrary {

    public void recordBadCase(String prompt, String content,
                               ScoreResult score, String issue) {
        BadCase badCase = new BadCase();
        badCase.setPromptHash(hashPrompt(prompt));
        badCase.setContent(content);
        badCase.setTotalScore(score.total());
        badCase.setWeaknesses(score.weaknesses());
        badCase.setIssue(issue);
        badCase.setRecordedAt(Instant.now());
        badCaseRepository.save(badCase);

        // 自动分析坏例模式
        analyzePattern(badCase);
    }

    public List<BadCasePattern> getTopPatterns() {
        // 返回最常见的失败模式 TOP 5
        return badCaseRepository.findTopPatterns(PageRequest.of(0, 5));
    }
}

质量趋势

CorpAssist 场景

内容质量评估系统上线 5 个月，平均分从 3.7 提升到 4.25。关键动作：1) 第 1 月收集 45 个坏例，发现"数据准确性"是最常见的低分项；2) 第 2 月在 Prompt 中加入"请核实数据的时效性"要求，准确率提升；3) 第 3 月优化 Few-shot 示例后"品牌一致性"评分明显改善；4) 第 4 月坏例数降至 20，主要问题转向"行动号召不够有力"；5) 持续迭代后月度坏例降至 15，平均分稳定在 4.2 以上。

本章结论

  • LLM-as-Judge 5 维度自动打分，总分从 3.7 提升到 4.25
  • 人工 Rubric 5 级评分覆盖 5 个评估项
  • 坏例库记录低分特征，驱动 Prompt 优化
  • 5 个月质量趋势持续上升，坏例数从 45 降至 15
