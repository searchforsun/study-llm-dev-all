为什么学这章

相同的内容给不同用户看，效果完全不同。千人千面个性化让内容根据用户画像（行业/角色/偏好）动态调整，大幅提升内容的相关性和转化率。

学完后能做什么

用 Redis 管理用户画像并存储偏好；实现动态模板注入（根据画像选择不同的 Prompt 变量）；配置 A/B 实验比较不同内容版本的效果；实现效果评估（点击率/转化率/NPS）。

本章要点速记

1) 用户画像 Redis Hash 存储：行业、角色、偏好风格、历史互动；2) 动态模板注入：根据画像选择不同的 Prompt 变量值（如电商用户用"限时优惠"，企业用户用"降本增效"）；3) A/B 实验：50% 流量走版本 A，50% 走版本 B，对比转化率；4) 效果评估：点击率提升 35%、转化率提升 22%、NPS 提升 12 分。

用户画像 Redis

// 用户画像管理
@Component
public class UserProfileService {

    private final RedisTemplate<String, Object> redisTemplate;

    public UserProfile getProfile(String userId) {
        String key = "user:profile:" + userId;
        Map<Object, Object> entries = redisTemplate
            .opsForHash().entries(key);
        if (entries.isEmpty()) {
            return UserProfile.defaultProfile();
        }
        return UserProfile.builder()
            .userId(userId)
            .industry((String) entries.get("industry"))
            .role((String) entries.get("role"))
            .preferredStyle((String) entries.get("preferredStyle"))
            .tonePreference((String) entries.get("tonePreference"))
            .interactionCount(
                Integer.parseInt((String) entries.getOrDefault("interactionCount", "0")))
            .lastCategory((String) entries.get("lastCategory"))
            .build();
    }

    public void updateInteraction(String userId, String category,
                                   boolean clicked) {
        String key = "user:profile:" + userId;
        redisTemplate.opsForHash().increment(key, "interactionCount", 1);
        redisTemplate.opsForHash().put(key, "lastCategory", category);
        if (clicked) {
            redisTemplate.opsForHash().increment(key, "clickCount", 1);
        }
        redisTemplate.expire(key, 30, TimeUnit.DAYS);
    }

    public String getPreferredHook(String userId) {
        UserProfile profile = getProfile(userId);
        // 根据画像选择最合适的"钩子"文案
        return switch (profile.getIndustry()) {
            case "电商" -> "限时优惠，立即抢购";
            case "金融" -> "稳健增值，财富护航";
            case "教育" -> "提升技能，赢在未来";
            case "SaaS" -> "降本增效，驱动增长";
            default -> "了解更多";
        };
    }
}

动态模板注入与 A/B 实验

// 动态模板注入
@Component
public class DynamicTemplateInjector {

    public String injectProfile(String templateId, UserProfile profile) {
        ContentTemplate template = templateRepository.findById(templateId)
            .orElseThrow();

        Map<String, String> variables = new HashMap<>();
        // 根据用户画像填充变量
        variables.put("{{industry}}", profile.getIndustry());
        variables.put("{{role}}", getRoleGreeting(profile.getRole()));
        variables.put("{{hook}}", getHookByProfile(profile));
        variables.put("{{tone}}", profile.getTonePreference());
        variables.put("{{style}}", profile.getPreferredStyle());

        String content = template.getContent();
        for (Map.Entry<String, String> entry : variables.entrySet()) {
            content = content.replace(entry.getKey(), entry.getValue());
        }
        return content;
    }

    private String getRoleGreeting(String role) {
        return switch (role) {
            case "CTO" -> "尊敬的CTO";
            case "PM" -> "尊敬的产品经理";
            case "Developer" -> "亲爱的开发者";
            default -> "尊敬的客户";
        };
    }
}

// A/B 实验管理器
@Component
public class AbExperimentManager {

    private final Map<String, Experiment> experiments = new ConcurrentHashMap<>();

    public String assignVariant(String userId, String experimentId) {
        Experiment exp = experiments.get(experimentId);
        if (exp == null) return "control";

        // 基于用户 ID 哈希分组
        int group = Math.abs(userId.hashCode()) % 100;
        if (group < exp.getControlPercent() * 100) {
            return "control";
        }
        return "treatment";
    }

    public void recordResult(String userId, String experimentId,
                              String variant, String metric, double value) {
        String key = "abtest:" + experimentId + ":" + variant;
        redisTemplate.opsForHash().increment(key, "count", 1);
        redisTemplate.opsForHash().increment(key, metric + "_sum",
            Double.doubleToRawLongBits(value));
    }

    // A/B 测试报告
    // | 版本 | 展示量 | 点击率 | 转化率 | NPS |
    // | A(control)  | 5000 | 2.3% | 0.8% | 42 |
    // | B(treatment)| 5000 | 3.1% | 1.2% | 54 |
    // | 提升幅度    | -    | +35% | +50% | +12 |
}
