用户画像的 Redis 存储结构

用户画像是个性化的数据基础——没有画像就谈不上个性化。CorpAssist 的用户画像系统使用 Redis 5 种数据结构的组合来高效存储用户的不同维度信息。基础属性（用户注册时填写的静态信息）使用 Hash 结构,key="user:{userId}:profile",field=name/age/gender/city/memberLevel/registerTime,value 使用 JSON 字符串或直接字符串。Hash 的优势是可以单独读取或更新某个 field,不需要读取整个对象。标签集合（通过用户行为计算出的动态标签）使用 Set 结构,key="user:{userId}:tags",member 格式为 "{category}:{value}",如 "interest:tech"、"lifestyle:fitness"、"purchase_freq:high"。Set 天然支持集合运算（并集/交集/差集）,方便在模板选择时做用户群体的交集分析。行为序列（用户的购买记录/阅读记录/点击记录）使用 Sorted Set 结构,key="user:{userId}:bought" 或 "user:{userId}:read",score=行为时间戳（double 类型,microsecond 精度）,member=SKU ID 或内容 ID。Zset 支持按时间范围查询（ZRANGEBYSCORE）和最近 N 条查询（ZREVRANGE 0 9）,还可以通过 ZREMRANGEBYSCORE 清理 90 天前的历史数据。

画像数据的来源有四个渠道:用户注册/完善资料时写入的基础属性（低频率,月均 1-2 次更新）；用户浏览/点击/分享内容时计算的交互标签（中等频率,日均 5-20 条事件）；用户在电商渠道的购买/加购/收藏记录（低频率,月均 1-3 条事件）；通过第三方平台（如微信/支付宝）回传的用户画像数据（通过 Webhook 接口接收,不定频率）。数据更新的实时性要求不同——基础属性可以容忍秒级延迟（通过 Redis TTL 5 分钟缓存后写回数据库持久化）,标签和行为序列需要实时更新（5 秒内可见）。写操作使用 Redis Pipeline 批量提交,减少网络往返:一次用户交互事件写入 3-5 个 key（标签更新 + 行为序列追加 + 基础属性计数器递增）,Pipeline 批量执行的耗时约 2-5ms。画像数据的 TTL 策略:基础属性 key="user:{userId}:profile" 永久保留（数据由数据库持久化,Redis 作为缓存）；标签 key 保留 30 天（每 30 天根据最近行为重新计算）；行为序列 key 保留 90 天（使用 ZREMRANGEBYSCORE 定期清理 90 天前数据）。

import org.springframework.data.redis.core.*;
import org.springframework.stereotype.Service;
import java.time.Instant;
import java.util.*;
import java.util.concurrent.TimeUnit;
import java.util.stream.Collectors;

@Service
public class UserProfileService {

    private final RedisTemplate redis;
    private static final long PROFILE_TTL = 300;      // 5 分钟缓存
    private static final long TAG_TTL = 2592000;       // 30 天
    private static final long HISTORY_TTL = 7776000;   // 90 天

    private static final String KEY_PROFILE = "user:%s:profile";
    private static final String KEY_TAGS = "user:%s:tags";
    private static final String KEY_BOUGHT = "user:%s:bought";
    private static final String KEY_READ = "user:%s:read";

    public UserProfileService(RedisTemplate redis) {
        this.redis = redis;
    }

    // 获取用户完整画像 (聚合多个 key)
    public UserProfile getProfile(String userId) {
        BoundHashOperations profileOps =
            redis.boundHashOps(KEY_PROFILE.formatted(userId));

        Map basic = profileOps.entries();
        if (basic.isEmpty()) {
            return null;  // 用户不存在
        }

        Set tags = redis.opsForSet()
            .members(KEY_TAGS.formatted(userId));

        // 最近 30 天购买记录
        Set> recentBought =
            redis.opsForZSet()
                .reverseRangeByScoreWithScores(
                    KEY_BOUGHT.formatted(userId),
                    Instant.now().minusSeconds(30 * 86400).toEpochMilli(),
                    Instant.now().toEpochMilli(),
                    0, 20);

        return new UserProfile(
            userId,
            basic.get("name"),
            Integer.parseInt(basic.getOrDefault("age", "0")),
            basic.getOrDefault("gender", "unknown"),
            basic.getOrDefault("city", ""),
            basic.getOrDefault("memberLevel", "normal"),
            tags,
            recentBought.stream()
                .map(t -> new PurchaseRecord(t.getValue(),
                    t.getScore().longValue()))
                .collect(Collectors.toList())
        );
    }

    // 更新用户标签 (增量)
    public void addTags(String userId, Set newTags) {
        redis.opsForSet().add(KEY_TAGS.formatted(userId),
            newTags.toArray(new String[0]));
        redis.expire(KEY_TAGS.formatted(userId), TAG_TTL, TimeUnit.SECONDS);
    }

    // 移除标签
    public void removeTags(String userId, Set tags) {
        redis.opsForSet().remove(KEY_TAGS.formatted(userId),
            tags.toArray(new String[0]));
    }

    // 记录购买行为
    public void recordPurchase(String userId, String skuId) {
        long now = Instant.now().toEpochMilli();
        redis.opsForZSet().add(KEY_BOUGHT.formatted(userId), skuId, now);
        redis.expire(KEY_BOUGHT.formatted(userId), HISTORY_TTL,
            TimeUnit.SECONDS);

        // 更新购买相关的标签 (自动)
        addTags(userId, Set.of("purchase_freq:active", "last_purchase:recent"));
    }

    // 记录阅读/交互行为
    public void recordInteraction(String userId, String contentId,
                                  String contentType) {
        long now = Instant.now().toEpochMilli();
        redis.opsForZSet().add(KEY_READ.formatted(userId), contentId, now);
        redis.expire(KEY_READ.formatted(userId), HISTORY_TTL,
            TimeUnit.SECONDS);

        // 根据内容类型更新兴趣标签
        addTags(userId, Set.of("interest:" + contentType));
    }
