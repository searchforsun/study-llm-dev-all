为什么学这章

AI 生成的内容不能直接发布——需要安全检测和人工复核。本章实现完整的内容安全检测（Advisor 链）、人工复核（Flowable 工作流）、驳回重写的闭环流程。

学完后能做什么

配置 Spring AI Advisor 链实现内容安全检测；基于 Flowable 实现人工复核工作流；实现驳回重写和重新提交；设计审核状态机。

本章要点速记

1) Advisor 链依次执行：敏感词过滤→有害内容检测→品牌合规检查→版权检查；2) Flowable 工作流管理"提交→审核→通过/驳回→发布"生命周期；3) 驳回重写时保留审核意见和原始版本，Agent 根据意见修改；4) 审核状态机：DRAFT → SUBMITTED → REVIEWING → APPROVED/REJECTED → PUBLISHED。

内容安全检测（Advisor 链）

java
// 内容安全 Advisor 链
@Component
public class ContentSafetyAdvisorChain {

    @Bean
    public ChatClient safeChatClient(ChatClient.Builder builder,
                                      List<Advisor> advisors) {
        return builder
            .defaultAdvisors(advisors)
            .build();
    }

    @Bean
    public List<Advisor> contentSafetyAdvisors() {
        return List.of(
            new SensitiveWordFilter(),      // 1. 敏感词过滤
            new HarmfulContentDetector(),    // 2. 有害内容检测
            new BrandComplianceChecker(),    // 3. 品牌合规检查
            new CopyrightChecker()           // 4. 版权来源检查
        );
    }
}

// 敏感词过滤器
@Component
public class SensitiveWordFilter implements Advisor {

    private final Set<String> sensitiveWords = Set.of(
        "敏感词A", "敏感词B", "敏感词C");

    @Override
    public AdviceResponse before(AdviceRequest request) {
        String content = request.getContent();
        for (String word : sensitiveWords) {
            if (content.contains(word)) {
                return new AdviceResponse(false,
                    "内容包含敏感词: " + word);
            }
        }
        return new AdviceResponse(true);
    }
}

// 有害内容检测器（基于 LLM）
@Component
public class HarmfulContentDetector implements Advisor {

    private final ChatClient detectorClient;

    @Override
    public AdviceResponse before(AdviceRequest request) {
        String result = detectorClient.prompt()
            .user(u -> u.text("""
                检查以下内容是否包含：暴力、色情、歧视、虚假宣传。
                内容：{content}
                输出 JSON：{"safe":true/false,"reason":"","riskLevel":"NONE/LOW/MEDIUM/HIGH"}
                """)
                .param("content", request.getContent())
            )
            .call()
            .content();

        DetectionResult dr = parseResult(result);
        if (!dr.isSafe()) {
            return new AdviceResponse(false,
                "有害内容检测未通过: " + dr.getReason());
        }
        return new AdviceResponse(true);
    }
}

人工复核（Flowable/Activity）

// Flowable 审核工作流
@Component
public class ReviewWorkflowService {

    private final RuntimeService runtimeService;
    private final TaskService taskService;

    public String startReview(String contentId, String reviewerId) {
        Map<String, Object> variables = new HashMap<>();
        variables.put("contentId", contentId);
        variables.put("reviewerId", reviewerId);
        variables.put("status", "SUBMITTED");
        variables.put("reviewCount", 0);

        ProcessInstance instance = runtimeService
            .startProcessInstanceByKey("contentReview", variables);
        return instance.getId();
    }

    public void approve(String taskId, String comment) {
        Map<String, Object> variables = new HashMap<>();
        variables.put("status", "APPROVED");
        variables.put("comment", comment);
        taskService.complete(taskId, variables);
    }

    public void reject(String taskId, String comment) {
        Map<String, Object> variables = new HashMap<>();
        variables.put("status", "REJECTED");
        variables.put("comment", comment);
        taskService.complete(taskId, variables);
    }
}

// 审核状态机
// DRAFT --> SUBMITTED --> REVIEWING --> APPROVED --> PUBLISHED
//                          REVIEWING --> REJECTED --> DRAFT (修改后重新提交)
// REJECTED --> SUBMITTED (修改后重新提交)

// Flowable BPMN 定义 (content-review.bpmn20.xml)

// 驳回重写服务
@Component
public class RewriteService {

    private final ChatClient rewriteClient;

    public String rewrite(String original, String reviewComment) {
        return rewriteClient.prompt()
            .user(u -> u.text("""
                请根据审核意见修改以下内容。

                原始内容：
                {original}

                审核意见：
                {comment}

                要求：
                - 保留原始内容的核心信息
                - 针对审核意见逐条修改
                - 保持品牌语调一致
                - 输出修改后的完整内容
                """)
                .param("original", original)
                .param("comment", reviewComment)
            )
            .call()
            .content();
    }

    // 版本管理：保留每次修改的版本
    public ContentVersion saveVersion(String contentId, String content,
                                       String reviewComment) {
        ContentVersion version = new ContentVersion();
        version.setContentId(contentId);
        version.setContent(content);
        version.setReviewComment(reviewComment);
        version.setVersionNumber(getNextVersion(contentId));
        version.setCreatedAt(Instant.now());
        return versionRepository.save(version);
    }
}
