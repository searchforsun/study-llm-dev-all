数据集加载与字段映射

Hugging Face Datasets 库原生支持 JSONL 加载。关键步骤是字段映射：将 JSONL 中的 instruction/input/output 转换成模型 chat template 格式。

python 
# load_dataset.py — 加载并映射 JSONL 到 chat format
from datasets import load_dataset
from transformers import AutoTokenizer

model_name = "Qwen/Qwen2.5-7B-Instruct"
tokenizer = AutoTokenizer.from_pretrained(model_name)

def format_corpassist(example):
    """将 JSONL 三字段映射为 chat template 格式"""
    instruction = example.get("instruction", "")
    input_text = example.get("input", "")
    output = example.get("output", "")

    # 构建 messages 格式
    user_content = f"{instruction}: {input_text}" if input_text else instruction
    messages = [
        {"role": "user", "content": user_content},
        {"role": "assistant", "content": output},
    ]
    # 用 tokenizer.apply_chat_template 格式化
    text = tokenizer.apply_chat_template(
        messages, tokenize=False, add_generation_prompt=False,
    )
    return {"text": text}

# 加载 JSONL
dataset = load_dataset("json", data_files="data/corpassist_v2_scrubbed.jsonl")
# 映射格式
dataset = dataset.map(format_corpassist)
# 分割 train/val
dataset = dataset["train"].train_test_split(test_size=0.1, seed=42)
print(f"Train: {len(dataset['train'])}, Val: {len(dataset['test'])}")
print(f"Example:\n{dataset['train'][0]['text'][:200]}...")

Chat Template 说明

不同模型的 chat template 不同。Qwen2.5 的 template 会添加 <|im_start|>user\n...<|im_end|>\n<|im_start|>assistant\n...<|im_end|> 等特殊 token。必须使用 apply_chat_template 方法而不是手动拼接，否则 loss 计算会包含特殊 token 导致训练异常。

完整训练脚本

训练脚本 是流水线的核心。使用 SFTTrainer 可以简化大量样板代码。

python 
# train_lora.py — CorpAssist LoRA 训练全流程
import os, torch, json
from datasets import load_dataset
from transformers import (
    AutoModelForCausalLM, AutoTokenizer,
    TrainingArguments, BitsAndBytesConfig,
)
from peft import LoraConfig, get_peft_model
from trl import SFTTrainer

# ---- 配置 ----
MODEL_NAME = "Qwen/Qwen2.5-7B-Instruct"
DATA_PATH = "data/corpassist_v2_scrubbed.jsonl"
OUTPUT_DIR = "./outputs/corpassist-lora-v1"

# ---- 1. 加载 Tokenizer ----
tokenizer = AutoTokenizer.from_pretrained(MODEL_NAME)
tokenizer.padding_side = "right"

# ---- 2. 加载 Base Model（4-bit QLoRA） ----
bnb_config = BitsAndBytesConfig(
    load_in_4bit=True,
    bnb_4bit_compute_dtype=torch.bfloat16,
    bnb_4bit_quant_type="nf4",
)
model = AutoModelForCausalLM.from_pretrained(
    MODEL_NAME,
    quantization_config=bnb_config,
    device_map="auto",
    torch_dtype=torch.bfloat16,
)

# ---- 3. LoRA 配置 ----
lora_config = LoraConfig(
    r=16,
    lora_alpha=32,
    target_modules=["q_proj", "k_proj", "v_proj", "o_proj"],
    lora_dropout=0.05,
    bias="none",
    task_type="CAUSAL_LM",
)
model = get_peft_model(model, lora_config)
model.print_trainable_parameters()  # 应显示 ~0.2% 参数可训练

# ---- 4. 加载数据集 ----
def format_fn(examples):
    texts = []
    for inst, inp, out in zip(
        examples["instruction"], examples["input"], examples["output"]
    ):
        user_content = f"{inst}: {inp}" if inp else inst
        messages = [
            {"role": "user", "content": user_content},
            {"role": "assistant", "content": out},
        ]
        texts.append(tokenizer.apply_chat_template(messages, tokenize=False))
    return {"text": texts}

dataset = load_dataset("json", data_files=DATA_PATH)
dataset = dataset.map(format_fn, batched=True)
dataset = dataset["train"].train_test_split(test_size=0.1, seed=42)

# ---- 5. Training Arguments ----
training_args = TrainingArguments(
    output_dir=OUTPUT_DIR,
    per_device_train_batch_size=4,
    gradient_accumulation_steps=4,
    learning_rate=2e-4,
    warmup_steps=20,
    num_train_epochs=3,
    logging_steps=10,
    save_steps=100,           # 每 100 步保存 checkpoint
    save_total_limit=2,       # 最多保留 2 个 checkpoint
    evaluation_strategy="steps",
    eval_steps=100,
    bf16=True,
    report_to="wandb",        # 记录到 W&B
    run_name="corpassist-lora-v1",
    remove_unused_columns=False,
)

# ---- 6. SFTTrainer ----
trainer = SFTTrainer(
    model=model,
    train_dataset=dataset["train"],
    eval_dataset=dataset["test"],
    tokenizer=tokenizer,
    args=training_args,
    dataset_text_field="text",
    max_seq_length=2048,
)

# ---- 7. 训练 ----
trainer.train()
trainer.save_model(OUTPUT_DIR)
tokenizer.save_pretrained(OUTPUT_DIR)
print(f"模型保存至: {OUTPUT_DIR}")

实验追踪：W&B 集成

W&B 实验追踪 是生产环境必不可少的环节。它记录每次训练的超参、loss 曲线、学习率和显存占用，便于对比不同实验。

不推荐
  • 训练脚本里 print loss 就完事
  • 超参先写死在代码里，不用 config 文件
  • 不看 loss 曲线，只看最终 loss 判断模型好坏
  • 多个实验没有统一追踪，靠文件名区分
推荐
  • W&B 或 MLflow 记录每次实验
  • 超参写在 YAML config 中，用 argparse 加载
  • 观察 loss 曲线的收敛趋势和过拟合拐点
  • 实验标签化（语料版本 + 超参缩写）
python 
# wandb_config.py — W&B 初始化与超参记录
import wandb
import yaml

def init_wandb(config_path: str):
    with open(config_path) as f:
        config = yaml.safe_load(f)

    run = wandb.init(
        project="corpassist-ft",
        name=config["run_name"],
        config=config,
        tags=["lora", "qwen2.5-7b", "v2-corpus"],
    )
    return run

# 使用示例
# yaml config:
# run_name: corpassist-lora-v2
# model: Qwen/Qwen2.5-7B-Instruct
# lora_r: 16
# lora_alpha: 32
# learning_rate: 2e-4
# batch_size: 16
# epochs: 3

Checkpoint 保存与恢复

Checkpoint 保存 是训练可靠性的重要保障。SFTTrainer 支持自动 checkpoint，训练中断后可从最新 checkpoint 恢复。

bash 
# 从 checkpoint 恢复训练
python train_lora.py --resume_from_checkpoint ./outputs/corpassist-lora-v1/checkpoint-500

# 查看训练曲线（CLI）
wandb online
# 然后打开 W&B 网页查看
python 
# checkpoint_callback.py — 自定义保存策略
from transformers import TrainerCallback

class CustomSaveCallback(TrainerCallback):
    """在 eval loss 新低时额外保存"""
    def __init__(self, output_dir):
        self.best_loss = float("inf")
        self.output_dir = output_dir

    def on_evaluate(self, args, state, control, metrics, **kwargs):
        eval_loss = metrics.get("eval_loss")
        if eval_loss and eval_loss < self.best_loss:
            self.best_loss = eval_loss
            print(f"New best eval loss: {eval_loss:.4f}, saving...")
            # 保存最佳 checkpoint
            kwargs["model"].save_pretrained(
                f"{self.output_dir}/best-checkpoint"
            )

产物管理：Adapter 保存与上传

Adapter artifact 是训练的最终产物。一个 LoRA adapter 包含：

  • adapter_config.json — LoRA 超参（r, alpha, target_modules）
  • adapter_model.safetensors — 适配器权重（~16MB for 7B r=16）
  • tokenizer_config.json 和 tokenizer.json

生产环境中，adapter 应该上传到模型注册中心而非留在本地。

python 
# upload_adapter.py — 上传 adapter 到 Hugging Face Hub
from huggingface_hub import HfApi

api = HfApi()
adapter_path = "./outputs/corpassist-lora-v1"
repo_id = "corp-ai/corpassist-tone-v1-lora"

api.create_repo(repo_id, private=True, exist_ok=True)
api.upload_folder(
    folder_path=adapter_path,
    repo_id=repo_id,
    repo_type="model",
    commit_message="CorpAssist 话术 LoRA v1 - 基于 Qwen2.5-7B",
)
print(f"已上传至: https://huggingface.co/{repo_id}")

本章结论

  • 流水线结构：加载 → 映射 → Tokenize → 训练 → 保存 → 上传。
  • 字段映射：必须用 apply_chat_template 而非手动拼接。
  • SFTTrainer：简化 SFT 训练，内置 LoRA 支持和 W&B 集成。
  • 实验追踪：W&B 记录超参和训练曲线，便于对比。
  • Checkpoint：自动保存 + 断点恢复 + 最佳模型保留。
