"""RAGAS 评测脚本"""

from ragas import evaluate
from ragas.metrics import faithfulness, answer_relevancy, context_precision
from datasets import Dataset
import pandas as pd

def load_test_set(path: str = "data/rag_test_set.json") -> list[dict]:
    """加载测试集（含 question/answer/contexts/ground_truth）"""
    import json
    with open(path) as f:
        return json.load(f)

def run_ragas_eval(
    questions: list[str],
    answers: list[str],
    contexts: list[list[str]],
    ground_truth: list[str],
) -> dict:
    """运行 RAGAS 评测并返回指标"""
    data = {
        "question": questions,
        "answer": answers,
        "contexts": contexts,
        "ground_truth": ground_truth,
    }
    dataset = Dataset.from_dict(data)
    result = evaluate(
        dataset,
        metrics=[faithfulness, answer_relevancy, context_precision],
    )
    return {
        "faithfulness": result["faithfulness"],
        "answer_relevancy": result["answer_relevancy"],
        "context_precision": result["context_precision"],
    }

def test_rag_pipeline_quality():
    """pytest 集成：RAG 管线质量评测"""
    test_set = load_test_set()
    # 使用 pytest fixtures 模拟 RAG 管线运行
    questions = [t["question"] for t in test_set]
    answers = [t.get("answer", "") for t in test_set]
    contexts = [t.get("retrieved_contexts", []) for t in test_set]
    ground_truth = [t["ground_truth"] for t in test_set]

    metrics = run_ragas_eval(questions, answers, contexts, ground_truth)

    # 断言指标达到阈值
    assert metrics["faithfulness"] >= 0.85, (
        f"Faithfulness {metrics['faithfulness']:.3f} < 0.85"
    )
    assert metrics["answer_relevancy"] >= 0.80, (
        f"Answer Relevancy {metrics['answer_relevancy']:.3f} < 0.80"
    )
    assert metrics["context_precision"] >= 0.75, (
        f"Context Precision {metrics['context_precision']:.3f} < 0.75"
    )

召回率 / MRR / NDCG 基准

除 RAGAS 框架外，检索阶段还需要单独测量检索指标。以下是 CorpAssist 的标准评测基准。

python
"""检索指标基准评测"""

from typing import List, Set
import numpy as np

class RetrievalBenchmark:
    """检索指标基准"""

    def __init__(self, test_queries: List[str], qrels: dict):
        self.queries = test_queries
        self.qrels = qrels  # {qid: [relevant_doc_ids]}

    def recall_at_k(
        self, relevant: Set[str], results: List[str], k: int
    ) -> float:
        if not relevant:
            return 1.0
        return sum(1 for d in results[:k] if d in relevant) / len(relevant)

    def mrr_at_k(
        self, relevant: Set[str], results: List[str], k: int
    ) -> float:
        for i, d in enumerate(results[:k]):
            if d in relevant:
                return 1.0 / (i + 1)
        return 0.0

    def ndcg_at_k(
        self, relevant: Set[str], results: List[str], k: int
    ) -> float:
        dcg = 0.0
        for i, d in enumerate(results[:k]):
            rel = 1.0 if d in relevant else 0.0
            dcg += rel / np.log2(i + 2)
        idcg = sum(1.0 / np.log2(i + 2) for i in range(min(len(relevant), k)))
        return dcg / idcg if idcg > 0 else 0.0

    def evaluate(self, retriever, k: int = 10) -> dict:
        """在测试集上评测检索器"""
        recall_scores, mrr_scores, ndcg_scores = [], [], []
        for qid, query in enumerate(self.queries):
            results = retriever.search(query, top_k=k)
            result_ids = [r["doc_id"] for r in results]
            relevant = set(self.qrels.get(str(qid), []))
            recall_scores.append(self.recall_at_k(relevant, result_ids, k))
            mrr_scores.append(self.mrr_at_k(relevant, result_ids, k))
            ndcg_scores.append(self.ndcg_at_k(relevant, result_ids, k))

        return {
            f"recall@{k}": round(np.mean(recall_scores), 4),
            f"mrr@{k}": round(np.mean(mrr_scores), 4),
            f"ndcg@{k}": round(np.mean(ndcg_scores), 4),
        }

if __name__ == "__main__":
    # 模拟基准测试
    qrels = {
        "0": ["doc_a1", "doc_a2"],
        "1": ["doc_b1"],
        "2": ["doc_c1", "doc_c2", "doc_c3"],
    }
    benchmark = RetrievalBenchmark(
        ["年假几天", "报销流程", "绩效考核"],
        qrels,
    )
    # 假设有一个检索器
    print("基准评测配置: k=10")
    print("生产标准: recall≥0.85, mrr≥0.90, ndcg≥0.85")

坏例归因四步法

当 RAG 回答出错时，需要在检索→分块→Prompt→LLM 四个环节中定位根因。CorpAssist 的坏例归因流程：

    A[坏例] --> B[Step 1: 检索检查]
    B --> C{检索到的文档
是否包含答案?}
    C -->|否| D[检索问题: 查 Embedding/BM25]
    C -->|是| E[Step 2: 分块检查]
    E --> F{chunk 是否完整包含
答案所需信息?}
    F -->|否| G[分块问题: 调参数/元数据]
    F -->|是| H[Step 3: Prompt 检查]
    H --> I{Prompt 是否明确约束
模型行为?}
    I -->|否| J[Prompt 问题: 优化指令]
    I -->|是| K[Step 4: LLM 检查]
    K --> L[LLM 问题: 换模型/调温度/加few-shot]

"""坏例归因四步法实现"""

from dataclasses import dataclass
from typing import Optional

@dataclass
class BadCase:
    question: str
    answer: str
    retrieved_docs: list[dict]
    ground_truth: str
    root_cause: Optional[str] = None
    fix: Optional[str] = None

class BadCaseAnalyzer:
    """坏例归因分析器"""

    def __init__(self, retriever, chunker, llm):
        self.retriever = retriever
        self.chunker = chunker
        self.llm = llm

    def analyze(self, case: BadCase) -> BadCase:
        """四步法分析坏例根因"""

        # Step 1: 检索检查
        docs = self.retriever.search(case.question, top_k=10)
        docs_contain_answer = any(
            case.ground_truth in d.get("text", "")
            for d in docs
        )
        if not docs_contain_answer:
            case.root_cause = "检索失败"
            case.fix = "检查 Embedding 模型/BM25 参数/查询重写"
            return case

        # Step 2: 分块检查
        relevant_chunks = [
            d for d in docs if case.ground_truth in d.get("text", "")
        ]
        chunk_complete = any(
            len(d.get("text", "")) > len(case.ground_truth) * 2
            for d in relevant_chunks
        )
        if not chunk_complete:
            case.root_cause = "分块截断"
            case.fix = "增大 chunk_size / overlap 或语义分块"
            return case

        # Step 3: Prompt 检查
        prompt_ok = self._check_prompt(case.question)
        if not prompt_ok:
            case.root_cause = "Prompt 约束不足"
            case.fix = "在 System Prompt 中加入强制引用和拒答指令"
            return case

        # Step 4: LLM 检查
        case.root_cause = "LLM 生成问题"
        case.fix = "降低 temperature(0.1→0.01) / 换更强模型 / 加 few-shot"
        return case

    def _check_prompt(self, question: str) -> bool:
        """检查 Prompt 是否有明确的约束"""
        # 模拟检查
        return True

if __name__ == "__main__":
    analyzer = BadCaseAnalyze(None, None, None)
    case = BadCase(
        question="年假几天",
        answer="年假10天",
        retrieved_docs=[{"text": "员工每年享有 15 天带薪年假", "doc_id": "d1"}],
        ground_truth="15天",
    )
    result = analyzer.analyze(case)
    print(f"Root cause: {result.root_cause}, Fix: {result.fix}")
    # 输出: Root cause: 检索失败, Fix: 检查 Embedding 模型/BM25 参数/查询重写

数据分布：在 CorpAssist 分析的 200 个坏例中，根因分布为：检索失败 38%、分块截断 22%、Prompt 约束不足 17%、LLM 幻觉 23%。值得注意的是，超过一半的坏例（检索 + 分块 = 60%）根源在「信息获取」而非「信息生成」。这提示团队应优先优化检索管线而非调 Prompt。

CorpAssist 场景

员工问「加班费怎么算」。系统回答「加班按 1.5 倍工资计算」。但正确答案是「工作日 1.5 倍，休息日 2 倍，节假日 3 倍」。四步法归因：Step 1 检索——检索到的文档 chunk 包含「加班费按以下标准计算：工作日 150%，休息日 200%，节假日 300%」。Step 2 分块——但 chunk_size=256 只截取到「工作日 150%」，后续内容在下一个 chunk 未被送入 LLM。根因：分块截断。修复：chunk_size 增至 384 或 overlap 增至 80。

from typing import List, Dict

class IncrementalIndexer:
    """增量索引器"""

    def __init__(self, milvus_client, es_client):
        self.milvus = milvus_client
        self.es = es_client
        self.batch_size = 50

    def check_changes(self) -> List[Dict]:
        """检查自上次索引后的文档变更"""
        # 从 PostgreSQL 读取变更日志
        # changes = db.query("SELECT * FROM doc_changes WHERE synced=0 LIMIT %s", self.batch_size)
        return []  # 模拟

    def index_changes(self, changes: List[Dict]) -> int:
        """增量索引变更文档"""
        indexed = 0
        for change in changes:
            doc_id = change["doc_id"]
            action = change["action"]  # add / update / delete

            if action == "add" or action == "update":
                # 调用索引任务（复用全量索引逻辑）
                from src.index import index_document
                index_document(doc_id)
                indexed += 1

            elif action == "delete":
                # 从 Milvus 和 ES 中删除
                self.milvus.delete(collection="corp_chunks", filter=f"doc_id == '{doc_id}'")
                self.es.delete_by_query(
                    index="corp_index",
                    body={"query": {"term": {"doc_id": doc_id}}},
                )
                indexed += 1

            # 标记变更已处理
            # db.execute("UPDATE doc_changes SET synced=1 WHERE doc_id=%s", doc_id)

        return indexed

    def swap_collections(self, new_milvus: str, new_es: str):
        """蓝绿切换——将流量路由到新索引"""
        # 1. 预热新索引（确保数据就绪）
        # 2. 修改检索服务的配置指向新 collection/index
        # 3. 验证新索引的检索指标
        # 4. 切流
        print(f"Switching to Milvus: {new_milvus}, ES: {new_es}")

面试可能问

Q1: 你用什么指标评估 RAG 系统的效果？

分两层。检索层：Recall@10（目标 ≥85%）、MRR@10（≥0.90）、NDCG@10（≥0.85）。回答层：RAGAS 的 faithfulness（≥0.85）、answer_relevancy（≥0.80）、context_precision（≥0.75）。外加业务指标如用户满意度评分、答案采纳率。在 CorpAssist 的实践中，当检索 NDCG@10 达到 0.85 时，答案采纳率约 83%，而当 NDCG 降至 0.72 时，采纳率降至 61%。

Q2: 发现一个坏例，怎么快速定位是检索问题还是生成问题？

用「人间检验法」：把检索到的 Top-3 文档直接给一个人类同事看，问他能不能回答用户的问题。如果能，说明信息在手，根因在生成侧（Prompt/LLM）；如果不能，说明信息不在，根因在检索侧（检索精度或分块策略）。这个检验方法 30 秒就能执行，不需要任何系统日志分析。在我们团队，每个坏例先用「人间检验」再走正式归因流程。

Q3: 增量索引和全量重建怎么配合？

增量索引处理日常的文档增/删/改，每 30 分钟执行一次，每次约 30 秒。全量重建每周日凌晨 3 点执行一次，重建完整索引。重建期间流量路由到旧索引（蓝绿部署），新索引构建完成后用一小时验证检索指标（NDCG@10 降幅 < 0.02），再切换流量。如果全量重建失败，自动回滚到旧索引。这样增量索引保证实时性，全量重建保证一致性。

判断练习

如果 RAGAS 的 faithfulness 是 0.72 但 answer_relevancy 是 0.90，说明什么问题？

说明 LLM 的回答确实切题（相关性高），但没有基于检索到的文档内容（忠实度低）。这通常是 Prompt 缺少强制约束——模型用了自己的知识而非检索文档。解决方案：在 System Prompt 中加入「必须基于检索到的文档内容回答，不能使用自己的知识」指令；降低 temperature；引入 Citation Guard 强制引用。

蓝绿部署时，如何验证新索引的检索质量？

三个步骤。1) 自动化：在测试集上运行 retrieval benchmark，比较新旧索引的 Recall/MRR/NDCG，差异 < 0.02 才通过。2) 抽样：从新索引中随机抽取 50 条 query，人工检查检索结果质量。3) 金丝雀：将 5% 的流量切到新索引，监控 24 小时的用户满意度、拒答率、幻觉率。全部通过后再全量切流。我们遇到过新索引在某类文档上 Recall 降了 0.05 但整体指标没变的情况，金丝雀阶段发现了这个回归。
