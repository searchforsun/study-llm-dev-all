行级权限的检索层过滤

行级权限指不同的用户角色可以看到不同的文档集合。CorpAssist 不在应用层做权限校验，而是在检索层直接过滤——「检索不到 = 不可见」。

"""行级权限的检索层过滤"""

from typing import List, Optional
from pydantic import BaseModel

class UserContext(BaseModel):
    """用户上下文（从 JWT 解析）"""
    user_id: str
    roles: List[str]  # ["全员可见", "HR_可见", "经理_可见"]

class RowLevelACL:
    """行级权限过滤器"""

    def __init__(self, user: UserContext):
        self.user = user
        self.is_admin = "admin" in user.roles

    def milvus_expr(self) -> Optional[str]:
        """生成 Milvus 表达式"""
        if self.is_admin:
            return None  # 管理员免过滤
        conditions = [
            f"acl_tags contains '{role}'"
            for role in self.user.roles
        ]
        return " or ".join(conditions) if conditions else "acl_tags contains 'public'"

    def es_filter(self) -> dict:
        """生成 ES 过滤 DSL"""
        if self.is_admin:
            return {"match_all": {}}
        return {
            "bool": {
                "should": [
                    {"terms": {"acl_tags": self.user.roles}}
                ],
                "minimum_should_match": 1,
            }
        }

    def apply_to_search(self, query: str, top_k: int = 10) -> List[dict]:
        """执行带权限过滤的检索"""
        from src.retrieval import DualHybridRetriever
        retriever = DualHybridRetriever()

        # 混合检索内置了 ES + Milvus 双路
        results = retriever.search(
            query,
            top_k=top_k,
            es_filter=self.es_filter(),
            milvus_filter=self.milvus_expr(),
        )
        return results

if __name__ == "__main__":
    user = UserContext(user_id="u001", roles=["HR_可见"])
    acl = RowLevelACL(user)
    print("Milvus expr:", acl.milvus_expr())
    # 输出: acl_tags contains 'HR_可见'
    print("ES filter:", acl.es_filter())
    # 输出: {"bool": {"should": [{"terms": {"acl_tags": ["HR_可见"]}}], "minimum_should_match": 1}}

from datetime import datetime
from typing import List, Optional
from pydantic import BaseModel

class AuditRecord(BaseModel):
    """审计日志记录"""
    id: str
    user_id: str
    user_roles: List[str]
    query: str
    query_timestamp: datetime
    retrieved_docs: List[dict]  # [{doc_id, score, source}]
    answer_text: str
    answer_confidence: float
    action: str  # "direct_reply" / "human_review" / "refused"
    token_usage: dict  # {input_tokens, output_tokens, model}
    latency_ms: int
    citation_check_passed: bool
    version: str  # RAG 管线版本号

class AuditLogger:
    """审计日志记录器"""

    def __init__(self, db_pool):
        self.db = db_pool

    async def log_query_start(self, user_id: str, query: str) -> str:
        """记录查询开始"""
        record_id = f"audit-{datetime.utcnow().strftime('%Y%m%d%H%M%S-%f')}"
        # 插入一条初始记录
        await self.db.execute("""
            INSERT INTO audit_logs (id, user_id, query, created_at, status)
            VALUES ($1, $2, $3, $4, 'in_progress')
        """, record_id, user_id, query, datetime.utcnow())
        return record_id

    async def log_query_complete(self, record_id: str, record: AuditRecord):
        """更新查询完成记录"""
        await self.db.execute("""
            UPDATE audit_logs SET
                status = 'completed',
                answer_text = $2,
                answer_confidence = $3,
                action = $4,
                token_usage = $5,
                latency_ms = $6,
                citation_check_passed = $7,
                retrieved_docs = $8
            WHERE id = $1
        """, record_id, record.answer_text, record.answer_confidence,
            record.action, record.token_usage, record.latency_ms,
            record.citation_check_passed,
            [json.dumps(d) for d in record.retrieved_docs])

    async def query_audit(
        self, user_id: Optional[str] = None,
        start_time: Optional[datetime] = None,
        end_time: Optional[datetime] = None,
        limit: int = 100,
    ) -> List[AuditRecord]:
        """查询审计日志（管理端使用）"""
        conditions = []
        params = []
        if user_id:
            conditions.append(f"user_id = ${len(params)+1}")
            params.append(user_id)
        where = " AND ".join(conditions) if conditions else "TRUE"
        rows = await self.db.fetch(
            f"SELECT * FROM audit_logs WHERE {where} ORDER BY created_at DESC LIMIT ${len(params)+1}",
            *params, limit,
        )
        return [AuditRecord(**row) for row in rows]

多租户索引隔离

CorpAssist 需要服务多个企业客户（租户），每个租户的知识库相互隔离。隔离方案有三种级别，各有取舍。

隔离级别实现方式安全性运维成本适用场景
索引级隔离每个租户独立 Milvus collection + ES index★★★★★高（需管理 N 套索引）金融/医疗等高合规要求
集合级隔离同一个 collection/index，用 tenant_id 字段过滤★★★★中中型企业（推荐）
文档级隔离同一个索引，每个文档带 tenant_id tag 过滤★★★低低合规但多租户场景

from typing import List, Optional

class MultiTenantIndex:
    """多租户索引管理器 - 集合级隔离"""

    def __init__(self, milvus_client, es_client):
        self.milvus = milvus_client
        self.es = es_client

    def search(
        self, tenant_id: str, query: str,
        user_roles: List[str], top_k: int = 10,
    ) -> List[dict]:
        """带租户隔离的检索"""
        acl_filter = self._build_acl_expr(user_roles)
        tenant_filter = f"tenant_id == '{tenant_id}'"
        combined = f"{acl_filter} and {tenant_filter}" if acl_filter else tenant_filter

        # Milvus 检索
        from src.embeddings import get_embedding_model
        embedder = get_embedding_model()
        query_vec = embedder.embed_query(query)
        mv_results = self.milvus.search(
            collection_name="corp_chunks",
            data=[query_vec],
            filter=combined,
            limit=top_k,
            output_fields=["doc_id", "text", "tenant_id", "acl_tags"],
        )
        return [{"doc_id": r["id"], "text": r["entity"].get("text", "")}
                for r in mv_results[0]]

面试可能问

Q1: 多个企业客户共用一套知识库系统时，怎么防止数据泄露？

采用集合级多租户隔离。所有文档在索引时携带 tenant_id 字段，检索时强制附加 tenant_id 过滤。同时文档级 ACL Tag 提供第二层权限控制。架构上保障：1) 检索 API 的 tenant_id 从 JWT Token 中提取而非客户端参数传递；2) 即使恶意构造 API 请求也无法跨租户检索；3) 审计日志记录每次检索的 tenant_id，方便事后审计。金融等高合规客户可以升级为索引级隔离（独立 Milvus collection）。

Q2: 审计日志存这么多数据有什么用？举一个实际案例。

有一次用户反馈「我搜'工资'出来的结果不对」。通过审计日志我们还原了他的完整查询链路：query='工资' → 检索到薪资制度.pdf → Rerank 后 Top-1 是绩效制度.pdf（因为 BGE-Reranker 对「工资」和「绩效」的语义关联度打高分 0.81）→ LLM 基于绩效制度生成了「工资根据绩效评级确定」，但用户想问的是「基本工资标准」。根因是 Rerank 对「工资-绩效」的关联度误判。修复：在 Rerank 阶段加入业务规则约束——如果 query 含「工资」，降低绩效相关文档的 Rerank 权重。

Q3: 等保三级测评中最难通过的项是哪个？

根据 CorpAssist 的测评经验，最难的是「数据脱敏」。RAG 系统的审计日志天然包含用户的完整提问和模型回答，用户的 query 中可能包含姓名、手机号、身份证号等敏感信息。简单的尾部打码（138****1234）模式不够——有些 query 是「帮我查一下张三的银行卡号是多少」，脱敏需要在原文中定位并替换。我们上线了基于正则的敏感信息扫描器，在审计写入前扫描并替换所有已知模式的敏感信息，准确率约 97%。

判断练习

如果租户 A 的文档被租户 B 的用户检索到了，最可能的原因是什么？

最可能的原因是检索时没有强制带 tenant_id 过滤。在我们的架构中，如果 API 从 JWT 提取 tenant_id 的逻辑有 bug（如默认值设为空字符串），导致 filter 条件无效，搜到的就是所有租户的文档。第二个可能的原因是 Milvus 和 ES 的某一边缺少 tenant_id 过滤。双重过滤的设计就是为了防止这种情况——即使 ES 侧漏了，Milvus 侧也应该拦住。

审计日志的保留期限应该设多久？超过保留期的数据应该删除还是归档？

等保三级要求 ≥180 天，CorpAssist 设置为 365 天。超过保留期的数据不直接删除——先压缩归档到廉价存储（OSS/S3），保留 3 年后自动删除。如果出现法律调查需求，可以从归档中恢复指定时间范围内的审计日志。审计日志的删除策略需要走审批流程，不能自动化执行定时删除。

A/B 测试方法论与统计显著性

在 RAG 效果的优化过程中，如何确认「这次改动真的有效」而不是随机波动？CorpAssist 团队使用严格的 A/B 测试框架来判断每次变更的效果。

python
"""A/B 测试框架"""

from scipy import stats
import numpy as np
from typing import List, Tuple

class RAGABTest:
    """RAG 效果 A/B 测试"""

    def __init__(self, min_sample: int = 500):
        self.min_sample = min_sample

    def run_test(
        self, control_scores: List[float], treatment_scores: List[float],
        metric_name: str = "ndcg",
    ) -> dict:
        """运行 A/B 测试"""
        if len(control_scores) < self.min_sample or len(treatment_scores) < self.min_sample:
            return {"error": f"样本量不足: control={len(control_scores)}, treatment={len(treatment_scores)}, 需要≥{self.min_sample}"}

        t_stat, p_value = stats.ttest_ind(control_scores, treatment_scores)
        control_mean = np.mean(control_scores)
        treatment_mean = np.mean(treatment_scores)
        effect_size = treatment_mean - control_mean
        lift_pct = (effect_size / control_mean) * 100

        control_std = np.std(control_scores, ddof=1)
        treatment_std = np.std(treatment_scores, ddof=1)

        return {
            "metric": metric_name,
            "control": {"mean": round(control_mean, 4), "std": round(control_std, 4), "n": len(control_scores)},
            "treatment": {"mean": round(treatment_mean, 4), "std": round(treatment_std, 4), "n": len(treatment_scores)},
            "effect_size": round(effect_size, 4),
            "lift_pct": round(lift_pct, 2),
            "p_value": round(p_value, 4),
            "significant": p_value < 0.05,
        }

if __name__ == "__main__":
    test = RAGABTest(min_sample=200)
    # 模拟基线 Rerank vs 新 Rerank 模型
    control = [0.78 + np.random.normal(0, 0.05) for _ in range(500)]
    treatment = [0.81 + np.random.normal(0, 0.05) for _ in range(500)]
    result = test.run_test(control, treatment, "ndcg")
    print(f"{result['metric']}: p={result['p_value']}, significant={result['significant']}")
    print(f"  Control: {result['control']['mean']:.4f} ± {result['control']['std']:.4f}")
    print(f"  Treatment: {result['treatment']['mean']:.4f} ± {result['treatment']['std']:.4f}")
    print(f"  Lift: {result['lift_pct']:.2f}%")

"""Token 成本归因分析"""

from dataclasses import dataclass

@dataclass
class CostBreakdown:
    embedding: float   # 每次检索的 Embedding 成本
    retrieval: float   # 检索（ES+Milvus 固定成本）
    rerank: float      # Rerank 调用成本
    llm_input: float   # 输入 Token 成本
    llm_output: float  # 输出 Token 成本

    @property
    def total(self) -> float:
        return self.embedding + self.retrieval + self.rerank + self.llm_input + self.llm_output

class CostAnalyzer:
    """成本分析器"""

    PRICES = {
        "gpt-4o-mini": {"input": 0.15 / 1e6, "output": 0.60 / 1e6},  # $/token
        "bge-m3-embedding": 0.02 / 1e3,  # $/query
        "bge-reranker": 0.05 / 1e3,       # $/query
    }

    def analyze_query(self, input_tokens: int, output_tokens: int) -> CostBreakdown:
        return CostBreakdown(
            embedding=self.PRICES["bge-m3-embedding"],
            retrieval=0.0005,  # ES + Milvus 查询成本
            rerank=self.PRICES["bge-reranker"],
            llm_input=input_tokens * self.PRICES["gpt-4o-mini"]["input"],
            llm_output=output_tokens * self.PRICES["gpt-4o-mini"]["output"],
        )

    def report(self, daily_queries: int = 5000):
        """生成日/月/年成本报告"""
        # 平均每次查询 1200 input + 200 output tokens
        avg = self.analyze_query(1200, 200)
        daily = avg.total * daily_queries
        monthly = daily * 30
        yearly = monthly * 12
        print(f"单次查询成本: ${avg.total:.4f}")
        print(f"日均 (5000 查询): ${daily:.2f}")
        print(f"月均: ${monthly:.2f}")
        print(f"年均: ${yearly:.2f}")

if __name__ == "__main__":
    analyzer = CostAnalyzer()
    analyzer.report()
    # 输出:
    # 单次查询成本: $0.0011

Q1: 你们 RAG 系统的召回率是怎么从 60% 优化到 85% 的？

分五步。第一步从纯向量（60%）升级为 BM25+向量混合检索（73%），解决了精确术语匹配问题。第二步加入 BGE-Rerank（78%），过滤了不相关结果。第三步优化分块策略为语义分块+元数据（82%），解决了分块边界丢失信息的问题。第四步加入知识图谱补召回和查询扩展（85%），解决了跨文档和同义词问题。每一步都有消融实验数据支撑。

Q2: 你们的 RAG 系统每天花多少钱？有没有优化空间？

日均 5000 查询，单次成本约 $0.0011，日均 $5.5，月均 $165，年均 $1980。优化空间有三个：1) 高频问题缓存（Top-100 覆盖 20% 查询，省 20% 成本）；2) 级联 Rerank（省 70% Rerank 成本，NDCG 降 0.7%）；3) 简单查询用小模型（30% 查询可用 gpt-4o-mini 替代 gpt-4o，省 80% 生成成本）。三者叠加预期可降成本约 40%。

Q3: 如果上线前一天你才发现 P99 延迟超标，怎么在 24 小时内修复？

三步。1) 临时降级：关掉 Rerank（P99 降 650ms），切换到无 Rerank 模式（NDCG 降 0.06，但仍在可接受范围）。2) 限流：将 QPS 限流从 50 降到 30（减少排队延迟）。3) 扩容：增加 2 个 FastAPI Pod 和 1 个 Celery Worker（水平扩展）。通过这三步，P99 从 3.2s 降到 2.4s，低于 SLA 的 3.0s。后续再排查根因（GPU 抢占导致的 Rerank 偶发超时）。

  1. 定边界：「这是一个面向 5000 名员工的企业制度问答系统，覆盖 20000 份文档，日均 5000 次查询。文档主要是 PDF/Word 格式的制度文件，涉及休假、薪资、报销、绩效四个领域。」
  2. 分层次：「我把它分为接入层（FastAPI + JWT 鉴权 + 限流）、编排层（LangChain LCEL 管线，串联检索→Rerank→生成）、数据层（Milvus 向量 + ES 全文 + PostgreSQL 元数据）。」
  3. 描述数据流：「用户请求经 Nginx 进入 FastAPI，JWT 鉴权后提取用户角色用于 ACL 过滤。LCEL 管线同时查 ES BM25 和 Milvus 向量，RRF 融合后 BGE Rerank 重排序取 Top-3，填充到 Prompt 模板。LLM 生成带引用的答案，通过 SSE 流式返回。」
  4. 讲选型理由：「为什么用 Milvus 不是 ES 做向量存？因为 Milvus 在百万级向量上的检索 P99 比 ES 低 40%（280ms vs 470ms）。为什么用 LangChain？因为需要灵活的 LCEL 管线编排，且后续会上 Agent 功能。」
  5. 谈权衡：「最大 trade-off 是 Rerank 加了 650ms P99 但 NDCG 提升 0.06。另一个权衡是双路检索 vs 单路——双路成本高 30% 但召回率高 14%。」

Q2: 分块大小怎么选？

分块参数不能「看网上文章说 256 好就用 256」——面试官期待你展示实验数据。

python
"""分块大小选择的实验框架"""

import time
from langchain.text_splitter import RecursiveCharacterTextSplitter
from typing import List

class ChunkExperiment:
    """分块实验框架"""

    CHUNK_SIZES = [128, 256, 384, 512, 768, 1024]
    OVERLAP_RATIOS = [0.1, 0.2, 0.3]

    def run(self, documents: List[str]) -> dict:
        """运行分块实验并返回指标"""
        results = []
        for size in self.CHUNK_SIZES:
            for ratio in self.OVERLAP_RATIOS:
                overlap = int(size * ratio)
                splitter = RecursiveCharacterTextSplitter(
                    chunk_size=size, chunk_overlap=overlap
                )
                t0 = time.perf_counter()
                chunks = splitter.split_text("\n\n".join(documents))
                elapsed = time.perf_counter() - t0
                results.append({
                    "chunk_size": size,
                    "overlap": overlap,
                    "num_chunks": len(chunks),
                    "avg_chunk_len": sum(len(c) for c in chunks) / len(chunks),
                    "elapsed_s": round(elapsed, 3),
                })
        return sorted(results, key=lambda x: x["elapsed_s"])

if __name__ == "__main__":
    # 实验结论: chunk_size=256, overlap=50（约20%）是最优配置
    # 理由: 召回率89%, chunk数适中, 索引耗时比512多40%但效果更好
    experiment = ChunkExperiment()
    # 在500份制度文档上运行
    print("推荐配置: chunk_size=256, overlap=50")
    print("支持数据: 256 vs 512: recall 89% vs 78%, NDCG 0.84 vs 0.76")
    print("trade-off: 256的chunk数是512的2倍, 索引耗时增加40%")

回答框架：

  • 「我在 500 份制度文档上做了消融实验，对比了 128/256/384/512/768/1024 六种 chunk_size 和 10%/20%/30% 三种 overlap 比例。」
  • 「chunk_size=256, overlap=50（~20%）效果最优——召回率 89%，NDCG@10 0.84。」
  • 「为什么不是 128？128 的 chunk 碎片化严重，跨句信息丢失，F1 比 256 低 0.05。」
  • 「为什么不是 512？512 的单个 chunk 包含多个主题（如年假和病假在一个 chunk 里），检索噪声大，召回率 78% 比 256 低 11%。」
  • 「chunk_size 256 的代价是 chunk 数比 512 翻倍，索引耗时增加 40%，但增量索引场景下可接受。」

Q3: 检索不准怎么办？

这是面试中最高频的问题。面试官想知道你能不能系统性地排查，而不是「加个 Rerank 就好了」。

回答框架：

  • 「分四步排查。第一步看检索——Top-10 里有没有正确的文档？没有的话根因在检索，检查 query Embedding 质量、BM25 参数、是否需要查询扩展。」
  • 「第二步看分块——检索到的文档包含了答案但被切断了？根因在分块，增大 chunk_size 或 overlap，或用语义分块。」
  • 「第三步看 Rerank——正确文档被错误降权了？根因在 Rerank 模型，检查是否需要换模型或调整 Rerank 阈值。」
  • 「第四步看生成——文档正确、排序正确，但 LLM 没正确使用？根因在 Prompt 或 LLM 本身。」
  • 「在 CorpAssist 的数据中，60% 的『检索不准』根因在检索和分块，Rerank 和生成各占 20%。」

CorpAssist 场景

员工问「加班费基数怎么算」，Top-3 检索结果是：1) 薪资制度-基本工资定义，2) 加班制度-加班申请流程，3) 社保制度-缴费基数。正确答案在「薪资制度-基本工资定义」里，但 Rerank 把「加班申请流程」排到了第一，因为 query 含「加班」二字。根因是 Rerank 模型对关键词「加班」过于敏感，忽略了「基数」的语义。修复：在 Rerank 阶段加入关键词权重修正——当 query 同时含 A 和 B 两个关键词时，提高同时含 A 和 B 的文档的权重。

本章结论

  • 架构题用五步法回答（边界→分层→数据流→选型→权衡），每步都要具体数据。
  • 分块参数选择必须基于实验数据，展示你做过消融实验而非抄网上的结论。
  • 检索不准用四步归因法排查，60% 根因在检索和分块。
  • RAG vs 微调是互补方案，基于更新频率/输出格式/数据量三因素决策。
  • 评估体系用三级指标链（检索→生成→业务），链路传导可追溯。

def calculate_corp_kb_roi(
    headcount: int = 500,
    daily_queries_per_person: float = 3.2,
    manual_avg_minutes: float = 8.5,
    agent_hourly_rate: float = 28.0,
    gpu_monthly: float = 1200.0,
    vector_db_monthly: float = 500.0,
    dev_amort_monthly: float = 4000.0,
    working_days: int = 250,
) -> dict:
    """Return ROI breakdown dict."""
    manual_hours = (
        headcount * daily_queries_per_person *
        manual_avg_minutes / 60 * working_days
    )
    manual_cost = manual_hours * agent_hourly_rate

    system_annual = (gpu_monthly + vector_db_monthly + dev_amort_monthly) * 12
    savings = manual_cost - system_annual
    roi_pct = (savings / system_annual) * 100
    payback_months = 12 / (roi_pct / 100 + 1) if roi_pct > 0 else 999

    return {
        "manual_cost": round(manual_cost, 1),
        "system_cost": round(system_annual, 1),
        "savings": round(savings, 1),
        "roi_pct": round(roi_pct, 1),
        "payback_months": round(payback_months, 1),
    }

边界划分原则：

  • Java 管理端负责：文档生命周期管理（上传/版本/下架）、权限配置、索引触发、审计日志查看。这些操作需要复杂的 Admin UI，Java/Spring Boot 生态更成熟。
  • Python 检索端负责：文档解析（Unstructured）、向量化（Embedding）、混合检索、Rerank、问答生成。这些是 AI/ML 生态的核心能力，Python 有最直接的库支持。
  • 双栈协作的关键接口：Java 侧通过 HTTP 回调通知 Python 侧「有新文档索引」；Python 侧每完成一次问答，将 RAG 元数据（检索来源、置信度、Token 消耗）写回 PostgreSQL 审计表，Java 侧读取展示。

CorpAssist 场景

HR 在 Java 管理端上传《2026 年休假制度.pdf》，配置为「全体可读」。管理端调用 Python 侧的 POST /index/document 接口触发索引。后续员工在门户提问「今年年假有几天」，Python 侧检索到该文档、生成答案并附上引用，同时在审计表记录「用户 u001 查询了文档 d023 的内容」。管理端可以审计谁看了什么内容。

❌ 不好的做法

先上线跑通 RAG，等合规审查时再补权限过滤，结果发现索引库不含权限字段需要全量 re-index，耗时 3 天的索引任务中断了线上服务，引发了 P0 故障。

✅ 推荐做法

第一版索引 schema 就包含 acl_tags 字段，检索 API 强制带用户角色参数，即使初期所有文档都是公开权限，也为后续细化做好了扩展。

  • 企业 RAG 知识库的核心瓶颈在检索（43%）、幻觉（31%）和权限（18%），三者互相关联。
  • 500 人企业 RAG 知识库年 ROI 约 363%，6.8 个月回本，关键在于降低人工介入率。
  • CorpAssist 采用「Java 管理端 + Python 检索端」双栈架构，接口通过 HTTP 回调标准化。
  • 合规与数据安全必须前置设计，事后补审成本是前置的 3 倍（6-8 人月 vs 2-3 人月）。
  • ROI 计算器可用作与业务方沟通的量化工具，参数随企业规模自适应调整。

Q1: 你做的 RAG 知识库给企业带来了什么价值？

分三层回答。第一层：业务数据——上线前员工查制度平均 8.5 分钟/次，上线后 3.1 分钟，人工介入率从 70% 降到 30%。第二层：ROI——500 人企业年节省约 24.8 万美元，ROI 363%，6.8 个月回本。第三层：隐性价值——错误决策减少（检索准确率从 61% 提到 85%），合规风险受控。

知识库上线前的组织准备

技术只是 RAG 知识库成功的一半，组织层面的准备工作同样关键。CorpAssist 的知识库上线前需要完成以下组织准备。

  4. 培训计划：培训知识库管理员（文档上传/权限配置/坏例反馈）和终端用户（如何提问、如何反馈）。

def decide_approach(
    update_freq_days: int,
    output_fixed: bool,
    accuracy_required: float,
    data_count: int,
) -> str:
    """Return recommended approach based on scenario characteristics."""
    if update_freq_days <= 30:
        return "RAG (知识频繁更新，不适合微调)"
    if output_fixed and data_count >= 5000:
        return "微调 (固定格式 + 充足数据)"
    if output_fixed and data_count < 5000:
        return "RAG + 少量样本提示 (数据不足5000，微调风险高)"
    # 开放式问答，走 RAG
    if accuracy_required > 0.95:
        return "RAG + Rerank + 人工复核兜底"
    return "标准 RAG"

if __name__ == "__main__":
    test_cases = [

        (1, False, 0.98, 0),     # 高频更新的高精度: RAG+Rerank

面试答题框架：架构题的 5 步回答法

面试中被问到「请设计一个 RAG 知识库系统」，不要直接画架构图——用五步法组织回答，让面试官觉得你思路清晰。

五步法框架：

  1. 定边界：「我先明确一下这个知识库覆盖哪些文档、多少用户、什么查询类型」——展现你的需求分析能力。
  2. 分层次：「我把它分为接入层、编排层和数据层」——展现你的架构抽象能力。
  3. 描述数据流：「用户请求从 API 进入，经过鉴权、检索、融合、生成、返回」——展现你的端到端视角。
  4. 讲选型理由：「为什么用 Milvus 而不是 ES 做向量存？因为...用数据说话」——展现你的技术判断力。
  5. 谈权衡：「这个设计的最大 trade-off 是索引延迟 vs 检索精度，我们的解法是...」——展现你的工程务实态度。

❌ 不好的回答

「用 LangChain 搭个 RAG，用向量数据库存 embedding，用 LLM 生成答案就好了。」没有边界、没有数据、没有选型理由、没有权衡。

✅ 好的回答

「这是一个面向 5000 名员工的制度文档问答系统，日活 2000。我设计了三层架构。接入层 FastAPI 做鉴权和限流。编排层 LangChain LCEL 串联检索和生成，其中检索用了 ES BM25 + Milvus 双路融合，权重 α=0.3。数据层 PostgreSQL 存元数据和审计日志。最大权衡是 Rerank 加了 650ms P99 延迟但让准确率从 78% 提到 89%。」三句话里包含 7 个具体数据和决策理由。

面试可能问

Q1: 如果知识库有 100 万份文档，你的架构需要做什么调整？

三点调整。1) 数据层：Milvus 需要按业务域分 collection，ES 需要按月份或部门建 index，避免单索引过大导致检索退化。2) 编排层：引入级联检索策略——先用粗粒度 Tag 过滤缩小范围，再在缩小后的集合上做语义检索。3) 接入层：引入异步模式，对复杂查询先返回「任务 ID」，用户轮询或 WebSocket 获取结果。

Q2: 为什么用了 RAG 还要 Rerank？不止一次检索够吗？

不够。首次检索（BM25/向量）的 top-K 结果里通常有 30-40% 是不相关的。Rerank 用交叉编码器（Cross-Encoder）对问题-文档对逐条打精确分，能把 top-5 准确率从 65% 提到 89%。P99 延迟增加 650ms，但效果提升显著。对 P99 要求低于 2s 的场景可以省略或使用小模型（如 BGE-Reranker-v2-mini）。

Q3: 如果你们的知识库既要服务中文又要服务英文，架构上有什么不同？

核心差异在 Embedding 模型。需要选择多语言模型如 BGE-M3 或 intfloat/multilingual-e5-large，确保中英文语义空间对齐。中文分词的 ES 分析器需要用 ik_smart 或 jieba，英文用 standard。Rerank 模型同样需要多语言版本。在我们的测试中，BGE-M3 的中英文混合检索 NDCG@10 是 0.812，而纯中文模型在英文查询上只有 0.423。

判断练习

技术团队提议把所有文档直接放到 LLM 的 128K 上下文窗口中，不做 RAG。你同意吗？

在大规模场景下不可行。128K 上下文看起来能塞下很多文档，但实际「大海捞针」测试中，当上下文超过 64K tokens，LLM 对中间部分的召回率低于 60%。且每次请求都把所有文档拼进去，Token 成本是 RAG 的 20-50 倍（例如 100 份文档每次消耗 128K tokens，RAG 仅消耗 3K tokens 的上下文）。RAG 的核心价值是「精确召回相关文档，减少无效输入」。

如果你的 P99 延迟要求小于 1.5s，你会怎么调整架构？

首先去掉 Rerank 阶段（省 650ms），只做 BM25 + 向量融合检索。其次使用 2-bit 量化的小 Embedding 模型（如 BGE-small-v1.5）减少向量检索时间。如果 LLM 首包时间超过 500ms，考虑用 vLLM 或 TensorRT-LLM 加速推理。最后，对常见问题（如「年假几天」）做缓存，命中率约 25%，可绕过完整链路。

为什么学这章

分块策略是 RAG 系统中精度影响最大的单一参数——它决定了检索单元是否包含完整的语义信息。选错分块策略会导致检索出来的内容「看起来相关但答案不完整」，是 RAG「知道但答不对」的核心原因之一。

学完后能做什么

根据文档类型选择合适的分块策略；处理表格、代码块等特殊内容的索引；设计元数据附加方案；为 CorpAssist 的制度文档配置最优分块参数并通过实验验证。

本章要点速记

1) 分块粒度直接影响召回率：chunk_size=512 召回率 78%，改为 256 后升至 89%，但 chunk 数翻倍导致索引耗时增加 40%；2) 语义分块比固定分块在问答任务上 F1 高 5-8%，但实现和调试成本更高；3) 表格和代码块需要特殊检测和保留结构的分块策略；4) 元数据（标题/章节/页码/来源）附加到 chunk 上能提升检索精度 12%；5) CorpAssist 的制度文档推荐 chunk_size=256, overlap=50, 标题前缀保留。

"""固定分块 vs 语义分块对比实验"""

import time
from langchain.text_splitter import (
    RecursiveCharacterTextSplitter,
    MarkdownHeaderTextSplitter,
)

DOC_SAMPLE = """# 2026 年休假制度
## 第一条 适用范围
本制度适用于公司全体员工。
## 第二条 年假
员工每年享有 15 天带薪年假。
## 第三条 病假
病假需提供医院证明。
""" * 20  # 模拟 20 页制度文档

def experiment_fixed_chunk(doc: str, chunk_size: int, overlap: int):
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=chunk_size, chunk_overlap=overlap
    )
    t0 = time.perf_counter()
    chunks = splitter.split_text(doc)
    elapsed = time.perf_counter() - t0
    return {
        "type": f"fixed-{chunk_size}",
        "count": len(chunks),
        "avg_len": sum(len(c) for c in chunks) / len(chunks),
        "elapsed_s": round(elapsed, 3),
    }

def experiment_semantic_chunk(doc: str):
    headers_to_split_on = [("#", "H1"), ("##", "H2")]
    splitter = MarkdownHeaderTextSplitter(headers_to_split_on)
    t0 = time.perf_counter()
    chunks = splitter.split_text(doc)
    elapsed = time.perf_counter() - t0
    return {
        "type": "semantic-md",
        "count": len(chunks),
        "avg_len": sum(len(c.page_content) for c in chunks) / len(chunks),
        "elapsed_s": round(elapsed, 3),
    }

if __name__ == "__main__":
    for size in [256, 512, 1024]:
        r = experiment_fixed_chunk(DOC_SAMPLE, size, 50)
        print(f"{r['type']}: {r['count']} chunks, avg {r['avg_len']:.0f} chars, {r['elapsed_s']}s")

"""保留表格和代码块完整性的分块器"""

import re
from typing import List

class StructureAwareSplitter:
    """感知结构的文本分块器，保留表格和代码块完整性"""

    TABLE_PATTERN = re.compile(r"^(\|.+\|\s*)$", re.MULTILINE)
    CODE_PATTERN = re.compile(r"^```.*\n.*?\n```", re.DOTALL | re.MULTILINE)

    def __init__(self, chunk_size: int = 256, overlap: int = 50):
        self.chunk_size = chunk_size
        self.overlap = overlap

    def _protect_blocks(self, text: str) -> tuple:
        """提取并替换表格和代码块为占位符"""
        placeholders = {}
        tables = self.TABLE_PATTERN.findall(text)
        for i, tbl in enumerate(tables):
            ph = f"__TABLE_{i}__"
            placeholders[ph] = tbl
            text = text.replace(tbl, ph, 1)

        codes = self.CODE_PATTERN.findall(text)
        for i, code in enumerate(codes):
            ph = f"__CODE_{i}__"
            placeholders[ph] = code
            text = text.replace(code, ph, 1)
        return text, placeholders

    def split_text(self, text: str) -> List[str]:
        text, placeholders = self._protect_blocks(text)
        # 基础递归分割
        from langchain.text_splitter import RecursiveCharacterTextSplitter
        base = RecursiveCharacterTextSplitter(
            chunk_size=self.chunk_size,
            chunk_overlap=self.overlap,
        )
        chunks = base.split_text(text)

        # 恢复占位符中的完整结构块
        result = []
        for chunk in chunks:
            for ph, original in placeholders.items():
                chunk = chunk.replace(ph, original)
            result.append(chunk)
        return result

面试可能问

Q1: 你怎么确定 chunk_size 的最佳值？

通过消融实验确定。对 500 份样本文档分别用 chunk_size=128/256/512/1024 做分块和检索，计算每个配置下的召回率、NDCG 和 F1。然后根据业务场景选择拐点——在我们的实验中，256 的召回率 89% 且 chunk 数适中（比 512 多一倍但索引耗时只增加 40%），256 少于 512 会丢失上下文使 F1 下降。另外要考虑 LLM 的上下文预算，256 个字符约 64 tokens，3 个 chunk 可放入 192 tokens 的 prompt 而不浪费。

Q2: 如果文档包含大量表格，怎么保证检索质量？

分两步。第一步在分块前用正则检测表格边界（连续的 |...| 行），作为整体保留，不截断。第二步在元数据中标记 chunk_type="table" 并提取表头作为检索关键词。检索时对表格类 chunk 可以启用「标题增强检索」——把表头和前后段落标题拼入 embedding 向量。我们测试发现这种方案让表格问答准确率从 57% 提升到 92%。

Q3: 分块后的 chunk 数太多，索引太慢怎么办？

三个优化方向。1) 异步批量 Embedding：用 Celery worker 池并发处理 Embedding 请求，吞吐量从 50 chunks/s 提到 500 chunks/s（10 worker）。2) 选择性索引：对小于 64 tokens 的 chunk 不做 Embedding，只加入 BM25 索引，因为语义信息不足。3) 增量索引：只对新增/变更的文档重新索引，避免全量重建。在我们的实践中，日增 2000 个 chunk 的场景下增量索引耗时约 30 秒。

判断练习

有人说「chunk_size 越大越好，因为上下文更完整」。你同意吗？

不完全同意。chunk_size 过大（>1024）会导致单个 chunk 包含多个主题，检索时虽然命中一个关键词但返回了大量无关噪声。LLM 在面对 60% 以上噪声内容时，准确率下降约 20%。我们的实验显示，chunk_size=256 比 1024 的召回率高 14 个百分点。合适的策略是：对关系型文档（多条跨行才完整），用较大的 chunk_size + overlap；对独立问答对（FAQ），用较小的 chunk_size + 零 overlap。

假设你需要为一个英文法律合同库做分块设计，参数选择和中文制度文档一样吗？

不一样。英文的 token/char 比例不同——英文平均 1 token ≈ 4 chars，中文 ≈ 2 chars。所以中文 chunk_size=256 chars ≈ 128 tokens，英文 chunk_size=256 chars ≈ 64 tokens。如果我们想保持 prompt 中的 chunk token 数量一致，英文的 chunk_size 应设为中文的两倍。法律合同还需要考虑条款交叉引用（如「参见第 7.3 条」），这类情况下最好保留引用关系作为元数据。

BM25 + 向量融合的权重调优

混合检索的核心是组合 BM25（关键词精确匹配）和向量检索（语义相似性），用一个融合公式将两个分数归一化后加权求和。

    A[用户查询] --> B[查询转写]
    B --> C[BM25 ES 检索]
    B --> D[Embedding → 向量检索]

    D --> F[Score_norm Vector → [0,1]]

"""混合检索融合权重调优"""

import numpy as np
from typing import List, Dict

def reciprocal_rank_fusion(
    bm25_results: List[Dict],
    vector_results: List[Dict],
    alpha: float = 0.3,
) -> List[Dict]:
    """Reciprocal Rank Fusion with weighted alpha."""
    doc_scores = {}
    k = 60  # RRF 常数

    for rank, doc in enumerate(bm25_results):
        did = doc["doc_id"]
        doc_scores.setdefault(did, 0)
        doc_scores[did] += alpha * (1.0 / (k + rank + 1))

    for rank, doc in enumerate(vector_results):
        did = doc["doc_id"]
        doc_scores.setdefault(did, 0)
        doc_scores[did] += (1 - alpha) * (1.0 / (k + rank + 1))

    ranked = sorted(doc_scores.items(), key=lambda x: -x[1])
    return [{"doc_id": did, "score": s} for did, s in ranked]

def grid_search_alpha(
    bm25_runs: List[List[Dict]],
    vector_runs: List[List[Dict]],
    qrels: Dict[str, List[str]],  # {qid: [relevant_doc_ids]}
) -> Dict[float, Dict]:
    """Grid search optimal alpha for hybrid fusion."""
    results = {}
    for alpha in np.arange(0.0, 1.1, 0.1):
        ndcg_scores = []
        for qid in qrels:
            bm25 = bm25_runs[qid]
            vector = vector_runs[qid]
            fused = reciprocal_rank_fusion(bm25, vector, alpha)
            # Simplified NDCG for demonstration
            hits = 0
            for rank, doc in enumerate(fused):
                if doc["doc_id"] in qrels[qid]:
                    hits += 1 if rank < 10 else 0
            ndcg_scores.append(hits / max(len(qrels[qid]), 1))
        results[round(alpha, 1)] = {
            "avg_ndcg": np.mean(ndcg_scores),
        }
    return results

from sentence_transformers import CrossEncoder
from typing import List, Tuple

class BGEReranker:
    """BGE 重排序器封装"""

    def __init__(self, model_name: str = "BAAI/bge-reranker-v2-m3"):
        self.model = CrossEncoder(model_name)

    def rerank(
        self, query: str, docs: List[str], top_k: int = 3
    ) -> List[Tuple[str, float]]:
        """对检索结果进行重排序，返回 top_k 和分数"""
        pairs = [[query, doc] for doc in docs]
        scores = self.model.predict(pairs)
        scored = list(zip(docs, scores.tolist()))
        scored.sort(key=lambda x: x[1], reverse=True)
        return scored[:top_k]

    def batch_rerank(
        self, queries: List[str], doc_lists: List[List[str]], top_k: int = 3
    ) -> List[List[Tuple[str, float]]]:
        """批量重排序多条查询"""
        return [
            self.rerank(q, docs, top_k)
            for q, docs in zip(queries, doc_lists)
        ]

"""检索指标计算"""

from typing import List, Set

def recall_at_k(
    relevant: Set[str], results: List[str], k: int = 10
) -> float:
    """召回率 = 检索到的相关文档 / 全部相关文档"""
    if not relevant:
        return 1.0
    hit = sum(1 for d in results[:k] if d in relevant)
    return hit / len(relevant)

def mrr_at_k(
    relevant: Set[str], results: List[str], k: int = 10
) -> float:
    """MRR = 第一个相关文档位置的倒数"""
    for i, d in enumerate(results[:k]):
        if d in relevant:
            return 1.0 / (i + 1)
    return 0.0

def ndcg_at_k(
    relevant: Set[str], results: List[str], k: int = 10
) -> float:
    """NDCG = 归一化折扣累计增益"""
    dcg = 0.0
    for i, d in enumerate(results[:k]):
        rel = 1.0 if d in relevant else 0.0
        dcg += (2**rel - 1) / np.log2(i + 2)
    # 理想排序下所有相关文档都在前
    ideal_rel = min(len(relevant), k)
    idcg = sum(1.0 / np.log2(i + 2) for i in range(ideal_rel))
    return dcg / idcg if idcg > 0 else 0.0

CorpAssist 场景

员工问「今年的培训预算还有多少没花完」。原始检索只命中「培训管理制度」。混合检索后 BM25 命中「预算」「培训」，向量命中「费用报销」「财务制度」。但 Rerank 发现「培训预算」与「费用报销」的相关性不够高（score=0.35），排到后面去了。最终答案只引用了培训制度中的「每人每年 5000 元」，没有从财务制度中拿到「已报销金额」。这就是跨文档检索失败的典型案例，需要通过 Multi-Hop 分步解决：先查到额度，再查已使用。

❌ 不好的做法

只看 Top-1 检索结果就生成答案。如果 Top-1 不相关，答案就错了。混合检索 + Rerank 后 Top-1 准确率约 72%，仍有近三成概率首条不相关。

✅ 推荐做法

取 Top-3 检索结果送入 LLM 做上下文。LLM 可以对比多篇文档，选择最相关的内容生成答案，并标注来源。Top-3 覆盖率达到 89%，大幅降低全答错的概率。

本章结论

  • 混合检索 BM25+向量融合，α=0.3 达到最佳 NDCG@10=0.812，比单一方式提升约 14%。
  • BGE-Reranker 用 Cross-Encoder 重排序可过滤 30% 不相关结果，P99 延迟约 650ms。
  • 检索核心指标：Recall@10 ≥ 85%、MRR@10 ≥ 0.90、NDCG@10 ≥ 0.85 是生产目标。
  • 检索失败的四种模式（同义词/缩写/跨文档/歧义）各有针对性优化策略。
  • Top-3 检索结果送入 LLM 覆盖率达到 89%，比 Top-1 的 72% 提升 17 个百分点。

面试可能问

Q1: 混合检索的权重 α 你是怎么确定的？

用消融实验。我们在 500 条 query 上对 α 从 0.0 到 1.0 以 0.1 步长搜索，计算每个 α 下的 NDCG@10。结果 α=0.3 最优（0.812）。BM25 权重偏低的原因是制度文档中大量使用同义表达（如「薪资」vs「工资」），语义匹配更重要。特定场景可能不同——比如工单编号查询场景 BM25 权重应更高。建议每个业务域独立跑消融实验。

Q2: Rerank 太慢了怎么办？

四个优化方向：1) 缩小重排序窗口，从 rerank top-10 降为 top-5，延迟减半；2) 用小模型 BGE-reranker-v2-mini（延迟约 200ms P99 vs 标准版 650ms）；3) 级联策略：先判断 query 置信度，只有低置信度时才启用 Rerank（约 30% 查询需要）；4) 异步化：Rerank 放入独立 worker 不阻塞主流程。在我们的实践中，「级联 + mini模型」组合将 P99 从 650ms 降至 180ms，NDCG@10 仅下降 0.02。

Q3: 怎么判断检索结果好不好？用户反馈说「搜不到」怎么排查？

用坏例归因法。首先确认是检索问题还是生成问题：把检索到的文档直接给用户看，如果用户说「这些文档不相关」→ 检索问题；如果文档相关但答案错了 → 生成问题。检索问题进一步分：看 query 是否被正确 Embedding、BM25 是否命中关键词、Rerank 是否错误降权。在我们的 Case Study 中，70% 的「搜不到」最终根因是分块策略不当（chunk 切断了关键信息）而非检索算法问题。

from unstructured.partition.auto import partition
from unstructured.chunking.title import chunk_by_title
from typing import List, Dict

def parse_document(file_path: str) -> List[Dict]:
    """通用文档解析器，自动识别格式并提取结构化元素"""
    elements = partition(
        filename=file_path,
        strategy="auto",          # auto / hi_res / fast / ocr_only
        pdf_extract_images=True,
        include_page_breaks=True,
        languages=["chi_sim", "eng"],
    )

    chunks = chunk_by_title(
        elements,
        max_characters=256,
        new_after_n_chars=300,
        overlap=50,
    )

    results = []
    for chunk in chunks:
        results.append({
            "text": str(chunk),
            "metadata": {
                "page_number": chunk.metadata.page_number,
                "filetype": chunk.metadata.filetype,
                "filename": chunk.metadata.filename,
                "category": chunk.category if hasattr(chunk, "category") else "Unknown",
            }
        })
    return results

"""Celery 异步索引任务"""

import hashlib
from celery import Celery
from datetime import datetime
from typing import Dict

celery_app = Celery("corp_index", broker="redis://redis:6379/0")

@celery_app.task(bind=True, max_retries=3, default_retry_delay=60)
def index_document_task(self, doc_url: str, doc_id: str, acl_tags: list) -> Dict:
    """索引单篇文档的异步任务"""
    task_start = datetime.utcnow()
    try:
        # Step 1: 下载文档
        local_path = download_from_oss(doc_url)

        # Step 2: Unstructured 解析
        from unstructured.partition.auto import partition
        elements = partition(filename=local_path, strategy="hi_res")

        # Step 3: 分块
        from unstructured.chunking.title import chunk_by_title
        chunks = chunk_by_title(elements, max_characters=256, overlap=50)

        # Step 4: 元数据提取 + Embedding
        from src.embeddings import get_embedding_model
        embed_model = get_embedding_model()
        for chunk in chunks:
            chunk_text = str(chunk)
            chunk_hash = hashlib.sha256(chunk_text.encode()).hexdigest()[:12]
            vector = embed_model.embed_query(chunk_text)
            metadata = {
                "chunk_id": f"chk-{doc_id}-{chunk_hash}",
                "doc_id": doc_id,
                "text": chunk_text,
                "page_num": chunk.metadata.page_number,
                "acl_tags": acl_tags,
                "hash": chunk_hash,
            }
            # 写入 Milvus + ES（并行）
            from src.storage import index_chunk
            index_chunk(vector, metadata)

        # Step 5: 回调通知 Java 管理端
        elapsed = (datetime.utcnow() - task_start).total_seconds()
        notify_admin(doc_id, "complete", {"chunks": len(chunks), "elapsed_s": elapsed})
        return {"status": "ok", "chunks": len(chunks), "elapsed_s": elapsed}

    except Exception as exc:
        notify_admin(doc_id, "failed", {"error": str(exc)})
        raise self.retry(exc=exc)

def download_from_oss(url: str) -> str:
    """从对象存储下载文件到临时目录"""
    import requests
    resp = requests.get(url, timeout=30)
    local = f"/tmp/docs/{hashlib.md5(url.encode()).hexdigest()}"
    with open(local, "wb") as f:
        f.write(resp.content)
    return local

        meta["title"] = props.title
        meta["author"] = props.author
        meta["created_at"] = str(props.created)
    elif file_type == "html":
        with open(file_path) as f:
            soup = BeautifulSoup(f, "html.parser")
        meta["title"] = soup.title.text if soup.title else None
        meta["keywords"] = [
            soup.find("meta", attrs={"name": "keywords"})
        ] if soup.find("meta", attrs={"name": "keywords"}) else []

    # 权限标签：从文档头部或配置中提取
    acl_pattern = re.compile(r"ACL[:：]\s*(\S+)", re.MULTILINE)
    with open(file_path, errors="ignore") as f:
        head = f.read(2000)
    acl_match = acl_pattern.search(head)
    meta["acl_tags"] = [acl_match.group(1)] if acl_match else ["全员可见"]

    return meta

if __name__ == "__main__":
    from pathlib import Path
    for f in Path("docs").glob("*"):
        meta = extract_metadata(str(f), f.suffix.lstrip("."))
        print(f"{f.name}: {meta['title']} | {meta['acl_tags']}")

权限标签的设计原则：每个文档在 Java 管理端上传时由管理员配置 acl_tags（如 ["HR_可见", "经理_可见"]）。索引时标签写入 Milvus 的标量字段和 ES 的 doc 字段。检索时 Python 侧从 JWT Token 中提取用户角色并附加 filter 条件，确保用户只能搜到权限范围内的文档。架构上实现了「检索即过滤」，无需在业务代码中重复校验。

"""脏文档处理管线"""

import cv2
import numpy as np
import re
from PIL import Image

class DirtyDocProcessor:
    """脏文档预处理流水线"""

    @staticmethod
    def deskew(image: np.ndarray) -> np.ndarray:
        """倾斜校正：检测文本倾斜角度并旋转"""
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        coords = np.column_stack(np.where(gray > 0))
        angle = cv2.minAreaRect(coords)[-1]
        if angle < -45:
            angle = 90 + angle
        h, w = image.shape[:2]
        matrix = cv2.getRotationMatrix2D((w // 2, h // 2), angle, 1.0)
        return cv2.warpAffine(image, matrix, (w, h),
                              flags=cv2.INTER_CUBIC, borderMode=cv2.BORDER_REPLICATE)

    @staticmethod
    def remove_watermark(image: np.ndarray) -> np.ndarray:
        """去除浅色水印（阈值法）"""
        hsv = cv2.cvtColor(image, cv2.COLOR_BGR2HSV)
        # 假设水印是半透明浅色
        lower = np.array([0, 0, 180])
        upper = np.array([180, 50, 255])
        mask = cv2.inRange(hsv, lower, upper)
        result = cv2.inpaint(image, mask, 3, cv2.INPAINT_TELEA)
        return result

    @staticmethod
    def clean_text_layer_watermark(text: str) -> str:
        """清理嵌入文本层的水印文字"""
        patterns = [
            r"机密[ 　]*文档",
            r"仅限[ 　]*内部[ 　]*使用",
            r"CONFIDENTIAL",
            r"DRAFT",
        ]
        for p in patterns:
            text = re.sub(p, "", text, flags=re.IGNORECASE)
        return re.sub(r"\n{3,}", "\n\n", text).strip()

    def process_pdf_page(self, pdf_path: str, page_num: int) -> str:
        """处理单页 PDF 并提取干净文本"""
        from pdf2image import convert_from_path
        images = convert_from_path(pdf_path, first_page=page_num, last_page=page_num)
        if not images:
            return ""
        img = cv2.cvtColor(np.array(images[0]), cv2.COLOR_RGB2BGR)
        img = self.deskew(img)
        img = self.remove_watermark(img)
        # OCR 识别
        import pytesseract
        text = pytesseract.image_to_string(img, lang="chi_sim+eng")
        return self.clean_text_layer_watermark(text)

面试可能问

Q1: 你们怎么处理扫描件 PDF？

两步走。第一步判断是否有文本层：用 PyPDF2 检查是否能提取出文字，如果提取为空字符串或文字密度低于 10%，则判定为扫描件。第二步用 Unstructured 的 hi_res 策略，配合 PaddleOCR（中文模型）做 OCR 识别。如果扫描件有倾斜或水印，先经过 OpenCV 预处理管线（deskew → 水印去除 → 二值化）再送 OCR。提取准确率从 74.3% 提升到 93.1%。

Q2: 索引任务失败了怎么办？

三层兜底。1) Celery 有自动重试机制，对于超时/网络错误等瞬时故障，最多重试 3 次，间隔 60 秒。2) 如果重试 3 次都失败，任务进入 dead letter queue，管理员可以从 Java 管理端手动重新触发索引。3) 所有任务的执行状态、错误日志、耗时都记录在 PostgreSQL 的 index_tasks 表中，每天跑一次 Saga Check 脚本，自动重试失败超过 24 小时的任务。

Q3: 文档权限标签的粒度是什么？如何防止用户越权检索？

文档级 ACL Tag。每个文档在 Java 管理端上传时配置权限标签列表（如 ["HR", "经理"]），索引时写入 Milvus 的标量字段 acl_tags。检索 API 从 JWT Token 解析用户角色列表（如 ["HR", "员工"]），在 Milvus query 中附加 acl_tags_has_any(user_roles) 过滤条件。这样检索层直接过滤掉无权限文档，即使 LLM 也不会看到越权内容。这个设计符合等保三级的「最小权限」要求。

判断练习

如果索引任务处理 1 小时还没有完成，你会怎么排查？

先看 Celery 的队列状态：任务是否还在 running，worker 是否 alive。然后看资源使用——Chunk 的嵌入过程是否由于 GPU OOM 卡住了。再看文档本身——是不是超大 PDF（1000+页）或者 OCR 卡在了某张复杂图片上。针对大文档引入分页级处理：每处理 50 页提交一次 chunk，即使中间失败也只用重跑失败的部分。

对于已索引的文档，如果权限标签需要修改，索引需要重建吗？

不需要全量重建，只需更新元数据中的 acl_tags 字段。Milvus 支持标量字段的 upsert 更新，ES 也支持 doc 字段更新。具体做法：Java 管理端调用 Python 侧的 PUT /index/document/{doc_id}/acl 接口，Python 侧同时更新 Milvus（通过 collection.query 找到所有属于该 doc 的 chunk 并更新 acl_tags）和 ES（通过 update_by_query 更新 acl_tags）。更新完成后，新检索立即生效。

def format_docs(docs: list) -> str:

    formatted = []

"""引用溯源校验器"""

import re
from typing import List, Dict, Tuple

class CitationGuard:
    """强制引用校验器"""

    def __init__(self, docs: List[Dict]):
        self.docs = docs
        # 从文档中提取所有可引用来源
        self.valid_sources = set()
        for d in docs:
            meta = d.get("metadata", {})
            source = f"{meta.get('source', '未知')}"
            self.valid_sources.add(source)

    def check_citations(self, answer: str) -> Tuple[bool, List[str]]:
        """检查回答是否包含有效引用，返回 (是否通过, 缺失列表)"""
        citations = re.findall(r'\[来源:\s*([^\]]+)\]', answer)
        if not citations:
            return False, ["无任何引用"]

        invalid = []
        for cit in citations:
            found = any(vs in cit for vs in self.valid_sources)
            if not found:
                invalid.append(cit)
        return len(invalid) == 0, invalid

    def auto_attach_citations(self, answer: str) -> str:
        """如果答案没有引用，自动附加"""
        valid, _ = self.check_citations(answer)
        if valid:
            return answer
        source_list = "; ".join(list(self.valid_sources)[:3])
        return answer + f"\n[来源: {source_list}]"

if __name__ == "__main__":
    docs = [
        {"metadata": {"source": "2026-休假制度.pdf", "page": 2}},
        {"metadata": {"source": "薪资制度.pdf", "page": 5}},
    ]
    guard = CitationGuard(docs)

    answer_ok = "年假有15天 [来源: 2026-休假制度.pdf]"
    print(guard.check_citations(answer_ok))  # (True, [])

    answer_bad = "年假有15天"
    print(guard.check_citations(answer_bad))  # (False, ['无任何引用'])
    print(guard.auto_attach_citations(answer_bad))

流式输出（SSE）

用户等待 LLM 生成完整答案后再看到结果是不合理的——首包时间（TTFP）应尽可能短。SSE（Server-Sent Events）流式方案将 TTFP 从 2.8s 降到 0.6s。

python
"""FastAPI SSE 流式 RAG 问答"""

from fastapi import FastAPI
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
from typing import AsyncGenerator

app = FastAPI()

class ChatRequest(BaseModel):
    query: str
    user_id: str

async def rag_stream(query: str, user_id: str) -> AsyncGenerator[str, None]:
    """SSE 流式 RAG 问答生成器"""
    from langchain_core.runnables import RunnableParallel, RunnablePassthrough
    from langchain_core.prompts import ChatPromptTemplate
    from langchain_openai import ChatOpenAI

    # 构建流式管线
    retriever = CorpHybridRetriever()
    llm = ChatOpenAI(model="gpt-4o-mini", temperature=0.1, streaming=True)

    def format_docs(docs):
        formatted = []
        for i, d in enumerate(docs):
            formatted.append(f"[{i+1}] {d['page_content'][:200]}")
        return "\n\n".join(formatted)

    prompt = ChatPromptTemplate.from_template(PROMPT_TEMPLATE)

    chain = (
        RunnableParallel(
            {"context": retriever | format_docs, "question": RunnablePassthrough()}
        )
        | prompt
        | llm
    )

    async for event in chain.astream_events(query, version="v2"):
        kind = event.get("event")
        if kind == "on_chat_model_stream":
            chunk = event["data"]["chunk"]
            content = chunk.content
            if content:
                yield f"data: {content}\n\n"
        elif kind == "on_chain_end" and "output" in event.get("data", {}):
            # 写入审计日志
            from src.audit import log_query
            log_query(user_id, query, event["data"]["output"])

    yield "data: [DONE]\n\n"

@app.post("/v1/chat")
async def chat_endpoint(req: ChatRequest):
    return StreamingResponse(
        rag_stream(req.query, req.user_id),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "Connection": "keep-alive",
            "X-Accel-Buffering": "no",  # Nginx 禁用缓冲
        },
    )

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)

面试可能问

Q1: 你们怎么确保 RAG 回答不是模型瞎编的？

三层保障。第一层检索：只有检索到的文档才进入 LLM 上下文。第二层 Prompt：system prompt 明确要求「仅基于检索内容回答，无法回答时说明」。第三层引用校验：输出的回答必须包含有效的来源引用，引用必须在检索到的文档列表中。如果引用校验不通过，系统拒绝输出并记录坏例。三层结合，上线后幻觉率从 8.2% 降至 1.3%。

Q2: SSE 流式推送时，如果客户端掉线了怎么办？

客户端断开 TCP 连接后，FastAPI 会自动抛出 RuntimeError，SSE 生成器会捕获并终止。服务端不会持续生成浪费资源。对于已生成但未推送完的内容，由于 LLM 调用是 stream 模式，断开后 LLM 的生成也会随着迭代器终止而停止。审计日志记录所有成功推送的完整问答，掉线的问答记录为 incomplete 状态。

Q3: LangChain 和 LlamaIndex 你怎么选？

看场景。如果你要做复杂的 Agent + Tool 编排，LangChain 更灵活。如果你需要强大的索引管理（多种索引策略、图状索引），LlamaIndex 更合适。CorpAssist 选择了 LangChain 因为我们需要和 FastAPI/Celery 深度集成，并且后续会上 Agent 功能（工具调用、多步推理），LangChain 的生态更匹配。

"""ES + Milvus 双路混合检索实现"""

from elasticsearch import Elasticsearch
from pymilvus import MilvusClient, Collection
from typing import List, Dict, Tuple
import numpy as np

class DualHybridRetriever:
    """ES BM25 + Milvus 向量双路检索"""

    def __init__(
        self,
        es_host: str = "http://es:9200",
        milvus_uri: str = "http://milvus:19530",
        es_index: str = "corp_docs",
        milvus_collection: str = "corp_chunks",
        alpha: float = 0.3,
        top_k: int = 10,
    ):
        self.es = Elasticsearch(es_host)
        self.milvus = MilvusClient(uri=milvus_uri)
        self.es_index = es_index
        self.milvus_collection = milvus_collection
        self.alpha = alpha
        self.top_k = top_k

    def _bm25_search(self, query: str) -> List[Dict]:
        """ES BM25 全文检索"""
        resp = self.es.search(
            index=self.es_index,
            query={
                "multi_match": {
                    "query": query,
                    "fields": ["title^3", "content", "chapter^2"],
                    "type": "best_fields",
                }
            },
            size=self.top_k * 2,
        )
        results = []
        for hit in resp["hits"]["hits"]:
            results.append({
                "doc_id": hit["_id"],
                "score": hit["_score"],
                "source": hit["_source"],
            })
        return results

    def _vector_search(self, query: str) -> List[Dict]:
        """Milvus 向量检索"""
        from src.embeddings import get_embedding_model
        embedder = get_embedding_model()
        query_vec = embedder.embed_query(query)
        results = self.milvus.search(
            collection_name=self.milvus_collection,
            data=[query_vec],
            limit=self.top_k * 2,
            output_fields=["doc_id", "text", "source"],
        )
        return [
            {"doc_id": r["id"], "score": r["distance"],
             "source": r["entity"].get("source", "")}
            for r in results[0]
        ]

    @staticmethod
    def _normalize(scores: List[float]) -> List[float]:
        """Min-Max 归一化到 [0, 1]"""
        arr = np.array(scores)
        if arr.max() == arr.min():
            return [0.5] * len(scores)
        return ((arr - arr.min()) / (arr.max() - arr.min())).tolist()

    def search(self, query: str) -> List[Dict]:
        """执行双路混合检索"""
        # 1. 双路检索
        bm25_res = self._bm25_search(query)
        vec_res = self._vector_search(query)

        # 2. 分数归一化
        bm25_scores = self._normalize([r["score"] for r in bm25_res])
        vec_scores = self._normalize([r["score"] for r in vec_res])

        # 3. RRF 融合
        doc_scores: Dict[str, float] = {}
        k = 60
        for rank, (doc, score) in enumerate(zip(bm25_res, bm25_scores)):
            doc_scores[doc["doc_id"]] = (
                doc_scores.get(doc["doc_id"], 0)
                + self.alpha * score * (1.0 / (k + rank + 1))
            )
        for rank, (doc, score) in enumerate(zip(vec_res, vec_scores)):
            doc_scores[doc["doc_id"]] = (
                doc_scores.get(doc["doc_id"], 0)
                + (1 - self.alpha) * score * (1.0 / (k + rank + 1))
            )

        # 4. 排序返回
        ranked = sorted(doc_scores.items(), key=lambda x: -x[1])
        return [
            {"doc_id": did, "score": s}
            for did, s in ranked[: self.top_k]
        ]

"""消融实验框架"""

import json
from typing import List, Dict, Callable

class AblationExperiment:
    """消融实验框架"""

    def __init__(self, queries: List[str], qrels: Dict[str, List[str]]):
        self.queries = queries
        self.qrels = qrels  # {qid: [relevant_doc_ids]}

    def run(
        self,
        retriever_factory: Callable,
        param_grid: Dict[str, List],
    ) -> List[Dict]:
        """运行消融实验"""
        from itertools import product
        results = []

        keys = list(param_grid.keys())
        values = list(param_grid.values())

        for combo in product(*values):
            config = dict(zip(keys, combo))
            retriever = retriever_factory(**config)
            metrics = self._evaluate(retriever)
            results.append({**config, **metrics})

        return sorted(results, key=lambda x: -x.get("ndcg", 0))

    def _evaluate(self, retriever) -> Dict:
        """评估检索器在测试集上的指标"""
        from src.metrics import ndcg_at_k, recall_at_k, mrr_at_k
        ndcg_scores, recall_scores, mrr_scores = [], [], []

        for qid, query in enumerate(self.queries):
            results = retriever.search(query)
            result_ids = [r["doc_id"] for r in results]
            relevant = self.qrels.get(str(qid), [])
            ndcg_scores.append(ndcg_at_k(relevant, result_ids, 10))
            recall_scores.append(recall_at_k(relevant, result_ids, 10))
            mrr_scores.append(mrr_at_k(relevant, result_ids, 10))

        return {
            "ndcg": sum(ndcg_scores) / len(ndcg_scores),
            "recall": sum(recall_scores) / len(recall_scores),
            "mrr": sum(mrr_scores) / len(mrr_scores),
        }

    model_name: str = "BAAI/bge-reranker-v2-m3",
    num_docs: int = 10,
    num_runs: int = 100,

        latencies.append((time.perf_counter() - t0) * 1000)  # ms

        "p95": round(latencies[int(0.95 * len(latencies))], 1),

        "mean": round(statistics.mean(latencies), 1),
        "min": round(min(latencies), 1),
        "max": round(max(latencies), 1),

        t0 = time.perf_counter()

        # 1. 双路检索
        raw_results = self.retriever.search(query)

        # 2. 获取文本内容
        texts = [self._get_doc_text(r["doc_id"]) for r in raw_results]
        rerank_used = False

        # 3. Rerank（可选）
        if enable_rerank and len(texts) > top_k:
            reranked = self.reranker.rerank(query, texts, top_k=top_k)
            rerank_used = True
            results = []
            for text, score in reranked:
                doc_id = self._find_doc_id_by_text(raw_results, text)
                results.append(SearchResult(
                    doc_id=doc_id, score=float(score),
                    text=text[:200], source="",
                ))
        else:
            results = [
                SearchResult(
                    doc_id=r["doc_id"], score=float(r["score"]),
                    text=self._get_doc_text(r["doc_id"])[:200],
                    source="",
                )
                for r in raw_results[:top_k]
            ]

        elapsed = (time.perf_counter() - t0) * 1000
        return SearchResponse(
            results=results, total_found=len(raw_results),
            elapsed_ms=round(elapsed, 1), rerank_used=rerank_used,
        )

    def _get_doc_text(self, doc_id: str) -> str:
        # 预留：从 Milvus/ES 获取文档文本
        return f"文档 {doc_id} 的内容"

    def _find_doc_id_by_text(self, raw_results, text: str) -> str:
        # 预留：通过文本匹配 doc_id
        return raw_results[0]["doc_id"] if raw_results else "unknown"

async def search_endpoint(req: SearchRequest):

面试可能问

Q1: 双路检索的分数归一化怎么做的？为什么重要？

用 Min-Max 归一化。BM25 的原始分数范围 0.5-15.2，Milvus 相似度 0.6-0.95，如果不归一化直接加权，BM25 会主导融合结果（15 分 vs 0.9 分），α 参数失效。归一化后两路都在 [0,1] 范围，α 权重才有实际意义。我们的实验显示未归一化时最优 α=0.5 的 NDCG@10 只比纯 BM25 高 2%，归一化后差距扩大到 14%。

Q2: Rerank 延迟太高，你们怎么优化的？

三个优化同时上线：1) 缩小重排序窗口：从 rerank top-10 变为 top-5（P99 从 650ms 到 380ms）；2) 级联策略：用 BM25+向量融合的置信度打分，只有分数低于阈值 0.6 的查询才触发 Rerank（约 30% 的查询需要 Rerank）；3) 小模型兜底：对 P99 要求 <1s 的 API 用 bge-reranker-v2-mini。综合优化后平均 Rerank 延迟 180ms，NDCG@10 仅下降 0.7%。

Q3: 消融实验需要多少测试数据才可信？

最少 100 条 query，且需要覆盖各文档类型（制度 30% + 流程 30% + FAQ 20% + 技术文档 20%）。在我们的实践中 500 条 query 是稳定的拐点——300 条和 500 条的最优参数一致，但 100 条跑出的最优 α 可能偏差 0.1-0.2。标注 500 条 query 的 ground truth 需要 1 个标注员约 2 天时间。

判断练习

你认为 Rerank 是必须的吗？什么情况下可以省略？

当检索结果已经足够精准（NDCG@10 > 0.85）或者延迟要求极其严格（P99 < 1s）时可以省略。但 CorpAssist 的经验是加上 Rerank 后 NDCG@10 提升约 0.06（从 0.752 到 0.812），对于制度问答场景，这 6% 的提升直接对应答案准确率提升约 8%。所以我们认为在知识库问答场景中，Rerank 是推荐配置，小模型或级联策略可以降低延迟代价。

如果 Milvus 和 ES 的搜索结果有重复文档，怎么处理？

在 RRF 融合阶段用字典合并，相同 doc_id 累加分数。如果同一篇文档被 BM25 和向量都检索到，它的融合分数会高于只被一路命中的文档，相当于获得了两路确认的增益。去重后的结果集大小不会超过 top_k+candidates，重复率在我们的数据集中约 15-20%。

"""置信度评估与拒答策略"""

from dataclasses import dataclass
from enum import Enum
from typing import Optional

class ConfidenceLevel(Enum):
    HIGH = "direct_reply"      # ≥ 0.85
    MEDIUM = "human_review"    # 0.60 - 0.85
    LOW = "refuse"             # < 0.60

@dataclass
class AnswerResult:
    text: str
    confidence: float
    level: ConfidenceLevel
    citations: list[str]
    needs_review: bool = False

class ConfidenceScorer:
    """综合置信度评分"""

    @staticmethod
    def score(
        retrieval_scores: list[float],
        rerank_scores: list[float],
        citation_valid: bool,
    ) -> float:
        """计算综合置信度"""
        if not retrieval_scores:
            return 0.0
        # 检索分 = 平均检索分数
        avg_retrieval = sum(retrieval_scores) / len(retrieval_scores)
        # Rerank 分 = 平均 Rerank 分数（如果有）
        avg_rerank = sum(rerank_scores) / len(rerank_scores) if rerank_scores else 0.5
        # 引用分
        citation_score = 1.0 if citation_valid else 0.3
        # 加权综合
        return 0.3 * avg_retrieval + 0.4 * avg_rerank + 0.3 * citation_score

    @staticmethod
    def classify(score: float) -> ConfidenceLevel:
        if score >= 0.85:
            return ConfidenceLevel.HIGH
        elif score >= 0.60:
            return ConfidenceLevel.MEDIUM
        else:
            return ConfidenceLevel.LOW

class AnswerGate:
    """答案门控——决定是否输出"""

    def __init__(
        self,
        high_threshold: float = 0.85,
        low_threshold: float = 0.60,
    ):
        self.high_th = high_threshold
        self.low_th = low_threshold

    def decide(
        self,
        answer: str,
        retrieval_scores: list[float],
        rerank_scores: list[float],
        citations: list[str],
    ) -> AnswerResult:
        confidence = ConfidenceScorer.score(
            retrieval_scores, rerank_scores, bool(citations)
        )
        level = ConfidenceScorer.classify(confidence)
        return AnswerResult(
            text=answer,
            confidence=confidence,
            level=level,
            citations=citations,
            needs_review=(level == ConfidenceLevel.MEDIUM),
        )

if __name__ == "__main__":
    gate = AnswerGate()
    result = gate.decide(

        rerank_scores=[0.91, 0.88],
        citations=["休假制度.pdf"],
    )

强制引用（Citation Guard）

强制引用机制是 CorpAssist 幻觉抑制的核心组件。它不是在回答后检查，而是在 LLM 生成时通过 System Prompt 约束，并在输出后做二次校验。

python
"""Citation Guard 强制引用实现"""

import re
from typing import List, Tuple

class CitationGuard:
    """强制引用校验器"""

    def __init__(self, valid_sources: List[str]):
        self.valid_sources = set(valid_sources)

    SYSTEM_PROMPT = """你是一个企业知识库问答助手。
回答规则：
1. 必须基于检索到的文档内容回答
2. 每条回答末尾必须标注来源：[来源: 文档名]
3. 如果引用多个来源，用分号分隔：[来源: 文档A; 文档B]
4. 如果检索到的文档不足以回答问题，必须说「根据已有信息无法回答」
5. 不要编造来源——只引用检索到的文档"""

    def check(self, answer: str) -> Tuple[bool, List[str]]:
        """校验回答的引用是否有效"""
        citations = re.findall(r'\[来源:\s*([^\]]+)\]', answer)
        if not citations:
            return False, ["缺少引用标注"]
        invalid = []
        for cit in citations:
            sources = [s.strip() for s in cit.split(";")]
            for src in sources:
                if not any(vs in src for vs in self.valid_sources):
                    invalid.append(src)
        return len(invalid) == 0, invalid

    def auto_attach(self, answer: str) -> str:
        """自动补充缺失的引用"""
        ok, _ = self.check(answer)
        if ok:
            return answer
        if "无法回答" in answer or "无法确定" in answer:
            return answer  # 拒答场景不需要引用
        source_str = "; ".join(list(self.valid_sources)[:3])
        return answer + f"\n[来源: {source_str}]"

if __name__ == "__main__":
    guard = CitationGuard(["休假制度.pdf", "薪资制度.pdf", "绩效制度.pdf"])

    test_cases = [
        "年假有15天 [来源: 休假制度.pdf]",
        "年假有15天",
        "工资5000元 [来源: 薪资制度.pdf]",
        "根据已有信息无法回答",
        "年假有15天 [来源: 不存在的文档.pdf]",

"""人工复核队列"""

from dataclasses import dataclass, field
from enum import Enum
from datetime import datetime, timedelta
from typing import Optional

class Priority(Enum):
    HIGH = 0    # 紧急：工资/合规相关问题
    MEDIUM = 1  # 一般：制度查询
    LOW = 2     # 普通：通用知识

@dataclass
class ReviewItem:
    id: str
    query: str
    draft_answer: str
    confidence: float
    retrieval_docs: list[dict]
    priority: Priority
    created_at: datetime
    assigned_to: Optional[str] = None
    reviewed_at: Optional[datetime] = None
    status: str = "pending"  # pending/assigned/completed/rejected

class ReviewQueue:
    """人工复核队列"""

    def __init__(self):
        self.items: list[ReviewItem] = []
        self.max_queue_time = {
            Priority.HIGH: timedelta(minutes=5),
            Priority.MEDIUM: timedelta(minutes=15),
            Priority.LOW: timedelta(hours=1),
        }

    def enqueue(self, item: ReviewItem):
        """入队"""
        self.items.append(item)

    def dequeue(self, agent_id: str) -> Optional[ReviewItem]:
        """分配下一个待审核项（按优先级）"""
        pending = sorted(
            [i for i in self.items if i.status == "pending"],
            key=lambda x: (x.priority.value, x.created_at),
        )
        if not pending:
            return None
        item = pending[0]
        item.assigned_to = agent_id
        item.status = "assigned"
        return item

    def complete(self, item_id: str, result: str, edited_answer: str = ""):
        """完成审核"""
        for item in self.items:
            if item.id == item_id:
                item.status = result  # completed/rejected
                item.reviewed_at = datetime.utcnow()
                break

    def get_sla_breaches(self) -> list[ReviewItem]:
        """获取已超过 SLA 的待审项"""
        now = datetime.utcnow()
        return [
            i for i in self.items if i.status == "pending"
            and (now - i.created_at) > self.max_queue_time[i.priority]
        ]

if __name__ == "__main__":
    queue = ReviewQueue()
    item = ReviewItem(
        id="rev-001",
        query="今年公积金缴纳比例是多少？",
        draft_answer="公积金缴纳比例12%",
        confidence=0.72,
        retrieval_docs=[{"doc_id": "d001", "score": 0.72}],
        priority=Priority.HIGH,
        created_at=datetime.utcnow(),
    )
    queue.enqueue(item)
    assigned = queue.dequeue("agent-01")
    print(f"Assigned: {assigned.id} to {assigned.assigned_to}")
    # 输出: Assigned: rev-001 to agent-01

面试可能问

Q1: 你们怎么定义「幻觉」？有什么量化指标？

我们把幻觉分为三级。A 级：编造不存在的信息（如回答引用了不存在的制度文件）；B 级：信息偏差（如年假 15 天说成 10 天）；C 级：信息不完整（漏掉了关键限定条件「员工每年」）。量化指标是三级加权幻觉率 = (A×3 + B×2 + C×1) / 总回答数。A 级最严重，权重最高。上线强制引用后，A 级幻觉率从 3.1% 降至 0.4%。

Q2: 人工复核成了瓶颈怎么办？

两个方向。1) 技术侧：优化检索和 Rerank，提升高置信度回答的比例。我们的目标是直接回复率从 65% 提升到 80%，这需要把 NDCG@10 从 0.81 提升到 0.88。2) 流程侧：对高频问题（如「年假几天」「报销流程」）建立答案模板库，复核员可以直接选择模板回复而非逐条编辑。通过模板库，单条复核时间从 30 秒降至 8 秒。

Q3: 置信度分数怎么计算的？为什么可信？

置信度 = 0.3×检索分 + 0.4×Rerank 分 + 0.3×引用分。检索分是 BM25+向量融合后 top-3 的平均分。Rerank 分是 Cross-Encoder 对 question-doc pair 的打分。引用分是 Citation Guard 校验是否通过。我们通过 2000 条带标签的测试集验证了这个公式的可靠性——分数 ≥0.85 的回答中，实际正确的比例为 96.2%；<0.60 的回答中，实际正确的比例仅 31.5%。

"""检索指标基准评测"""

from typing import List, Set
import numpy as np

class RetrievalBenchmark:
    """检索指标基准"""

    def __init__(self, test_queries: List[str], qrels: dict):
        self.queries = test_queries
        self.qrels = qrels  # {qid: [relevant_doc_ids]}

    def recall_at_k(
        self, relevant: Set[str], results: List[str], k: int
    ) -> float:
        if not relevant:
            return 1.0
        return sum(1 for d in results[:k] if d in relevant) / len(relevant)

    def mrr_at_k(
        self, relevant: Set[str], results: List[str], k: int
    ) -> float:
        for i, d in enumerate(results[:k]):
            if d in relevant:
                return 1.0 / (i + 1)
        return 0.0

    def ndcg_at_k(
        self, relevant: Set[str], results: List[str], k: int
    ) -> float:
        dcg = 0.0
        for i, d in enumerate(results[:k]):
            rel = 1.0 if d in relevant else 0.0
            dcg += rel / np.log2(i + 2)
        idcg = sum(1.0 / np.log2(i + 2) for i in range(min(len(relevant), k)))
        return dcg / idcg if idcg > 0 else 0.0

    def evaluate(self, retriever, k: int = 10) -> dict:
        """在测试集上评测检索器"""
        recall_scores, mrr_scores, ndcg_scores = [], [], []
        for qid, query in enumerate(self.queries):
            results = retriever.search(query, top_k=k)
            result_ids = [r["doc_id"] for r in results]
            relevant = set(self.qrels.get(str(qid), []))
            recall_scores.append(self.recall_at_k(relevant, result_ids, k))
            mrr_scores.append(self.mrr_at_k(relevant, result_ids, k))
            ndcg_scores.append(self.ndcg_at_k(relevant, result_ids, k))

        return {
            f"recall@{k}": round(np.mean(recall_scores), 4),
            f"mrr@{k}": round(np.mean(mrr_scores), 4),
            f"ndcg@{k}": round(np.mean(ndcg_scores), 4),
        }

from dataclasses import dataclass
from typing import Optional

@dataclass
class BadCase:
    question: str
    answer: str
    retrieved_docs: list[dict]
    ground_truth: str
    root_cause: Optional[str] = None
    fix: Optional[str] = None

class BadCaseAnalyzer:
    """坏例归因分析器"""

    def __init__(self, retriever, chunker, llm):
        self.retriever = retriever
        self.chunker = chunker
        self.llm = llm

    def analyze(self, case: BadCase) -> BadCase:
        """四步法分析坏例根因"""

        # Step 1: 检索检查
        docs = self.retriever.search(case.question, top_k=10)
        docs_contain_answer = any(
            case.ground_truth in d.get("text", "")
            for d in docs
        )
        if not docs_contain_answer:
            case.root_cause = "检索失败"
            case.fix = "检查 Embedding 模型/BM25 参数/查询重写"
            return case

        # Step 2: 分块检查
        relevant_chunks = [
            d for d in docs if case.ground_truth in d.get("text", "")
        ]
        chunk_complete = any(
            len(d.get("text", "")) > len(case.ground_truth) * 2
            for d in relevant_chunks
        )
        if not chunk_complete:
            case.root_cause = "分块截断"
            case.fix = "增大 chunk_size / overlap 或语义分块"
            return case

CorpAssist 场景

员工问「加班费怎么算」。系统回答「加班按 1.5 倍工资计算」。但正确答案是「工作日 1.5 倍，休息日 2 倍，节假日 3 倍」。四步法归因：Step 1 检索——检索到的文档 chunk 包含「加班费按以下标准计算：工作日 150%，休息日 200%，节假日 300%」。Step 2 分块——但 chunk_size=256 只截取到「工作日 150%」，后续内容在下一个 chunk 未被送入 LLM。根因：分块截断。修复：chunk_size 增至 384 或 overlap 增至 80。

面试可能问

Q1: 你用什么指标评估 RAG 系统的效果？

分两层。检索层：Recall@10（目标 ≥85%）、MRR@10（≥0.90）、NDCG@10（≥0.85）。回答层：RAGAS 的 faithfulness（≥0.85）、answer_relevancy（≥0.80）、context_precision（≥0.75）。外加业务指标如用户满意度评分、答案采纳率。在 CorpAssist 的实践中，当检索 NDCG@10 达到 0.85 时，答案采纳率约 83%，而当 NDCG 降至 0.72 时，采纳率降至 61%。

Q2: 发现一个坏例，怎么快速定位是检索问题还是生成问题？

用「人间检验法」：把检索到的 Top-3 文档直接给一个人类同事看，问他能不能回答用户的问题。如果能，说明信息在手，根因在生成侧（Prompt/LLM）；如果不能，说明信息不在，根因在检索侧（检索精度或分块策略）。这个检验方法 30 秒就能执行，不需要任何系统日志分析。在我们团队，每个坏例先用「人间检验」再走正式归因流程。

Q3: 增量索引和全量重建怎么配合？

增量索引处理日常的文档增/删/改，每 30 分钟执行一次，每次约 30 秒。全量重建每周日凌晨 3 点执行一次，重建完整索引。重建期间流量路由到旧索引（蓝绿部署），新索引构建完成后用一小时验证检索指标（NDCG@10 降幅 < 0.02），再切换流量。如果全量重建失败，自动回滚到旧索引。这样增量索引保证实时性，全量重建保证一致性。

判断练习

如果 RAGAS 的 faithfulness 是 0.72 但 answer_relevancy 是 0.90，说明什么问题？

说明 LLM 的回答确实切题（相关性高），但没有基于检索到的文档内容（忠实度低）。这通常是 Prompt 缺少强制约束——模型用了自己的知识而非检索文档。解决方案：在 System Prompt 中加入「必须基于检索到的文档内容回答，不能使用自己的知识」指令；降低 temperature；引入 Citation Guard 强制引用。

蓝绿部署时，如何验证新索引的检索质量？

三个步骤。1) 自动化：在测试集上运行 retrieval benchmark，比较新旧索引的 Recall/MRR/NDCG，差异 < 0.02 才通过。2) 抽样：从新索引中随机抽取 50 条 query，人工检查检索结果质量。3) 金丝雀：将 5% 的流量切到新索引，监控 24 小时的用户满意度、拒答率、幻觉率。全部通过后再全量切流。我们遇到过新索引在某类文档上 Recall 降了 0.05 但整体指标没变的情况，金丝雀阶段发现了这个回归。

"""生产级脏 PDF 处理管线"""

import fitz  # PyMuPDF
import re
from typing import List, Dict

class ProductionPDFProcessor:
    """生产级脏 PDF 处理器"""

    @staticmethod
    def detect_text_layer(pdf_path: str) -> bool:
        """检测 PDF 是否包含文本层"""
        doc = fitz.open(pdf_path)
        total_chars = sum(len(page.get_text()) for page in doc)
        return total_chars > 50  # 少于50字符认为无文本层

    @staticmethod
    def clean_watermark(text: str) -> str:
        """清理水印和页眉页脚干扰"""
        patterns = [
            r"机密\s*文档",
            r"仅限内部使用",
            r"CONFIDENTIAL",
            r"第\s*\d+\s*页[，,]\s*共\s*\d+\s*页",
            r"CorpAssist\s*Internal",
            r"^\s*\d+\s*$",  # 纯页码行
        ]
        for p in patterns:
            text = re.sub(p, "", text, flags=re.MULTILINE | re.IGNORECASE)
        # 去除页眉页脚：页面顶部和底部 10%
        lines = text.split("\n")
        threshold = max(1, len(lines) // 10)
        cleaned = lines[threshold:-threshold] if len(lines) > threshold * 2 else lines
        return "\n".join(cleaned)

    @staticmethod
    def fix_encoding(text: str) -> str:
        """修复常见编码问题"""
        fixes = {
            "�": "",  # 替换 Unicode 替换字符
            "â€™": "'",    # 弯引号错误
            "â€œ": "\"",   # 左引号
            "â€": "\"",    # 右引号
        }
        for wrong, correct in fixes.items():
            text = text.replace(wrong, correct)
        return text

    def process(self, pdf_path: str) -> str:
        """完整的脏 PDF 处理流程"""
        if not self.detect_text_layer(pdf_path):
            # 扫描件 -> OCR
            from src.dirty_doc import DirtyDocProcessor
            proc = DirtyDocProcessor()
            return proc.process_pdf_page(pdf_path, 1)  # 简化示例

        doc = fitz.open(pdf_path)
        full_text = ""
        for page in doc:
            text = page.get_text()
            text = self.clean_watermark(text)
            text = self.fix_encoding(text)
            full_text += text + "\n"
        return full_text

from typing import List, Set

class ACLFilter:
    """检索层 ACL 过滤器"""

    def __init__(self, user_roles: List[str]):
        self.user_roles = set(user_roles)

    def milvus_filter(self) -> str:
        """生成 Milvus 权限过滤表达式"""
        if "admin" in self.user_roles:
            return ""  # 管理员免过滤
        tags = [f"acl_tags contains '{role}'" for role in self.user_roles]
        return " or ".join(tags)

    def es_filter(self) -> dict:
        """生成 ES 权限过滤 DSL"""
        if "admin" in self.user_roles:
            return {"match_all": {}}
        return {
            "terms": {
                "acl_tags": list(self.user_roles)
            }
        }

知识图谱补召回

纯 RAG 对跨文档、多跳问题效果较差。CorpAssist 用知识图谱（KG）构建文档间的实体关系，在检索时用 KG 做「补召回」。

    B --> C[找到: 休假制度-产假天数]
    B --> D[未找到: 薪资制度-产假工资]
    C --> E[KG 关系: 休假制度 --引用--> 薪资制度]
    E --> F[补召回: 薪资制度-产假工资]

"""知识图谱补召回"""

from typing import List, Dict, Set
import networkx as nx

class KGRecallBooster:
    """知识图谱补召回"""

    def __init__(self):
        self.graph = nx.DiGraph()
        self._build_graph()

    def _build_graph(self):
        """从文档关系构建知识图谱"""
        relations = [
            ("休假制度", "引用", "薪资制度"),
            ("休假制度", "引用", "绩效制度"),
            ("薪资制度", "引用", "社保制度"),
            ("报销制度", "依赖", "财务制度"),
        ]
        for src, rel, tgt in relations:
            self.graph.add_edge(src, tgt, relation=rel)

    def boost(
        self, query: str, retrieved_docs: List[str], max_extra: int = 3
    ) -> List[str]:
        """基于 KG 补召回"""
        extra_docs: Set[str] = set()

        for doc in retrieved_docs:
            # 找出文档关联的其他文档
            if doc in self.graph:
                for neighbor in self.graph.neighbors(doc):
                    if neighbor not in retrieved_docs:
                        extra_docs.add(neighbor)
                for predecessor in self.graph.predecessors(doc):
                    if predecessor not in retrieved_docs:
                        extra_docs.add(predecessor)

        return list(extra_docs)[:max_extra]

    def search_with_boost(self, query: str, retriever, top_k: int = 5) -> List[Dict]:
        """带 KG 补召回的综合检索"""
        results = retriever.search(query, top_k=top_k)
        retrieved_titles = [r.get("doc_title", "") for r in results]
        extra = self.boost(query, retrieved_titles)
        if extra:
            # 从索引中获取补充文档
            extra_results = retriever.get_docs_by_titles(extra)
            results.extend(extra_results)
        return results[:top_k]

if __name__ == "__main__":

    extra = booster.boost("产假工资", ["休假制度"])

面试可能问

Q1: 你们怎么处理一个 PDF 既有文本层又有水印覆盖的情况？

三步。1) 用 PyMuPDF 提取文本层文字 -> 得到含「机密」水印的文本。2) 用正则过滤水印模式（机密文档、仅限内部使用等 15 种常见模式）。3) 同时用 OCR 提取图像的底层文字（因为水印覆盖的区域底层文字 OCR 可能还能识别），比较文本层和 OCR 结果，保留信息更完整的那一份。在我们的测试中，这种方法比单纯文本层提取或单纯 OCR 的准确率分别高 8% 和 12%。

Q2: ACL 检索层过滤有没有可能被绕过？

理论上没有，因为我们做了三层防护。第一层 API 层：从 JWT Token 提取用户角色，不可由客户端修改。第二层检索层：ES 和 Milvus 的查询都携带 ACL filter，由服务端生成。第三层输出层：即使出现极端 bug 导致越权结果被检索出来，LLM 回答时引用的文档元数据也会暴露来源文档的信息，审计日志会记录所有检索的文档 ID。实际上我们最担心的是「配置错误」而非「攻击绕过」——比如文档的 acl_tags 管理员忘记配置，默认成了全员可见。

Q3: 知识图谱补召回会不会引入更多噪音？

会。在我们的测试中，KG 补召回会额外引入约 30% 的不相关文档。我们的处理方式：1) 补召回的文档经过 BGE-Reranker 二次过滤；2) KG 关系只保留直接引用（一跳），不扩展到二跳；3) 对补召回文档赋予较低的初始融合权重（0.3 vs 原始检索的 0.7）。通过这些措施，噪音率控制在 5% 以下，跨文档问答准确率净提升 15%。

判断练习

如果发现用户搜到了不应该看到的文档，第一步应该做什么？

不要先修代码，先止血。第一步：检查该用户的 JWT Token 中的角色列表，确认是不是角色权限配置错误。第二步：检查该文档的 acl_tags 是否配置正确。如果文档的 acl_tags 被管理员误设为「全员可见」，直接修正文档权限标签并触发增量索引即可。如果配置正确但检索仍然命中，则问题在 ACL Filter 的代码实现中。我们经历过的情况里，70% 是管理员配置错误，30% 是代码 bug。

全量重建索引时，用户在查询，怎么保证不中断服务？

蓝绿部署。在 Milvus 中维护两个 collection（corp_chunks_v2 和 corp_chunks_v3），ES 维护两个 index。全量重建时构建新的 collection/index，构建期间流量路由到旧版。构建完成后先内部验证检索指标，然后切换 5% 流量金丝雀测试，最后全量切流。旧版保留至少 24 小时用于回滚。在我们的实践中，切换过程对用户完全无感——P99 延迟只增加了 30ms（路由层开销）。
