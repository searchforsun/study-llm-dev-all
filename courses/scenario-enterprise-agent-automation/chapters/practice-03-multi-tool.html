Python：并行工具调用

python
# parallel_tool_executor.py
import asyncio
from typing import List, Dict, Any, Callable

class ParallelToolExecutor:
    def __init__(self, timeout_sec: int = 30):
        self.timeout = timeout_sec

    async def execute_tool(self, tool_call: Callable,
                           args: dict) -> Dict[str, Any]:
        try:
            result = await asyncio.wait_for(
                tool_call(**args), timeout=self.timeout)
            return {"success": True, "result": result}
        except asyncio.TimeoutError:
            return {"success": False, "error": "timeout"}
        except Exception as e:
            return {"success": False, "error": str(e)}

    async def execute_parallel(self, calls: List[tuple]) -> List[Dict]:
        """并行执行多个工具调用"""
        tasks = [self.execute_tool(fn, args) for fn, args in calls]
        return await asyncio.gather(*tasks)

    def execute_serial(self, calls: List[tuple]) -> List[Dict]:
        """串行执行，前一步结果可影响后一步"""
        results = []
        for fn, args in calls:
            # 可以将前一步结果注入 args
            result = asyncio.run(self.execute_tool(fn, args))
            results.append(result)
            if not result["success"]:
                break  # 串行调用中断
        return results

# 使用示例：会议安排
async def arrange_meeting():
    executor = ParallelToolExecutor()

    # 串行：先查主会议室
    room_result = await executor.execute_tool(
        calendar.query, {"room": "A201", "time": "2025-06-01T15:00"})
    if not room_result["success"]:
        return "会议室已被占用"

    # 并行：查所有参会人
    attendees = ["zhangsan", "lisi", "wangwu"]
    parallel_calls = [
        (calendar.query, {"user": u, "time": "2025-06-01T15:00"})
        for u in attendees
    ]
    cal_results = await executor.execute_parallel(parallel_calls)

    # 处理部分失败
    failed = [r for r in cal_results if not r["success"]]
    if failed:
        return f"部分参会人日历查询失败: {len(failed)}人"
    return "全员空闲，可以创建会议"

Java：CompletableFuture 并行调用

java
// ParallelToolService.java
@Service
public class ParallelToolService {

    private final ExecutorService executor =
        Executors.newFixedThreadPool(10);
    private final Duration timeout = Duration.ofSeconds(30);

    public List<ToolResult> executeParallel(List<ToolCall> calls) {
        var futures = calls.stream()
            .map(call -> CompletableFuture
                .supplyAsync(() -> executeSingle(call), executor)
                .orTimeout(timeout.toMillis(), TimeUnit.MILLISECONDS)
                .exceptionally(ex -> ToolResult.failure(call.id(), ex.getMessage())))
            .toList();

        return futures.stream()
            .map(CompletableFuture::join)
            .toList();
    }

    private ToolResult executeSingle(ToolCall call) {
        try {
            String result = toolRegistry.call(call.name(), call.args());
            return ToolResult.success(call.id(), result);
        } catch (Exception e) {
            return ToolResult.failure(call.id(), e.getMessage());
        }
    }

    // 串行执行，失败即终止
    public List<ToolResult> executeSerial(List<ToolCall> calls) {
        var results = new ArrayList<ToolResult>();
        for (ToolCall call : calls) {
            ToolResult result = executeSingle(call);
            results.add(result);
            if (!result.isSuccess()) break;
        }
        return results;
    }
}

工具错误处理与重试

tool retry 处理工具调用的瞬时失败。办公工具的常见失败模式：

失败类型概率处理策略
网络超时（5xx）高指数退避重试 3 次
限流（429）中等待后重试，增加间隔
参数错误（4xx）低记录日志，返回 Agent 修正
数据不存在（404）低返回空结果，继续执行
python
# resilient_tool.py
import time
from functools import wraps

def resilient_tool(max_retries=3, base_delay=1.0,
                   retryable_exceptions=(TimeoutError, ConnectionError)):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_error = None
            for attempt in range(max_retries):
                try:
                    return func(*args, **kwargs)
                except retryable_exceptions as e:
                    last_error = e
                    if attempt < max_retries - 1:
                        delay = base_delay * (2 ** attempt)
                        time.sleep(delay)
                except Exception as e:
                    # 非可重试异常，立即返回错误
                    return {"error": str(e), "fatal": True}
            return {"error": str(last_error), "retries_exhausted": True}
        return wrapper
    return decorator

@resilient_tool(max_retries=3)
def call_email_api(to: str, subject: str, body: str) -> dict:
    return email_client.send(to, subject, body)

partial failure 是并行工具调用的典型问题：部分工具成功、部分失败。Agent 应处理这种「部分成功」状态，例如：会议创建成功了但通知邮件失败了，应标记「部分完成」并记录异常。

工具超时与隔离设计

每个工具的失败不应该影响其他工具的正常执行。工具隔离（Tool Isolation）通过以下机制实现：

  • 线程隔离：每个工具在独立的线程池中执行，一个工具阻塞不阻塞其他工具
  • 超时隔离：每个工具有独立的超时配置，超时不影响其他正在执行的工具
  • 熔断隔离：每个工具有独立的熔断状态，一个工具熔断不影响其他工具
java
// ToolIsolationConfig.java
@Configuration
public class ToolIsolationConfig {

    @Bean
    public ExecutorService calendarExecutor() {
        return Executors.newFixedThreadPool(5,
            new ThreadFactoryBuilder().setNameFormat("cal-tool-%d").build());
    }

    @Bean
    public ExecutorService emailExecutor() {
        return Executors.newFixedThreadPool(3,
            new ThreadFactoryBuilder().setNameFormat("email-tool-%d").build());
    }

    @Bean
    public ExecutorService ticketExecutor() {
        return Executors.newFixedThreadPool(3,
            new ThreadFactoryBuilder().setNameFormat("ticket-tool-%d").build());
    }
}

// 使用示例
@Service
public class IsolatedToolService {

    private final ExecutorService calendarExec;

    public CompletableFuture<ToolResult> queryCalendar(String user, String time) {
        return CompletableFuture
            .supplyAsync(() -> calendarClient.query(user, time), calendarExec)
            .orTimeout(10, TimeUnit.SECONDS);
    }
}

Python 侧用 asyncio 的 Semaphore 实现线程隔离：

python
import asyncio

class ToolPool:
    def __init__(self, name: str, max_concurrent: int = 5):
        self.name = name
        self.semaphore = asyncio.Semaphore(max_concurrent)

    async def execute(self, fn, *args, **kwargs):
        async with self.semaphore:
            return await fn(*args, **kwargs)

# 每个工具有独立的连接池
calendar_pool = ToolPool("calendar", max_concurrent=5)
email_pool = ToolPool("email", max_concurrent=3)

async def safe_query_calendar(user: str):
    return await calendar_pool.execute(calendar_api.query, user)

办公工具编排最佳实践

office tools 的编排原则：

  • 读优于写：查询类工具优先并行执行，修改类工具串行执行
  • 先检查再操作：创建/发送前先查询权限和状态
  • 幂等设计：每个写操作带幂等键，重复调用不产生副作用
  • 部分成功处理：并行调用中部分失败不影响已成功的部分
  • 隔离设计：每个工具有独立的线程池和超时配置
不推荐 — 全串行或全并行
  • 所有工具串行调用，延迟叠加
  • 无依赖工具也串行，浪费等待时间
  • 一个工具失败导致整个流程中止
  • 所有工具共享线程池，一个阻塞全阻塞
推荐 — 按依赖编排+隔离
  • 无依赖的查询并行执行（查多人日历）
  • 有依赖的步骤串行执行（查完再创建）
  • 部分失败仅影响相关分支
  • 每个工具独立线程池+超时

本章结论

  • 多工具编排分并行和串行，按依赖关系决定
  • 并行调用用 asyncio（Python）或 CompletableFuture（Java）
  • 工具失败用指数退避重试，不可重试异常记录后返回
  • 部分失败是并行调用的常态，需设计兼容策略
  • 读操作并行、写操作串行+幂等键

常见问题（FAQ）

并行调用的并发数设多少合适？
取决于 API 限流和系统资源。办公场景建议硬件限制为 10。如果调用企业微信/钉钉 API，注意它们的 QPS 限制（通常 5-20/s），并发数不要超过限流阈值。
并行调用中一个失败，其它要不要回滚？
取决于业务场景。查询类失败无需回滚。写操作类（如批量创建工单）建议用 Saga 模式：先记录已成功部分，失败部分补偿或人工处理。
工具调用顺序是谁决定的？Agent 还是代码？
推荐 Agent 决定「调什么」，代码决定「怎么调」。Agent 的 Planner 决定工具调用顺序和参数，工具的执行引擎决定并行/串行和执行策略。
