Supervisor-Worker 架构

supervisor agent 负责拆解任务并分派给 worker agent，role division 是核心设计原则。

Python：Supervisor-Worker 实现

python
# multi_agent_supervisor.py
from langgraph.graph import StateGraph, END
from typing import TypedDict, List, Dict

class AgentTask(TypedDict):
    worker_id: str
    task: str
    params: Dict
    status: str
    result: str

class MultiAgentState(TypedDict):
    user_input: str
    plan: List[AgentTask]
    results: Dict[str, str]
    conflicts: List[str]
    final_output: str

def supervisor_node(state: MultiAgentState) -> dict:
    """Supervisor：分析输入，拆解任务，分派给 Worker"""
    prompt = f"分析采购请求，拆解为并行子任务：{state['user_input']}"
    plan = llm.invoke(prompt)
    tasks = []
    for item in parse_plan(plan):
        tasks.append(AgentTask(
            worker_id=item["worker"],
            task=item["description"],
            params=item["params"],
            status="pending",
            result=""
        ))
    return {"plan": tasks}

def worker_node(state: MultiAgentState, worker_id: str) -> dict:
    """Worker：执行指定任务"""
    task = next(t for t in state["plan"] if t["worker_id"] == worker_id)
    try:
        result = execute_worker_task(worker_id, task)
        return {"results": {worker_id: result},
                "plan": [update_status(task, "done")]}
    except Exception as e:
        return {"results": {worker_id: f"ERROR: {str(e)}"},
                "plan": [update_status(task, "failed")]}

def aggregator_node(state: MultiAgentState) -> dict:
    """聚合所有 Worker 结果，检查冲突"""
    conflicts = []
    # 检查预算 vs 政策冲突
    budget = state["results"].get("budget_worker", "")
    compliance = state["results"].get("compliance_worker", "")
    if "超预算" in budget and "合规" in compliance:
        conflicts.append("预算不足但合规允许，需人工审核")
    return {"conflicts": conflicts}

def router(state: MultiAgentState) -> str:
    if state["conflicts"]:
        return "human_review"
    return "output"

# 构建多 Agent Graph
builder = StateGraph(MultiAgentState)
builder.add_node("supervisor", supervisor_node)
builder.add_node("budget_worker", lambda s: worker_node(s, "budget_worker"))
builder.add_node("inventory_worker", lambda s: worker_node(s, "inventory_worker"))
builder.add_node("compliance_worker", lambda s: worker_node(s, "compliance_worker"))
builder.add_node("aggregator", aggregator_node)
builder.add_node("human_review", human_approval_node)
builder.add_node("output", output_node)

builder.set_entry_point("supervisor")
builder.add_edge("supervisor", "budget_worker")
builder.add_edge("supervisor", "inventory_worker")
builder.add_edge("supervisor", "compliance_worker")
builder.add_edge("budget_worker", "aggregator")
builder.add_edge("inventory_worker", "aggregator")
builder.add_edge("compliance_worker", "aggregator")
builder.add_conditional_edges("aggregator", router)
builder.add_edge("human_review", "output")
builder.add_edge("output", END)

// AgentMessageBus.java
@Component
public class AgentMessageBus {

    private final RocketMQTemplate mqTemplate;

    public AgentMessageBus(RocketMQTemplate mqTemplate) {
        this.mqTemplate = mqTemplate;
    }

    public void dispatchTask(AgentTask task) {
        // 根据 worker_id 路由到不同队列
        String destination = "agent-task:" + task.getWorkerId();
        mqTemplate.syncSend(destination, task);
    }

    @RocketMQMessageListener(topic = "agent-result", consumerGroup = "aggregator")
    public class ResultListener implements RocketMQListener<WorkerResult> {
        @Override
        public void onMessage(WorkerResult result) {
            // 聚合 Worker 结果
            aggregator.collect(result);
        }
    }
}

public class BudgetWorker {

冲突解决

conflict resolve 处理多 Agent 决策不一致。典型冲突类型：

冲突类型示例解决策略
预算 vs 需求预算充足但政策限制走人工审批升级
数据不一致两个系统库存数据不同以 ERP 为准，记录差异
权限冲突A Agent 允许、B Agent 拒绝按最小权限原则，取严格
时间冲突会议 Agent 和出差 Agent 约同一时间优先级排序 + 人工确认

handoff agent 是 Agent 间交接任务的模式。一个 Worker 完成后将上下文传递给下一个 Worker 或 Supervisor。

Agent 注册与发现机制

在微服务架构中，多 Agent 需要服务注册与发现机制。每个 Worker Agent 启动时向注册中心注册自己的能力和地址，Supervisor 通过查询注册中心找到合适的 Worker。

python
# agent_registry.py
from dataclasses import dataclass, field
from typing import Dict, List
import redis

@dataclass
class AgentCapability:
    agent_id: str
    agent_type: str  # supervisor / worker
    capabilities: List[str]  # ["budget_query", "inventory_query"]
    endpoint: str  # gRPC / HTTP 地址
    status: str = "active"  # active / busy / offline

class AgentRegistry:
    def __init__(self, redis_client):
        self.redis = redis_client

    def register(self, capability: AgentCapability):
        key = f"agent:{capability.agent_id}"
        self.redis.hset(key, mapping={
            "type": capability.agent_type,
            "capabilities": ",".join(capability.capabilities),
            "endpoint": capability.endpoint,
            "status": capability.status
        })
        self.redis.expire(key, 30)  # 30 秒心跳

    def discover(self, required_capability: str) -> List[AgentCapability]:
        """发现具有特定能力的 Agent"""
        agents = []
        for key in self.redis.scan_iter("agent:*"):
            data = self.redis.hgetall(key)
            if required_capability in data.get("capabilities", ""):
                agents.append(AgentCapability(
                    agent_id=key.split(":")[1],
                    agent_type=data["type"],
                    capabilities=data["capabilities"].split(","),
                    endpoint=data["endpoint"],
                    status=data["status"]
                ))
        return agents

不推荐 — 单 Agent 做所有事
  • 一个 Agent 既查预算又查库存还做合规
  • Agent 上下文被多种职责撑爆
  • 修改一个职责影响整个 Agent
推荐 — 职责分离
  • 每个 Agent 单一职责（查询/执行/审核）
  • Supervisor 管理生命周期
  • 消息总线解耦通信

本章结论

  • Supervisor-Worker 是多 Agent 的标准架构模式
  • 消息总线（RocketMQ/Kafka）解耦 Agent 间通信
  • 冲突解决策略：审计优先、最小权限、人工升级
  • 每个 Worker Agent 单一职责，可独立部署和扩缩
  • Python 做编排，Java 做治理的双栈分工

常见问题（FAQ）

Supervisor Agent 自身会不会成为瓶颈？
会。Supervisor 的决策质量决定整体效果。建议：Supervisor 只做任务分派不做事执行；给 Supervisor 设置 max_steps 限制；支持 Supervisor 超时后由人工接管。
Worker Agent 数量太多怎么办？
按职能分组，每组一个 Agent。例如「财务组」包含预算查询 Agent 和费用报销 Agent。分组后的 Agent 数量控制在 10 个以内。
Agent 间通信延迟怎么处理？
消息总线（Kafka/RocketMQ）自身有延迟，通常毫秒级。如果 Agent 需要同步等待结果，设置超时时间并在超时后走降级（如使用缓存数据）。
