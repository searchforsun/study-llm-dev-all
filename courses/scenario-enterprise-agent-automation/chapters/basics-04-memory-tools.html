短期与长期记忆分层

short long memory 是 Agent 记忆系统的核心。短期记忆保存当前会话上下文，长期记忆存储用户偏好和跨会话知识。

memory tier 设计：短期存会话上下文，长期存用户偏好和摘要。

memory overflow 是常见问题：Agent 会话历史无限增长，最终撑爆上下文窗口。解决方案是分层 + 摘要 + 滑动窗口。

python
# memory_manager.py
from collections import deque
from typing import List, Dict
import json

class MemoryManager:
    def __init__(self, window_size: int = 20):
        self.short_term = deque(maxlen=window_size)  # 滑动窗口
        self.long_term: Dict[str, str] = {}  # 用户偏好
        self.summary: str = ""

    def add_short(self, item: dict):
        self.short_term.append(item)

    def summarize_and_archive(self):
        """将短期记忆摘要后归档到长期"""
        if len(self.short_term) >= self.short_term.maxlen:
            recent = list(self.short_term)[-5:]  # 保留最近 5 条
            old = list(self.short_term)[:-5]
            self.summary = self._generate_summary(old)
            self.short_term = deque(recent, maxlen=20)

    def get_context(self) -> str:
        """组装最终上下文：摘要+短期+用户偏好"""
        parts = []
        if self.summary:
            parts.append(f"[历史摘要] {self.summary}")
        if self.short_term:
            parts.append(f"[当前会话] {json.dumps(list(self.short_term))}")
        if self.long_term:
            parts.append(f"[用户偏好] {json.dumps(self.long_term)}")
        return "\n".join(parts)

    def _generate_summary(self, history: list) -> str:
        # 调用轻量 LLM 生成摘要
        return f"用户已完成 {len(history)} 轮交互"

Java：Redis 滑动窗口实现

java
// MemoryService.java
@Service
public class MemoryService {
    private final StringRedisTemplate redis;
    private final int WINDOW_SIZE = 20;
    private final Duration TTL = Duration.ofHours(1);

    public void addMessage(String sessionId, String message) {
        String key = "session:" + sessionId;
        redis.opsForList().rightPush(key, message);
        redis.opsForList().trim(key, 0, WINDOW_SIZE - 1);
        redis.expire(key, TTL);
    }

    public List<String> getContext(String sessionId) {
        String key = "session:" + sessionId;
        return redis.opsForList().range(key, 0, -1);
    }

    public void archiveSummary(String sessionId, String summary) {
        // 将摘要写入长期存储（MySQL / Vector DB）
        String archiveKey = "archive:" + sessionId;
        redis.opsForValue().set(archiveKey, summary, Duration.ofDays(30));
    }
}

mcp office 场景下，工具契约（tool contract）定义了工具的名称、参数、返回值和错误码。契约必须显式声明，Agent 才能正确调用：

python
# mcp_tool_registry.py
from dataclasses import dataclass, field
from typing import Dict, Any, List

@dataclass
class ToolContract:
    name: str
    description: str
    parameters: Dict[str, str]  # 参数名 -> 类型
    required: List[str]
    returns: str
    permission: str  # read/write/admin
    timeout_sec: int = 30

class MCPToolRegistry:
    def __init__(self):
        self.tools: Dict[str, ToolContract] = {}

    def register(self, contract: ToolContract):
        self.tools[contract.name] = contract

    def validate_call(self, tool_name: str, args: Dict[str, Any],
                      user_role: str) -> bool:
        contract = self.tools.get(tool_name)
        if not contract:
            return False
        # 检查权限
        if contract.permission == "admin" and user_role != "admin":
            return False
        # 检查必需参数
        for param in contract.required:
            if param not in args:
                return False
        return True

# 注册办公工具
registry = MCPToolRegistry()
registry.register(ToolContract(
    name="send_email",
    description="发送邮件",
    parameters={"to": "string", "subject": "string", "body": "string"},
    required=["to", "subject"],
    returns="message_id",
    permission="write",
    timeout_sec=30
))

工具结果缓存策略

短期内重复调用同一工具会浪费 Token 和 API 配额。工具结果缓存（Tool Result Caching）有效减少重复调用：

python
# tool_cache.py
import hashlib
import json
import time
from typing import Dict, Any, Optional

class ToolResultCache:
    def __init__(self, ttl_sec: int = 60):
        self.cache: Dict[str, tuple] = {}
        self.ttl = ttl_sec

    def _make_key(self, tool_name: str, args: dict) -> str:
        raw = f"{tool_name}:{json.dumps(args, sort_keys=True)}"
        return hashlib.md5(raw.encode()).hexdigest()

    def get(self, tool_name: str, args: dict) -> Optional[str]:
        key = self._make_key(tool_name, args)
        if key in self.cache:
            result, timestamp = self.cache[key]
            if time.time() - timestamp < self.ttl:
                return result
            del self.cache[key]
        return None

    def set(self, tool_name: str, args: dict, result: str):
        key = self._make_key(tool_name, args)
        self.cache[key] = (result, time.time())

    def invalidate(self, tool_name: str):
        """工具数据变化时清除相关缓存"""
        keys_to_delete = [k for k in self.cache if k.startswith(tool_name)]
        for k in keys_to_delete:
            del self.cache[k]

# 使用示例
tool_cache = ToolResultCache(ttl_sec=120)

def cached_tool_call(tool_name: str, args: dict) -> str:
    cached = tool_cache.get(tool_name, args)
    if cached:
        return cached  # 命中缓存
    result = mcp_tools.call(tool_name, args)
    if not result.get("error"):
        tool_cache.set(tool_name, args, result)
    return result

缓存策略的选择：只读工具（查余额、查日历）缓存 60-120s；写操作（创建、提交）不缓存；查询结果带「数据时间戳」标识。

工具权限模型

tool permission 是防止 Agent 滥用工具的关键。每个工具按角色分层授权：

角色可读可写审批
普通员工个人日历、个人邮箱个人签名、个人请假超 3 天请假
部门经理部门日历、下属信息审批请假、发送部门通知超 1 万元付款
HR全员考勤、组织架构修改状态、发送全员通知敏感操作
不推荐 — 无权限控制
  • 所有工具对所有人开放
  • Agent 调用前不检查角色
  • 权限逻辑散落在 Prompt 里
推荐 — 分层授权
  • 每个工具标记 permission 等级
  • Agent 调用前校验角色
  • 权限配置在 YAML 中集中管理

常见问题（FAQ）

短期记忆和长期记忆的界限在哪里？
短期：当前会话上下文（通常 20 轮对话或 4096 tokens）。长期：跨会话的用户偏好、知识库、历史摘要。超过短期窗口的自动摘要归档到长期。
MCP Server 和传统 API Gateway 有什么区别？
MCP 专门为 LLM Agent 设计：工具描述（Function Calling Schema）是 MCP Server 的一部分，Agent 可以动态发现工具。传统 API Gateway 面向人类开发者，没有工具发现机制。
工具契约变了怎么办？
工具契约版本化。Agent 启动时从 MCP Server 拉取最新契约。契约不兼容时应有迁移期，期间新旧版本同时支持。
