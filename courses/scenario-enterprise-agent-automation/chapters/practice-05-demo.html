Python：Agent 监控埋点

python
# agent_monitor.py
import time
from dataclasses import dataclass, field
from typing import List, Dict
from datetime import datetime

@dataclass
class StepMetrics:
    step_id: int
    node_name: str
    tool_calls: List[Dict]
    input_tokens: int
    output_tokens: int
    duration_ms: float
    success: bool
    error: str = ""

class AgentMonitor:
    def __init__(self):
        self.steps: List[StepMetrics] = []

    def start_step(self, node_name: str) -> dict:
        return {
            "step_id": len(self.steps) + 1,
            "node_name": node_name,
            "start_time": time.time()
        }

    def end_step(self, step: dict, success: bool,
                 tool_calls=None, tokens=None, error=""):
        duration = (time.time() - step["start_time"]) * 1000
        metrics = StepMetrics(
            step_id=step["step_id"],
            node_name=step["node_name"],
            tool_calls=tool_calls or [],
            input_tokens=tokens.get("input", 0) if tokens else 0,
            output_tokens=tokens.get("output", 0) if tokens else 0,
            duration_ms=round(duration, 2),
            success=success,
            error=error
        )
        self.steps.append(metrics)
        return metrics

    def summary(self) -> dict:
        total_tokens = sum(s.input_tokens + s.output_tokens for s in self.steps)
        total_duration = sum(s.duration_ms for s in self.steps)
        success_rate = sum(1 for s in self.steps if s.success) / len(self.steps)

        return {
            "total_steps": len(self.steps),
            "total_tokens": total_tokens,
            "total_duration_ms": round(total_duration, 2),
            "success_rate": round(success_rate * 100, 1),
            "step_details": [
                {"step": s.step_id, "node": s.node_name,
                 "tokens": s.input_tokens + s.output_tokens,
                 "duration_ms": s.duration_ms, "success": s.success}
                for s in self.steps
            ]
        }

# LangGraph 集成
monitor = AgentMonitor()

def monitored_node(node_func):
    def wrapper(state):
        step = monitor.start_step(node_func.__name__)
        try:
            result = node_func(state)
            monitor.end_step(step, True)
            return result
        except Exception as e:
            monitor.end_step(step, False, error=str(e))
            raise
    return wrapper

Java：Micrometer 监控

java
// AgentMetricsService.java
@Configuration
public class AgentMetricsService {

    private final MeterRegistry meterRegistry;

    public AgentMetricsService(MeterRegistry meterRegistry) {
        this.meterRegistry = meterRegistry;
    }

    // 步数计数器
    public void recordStep(String agentId, String nodeName, boolean success) {
        meterRegistry.counter("agent.steps.total",
            "agent", agentId, "node", nodeName).increment();
        if (!success) {
            meterRegistry.counter("agent.steps.failed",
                "agent", agentId).increment();
        }
    }

    // Token 消耗记录
    public void recordTokenUsage(String agentId,
                                  int inputTokens, int outputTokens) {
        meterRegistry.counter("agent.tokens.input",
            "agent", agentId).increment(inputTokens);
        meterRegistry.counter("agent.tokens.output",
            "agent", agentId).increment(outputTokens);
    }

    // 工具调用延迟
    public void recordToolLatency(String toolName, long durationMs) {
        meterRegistry.timer("agent.tool.latency",
            "tool", toolName).record(Duration.ofMillis(durationMs));
    }
}

yaml
# prometheus-alerts.yml
groups:
  - name: agent-alerts
    rules:
      - alert: AgentHighLatency
        expr: agent_step_duration_seconds{quantile="0.99"} > 30
        for: 5m
        annotations:
          summary: "Agent 99 分位延迟超过 30s"
      - alert: AgentTokenSurge
        expr: rate(agent_tokens_total[5m]) > 10000
        annotations:
          summary: "Token 消耗异常激增"

常见问题（FAQ）

演示时 Agent 调用失败怎么办？
准备 fallback 演示方式：截屏展示之前的正确执行记录和监控数据。演示前用 Mock 模式跑通全链路，确保演示环境独立、不受外部 API 影响。
Token 消耗演示数据怎么量化？
在监控面板中展示：input_tokens、output_tokens、总消耗和预估成本。可以用成本计算器（如 0.002 元/千 tokens）把 Token 转化为人民币展示。
演示环境需要和线上一样吗？
不需要也可以。Demo 环境建议用独立的 Mock 工具，避免依赖外部系统。但演示的 Agent 流程逻辑必须和生产一致。
